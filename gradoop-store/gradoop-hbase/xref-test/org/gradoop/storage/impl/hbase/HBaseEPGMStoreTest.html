328   /**
329    * Test the truncate tables functionality.
330    */
331   @Test
332   public void truncateTablesTest() throws IOException {
333     HBaseEPGMStore store = createEmptyEPGMStore("truncateTest");
334     AsciiGraphLoader<EPGMGraphHead, EPGMVertex, EPGMEdge> loader = getMinimalFullFeaturedGraphLoader();
335     checkIfStoreIsEmpty("Store was not empty before writing anything.", store);
336     // Now write something to the store, check if it was written (i.e. the store is not empty).
337     EPGMGraphHead graphHead = loader.getGraphHeads().iterator().next();
338     EPGMVertex vertex = loader.getVertices().iterator().next();
339     EPGMEdge edge = loader.getEdges().iterator().next();
340     store.writeGraphHead(graphHead);
341     store.writeVertex(vertex);
342     store.writeEdge(edge);
343     store.flush();
344     validateGraphHead(store, graphHead);
345     validateVertex(store, vertex);
346     validateEdge(store, edge);
347     // Now truncate and check if the store if empty afterwards.
348     store.truncateTables();
349     checkIfStoreIsEmpty("Store was not empty after truncating.", store);
350     // Finally check if we can write elements.
351     store.writeGraphHead(graphHead);
352     store.writeVertex(vertex);
353     store.writeEdge(edge);
354     store.close();
355   }

357   /**
358    * Check if all tables of are store are empty.
359    *
360    * @param message The messenge used in the assertion.
361    * @param store The HBase store to check.
362    * @throws IOException when accessing the store fails.
363    */
364   private void checkIfStoreIsEmpty(String message, HBaseEPGMStore store) throws IOException {
365     for (ClosableIterator<?> space : Arrays.asList(store.getGraphSpace(),
366       store.getVertexSpace(), store.getEdgeSpace())) {
367       boolean hasNext = space.hasNext();
368       // Make sure to close the iterator before the assertion.
369       space.close();
370       assertFalse(hasNext, message);
371     }
372   }

374   private AsciiGraphLoader<EPGMGraphHead, EPGMVertex, EPGMEdge> getMinimalFullFeaturedGraphLoader() {
375     String asciiGraph = ":G{k:\"v\"}[(v:V{k:\"v\"}),(v)-[:e{k:\"v\"}]->(v)]";
376     return AsciiGraphLoader.fromString(asciiGraph, getEPGMElementFactoryProvider());
377   }

395   private void validateEdge(HBaseEPGMStore graphStore, EPGMEdge originalEdge) throws IOException {
396

398

401
