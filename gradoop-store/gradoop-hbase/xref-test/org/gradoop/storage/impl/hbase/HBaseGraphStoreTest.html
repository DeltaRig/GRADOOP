540       .stream()
541       .filter(g -> g.getLabel().equals("Community"))
542       .filter(g -> g.getPropertyValue(PROP_INTEREST).getString().equals("Hadoop") ||
543           g.getPropertyValue(PROP_INTEREST).getString().equals("Graphs"))
544       .collect(Collectors.toList());
545 
546     List<EPGMEdge> edges = getSocialEdges()
547       .stream()
548       .filter(e -> e.getLabel().matches(PATTERN_EDGE.pattern()) ||
549         (e.hasProperty(PROP_SINCE) && e.getPropertyValue(PROP_SINCE).getInt() < 2015))
550       .collect(Collectors.toList());
551 
552     List<EPGMVertex> vertices = getSocialVertices()
553       .stream()
554       .filter(v -> v.getLabel().equals("Person"))
555       .collect(Collectors.toList())
556       .subList(1, 4);
557 
558     // Query the store
559     List<EPGMGraphHead> graphHeadResult = epgmStores[storeIndex].getGraphSpace(
560       Query.elements()
561         .fromAll()
562         .where(HBaseFilters.<EPGMGraphHead>labelIn("Community")
563           .and(HBaseFilters.<EPGMGraphHead>propEquals(PROP_INTEREST, "Hadoop")
564             .or(HBaseFilters.propEquals(PROP_INTEREST, "Graphs")))))
565       .readRemainsAndClose();
566 
567     List<EPGMEdge> edgeResult = epgmStores[storeIndex].getEdgeSpace(
568       Query.elements()
569         .fromAll()
570         // WHERE edge.label LIKE '^has.*$' OR edge.since < 2015
571         .where(HBaseFilters.<EPGMEdge>labelReg(PATTERN_EDGE)
572           .or(HBaseFilters.<EPGMEdge>propLargerThan(PROP_SINCE, 2015, true).negate())))
573       .readRemainsAndClose();
574 
575     final HBaseElementFilter<EPGMVertex> vertexFilter = HBaseFilters.<EPGMVertex>labelIn("Person")
576       .and(HBaseFilters.<EPGMVertex>propEquals(PROP_NAME, vertices.get(0).getPropertyValue("name"))
577         .or(HBaseFilters.<EPGMVertex>propEquals(PROP_NAME, vertices.get(1).getPropertyValue("name"))
578           .or(HBaseFilters.propEquals(PROP_NAME, vertices.get(2).getPropertyValue("name")))));
579 
580     List<EPGMVertex> vertexResult = epgmStores[storeIndex].getVertexSpace(
581       Query.elements()
582         .fromAll()
583         .where(vertexFilter))
584       .readRemainsAndClose();
585 
586     assertEquals(graphHeadResult.size(), 2);
587     assertEquals(edgeResult.size(), 21);
588     assertEquals(vertexResult.size(), 3);
589 
590     validateElementCollections(graphHeads, graphHeadResult);
591     validateElementCollections(vertices, vertexResult);
592     validateElementCollections(edges, edgeResult);
593   }
