16  package org.gradoop.storage.hbase.impl.io;

20  import org.apache.hadoop.mapreduce.Job;
21  import org.gradoop.flink.io.api.DataSink;
22  import org.gradoop.flink.model.impl.epgm.GraphCollection;

24  import org.gradoop.flink.util.GradoopFlinkConfig;
25  import org.gradoop.storage.hbase.impl.io.functions.BuildEdgeMutation;
26  import org.gradoop.storage.hbase.impl.io.functions.BuildGraphHeadMutation;
27  import org.gradoop.storage.hbase.impl.io.functions.BuildVertexMutation;
28  import org.gradoop.storage.hbase.impl.HBaseEPGMStore;
29  
30  import javax.annotation.Nonnull;
31  import java.io.IOException;
32  
33  /**
34   * Converts runtime representation of EPGM elements into persistent
35   * representations and writes them to HBase.
36   */
37  public class HBaseDataSink extends HBaseBase implements DataSink {
38  
39    /**
40     * Creates a new HBase data sink.
41     *
42     * @param epgmStore store implementation
43     * @param flinkConfig gradoop flink execute config
44     */
45    public HBaseDataSink(
46      @Nonnull HBaseEPGMStore epgmStore,
47      @Nonnull GradoopFlinkConfig flinkConfig
48    ) {
49      super(epgmStore, flinkConfig);
50    }
51  
52    @Override
53    public void write(LogicalGraph logicalGraph) throws IOException {
54      write(logicalGraph, false);
55    }
56  
57    @Override
58    public void write(GraphCollection graphCollection) throws IOException {
59      write(graphCollection, false);
60    }
61  
62    @Override
63    public void write(LogicalGraph logicalGraph, boolean overwrite) throws IOException {
64      write(logicalGraph.getCollectionFactory().fromGraph(logicalGraph), overwrite);
65    }
66  
67    @Override
68    public void write(GraphCollection graphCollection, boolean overWrite) throws IOException {
69      if (overWrite) {
70        getStore().truncateTables();
71      }
72  
73      // transform graph data to persistent graph data and write it
74      writeGraphHeads(graphCollection);
75  
76      // transform vertex data to persistent vertex data and write it
77      writeVertices(graphCollection);
78  
79      // transform edge data to persistent edge data and write it
80      writeEdges(graphCollection);
81    }
82  
83    /**
84     * Converts runtime graph data to persistent graph data (including vertex
85     * and edge identifiers) and writes it to HBase.
86     *
87     * @param collection Graph collection
88     * @throws IOException if fetching mapreduce instance failed
89     */
90    private void writeGraphHeads(final GraphCollection collection)
91      throws IOException {
92  
93      // write (graph-data) to HBase table
94      Job job = Job.getInstance();
95      job.getConfiguration()
96        .set(TableOutputFormat.OUTPUT_TABLE, getHBaseConfig().getGraphTableName().getNameAsString());
97  
98      collection.getGraphHeads()
99        .map(new BuildGraphHeadMutation(getHBaseConfig().getGraphHeadHandler()))
100       .output(new HadoopOutputFormat<>(new TableOutputFormat<>(), job));
101   }
102 
103   /**
104    * Converts runtime vertex data to persistent vertex data (includes
105    * incoming and outgoing edge data) and writes it to HBase.
106    *
107    * @param collection Graph collection
108    * @throws IOException if fetching mapreduce instance failed
109    */
110   private void writeVertices(final GraphCollection collection) throws IOException {
111 
112     // write (vertex-data) to HBase table
113     Job job = Job.getInstance();
114     job.getConfiguration()
115       .set(TableOutputFormat.OUTPUT_TABLE, getHBaseConfig().getVertexTableName().getNameAsString());
116 
117     collection.getVertices()
118       .map(new BuildVertexMutation(getHBaseConfig().getVertexHandler()))
119       .output(new HadoopOutputFormat<>(new TableOutputFormat<>(), job));
120   }
121 
122   /**
123    * Converts runtime edge data to persistent edge data (includes
124    * source/target vertex data) and writes it to HBase.
125    *
126    * @param collection Graph collection
127    * @throws IOException if fetching mapreduce instance failed
128    */
129   private void writeEdges(final GraphCollection collection) throws IOException {
130 
131     // write (edge-data) to HBase table
132     Job job = Job.getInstance();
133     job.getConfiguration()
134       .set(TableOutputFormat.OUTPUT_TABLE, getHBaseConfig().getEdgeTableName().getNameAsString());
135 
136     collection.getEdges()
137       .map(new BuildEdgeMutation(getHBaseConfig().getEdgeHandler()))
138       .output(new HadoopOutputFormat<>(new TableOutputFormat<>(), job));
139   }
140 }
