66    /**
67     * Creates a HBase Filter object to return only graph elements that are equal to the given
68     * GradoopIds.
69     *
70     * @param elementIds a set of graph element GradoopIds to filter
71     * @param isSpreadingByteUsed indicates if the handlers using a spreading byte as prefix
72     *                            of each row key
73     * @return a HBase Filter object
74     */
75    public static Filter getIdFilter(GradoopIdSet elementIds, boolean isSpreadingByteUsed) {
76      Filter filter;
77      if (isSpreadingByteUsed) {
78        // a spreading byte is used as prefix in row key, therefore a FuzzyRowFilter has to be used
79        byte[] dummyByte = new byte[1];
80  
81        // instantiate a list of byte array pairs for the fuzzy row filter
82        List<Pair<byte[], byte[]>> fuzzyList = new ArrayList<>();
83  
84        // create a byte array and set the first to '1' to identify the position of dynamic bytes
85        byte[] fuzzyInfo = new byte[13];
86        fuzzyInfo[0] = (byte) 1;
87  
88        for (GradoopId gradoopId : elementIds) {
89          // for each id create a 13 byte long array from a dummy byte and the 12 byte gradoop id
90          byte[] rowKey = Bytes.add(dummyByte, gradoopId.toByteArray());
91          fuzzyList.add(new Pair<>(rowKey, fuzzyInfo));
92        }
93        filter = new FuzzyRowFilter(fuzzyList);
94      } else {
95        // no spreading byte in row key, therefore a list of RowFilters has to be used
96  
97        FilterList filterList = new FilterList(FilterList.Operator.MUST_PASS_ONE);
98  
99        for (GradoopId gradoopId : elementIds) {
100         RowFilter rowFilter = new RowFilter(
101           CompareFilter.CompareOp.EQUAL,
102           new BinaryComparator(gradoopId.toByteArray())
103         );
104         filterList.addFilter(rowFilter);
105       }
106       filter = filterList;
107     }
108     return filter;
109   }

111   /**
112    * Creates a HBase Filter object representation of labelIn predicate
113    *
114    * @param labels set of labels to filter for
115    * @param negate flag to define if this filter should be negated
116    * @return the HBase filter representation
117    */
118   public static Filter getLabelInFilter(@Nonnull Set<String> labels, boolean negate) {
119     // Handle negation
120     CompareFilter.CompareOp compareOp = negate ? CompareFilter.CompareOp.NOT_EQUAL :
121       CompareFilter.CompareOp.EQUAL;
122     FilterList.Operator listOperator = negate ? FilterList.Operator.MUST_PASS_ALL :
123       FilterList.Operator.MUST_PASS_ONE;
124 
125     FilterList filterList = new FilterList(listOperator);
126 
127     labels.stream()
128       .map(label -> new SingleColumnValueFilter(CF_META_BYTES, COL_LABEL_BYTES, compareOp,
129         Bytes.toBytesBinary(label)))
130       .forEach(filterList::addFilter);
131 
132     return filterList;
133   }

151   /**
152    * Creates a HBase Filter object representation of propEquals predicate
153    *
154    * @param key the property key to filter for
155    * @param value the value to filter for
156    * @param negate flag to define if this filter should be negated
157    * @return the HBase filter representation
158    */
159   public static Filter getPropEqualsFilter(@Nonnull String key, @Nonnull PropertyValue value,
160     boolean negate) {
161     // Handle negation
162     CompareFilter.CompareOp compareOp = negate ? CompareFilter.CompareOp.NOT_EQUAL :
163       CompareFilter.CompareOp.EQUAL;
164     FilterList.Operator listOperator = negate ? FilterList.Operator.MUST_PASS_ONE :
165       FilterList.Operator.MUST_PASS_ALL;
166 
167     FilterList filterList = new FilterList(listOperator);
168 
169     SingleColumnValueFilter valueFilter = new SingleColumnValueFilter(
170       CF_PROPERTY_VALUE_BYTES,
171       Bytes.toBytesBinary(key),
172       compareOp,
173       PropertyValueUtils.BytesUtils.getRawBytesWithoutType(value));
174 
175     // Define that the entire row will be skipped if the column is not found
176     valueFilter.setFilterIfMissing(true);
177 
178     SingleColumnValueFilter typeFilter = new SingleColumnValueFilter(
179       CF_PROPERTY_TYPE_BYTES,
180       Bytes.toBytesBinary(key),
181       compareOp,
182       PropertyValueUtils.BytesUtils.getTypeByte(value));
183 
184     // Define that the entire row will be skipped if the column is not found
185     typeFilter.setFilterIfMissing(true);
186 
187     filterList.addFilter(valueFilter);
188     filterList.addFilter(typeFilter);
189     return filterList;
190   }

232   /**
233    * Creates a HBase Filter object representation of propLargerThan predicate
234    *
235    * @param key the property key to filter for
236    * @param min the property value that defines the minimum of the filter
237    * @param include a flag to define if a value that is equal to min should be included
238    * @param negate flag to define if this filter should be negated
239    * @return the HBase filter representation
240    */
241   public static Filter getPropLargerThanFilter(@Nonnull String key, @Nonnull PropertyValue min,
242     boolean include, boolean negate) {
243     // Handle negation
244     FilterList.Operator listOperator = negate ? FilterList.Operator.MUST_PASS_ONE :
245       FilterList.Operator.MUST_PASS_ALL;
246 
247     CompareFilter.CompareOp compareOp;
248     if (include) {
249       if (negate) {
250         compareOp = CompareFilter.CompareOp.LESS;
251       } else {
252         compareOp = CompareFilter.CompareOp.GREATER_OR_EQUAL;
253       }
254     } else {
255       if (negate) {
256         compareOp = CompareFilter.CompareOp.LESS_OR_EQUAL;
257       } else {
258         compareOp = CompareFilter.CompareOp.GREATER;
259       }
260     }
261 
262     FilterList filterList = new FilterList(listOperator);
263 
264     SingleColumnValueFilter valueFilter = new SingleColumnValueFilter(
265       CF_PROPERTY_VALUE_BYTES,
266       Bytes.toBytesBinary(key),
267       compareOp,
268       new BinaryComparator(PropertyValueUtils.BytesUtils.getRawBytesWithoutType(min)));
269 
270     // Define that the entire row will be skipped if the column is not found
271     valueFilter.setFilterIfMissing(true);
272 
273     SingleColumnValueFilter typeFilter = new SingleColumnValueFilter(
274       CF_PROPERTY_TYPE_BYTES,
275       Bytes.toBytesBinary(key),
276       negate ? CompareFilter.CompareOp.NOT_EQUAL : CompareFilter.CompareOp.EQUAL,
277       PropertyValueUtils.BytesUtils.getTypeByte(min));
278 
279     // Define that the entire row will be skipped if the column is not found
280     typeFilter.setFilterIfMissing(true);
281 
282     filterList.addFilter(valueFilter);
283     filterList.addFilter(typeFilter);
284     return filterList;
285   }
