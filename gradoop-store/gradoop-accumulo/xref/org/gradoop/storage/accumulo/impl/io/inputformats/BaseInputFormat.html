137   @Override
138   public void open(GenericInputSplit split) throws IOException {
139     super.open(split);
140     try {
141       String user = (String) properties
142         .getOrDefault(GradoopAccumuloConfig.ACCUMULO_USER, AccumuloDefault.USER);
143       String password = (String) properties
144         .getOrDefault(GradoopAccumuloConfig.ACCUMULO_PASSWD, AccumuloDefault.PASSWORD);
145       String instance = (String) properties
146         .getOrDefault(GradoopAccumuloConfig.ACCUMULO_INSTANCE, AccumuloDefault.INSTANCE);
147       String zkHosts = (String) properties
148         .getOrDefault(GradoopAccumuloConfig.ZOOKEEPER_HOSTS, AccumuloDefault.INSTANCE);
149       String tableName = getTableName((String) properties
150         .getOrDefault(GradoopAccumuloConfig.ACCUMULO_TABLE_PREFIX, AccumuloDefault.TABLE_PREFIX));
151       Authorizations auth = (Authorizations) properties
152         .getOrDefault(GradoopAccumuloConfig.ACCUMULO_AUTHORIZATIONS,
153           AccumuloDefault.AUTHORIZATION);
154       int batchSize = (int) properties
155         .getOrDefault(GradoopAccumuloConfig.GRADOOP_BATCH_SCANNER_THREADS,
156           AccumuloDefault.BATCH_SCANNER_THREADS);
157       int iteratorPriority = (int) properties
158         .getOrDefault(GradoopAccumuloConfig.GRADOOP_ITERATOR_PRIORITY,
159           AccumuloDefault.ITERATOR_PRIORITY);
160 
161       Connector conn = new ZooKeeperInstance(instance, zkHosts)
162         .getConnector(user, new PasswordToken(password));
163 
164       List<Range> ranges = doSplits(
165         split.getTotalNumberOfSplits(),
166         tableName,
167         user,
168         password,
169         instance,
170         zkHosts,
171         auth);
172 
173       Map<String, String> options = new HashMap<>();
174       if (predicate != null && predicate.getReduceFilter() != null) {
175         options.put(AccumuloTables.KEY_PREDICATE, predicate.getReduceFilter().encode());
176       }
177       if (split.getSplitNumber() + 1 > ranges.size()) {
178         scanner = null;
179         iterator = Collections.emptyIterator();
180       } else {
181         scanner = conn.createBatchScanner(tableName, auth, batchSize);
182         attachIterator(scanner, iteratorPriority, options);
183         scanner.setRanges(Lists.newArrayList(ranges.get(split.getSplitNumber())));
184         iterator = scanner.iterator();
185       }
186 
187       initiate();
188     } catch (AccumuloException | AccumuloSecurityException | TableNotFoundException e) {
189       throw new IOException(e);
190     }
191   }
192 
193   @Override
194   public boolean reachedEnd() {
195     return !iterator.hasNext();
196   }
197 
198   @Override
199   public T nextRecord(T reuse) throws IOException {
200     return mapRow(iterator.next());
201   }
202 
203   @Override
204   public void close() throws IOException {
205     super.close();
206     if (scanner != null) {
207       scanner.close();
208     }
209   }

211   /**
212    * Split table into ranges according to {@link AccumuloRowInputFormat#getSplits} suggest
213    *
214    * @param maxSplit max split size
215    * @param tableName split table name
216    * @param user accumulo user
217    * @param password accumulo password
218    * @param instance accumulo instance name
219    * @param zkHosts zookeeper hosts
220    * @param auth accumulo access authorization
221    * @return split range collections
222    * @throws IOException on failure
223    * @throws AccumuloSecurityException on security error
224    */
225   @Nonnull
226   private List<Range> doSplits(
227     int maxSplit,
228     @Nonnull String tableName,
229     @Nonnull String user,
230     @Nonnull String password,
231     @Nonnull String instance,
232     @Nonnull String zkHosts,
233     @Nonnull Authorizations auth
234   ) throws IOException, AccumuloSecurityException {
235     AccumuloRowInputFormat format = new AccumuloRowInputFormat();
236     JobConf conf = new JobConf();
237     AccumuloRowInputFormat.setInputTableName(conf, tableName);
238     AccumuloRowInputFormat.setConnectorInfo(conf, user, new PasswordToken(password));
239     AccumuloRowInputFormat.setZooKeeperInstance(conf, ClientConfiguration.create()
240       .withInstance(instance)
241       .withZkHosts(zkHosts));
242     AccumuloRowInputFormat.setScanAuthorizations(conf, auth);
243     if (predicate != null && predicate.getQueryRanges() != null) {
244       AccumuloRowInputFormat.setRanges(conf, predicate.getQueryRanges());
245     }
246     InputSplit[] splits = format.getSplits(conf, maxSplit);
247     return Stream.of(splits)
248       .map(it -> (RangeInputSplit) it)
249       .map(RangeInputSplit::getRange)
250       .collect(Collectors.toList());
251   }
