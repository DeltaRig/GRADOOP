72  /**
73   * Default Accumulo EPGM graph store that handles reading and writing vertices and
74   * graphs from and to Accumulo, It is designed thread-safe.
75   * Store contains instances are divided by {@link GradoopAccumuloConfig#ACCUMULO_TABLE_PREFIX}

122   /**
123    * Creates an AccumuloEPGMStore based on the given parameters.
124    * Tables with given prefix will be auto-create if not exists
125    *
126    * @param config                      accumulo store configuration
127    * @throws AccumuloSecurityException  for security violations,
128    *                                    authentication failures,
129    *                                    authorization failures,
130    *                                    etc.
131    * @throws AccumuloException          generic Accumulo Exception for general accumulo failures.
132    */
133   public AccumuloEPGMStore(@Nonnull GradoopAccumuloConfig config)
134     throws AccumuloSecurityException, AccumuloException {
135     this.config = config;
136     this.conn = createConnector();
137     createTablesIfNotExists();
138     try {
139       graphWriter = conn.createBatchWriter(getGraphHeadName(), new BatchWriterConfig());
140       vertexWriter = conn.createBatchWriter(getVertexTableName(), new BatchWriterConfig());
141       edgeWriter = conn.createBatchWriter(getEdgeTableName(), new BatchWriterConfig());
142     } catch (TableNotFoundException e) {
143       throw new IllegalStateException(e); //should not be here
144     }
145   }

230   /**
231    * Drop all tables used by this store instance.
232    *
233    * @throws IOException when deleting tables fails.
234    */
235   public void dropTables() throws IOException {
236     for (String tableName : new String[] {
237       getVertexTableName(), getEdgeTableName(), getGraphHeadName()}) {
238       try {
239         dropTableIfExists(tableName);
240       } catch (AccumuloSecurityException | AccumuloException e) {
241         throw new IOException("Failed to delete table " + tableName, e);
242       }
243     }
244   }

297   @Nonnull
298   @Override
299   public ClosableIterator<EPGMGraphHead> getGraphSpace(
300     @Nullable ElementQuery<AccumuloElementFilter<EPGMGraphHead>> query,
301     int cacheSize
302   ) throws IOException {
303     if (query != null &&
304       query.getQueryRanges() != null &&
305       query.getQueryRanges().isEmpty()) {
306       return new EmptyClosableIterator<>();
307     }
308 
309     if (query != null) {
310       LOG.info(query.toString());
311     }
312 
313     BatchScanner scanner = createBatchScanner(
314       getGraphHeadName(),
315       GradoopGraphHeadIterator.class,
316       query);
317     Iterator<Map.Entry<Key, Value>> iterator = scanner.iterator();
318     if (!iterator.hasNext()) {
319       return new EmptyClosableIterator<>();
320     } else {
321       return new ClientClosableIterator<>(scanner,
322         new GradoopGraphHeadIterator(),
323         config.getGraphHandler(),
324         cacheSize);
325     }
326   }

390   /**
391    * Write an EPGM Element instance into table
392    *
393    * @param record  gradoop EPGM element
394    * @param writer  accumulo batch writer
395    * @param handler accumulo row handler
396    * @param <T>     element type
397    */
398   private <T extends Element> void writeRecord(
399     @Nonnull T record,
400     @Nonnull BatchWriter writer,
401     @Nonnull AccumuloRowHandler handler
402   ) {
403     Mutation mutation = new Mutation(record.getId().toString());
404     //noinspection unchecked
405     mutation = handler.writeRow(mutation, record);
406     try {
407       writer.addMutation(mutation);
408       if (autoFlush) {
409         writer.flush();
410       }
411     } catch (MutationsRejectedException e) {
412       throw new RuntimeException(e);
413     }
414   }

416   /**
417    * Create accumulo batch scanner with element predicate
418    *
419    * @param table  table name
420    * @param iterator iterator class
421    * @param predicate accumulo predicate
422    * @param <T> epgm element type
423    * @return batch scanner instance
424    * @throws IOException if create fail
425    */
426   private <T extends Element> BatchScanner createBatchScanner(
427     String table,
428     Class<? extends SortedKeyValueIterator<Key, Value>> iterator,
429     @Nullable ElementQuery<AccumuloElementFilter<T>> predicate
430   ) throws IOException {
431     Map<String, String> options = new HashMap<>();
432     if (predicate != null && predicate.getFilterPredicate() != null) {
433       options.put(AccumuloTables.KEY_PREDICATE, predicate.getFilterPredicate().encode());
434     }
435     BatchScanner scanner;
436     try {
437       scanner = conn.createBatchScanner(table,
438         config.get(GradoopAccumuloConfig.ACCUMULO_AUTHORIZATIONS,
439           AccumuloDefault.AUTHORIZATION),
440         config.get(GradoopAccumuloConfig.GRADOOP_BATCH_SCANNER_THREADS,
441           AccumuloDefault.BATCH_SCANNER_THREADS));
442       int priority =
443         config.get(GradoopAccumuloConfig.GRADOOP_ITERATOR_PRIORITY, AccumuloDefault
444           .ITERATOR_PRIORITY);
445       scanner.addScanIterator(new IteratorSetting(
446         /*iterator priority*/priority,
447         /*iterator class*/iterator,
448         /*args*/options));
449 
450       if (predicate == null ||
451         predicate.getQueryRanges() == null) {
452         scanner.setRanges(Lists.newArrayList(new Range()));
453       } else {
454         scanner.setRanges(Range.mergeOverlapping(predicate.getQueryRanges()
455           .stream()
456           .map(GradoopId::toString)
457           .map(Range::exact)
458           .collect(Collectors.toList())));
459       }
460       return scanner;
461 
462     } catch (TableNotFoundException e) {
463       throw new IOException(e);
464     }
465   }

467   /**
468    * Create tables (and their namespaces, if defined by table prefix) if not exists
469    *
470    * @throws AccumuloSecurityException  for security violations,
471    *                                    authentication failures,
472    *                                    authorization failures,
473    *                                    etc.
474    * @throws AccumuloException          generic Accumulo Exception for general accumulo failures.
475    */
476   private void createTablesIfNotExists() throws AccumuloSecurityException, AccumuloException {
477     String prefix =
478       config.get(GradoopAccumuloConfig.ACCUMULO_TABLE_PREFIX, AccumuloDefault.TABLE_PREFIX);
479     if (prefix.contains(".")) {
480       String namespace = prefix.substring(0, prefix.indexOf("."));
481       try {
482         if (!conn.namespaceOperations().exists(namespace)) {
483           conn.namespaceOperations().create(namespace);
484         }
485       } catch (NamespaceExistsException ignore) {
486         //ignore if it is exists, maybe create by another process or thread
487       }
488     }
489     for (String table : new String[] {
490       getVertexTableName(), getEdgeTableName(), getGraphHeadName()
491     }) {
492       try {
493         if (!conn.tableOperations().exists(table)) {
494           conn.tableOperations().create(table);
495         }
496       } catch (TableExistsException ignore) {
497         //ignore if it is exists, maybe create by another process or thread
498       }
499     }
500   }

502   /**
503    * Delete a table if it exists.
504    *
505    * @param tableName The table name.
506    * @throws AccumuloSecurityException if the user does not have persmissions to delete the table.
507    * @throws AccumuloException         if a general Accumulo error occurs.
508    */
509   private void dropTableIfExists(@Nonnull String tableName) throws AccumuloSecurityException,
510     AccumuloException {
511     if (conn.tableOperations().exists(tableName)) {
512       try {
513         conn.tableOperations().delete(tableName);
514       } catch (TableNotFoundException e) {
515         // We checked before if it exists, so this should not happen.
516         throw new IllegalStateException(e);
517       }
518     }
