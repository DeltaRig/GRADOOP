100   /**
101    * Creates tables, deletes them and checks if they were deleted.
102    *
103    * @throws AccumuloSecurityException when the user does not have permissions for some actions.
104    * @throws AccumuloException on general accumulo errors.
105    * @throws IOException when deleting tables fails.
106    */
107   @Test
108   public void deleteTablesTest() throws AccumuloSecurityException, AccumuloException, IOException {
109     final String table = "test_to_delete";
110     GradoopAccumuloConfig config = AccumuloTestSuite.getAcConfig(table);
111     AccumuloEPGMStore store = new AccumuloEPGMStore(config);
112     Connector connector = store.createConnector();
113     // Make sure that the tables were created.
114     assertTrue(connector.tableOperations().exists(store.getVertexTableName()));
115     assertTrue(connector.tableOperations().exists(store.getEdgeTableName()));
116     assertTrue(connector.tableOperations().exists(store.getGraphHeadName()));
117     // Delete tables.
118     store.dropTables();
119     // Check if they were deleted.
120     assertFalse(connector.tableOperations().exists(store.getVertexTableName()));
121     assertFalse(connector.tableOperations().exists(store.getEdgeTableName()));
122     assertFalse(connector.tableOperations().exists(store.getGraphHeadName()));
123   }

127    *
128    * @throws AccumuloSecurityException when the user does not have permissions for some actions.
129    * @throws AccumuloException on general accumulo errors.
130    */
131   @Test
132   public void truncateTablesTest() throws AccumuloSecurityException, AccumuloException,
133     IOException {
134     final String table = "test_to_truncate";
135     GradoopAccumuloConfig config = AccumuloTestSuite.getAcConfig(table);
136 
137     AccumuloEPGMStore graphStore = new AccumuloEPGMStore(config);
138 
139     AsciiGraphLoader<EPGMGraphHead, EPGMVertex, EPGMEdge> loader = getMinimalFullFeaturedGraphLoader();
140 
141     EPGMGraphHead graphHead = loader.getGraphHeads().iterator().next();
142     EPGMVertex vertex = loader.getVertices().iterator().next();
143     EPGMEdge edge = loader.getEdges().iterator().next();
144 
145     graphStore.writeGraphHead(graphHead);
146     graphStore.writeVertex(vertex);
147     graphStore.writeEdge(edge);
148 
149     // re-open
150     graphStore.close();
151     graphStore = new AccumuloEPGMStore(config);
152     graphStore.truncateTables();
153 
154     assertFalse(graphStore.getVertexSpace().hasNext());
155     assertFalse(graphStore.getEdgeSpace().hasNext());
156     assertFalse(graphStore.getGraphSpace().hasNext());
157     graphStore.close();

240     for (EPGMGraphHead g : graphHeads) {
241       graphStore.writeGraphHead(g);
242     }
243     for (EPGMEdge e : edges) {
244       graphStore.writeEdge(e);
245     }
246     for (EPGMVertex v : vertices) {
247       graphStore.writeVertex(v);
248     }
249 
250     graphStore.flush();
251 
252     // graph heads
253     validateElementCollections(graphHeads,
254       graphStore.getGraphSpace().readRemainsAndClose());
255     // vertices
256     validateElementCollections(vertices,
257       graphStore.getVertexSpace().readRemainsAndClose());
258     validateGraphElementCollections(vertices,
259       graphStore.getVertexSpace().readRemainsAndClose());
260     // edges
261     validateElementCollections(edges,
262       graphStore.getEdgeSpace().readRemainsAndClose());
263     validateGraphElementCollections(edges,
264       graphStore.getEdgeSpace().readRemainsAndClose());
265 
266     graphStore.close();

396   private AsciiGraphLoader<EPGMGraphHead, EPGMVertex, EPGMEdge> getMinimalFullFeaturedGraphLoader() {
397     String asciiGraph = ":G{k:\"v\"}[(v:V{k:\"v\"}),(v)-[:e{k:\"v\"}]->(v)]";
398 
399     return AsciiGraphLoader.fromString(asciiGraph, getConfig().getLogicalGraphFactory());
400   }
