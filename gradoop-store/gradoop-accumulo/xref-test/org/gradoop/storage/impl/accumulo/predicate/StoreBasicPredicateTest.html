50    /**
51     * Pick 3 person randomly then find vertex with label 'Person' and with same name (property) value
52     *
53     * @throws Throwable if error
54     */
55    @Test
56    public void findPersonByName() throws Throwable {
57      doTest(TEST01, (loader, store, config) -> {
58        //vertex label and property query
59        List<EPGMVertex> inputVertices = sample(loader.getVertices()
60          .stream()
61          .filter(it -> Objects.equals(it.getLabel(), "Person"))
62          .collect(Collectors.toList()), 3);
63  
64  
65        List<EPGMVertex> queryResult = store
66          .getVertexSpace(
67            Query.elements()
68              .fromAll()
69              .where(AccumuloFilters
70                .<EPGMVertex>labelIn("Person")
71                .and(inputVertices.stream()
72                  .map(it -> {
73                    assert it.getProperties() != null;
74                    String name = it.getProperties().get("name").getString();
75                    return (AccumuloElementFilter<EPGMVertex>) AccumuloFilters
76                      .<EPGMVertex>propEquals("name", name);
77                  })
78                  .reduce((a, b) -> Or.create(a, b))
79                  .orElse(it -> false))
80              ))
81          .readRemainsAndClose();
82  
83        validateElementCollections(inputVertices, queryResult);
84      });
85    }

96        List<EPGMVertex> inputVertices = loader.getVertices()
97          .stream()
98          .filter(it -> Objects.equals(it.getLabel(), "Person"))
99          .filter(it -> it.getProperties() != null)
100         .filter(it -> it.getProperties().get("age") != null)
101         .filter(it -> it.getProperties().get("age").getInt() >= 35)
102         .collect(Collectors.toList());
103 
104       List<EPGMVertex> queryResult = store
105         .getVertexSpace(
106           Query.elements()
107             .fromAll()
108             .where(AccumuloFilters.propLargerThan("age", 35, true)))
109         .readRemainsAndClose();
110       validateElementCollections(inputVertices, queryResult);
111     });

114   /**
115    * Find graph by property equality within a certain sample id range
116    *
117    * @throws Throwable if error
118    */
119   @Test
120   public void findGraphByIdsAndProperty() throws Throwable {
121     doTest(TEST03, (loader, store, config) -> {
122       List<EPGMGraphHead> samples = sample(new ArrayList<>(loader.getGraphHeads()), 3);
123 
124       GradoopIdSet sampleRange = GradoopIdSet.fromExisting(samples.stream()
125         .map(EPGMElement::getId)
126         .collect(Collectors.toList()));
127 
128       List<EPGMGraphHead> inputGraph = samples.stream()
129         .filter(it -> Objects.equals(it.getLabel(), "Community"))
130         .filter(it -> it.getPropertyValue("interest") != null)
131         .filter(it ->
132           Objects.equals(it.getPropertyValue("interest").getString(), "Hadoop") ||
133             Objects.equals(it.getPropertyValue("interest").getString(), "Graphs"))
134         .collect(Collectors.toList());
135 
136       ElementQuery<AccumuloElementFilter<EPGMGraphHead>> queryFormula = Query.elements()
137         .fromSets(sampleRange)
138         .where(AccumuloFilters.<EPGMGraphHead>labelIn("Community")
139           .and(AccumuloFilters.<EPGMGraphHead>propEquals("interest", "Hadoop")
140             .or(AccumuloFilters.propEquals("interest", "Graphs"))));
141 
142       List<EPGMGraphHead> query = store
143         .getGraphSpace(queryFormula)
144         .readRemainsAndClose();
145       GradoopTestUtils.validateElementCollections(inputGraph, query);
146     });
147   }
