52      doTest(TEST01, (loader, store, config) -> {
53        List<EPGMVertex> inputVertex = loader.getVertices().stream()
54          .filter(it ->
55            Objects.equals(it.getLabel(), "Person") ||
56              Objects.equals(it.getLabel(), "Tag"))
57          .collect(Collectors.toList());
58  
59        //vertex label query
60        List<EPGMVertex> queryResult = store
61          .getVertexSpace(
62            Query.elements()
63              .fromAll()
64              .where(AccumuloFilters.labelIn("Person", "Tag")))
65          .readRemainsAndClose();
66  
67        validateElementCollections(inputVertex, queryResult);
68      });
69    }
