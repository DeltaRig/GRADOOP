16  package org.gradoop.storage.common.predicate.filter.impl;
17  
18  import org.gradoop.common.model.impl.properties.PropertyValue;
19  import org.gradoop.storage.common.predicate.filter.api.ElementFilter;
20  
21  import javax.annotation.Nonnull;
22  
23  /**
24   * Predicate by property equality
25   * return element if and only if:
26   *  - contains property key
27   *  - property value equals to given value
28   *
29   * @param <FilterImpl> filter implement type
30   */
31  public abstract class PropEquals<FilterImpl extends ElementFilter>
32    implements ElementFilter<FilterImpl> {
33  
34    /**
35     * property key
36     */
37    private final String key;
38  
39    /**
40     * property value
41     */
42    private final PropertyValue value;
43  
44    /**
45     * property equals filter
46     *
47     * @param key property key
48     * @param value property value
49     */
50    public PropEquals(
51      @Nonnull String key,
52      @Nonnull Object value
53    ) {
54      this.key = key;
55      this.value = value instanceof PropertyValue ?
56        (PropertyValue) value : PropertyValue.create(value);
57    }
58  
59    @Override
60    public String toString() {
61      return String.format("e.prop.%1$s IS NOT NULL AND e.prop.%1$s=`%2$s`",
62        key, value);
63    }
64  
65    protected String getKey() {
66      return key;
67    }
68  
69    protected PropertyValue getValue() {
70      return value;
71    }
72  
73  }
