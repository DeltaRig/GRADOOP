45  /**
46   * A temporal graph data source for CSV files indexed by label.
47   * <p>
48   * The datasource expects files separated by label, e.g. in the following directory structure:
49   * <p>
50   * {@code
51   * csvRoot
52   * |- vertices
53   *   |- person
54   *     |- data.csv    # contains all vertices with label 'Person'
55   *   |- university
56   *     |- data.csv    # contains all vertices with label 'University'
57   * |- edges
58   *   |- knows
59   *     |- data.csv    # contains all edges with label 'knows'
60   *   |- studyAt
61   *     |- data.csv    # contains all edges with label 'studyAt'
62   * |- graphs
63   *   |- myGraph
64   *     |- data.csv
65   * |- metadata.csv   # Meta data for all data contained in the graph
66   * }
67   */
68  public class TemporalIndexedCSVDataSource extends CSVBase implements TemporalDataSource, DataSource {

119   /**
120    * Reads the graph data from (distributed) file system.
121    *
122    * @return a temporal indexed layout containing the graph/collection data
123    * @throws IOException in case of an error while reading
124    */
125   private TemporalIndexedLayout readIndexedCSV() throws IOException {
126     CSVMetaDataSource source = new CSVMetaDataSource();
127     CSVMetaData metaData = source.readLocal(getMetaDataPath(), hdfsConfig);
128     DataSet<Tuple3<String, String, String>> metaDataBroadcast =
129       source.readDistributed(getMetaDataPath(), getConfig());
130 
131     ExecutionEnvironment env = getConfig().getExecutionEnvironment();
132 
133     Map<String, DataSet<TemporalGraphHead>> graphHeads = metaData.getGraphLabels().stream()
134       .map(label -> Tuple2.of(label, env.readTextFile(getGraphHeadCSVPath(label))
135         .map(new CSVLineToTemporalGraphHead(getConfig().getTemporalGraphFactory().getGraphHeadFactory()))
136         .withBroadcastSet(metaDataBroadcast, BC_METADATA)
137         .filter(graphHead -> graphHead.getLabel().equals(label))))
138       .collect(Collectors.toMap(t -> t.f0, t -> t.f1));
139 
140     Map<String, DataSet<TemporalVertex>> vertices = metaData.getVertexLabels().stream()
141       .map(label -> Tuple2.of(label, env.readTextFile(getVertexCSVPath(label))
142         .map(new CSVLineToTemporalVertex(getConfig().getTemporalGraphFactory().getVertexFactory()))
143         .withBroadcastSet(metaDataBroadcast, BC_METADATA)
144         .filter(vertex -> vertex.getLabel().equals(label))))
145       .collect(Collectors.toMap(t -> t.f0, t -> t.f1));
146 
147     Map<String, DataSet<TemporalEdge>> edges = metaData.getEdgeLabels().stream()
148       .map(label -> Tuple2.of(label, env.readTextFile(getEdgeCSVPath(label))
149         .map(new CSVLineToTemporalEdge(getConfig().getTemporalGraphFactory().getEdgeFactory()))
150         .withBroadcastSet(metaDataBroadcast, BC_METADATA)
151         .filter(edge -> edge.getLabel().equals(label))))
152       .collect(Collectors.toMap(t -> t.f0, t -> t.f1));
153 
154     return new TemporalIndexedLayout(graphHeads, vertices, edges);
155   }
