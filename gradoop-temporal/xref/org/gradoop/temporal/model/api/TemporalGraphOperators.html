185   /**
186    * Compares two snapshots of this graph. Given two temporal predicates, this operation
187    * will check if a graph element (vertex or edge) was added, removed or persists in the second
188    * snapshot compared to the first snapshot. The predicates are applied on the valid times. To use
189    * transaction time dimension, use
190    * {@link TemporalGraphOperators#diff(TemporalPredicate, TemporalPredicate, TimeDimension)}.
191    * <p>
192    * This operation returns the union of both snapshots with the following changes:
193    * A property with key {@value Diff#PROPERTY_KEY}
194    * will be set on each graph element. Its value will be set to
195    * <ul>
196    *   <li>{@code 0}, if the element is present in both snapshots.</li>
197    *   <li>{@code 1}, if the element is present in the second, but not the first snapshot
198    *   (i.e. it was added since the first snapshot).</li>
199    *   <li>{@code -1}, if the element is present in the first, but not the second snapshot
200    *   (i.e. it was removed since the first snapshot).</li>
201    * </ul>
202    * Graph elements present in neither snapshot will be discarded.
203    * The resulting graph will not be verified, i.e. dangling edges could occur. Use the
204    * {@code verify()} operator to validate the graph. The graph head is preserved.
205    *
206    * @param firstSnapshot  The predicate used to determine the first snapshot.
207    * @param secondSnapshot The predicate used to determine the second snapshot.
208    * @return A temporal graph containing the union of vertex and edge sets of both snapshots,
209    * defined by the given two predicate functions. A property with key
210    * {@link Diff#PROPERTY_KEY} is set on each graph element with a numerical value (-1, 0, 1) defined above.
211    */
212   default TemporalGraph diff(TemporalPredicate firstSnapshot, TemporalPredicate secondSnapshot) {
213     return diff(firstSnapshot, secondSnapshot, TimeDimension.VALID_TIME);
214   }

217    * Compares two snapshots of this graph. Given two temporal predicates, this operation will check if a
218    * graph element (vertex or edge) was added, removed or persists in the second snapshot compared to the
219    * first snapshot. The predicates are applied on the specified time dimension.
220    * <p>
221    * This operation returns the union of both snapshots with the following changes:
222    * A property with key {@value Diff#PROPERTY_KEY} will be set on each graph element.
223    * Its value will be set to
224    * <ul>
225    *   <li>{@code 0}, if the element is present in both snapshots.</li>
226    *   <li>{@code 1}, if the element is present in the second, but not the first snapshot
227    *   (i.e. it was added since the first snapshot).</li>
228    *   <li>{@code -1}, if the element is present in the first, but not the second snapshot
229    *   (i.e. it was removed since the first snapshot).</li>
230    * </ul>
231    * Graph elements present in neither snapshot will be discarded.
232    * The resulting graph will not be verified, i.e. dangling edges could occur. Use the
233    * {@code verify()} operator to validate the graph. The graph head is preserved.
234    *
235    * @param firstSnapshot  The predicate used to determine the first snapshot.
236    * @param secondSnapshot The predicate used to determine the second snapshot.
237    * @param dimension      The time dimension that will be considered by the predicates.
238    * @return A temporal graph containing the union of vertex and edge sets of both snapshots, defined by the
239    * given two predicate functions. A property with key {@link Diff#PROPERTY_KEY} is set on each graph
240    * element with a numerical value (-1, 0, 1) defined above.
241    */
242   default TemporalGraph diff(TemporalPredicate firstSnapshot, TemporalPredicate secondSnapshot,

263   /**
264    * Evaluates the given query using the Temporal-GDL query engine. The engine uses default morphism
265    * strategies, which is vertex homomorphism and edge isomorphism. The vertex and edge data of the data graph
266    * elements is attached to the resulting vertices.
267    * <p>
268    * Note, that this method used no statistics about the data graph which may result in bad runtime
269    * performance. Use {@link #temporalQuery(String, String, TemporalGraphStatistics)} to provide statistics
270    * for the query planner.
271    * <p>
272    * In addition, the operator can be supplied with a construction pattern allowing the creation of new graph
273    * elements based on variable bindings of the match pattern. Consider the following example:
274    * <br>
275    * {@code graph.query(
276    * "MATCH (a:Author)-[:WROTE]->(:Paper)<-[:WROTE]-(b:Author) WHERE a <> b",
277    * "(a)-[:CO_AUTHOR]->(b)")}
278    * <p>
279    * The query pattern is looking for pairs of authors that worked on the same paper.
280    * The construction pattern defines a new edge of type CO_AUTHOR between the two entities.
281    *
282    * @param temporalGdlQuery A Temporal-GDL query as {@link String}
283    * @param constructionPattern Construction pattern in Temporal-GDL format
284    * @return graph collection containing the output of the construct pattern
285    */
286   default TemporalGraphCollection temporalQuery(String temporalGdlQuery, String constructionPattern) {
287     return temporalQuery(temporalGdlQuery, constructionPattern, new DummyTemporalGraphStatistics());
288   }

378   /**
379    * Grouping operator that aggregates valid times per group and sets it as new valid time.
380    * The grouped validFrom value will be computed by min over all validFrom values.
381    * The grouped validTo value will be computed by max over all validTo values.
382    *
383    * @param vertexGroupingKeys       property keys to group vertices
384    * @param vertexAggregateFunctions aggregate functions to apply on super vertices
385    * @param edgeGroupingKeys         property keys to group edges
386    * @param edgeAggregateFunctions   aggregate functions to apply on super edges
387    * @return summary graph
388    * @see KeyedGrouping
389    */
390   default TemporalGraph temporalGroupBy(List<KeyFunction<TemporalVertex, ?>> vertexGroupingKeys,
391     List<AggregateFunction> vertexAggregateFunctions, List<KeyFunction<TemporalEdge, ?>> edgeGroupingKeys,
392     List<AggregateFunction> edgeAggregateFunctions) {
393     // Add min/max valid time aggregations that will result in the new valid times
394     List<AggregateFunction> tempVertexAgg = new ArrayList<>(vertexAggregateFunctions);
395     tempVertexAgg.add(new MinVertexTime(TimeDimension.VALID_TIME, TimeDimension.Field.FROM)
396       .setAsValidTime(TimeDimension.Field.FROM));
397     tempVertexAgg.add(new MaxVertexTime(TimeDimension.VALID_TIME, TimeDimension.Field.TO)
398       .setAsValidTime(TimeDimension.Field.TO));
399     List<AggregateFunction> tempEdgeAgg = new ArrayList<>(edgeAggregateFunctions);
400     tempEdgeAgg.add(new MinEdgeTime(TimeDimension.VALID_TIME, TimeDimension.Field.FROM)
401       .setAsValidTime(TimeDimension.Field.FROM));
402     tempEdgeAgg.add(new MaxEdgeTime(TimeDimension.VALID_TIME, TimeDimension.Field.TO)
403       .setAsValidTime(TimeDimension.Field.TO));
404 
405     return callForGraph(new KeyedGrouping<>(vertexGroupingKeys, tempVertexAgg, edgeGroupingKeys,
406       tempEdgeAgg));
407   }

422   /**
423    * Updates edges of this graph to set their validity such that an edge is only valid if both of
424    * its adjacent vertices are valid at that time. Edges that can never be valid since the
