28  /**
29   * Factory for creating temporal vertex POJOs.
30   */
31  public class TemporalVertexFactory implements VertexFactory<TemporalVertex>, Serializable {
32  
33    @Override
34    public TemporalVertex createVertex() {
35      return initVertex(GradoopId.get());
36    }

38    @Override

41    }

44    public TemporalVertex createVertex(String label) {
45      return initVertex(GradoopId.get(), label);
46    }

50      return initVertex(id, label, null, null);
51    }

78    @Override
79    public TemporalVertex initVertex(GradoopId id, String label, Properties properties, GradoopIdSet graphIds) {
80      return new TemporalVertex(
81        Objects.requireNonNull(id, "Identifier is null."),
82        Objects.requireNonNull(label, "Label is null."),
83        properties,
84        graphIds,
85        null,
86        null
87      );
88    }
89  
90    @Override
91    public Class<TemporalVertex> getType() {
92      return TemporalVertex.class;
93    }

95  
96    /**
97     * Initializes a temporal vertex based on the given parameters. If the valid times are null,
98     * default values are used.
99     *
100    * @param id              vertex identifier
101    * @param label           vertex label
102    * @param properties      vertex properties
103    * @param graphIds        graphIds, that contain the vertex
104    * @param validFrom       begin of the elements validity as unix timestamp [ms] or null
105    * @param validTo         end of the elements validity as unix timestamp [ms] or null
106    * @return the temporal vertex instance
107    */
108   public TemporalVertex initVertex(GradoopId id, String label, Properties properties,
109     GradoopIdSet graphIds, Long validFrom, Long validTo) {
110     return new TemporalVertex(
111       Objects.requireNonNull(id, "Identifier is null."),
112       Objects.requireNonNull(label, "Label is null."),
113       properties,
114       graphIds,
115       validFrom,
116       validTo);
117   }
118 
119   /**
120    * Helper function to create a TPGM vertex from an EPGM vertex.
121    * The id, label and all other information will be inherited.
122    *
123    * @param vertex the EPGM vertex instance
124    * @return a TPGM vertex instance with default values at its valid times
125    */
126   public TemporalVertex fromNonTemporalVertex(Vertex vertex) {
127     return initVertex(vertex.getId(), vertex.getLabel(), vertex.getProperties(),
128       vertex.getGraphIds(), null, null);
129   }
