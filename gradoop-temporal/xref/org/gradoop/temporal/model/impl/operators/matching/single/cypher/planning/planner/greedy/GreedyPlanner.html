67  /**
68   * Greedy query planning for temporal queries
69   * @param <G> a temporal graph head type
70   * @param <LG> a logical graph type for temporal elements
71   * @param <GC> a graph collection type for temporal elements
72   */
73  public class GreedyPlanner<
74    G extends TemporalGraphHead,
75    LG extends BaseGraph<G, TemporalVertex, TemporalEdge, LG, GC>,
76    GC extends BaseGraphCollection<G, TemporalVertex, TemporalEdge, LG, GC>> {

107   /**
108    * Creates a new greedy planner.
109    *
110    * @param graph           search graph
111    * @param queryHandler    query handler
112    * @param graphStatistics search graph statistics
113    * @param vertexStrategy  morphism type for vertex mappings
114    * @param edgeStrategy    morphism type for edge mappings
115    */
116   public GreedyPlanner(LG graph, TemporalQueryHandler queryHandler,
117                        TemporalGraphStatistics graphStatistics, MatchStrategy vertexStrategy,
118                        MatchStrategy edgeStrategy) {
119     this.graph = graph;
120     this.queryHandler = queryHandler;
121     this.graphStatistics = graphStatistics;
122     this.vertexStrategy = vertexStrategy;
123     this.edgeStrategy = edgeStrategy;
124     this.cnfEstimation = new CNFEstimation(graphStatistics, queryHandler);
125     this.cnf = cnfEstimation.reorderCNF(queryHandler.getPredicates());
126   }

128   /**
129    * Computes the {@link TemporalPlanTableEntry} that wraps the {@link QueryPlan} with the minimum costs
130    * according to the greedy optimization algorithm.
131    *
132    * @return entry with minimum execution costs
133    */
134   public TemporalPlanTableEntry plan() {
135     PlanTable planTable = initPlanTable();
136 
137     while (planTable.size() > 1) {
138       PlanTable newPlans = evaluateJoins(planTable);
139 
140       if (newPlans.size() == 0) {
141         // No new plans where generated by joining but there are still multiple disconnected
142         // query graph components. In this case we need to do a cartesian product.
143         newPlans = evaluateCartesianProducts(planTable);
144       }
145       newPlans = evaluateFilter(newPlans);
146       newPlans = evaluateProjection(newPlans);
147 
148       // get plan with minimum costs and remove all plans covered by this plan
149       TemporalPlanTableEntry bestEntry = (TemporalPlanTableEntry) newPlans.min();
150       planTable.removeCoveredBy(bestEntry);
151       planTable.add(bestEntry);
152     }
153 
154     return (TemporalPlanTableEntry) planTable.get(0);
155   }

177   /**
178    * Creates an initial {@link TemporalPlanTableEntry} for each vertex in the query graph and adds it to the
179    * specified {@link PlanTable}. The entry wraps a query plan that filters vertices based on their
180    * predicates and projects properties that are required for further query planning.
181    *
182    * @param planTable plan table
183    */
184   private void createVertexPlans(PlanTable planTable) {
185     for (Vertex vertex : queryHandler.getVertices()) {
186       String vertexVariable = vertex.getVariable();
187       CNF allPredicates = this.cnf;
188 
189       CNF vertexPredicates = allPredicates.removeSubCNF(vertexVariable);
190       Set<String> projectionKeys = allPredicates.getPropertyKeys(vertexVariable);
191 
192       DataSet<TemporalVertex> vertices =
193         vertex.getLabel().equals(GradoopConstants.DEFAULT_VERTEX_LABEL) ?
194           graph.getVertices() : graph.getVerticesByLabel(vertex.getLabel());
195 
196       FilterAndProjectTemporalVerticesNode node =
197         new FilterAndProjectTemporalVerticesNode(vertices,
198           vertex.getVariable(), vertexPredicates, projectionKeys);
199 
200       planTable.add(new TemporalPlanTableEntry(VERTEX, Sets.newHashSet(vertexVariable), allPredicates,
201         new TemporalQueryPlanEstimator(new QueryPlan(node), queryHandler, graphStatistics, cnfEstimation)));
202     }
203   }
204 
205   /**
206    * Creates an initial {@link TemporalPlanTableEntry} for each edge in the query graph and adds it to the
207    * specified {@link PlanTable}. The entry wraps a {@link QueryPlan} that filters edges based on
208    * their predicates and projects properties that are required for further query planning.
209    *
210    * @param planTable plan table
211    */
212   private void createEdgePlans(PlanTable planTable) {
213     for (Edge edge : queryHandler.getEdges()) {
214       String edgeVariable = edge.getVariable();
215       String sourceVariable = queryHandler.getVertexById(edge.getSourceVertexId()).getVariable();
216       String targetVariable = queryHandler.getVertexById(edge.getTargetVertexId()).getVariable();
217 
218       CNF allPredicates = this.cnf;
219 
220       CNF edgePredicates = allPredicates.removeSubCNF(edgeVariable);
221 
222       Set<String> projectionKeys = allPredicates.getPropertyKeys(edgeVariable);
223 
224       boolean isPath = edge.getUpperBound() != 1;
225 
226       DataSet<TemporalEdge> edges =
227         edge.getLabel().equals(GradoopConstants.DEFAULT_EDGE_LABEL) ?
228           graph.getEdges() : graph.getEdgesByLabel(edge.getLabel());
229 
230       FilterAndProjectTemporalEdgesNode node = new FilterAndProjectTemporalEdgesNode(edges,
231         sourceVariable, edgeVariable, targetVariable, edgePredicates, projectionKeys, isPath);
232 
233       PlanTableEntry.Type type = edge.hasVariableLength() ? PATH : EDGE;
234 
235       planTable.add(new TemporalPlanTableEntry(type, Sets.newHashSet(edgeVariable), allPredicates,
236         new TemporalQueryPlanEstimator(new QueryPlan(node), queryHandler, graphStatistics, cnfEstimation)));
237     }
238   }

244   /**
245    * Evaluates which entries in the specified plan table can be joined. The joined entries
246    * are added to a new table which is returned.
247    *
248    * @param currentTable query plan table
249    * @return new table containing solely joined plans from the input table
250    */
251   private PlanTable evaluateJoins(PlanTable currentTable) {
252     PlanTable newTable = new PlanTable();
253 
254     for (int i = 0; i < currentTable.size(); i++) {
255       PlanTableEntry leftEntry = currentTable.get(i);
256       if (mayExtend(leftEntry)) {
257         for (int j = 0; j < currentTable.size(); j++) {
258           PlanTableEntry rightEntry = currentTable.get(j);
259           if (i != j) {
260             List<String> joinVariables = getOverlap(leftEntry, rightEntry);
261             if (joinVariables.size() > 0) {
262               if (rightEntry.getType() == PATH && joinVariables.size() == 2) {
263                 // evaluate join with variable length path on source and target vertex
264                 newTable.add(joinEntries(leftEntry, rightEntry, joinVariables.subList(0, 1)));
265                 newTable.add(joinEntries(leftEntry, rightEntry, joinVariables.subList(1, 2)));
266               } else {
267                 // regular join or join with variable length path on source or target vertex
268                 newTable.add(joinEntries(leftEntry, rightEntry, joinVariables));
269               }
270             }
271           }
272         }
273       }
274     }
275     return newTable;
276   }

278   /**
279    * Checks if the given entry may be extended. This is only the case for entries that represents
280    * either a vertex or a partial match graph.
281    *
282    * @param entry plan table entry
283    * @return true, iff the specified entry may be extended
284    */
285   private boolean mayExtend(PlanTableEntry entry) {
286     return entry.getType() == VERTEX || entry.getType() == GRAPH;
287   }

289   /**
290    * Computes the overlapping query variables of the specified entries.
291    *
292    * @param firstEntry  first entry
293    * @param secondEntry second entry
294    * @return variables that are available in both input entries
295    */
296   private List<String> getOverlap(PlanTableEntry firstEntry, PlanTableEntry secondEntry) {
297     Set<String> overlap = firstEntry.getAllVariables();
298     overlap.retainAll(secondEntry.getAllVariables());
299     return new ArrayList<>(overlap);
300   }

302   /**
303    * Joins the query plans represented by the specified plan table entries.
304    * <p>
305    * The method considers if the right entry is a variable length path and in that case
306    * creates an {@link ExpandEmbeddingsNode}. In any other case, a regular
307    * {@link JoinEmbeddingsNode} is used to join the query plans.
308    *
309    * @param leftEntry     left entry
310    * @param rightEntry    right entry
311    * @param joinVariables join variables
312    * @return an entry that represents the join of both input entries
313    */
314   private PlanTableEntry joinEntries(PlanTableEntry leftEntry, PlanTableEntry rightEntry,
315                                              List<String> joinVariables) {
316 
317     PlanNode node;
318     if (rightEntry.getType() == PATH) {
319       assert joinVariables.size() == 1;
320       node = createExpandNode(leftEntry, rightEntry, joinVariables.get(0));
321     } else {
322       node = new JoinEmbeddingsNode(leftEntry.getQueryPlan().getRoot(),
323         rightEntry.getQueryPlan().getRoot(), joinVariables, vertexStrategy, edgeStrategy);
324     }
325 
326     // update processed variables
327     HashSet<String> processedVariables = Sets.newHashSet(leftEntry.getProcessedVariables());
328     processedVariables.addAll(rightEntry.getProcessedVariables());
329     // create resulting predicates
330     // TODO: this might be moved to the join/expand node in issue #510
331     CNF predicates = mergePredicates(leftEntry, rightEntry);
332 
333     return new TemporalPlanTableEntry(GRAPH, processedVariables, predicates,
334       new TemporalQueryPlanEstimator(new QueryPlan(node), queryHandler, graphStatistics, cnfEstimation));
335   }

337   /**
338    * Creates an {@link ExpandEmbeddingsNode} from the specified arguments.
339    *
340    * @param leftEntry     left entry
341    * @param rightEntry    right entry
342    * @param startVariable vertex variable to expand from
343    * @return new expand node
344    */
345   private ExpandEmbeddingsNode createExpandNode(PlanTableEntry leftEntry, PlanTableEntry rightEntry,
346                                                 String startVariable) {
347 
348     String pathVariable = rightEntry.getQueryPlan().getRoot()
349       .getEmbeddingMetaData().getEdgeVariables().get(0);
350 
351     Edge queryEdge = queryHandler.getEdgeByVariable(pathVariable);
352     Vertex sourceVertex = queryHandler.getVertexById(queryEdge.getSourceVertexId());
353     Vertex targetVertex = queryHandler.getVertexById(queryEdge.getTargetVertexId());
354 
355     int lowerBound = queryEdge.getLowerBound();
356     int upperBound = queryEdge.getUpperBound();
357     ExpandDirection direction = sourceVertex.getVariable().equals(startVariable) ?
358       ExpandDirection.OUT : ExpandDirection.IN;
359     String endVariable = direction == ExpandDirection.OUT ?
360       targetVertex.getVariable() : sourceVertex.getVariable();
361 
362     return new ExpandEmbeddingsNode(leftEntry.getQueryPlan().getRoot(),
363       rightEntry.getQueryPlan().getRoot(),
364       startVariable, pathVariable, endVariable, lowerBound, upperBound, direction,
365       vertexStrategy, edgeStrategy);
366   }

372   /**
373    * The method checks if a filter can be applied on any of the entries in the specified table. If
374    * this is the case, a {@link FilterEmbeddingsNode} is added to the query plan represented by the
375    * affected entries.
376    *
377    * @param currentTable query plan table
378    * @return input table with possibly updated entries
379    */
380   private PlanTable evaluateFilter(PlanTable currentTable) {
381     PlanTable newTable = new PlanTable();
382 
383     for (PlanTableEntry entry : currentTable) {
384       Set<String> variables = Sets.newHashSet(entry.getProcessedVariables());
385       CNF predicates = entry.getPredicates();
386       CNF subCNF = predicates.removeSubCNF(variables);
387       if (subCNF.size() > 0) {
388         FilterEmbeddingsNode node = new FilterEmbeddingsNode(
389           entry.getQueryPlan().getRoot(), subCNF);
390         newTable.add(new TemporalPlanTableEntry(GRAPH, Sets.newHashSet(entry.getProcessedVariables()),
391           predicates,
392           new TemporalQueryPlanEstimator(new QueryPlan(node), queryHandler, graphStatistics, cnfEstimation)));
393       } else {
394         newTable.add(entry);
395       }
396     }
397 
398     return newTable;
399   }

405   /**
406    * The method checks if a filter can be applied on any of the entries in the specified table. If
407    * this is the case, a {@link ProjectEmbeddingsNode} is added to the query plan represented by the
408    * affected entries.
409    *
410    * @param currentTable query plan table
411    * @return input table with possibly updated entries
412    */
413   private PlanTable evaluateProjection(PlanTable currentTable) {
414     PlanTable newTable = new PlanTable();
415 
416     for (PlanTableEntry entry : currentTable) {
417       Set<Pair<String, String>> propertyPairs = entry.getPropertyPairs();
418       Set<Pair<String, String>> projectionPairs = entry.getProjectionPairs();
419 
420       Set<Pair<String, String>> updatedPropertyPairs = propertyPairs.stream()
421         .filter(projectionPairs::contains)
422         .collect(Collectors.toSet());
423 
424       if (updatedPropertyPairs.size() < propertyPairs.size()) {
425         ProjectEmbeddingsNode node = new ProjectEmbeddingsNode(entry.getQueryPlan().getRoot(),
426           new ArrayList<>(updatedPropertyPairs));
427         newTable.add(new TemporalPlanTableEntry(GRAPH,
428           Sets.newHashSet(entry.getProcessedVariables()), entry.getPredicates(),
429           new TemporalQueryPlanEstimator(new QueryPlan(node), queryHandler, graphStatistics, cnfEstimation)));
430       } else {
431         newTable.add(entry);
432       }
433     }
434     return newTable;
435   }

441   /**
442    * Evaluates which entries in the specified plan table can be joined. The joined entries
443    * are added to a new table which is returned.
444    *
445    * @param currentTable query plan table
446    * @return new table containing solely joined plans from the input table
447    */
448   private PlanTable evaluateCartesianProducts(PlanTable currentTable) {
449     PlanTable newTable = new PlanTable();
450     for (int i = 0; i < currentTable.size(); i++) {
451       PlanTableEntry leftEntry = currentTable.get(i);
452       for (int j = i + 1; j < currentTable.size(); j++) {
453         PlanTableEntry rightEntry = currentTable.get(j);
454         CNF joinPredicate = getJoinPredicate(leftEntry, rightEntry);
455         if (joinPredicate.size() > 0) {
456           newTable.add(createValueJoinEntry(leftEntry, rightEntry, joinPredicate));
457         } else {
458           // regular join or join with variable length path on source or target vertex
459           newTable.add(createCartesianProductEntry(leftEntry, rightEntry));
460         }
461       }
462     }
463     return newTable;
464   }

466   /**
467    * Computes the overlapping query variables of the specified entries.
468    *
469    * @param leftEntry  first entry
470    * @param rightEntry second entry
471    * @return variables that are available in both input entries
472    */
473   private CNF getJoinPredicate(PlanTableEntry leftEntry, PlanTableEntry rightEntry) {
474     Set<String> allVariables = leftEntry.getAllVariables();
475     allVariables.addAll(rightEntry.getAllVariables());
476 
477     CNF leftPredicates = new CNF(leftEntry.getPredicates());
478     CNF rightPredicates = new CNF(rightEntry.getPredicates());
479     leftPredicates.removeSubCNF(rightEntry.getProcessedVariables());
480     rightPredicates.removeSubCNF(leftEntry.getProcessedVariables());
481     CNF predicates = leftPredicates.and(rightPredicates).getSubCNF(allVariables);
482 
483     return new CNF(
484       predicates.getPredicates()
485         .stream()
486         .filter(p ->
487           p.size() == 1 && p.getPredicates().get(0).getComparator().equals(Comparator.EQ)
488         ).collect(Collectors.toList())
489     );
490   }

520   /**
521    * Creates an {@link ValueJoinNode} from the specified arguments.
522    *
523    * @param leftEntry     left entry
524    * @param rightEntry    right entry
525    * @param joinPredicate join predicate
526    * @return new value join node
527    */
528   private PlanTableEntry createValueJoinEntry(PlanTableEntry leftEntry,
529                                                       PlanTableEntry rightEntry,
530                                                       CNF joinPredicate) {
531 
532     List<Pair<String, String>> leftProperties = new ArrayList<>();
533     List<Pair<String, String>> rightProperties = new ArrayList<>();
534 
535     for (CNFElement e : joinPredicate.getPredicates()) {
536       ComparisonExpression comparison = e.getPredicates().get(0);
537 
538       Pair<String, String> joinExpression = extractJoinExpression(comparison.getLhs());
539       if (leftEntry.getAllVariables().contains(joinExpression.getKey())) {
540         /*if (!isTemporal(joinExpression)) {
541           leftProperties.add(joinExpression);
542         }*/
543         leftProperties.add(joinExpression);
544       } else {
545         /*if (!isTemporal(joinExpression)) {
546           rightProperties.add(joinExpression);
547         }*/
548         rightProperties.add(joinExpression);
549       }
550 
551       joinExpression = extractJoinExpression(comparison.getRhs());
552       if (leftEntry.getAllVariables().contains(joinExpression.getKey())) {
553         /*if (!isTemporal(joinExpression)) {
554           leftProperties.add(joinExpression);
555         }*/
556         leftProperties.add(joinExpression);
557       } else {
558         /*if (!isTemporal(joinExpression)) {
559           rightProperties.add(joinExpression);
560         }*/
561         rightProperties.add(joinExpression);
562       }
563     }
564 
565     ValueJoinNode node = new ValueJoinNode(
566       leftEntry.getQueryPlan().getRoot(),
567       rightEntry.getQueryPlan().getRoot(),
568       leftProperties, rightProperties,
569       vertexStrategy, edgeStrategy
570     );
571 
572     Set<String> processedVariables = leftEntry.getProcessedVariables();
573     processedVariables.addAll(rightEntry.getProcessedVariables());
574 
575     CNF predicates = mergePredicates(leftEntry, rightEntry);
576 
577     return new TemporalPlanTableEntry(
578       GRAPH,
579       processedVariables,
580       predicates,
581       new TemporalQueryPlanEstimator(new QueryPlan(node), queryHandler, graphStatistics, cnfEstimation)
582     );
583   }

585   /**
586    * Turns a QueryComparable into a {@code Pair<Variable, PropertyKey>}
587    *
588    * @param comparable query comparable
589    * @return join property
590    */
591   private Pair<String, String> extractJoinExpression(QueryComparable comparable) {
592     if (comparable instanceof PropertySelectorComparable) {
593       PropertySelectorComparable propertySelector = (PropertySelectorComparable) comparable;
594       return Pair.of(propertySelector.getVariable(), propertySelector.getPropertyKey());
595     }
596     if (comparable instanceof TimeSelectorComparable) {
597       TimeSelectorComparable timeSelector = (TimeSelectorComparable) comparable;
598       return Pair.of(timeSelector.getVariable(), timeSelector.getTimeField().name());
599     } else {
600       //TODO #580 Include ElementSelector -> ID needs to be projected as property
601       throw new RuntimeException("Comparable " + comparable + "cant be used for ValueJoin");
602     }
603   }

605   /**
606    * Creates a new predicate that includes only elements that exist in both input predicates
607    *
