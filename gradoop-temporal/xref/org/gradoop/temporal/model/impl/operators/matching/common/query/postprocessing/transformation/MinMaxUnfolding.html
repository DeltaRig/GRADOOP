36  /**
37   * Reformulates certain expressions involving MIN and MAX elements, according to the following rules:
38   * <ul>
39   * <li> a singleton clause [MAX(a,b,c) < / <= x] is reformulated to
40   * [a < / <= x] AND [b < / <= x] AND [c < / <= x] </li>
41   * <li> a n-ary disjunctive clause [comp_1,...,comp_(i-1), MIN(a,b,c) < / <= x, comp_(i+1),...,comp_n]
42   * is reformulated to
43   * [comp_1,...,comp_(i-1), a < / <= x, b < / <= x, c < / <= x, comp_(i+1),...,comp_n] </li>
44   * <li> a n-ary disjunctive clause [comp_1,...,comp_(i-1), x < / <= MAX(a,b,c), comp_(i+1),...,comp_n]
45   * is reformulated to
46   * [comp_1,...,comp_(i-1), x < / <= a, x < / <= b, x < / <= c, comp_(i+1),...,comp_n] </li>
47   * <li> a unary clause [x < / <= MIN(a,b,c)] is reformulated to
48   * [x < / <= a] AND [x < / <= b] AND [x < / <= c] </li>
49   * </ul>
50   *
51   * These rules are applied exhaustively.
52   * !!! This class assumes input CNFs to be normalized, i.e. not to contain > or >= !!!
53   */
54  public class MinMaxUnfolding implements QueryTransformation {
55  
56    @Override
57    public CNF transformCNF(CNF cnf) {
58      if (cnf.getPredicates().size() == 0) {
59        return cnf;
60      } else {
61        // apply transformations exhaustively
62        CNF oldCNF;
63        CNF newCNF = cnf;
64        do {
65          oldCNF = newCNF;
66          newCNF = newCNF.getPredicates().stream()
67            .map(this::unfoldNext)
68            .reduce(CNF::and).get();
69        } while (!newCNF.equals(oldCNF));
70        return newCNF;
71      }
72    }

74    /**
75     * Applies the rules to a single clause (not necessarily exhaustive, several calls may be necessary)
76     *
77     * @param clause clause to apply the rules to
78     * @return transformed clause
79     */
80    private CNF unfoldNext(CNFElement clause) {
81      // determine whether clause is singleton or not and and trigger the corresponding rule applications
82      if (clause.getPredicates().size() > 1) {
83        return new CNF(
84          clause.getPredicates().stream()
85            // no conjunctions within a disjunctive clause
86            .map(comp -> unfoldNext(comp, false))
87            .reduce(CNF::or)
88            .get()
89        );
90      } else {
91        return unfoldNext(clause.getPredicates().get(0), true);
92      }
93    }

95    /**
96     * Applies the rules to a single comparison (not necessarily exhaustive), depending
97     * on whether the comparison is part of a singleton clause or a n-ary one (n>1).
98     *
99     * @param comp              comparison to apply the rules to
100    * @param allowConjunctions true iff the result is allowed to be a n-ary conjunction (n > 1)
101    * @return transformed comparison as a CNF
102    */
103   private CNF unfoldNext(ComparisonExpression comp, boolean allowConjunctions) {
104     // unwrap comparison
105     ComparableExpression lhs = comp.getLhs().getWrappedComparable();
106     Comparator comparator = comp.getComparator();
107     ComparableExpression rhs = comp.getRhs().getWrappedComparable();
108 
109     // apply rules to comparison
110     if (lhs instanceof MinTimePoint) {
111       if (comparator == Comparator.LT || comparator == LTE) {
112         return exists(((MinTimePoint) lhs).getArgs(), comparator, rhs);
113       }
114     } else if (lhs instanceof MaxTimePoint) {
115       if (comparator == Comparator.LT || comparator == LTE) {
116         if (allowConjunctions) {
117           return forAll(((MaxTimePoint) lhs).getArgs(), comparator, rhs);
118         }
119       }
120     } else if (rhs instanceof MinTimePoint) {
121       if (comparator == Comparator.LT || comparator == Comparator.LTE) {
122         if (allowConjunctions) {
123           return forAll(lhs, comparator, ((MinTimePoint) rhs).getArgs());
124         }
125       }
126     } else if (rhs instanceof MaxTimePoint) {
127       if (comparator == Comparator.LT || comparator == Comparator.LTE) {
128         return exists(lhs, comparator, ((MaxTimePoint) rhs).getArgs());
129       }
130     }
131 
132     return new CNF(new ArrayList<>(
133       Collections.singletonList(new CNFElement(new ArrayList<>(Collections.singletonList(comp))))));
134 
135   }

137   /**
138    * Realizes an "exists" predicate as CNF.
139    * E.g., {@code exists((a,b,c), <=, x)} would yield
140    * a <= x OR b <= x OR c <= x
141    *
142    * @param args       left hand side, domain of the "exists"
143    * @param comparator comparator
144    * @param rhs        right hand side
145    * @return "exists" predicate as CNF
146    */
147   private CNF exists(List<TimePoint> args, Comparator comparator, ComparableExpression rhs) {
148     List<ComparisonExpression> comparisons = new ArrayList<>();
149     for (TimePoint arg : args) {
150       comparisons.add(new ComparisonExpression(new Comparison(arg, comparator, rhs),
151         new ComparableTPGMFactory()));
152     }
153     CNFElement singleClause = new CNFElement(comparisons);
154     return new CNF(new ArrayList<>(Collections.singletonList(singleClause)));
155   }
156 
157   /**
158    * Realizes an "exists" predicate as CNF.
159    * E.g., {@code exists(x, <= (a,b,c)} would yield
160    * x <= a OR x <= b OR x <= c
161    *
162    * @param lhs        left hand side
163    * @param comparator comparator
164    * @param args       right hand side, domain of the "exists"
165    * @return "exists" predicate as CNF
166    */
167   private CNF exists(ComparableExpression lhs, Comparator comparator, List<TimePoint> args) {
168     List<ComparisonExpression> comparisons = new ArrayList<>();
169     for (TimePoint arg : args) {
170       comparisons.add(new ComparisonExpression(new Comparison(lhs, comparator, arg),
171         new ComparableTPGMFactory()));
172     }
173     return new CNF(new ArrayList<>(Collections.singletonList(new CNFElement(comparisons))));
174   }

176   /**
177    * Realizes a "forall" predicate as CNF.
178    * E.g., {@code forAll((a,b,c), <= x)} would yield
179    * a <= x AND b <= x AND c <= x
180    *
181    * @param args       left hand side, domain of the "forall"
182    * @param comparator comparator
183    * @param rhs        right hand side
184    * @return "forall" predicate as CNF
185    */
186   private CNF forAll(List<TimePoint> args, Comparator comparator, ComparableExpression rhs) {
187     List<CNFElement> clauses = new ArrayList<>();
188     for (TimePoint arg : args) {
189       clauses.add(new CNFElement(new ArrayList<>(Collections.singletonList(new ComparisonExpression(
190         new Comparison(arg, comparator, rhs),
191         new ComparableTPGMFactory()
192       )))));
193     }
194     return new CNF(clauses);
195   }

197   /**
198    * Realizes a "forall" predicate as CNF.
199    * E.g., {@code forAll(x, <=, (a,b,c))} would yield
200    * x <= a AND x <= b AND x <= c
201    *
202    * @param lhs        left hand side
203    * @param comparator comparator
204    * @param args       right hand side, domain of the "forall"
205    * @return "forall" predicate as CNF
206    */
207   private CNF forAll(ComparableExpression lhs, Comparator comparator, List<TimePoint> args) {
208     List<CNFElement> clauses = new ArrayList<>();
209     for (TimePoint arg : args) {
210       clauses.add(new CNFElement(new ArrayList<>(Collections.singletonList(new ComparisonExpression(
211         new Comparison(lhs, comparator, arg),
212         new ComparableTPGMFactory()
213       )))));
214     }
215     return new CNF(clauses);
216   }
