55  /**
56   * Transformation that aims to infer new, informative bounds for temporal properties
57   * according to the clauses.
58   * Furthermore, contradictions are detected.
59   */
60  public class BoundsInference implements QueryTransformation {
61  
62    @Override
63    public CNF transformCNF(CNF cnf) throws QueryContradictoryException {
64      /*
65       * stores the newly inferred lower bounds for time selectors
66       */
67      HashMap<TimeSelector, Long> lowerBounds = new HashMap<>();
68      /*
69       * stores the newly inferred upper bounds for time selectors
70       */
71      HashMap<TimeSelector, Long> upperBounds = new HashMap<>();
72      // clauses (not) to consider
73      List<CNFElement> relevantClauses = cnf.getPredicates().stream()
74        .filter(this::isRelevantClause)
75        .collect(Collectors.toList());
76      List<CNFElement> otherClauses = cnf.getPredicates().stream()
77        .filter(clause -> !isRelevantClause(clause))
78        .collect(Collectors.toList());
79      List<ComparisonExpression> relevantComparisons = relevantClauses.stream()
80        .map(clause -> clause.getPredicates().get(0))
81        .collect(Collectors.toList());
82  
83      // stores all tuples (a,b) with a=b in relevant comparisons
84      HashMap<ComparableExpression, HashSet<ComparableExpression>> cEq = new HashMap<>();
85      // stores all tuples (a,b) with a<=b in relevant comparisons
86      HashMap<ComparableExpression, HashSet<ComparableExpression>> cLeq = new HashMap<>();
87      // stores all tuples (a,b) with a<b in relevant comparisons
88      HashMap<ComparableExpression, HashSet<ComparableExpression>> cLt = new HashMap<>();
89      // stores all tuples (a,b) with a!=b in relevant comparisons
90      HashMap<ComparableExpression, HashSet<ComparableExpression>> cNeq = new HashMap<>();
91  
92  
93      // init bounds
94      for (ComparisonExpression comp : relevantComparisons) {
95        ComparableExpression lhs = comp.getLhs().getWrappedComparable();
96        ComparableExpression rhs = comp.getRhs().getWrappedComparable();
97        Comparator comparator = comp.getComparator();
98        if (comp.getLhs() instanceof TimeSelectorComparable) {
99          lowerBounds.put((TimeSelector) lhs, TemporalElement.DEFAULT_TIME_FROM);
100         upperBounds.put((TimeSelector) lhs, TemporalElement.DEFAULT_TIME_TO);
101       }
102       if (comp.getRhs() instanceof TimeSelectorComparable) {
103         lowerBounds.put((TimeSelector) rhs, TemporalElement.DEFAULT_TIME_FROM);
104         upperBounds.put((TimeSelector) rhs, TemporalElement.DEFAULT_TIME_TO);
105       }
106       // init c-relations
107       if (comparator.equals(EQ)) {
108         cEq.putIfAbsent(lhs, new HashSet<>());
109         cEq.putIfAbsent(rhs, new HashSet<>());
110         cEq.get(lhs).add(rhs);
111         cEq.get(rhs).add(lhs);
112       } else if (comparator.equals(Comparator.LTE)) {
113         cLeq.putIfAbsent(lhs, new HashSet<>());
114         cLeq.get(lhs).add(rhs);
115       } else if (comparator.equals(Comparator.LT)) {
116         cLt.putIfAbsent(lhs, new HashSet<>());
117         cLt.get(lhs).add(rhs);
118       } else if (comparator.equals(Comparator.NEQ)) {
119         cNeq.putIfAbsent(lhs, new HashSet<>());
120         cNeq.get(lhs).add(rhs);
121       }
122     }
123 
124     // compute closures for =,<,<=
125     List<HashMap<ComparableExpression, HashSet<ComparableExpression>>> closures =
126       computeClosureRelations(cEq, cLeq, cLt);
127 
128     checkLtContradictions(closures.get(2));
129     // infer new bounds
130     List<HashMap<TimeSelector, Long>> newBounds = updateEq(closures.get(0), lowerBounds, upperBounds);
131     newBounds = updateLeq(closures.get(1), newBounds.get(0), newBounds.get(1));
132     newBounds = updateLt(closures.get(2), newBounds.get(0), newBounds.get(1));
133 
134     // check them for contradictions with NEQ clauses
135     checkNeqContradictions(cNeq, newBounds.get(0), newBounds.get(1));
136     // filter redundant clauses like a.tx_from <= a.tx_to and comparisons between selectors and literals
137     List<ComparisonExpression> remainingOldComparisons = filterRedundantComparisons(relevantComparisons);
138     // construct new clauses
139     List<ComparisonExpression> newComparisons = comparisons(newBounds.get(0), newBounds.get(1));
140 
141     // construct the CNF
142     List<CNFElement> remainingOldClauses = remainingOldComparisons.stream()
143       .map(comparison -> new CNFElement(Collections.singletonList(comparison)))
144       .collect(Collectors.toList());
145     List<CNFElement> newClauses = newComparisons.stream()
146       .map(comparison -> new CNFElement(Collections.singletonList(comparison)))
147       .collect(Collectors.toList());
148 
149     List<CNFElement> allClauses = otherClauses;
150     allClauses.addAll(remainingOldClauses);
151     allClauses.addAll(newClauses);
152     return new CNF(allClauses);
153   }

155   /**
156    * Constructs new comparisons from the inferred bounds
157    *
158    * @param lowerBounds inferred lower bounds
159    * @param upperBounds inferred upper bounds
160    * @return inferred comparisons
161    */
162   private List<ComparisonExpression> comparisons(HashMap<TimeSelector, Long> lowerBounds,
163                                                  HashMap<TimeSelector, Long> upperBounds) {
164     List<ComparisonExpression> newComparisons = new ArrayList<>();
165     // for all selectors, as lowerBounds has the same keys as upperBounds
166     for (Map.Entry<TimeSelector, Long> entry : lowerBounds.entrySet()) {
167       TimeSelector selector = entry.getKey();
168       Long lower = entry.getValue();
169       Long upper = upperBounds.get(selector);
170       if (lower.equals(upper)) {
171         newComparisons.add(new ComparisonExpression(
172           new Comparison(selector, EQ, new TimeLiteral(lower)),
173           new ComparableTPGMFactory()));
174       } else {
175         if (lower > TemporalElement.DEFAULT_TIME_FROM) {
176           // check if informative: lower bound of from is trivial lower bound of to
177           if (selector.getTimeProp().equals(TX_TO)) {
178             TimeSelector txFromSel = new TimeSelector(selector.getVariable(), TX_FROM);
179             if (lowerBounds.getOrDefault(txFromSel, TemporalElement.DEFAULT_TIME_FROM).equals(lower)) {
180               continue;
181             }
182           } else if (selector.getTimeProp().equals(VAL_TO)) {
183             TimeSelector valFromSel = new TimeSelector(selector.getVariable(), VAL_FROM);
184             if (lowerBounds.getOrDefault(valFromSel, TemporalElement.DEFAULT_TIME_FROM).equals(lower)) {
185               continue;
186             }
187           }
188           // informative => build comparison
189           newComparisons.add(new ComparisonExpression(
190             new Comparison(new TimeLiteral(lower), LTE, selector),
191             new ComparableTPGMFactory()));
192         }
193         if (upper < TemporalElement.DEFAULT_TIME_TO) {
194           // analagously as for lower bounds
195           // upper bound of to is trivial upper of from
196           if (selector.getTimeProp().equals(TimeSelector.TimeField.TX_FROM)) {
197             TimeSelector txToSel = new TimeSelector(selector.getVariable(), TX_TO);
198             if (upperBounds.getOrDefault(txToSel, TemporalElement.DEFAULT_TIME_TO).equals(upper)) {
199               continue;
200             }
201           } else if (selector.getTimeProp().equals(TimeSelector.TimeField.VAL_FROM)) {
202             TimeSelector valToSel = new TimeSelector(selector.getVariable(), VAL_TO);
203             if (upperBounds.getOrDefault(valToSel, TemporalElement.DEFAULT_TIME_TO).equals(upper)) {
204               continue;
205             }
206           }
207           // informative => build comparison
208           newComparisons.add(new ComparisonExpression(
209             new Comparison(selector, LTE, new TimeLiteral(upper)),
210             new ComparableTPGMFactory()));
211         }
212       }
213     }
214     return newComparisons;
215   }

218   /**
219    * Checks a set of comparisons for trivial comparisons like a.tx_from <= a.tx_to or
220    * l1 < l2 (l1, l2 literals).
221    * Furthermore, comparisons between selectors and literals are removed, as they are
222    * replaced by those gained from the bounds
223    *
224    * @param comparisons comparisons to filter
225    * @return @code{comparisons} without redundant comparisons
226    */
227   private List<ComparisonExpression> filterRedundantComparisons(List<ComparisonExpression> comparisons) {
228     List<ComparisonExpression> newComparisons = new ArrayList<>();
229 
230     // filter out redundant clauses
231     for (ComparisonExpression comp : comparisons) {
232       QueryComparable rhs = comp.getRhs();
233       QueryComparable lhs = comp.getLhs();
234       Comparator comparator = comp.getComparator();
235       // a.tx_from <= a.tx_to redundant
236       if (rhs instanceof TimeSelectorComparable && lhs instanceof TimeSelectorComparable &&
237         comparator.equals(Comparator.LTE)) {
238         if (!((TimeSelectorComparable) rhs).getVariable()
239           .equals(((TimeSelectorComparable) lhs).getVariable())) {
240           newComparisons.add(comp);
241         }
242       } else if (rhs instanceof TimeLiteralComparable && lhs instanceof TimeLiteralComparable) {
243         // comparisons between literals not informative.
244         // Contradictory comparisons have been filtered out before
245         // cf. @link{TrivialContradictions}
246         continue;
247       } else if ((rhs instanceof TimeLiteralComparable && lhs instanceof TimeSelectorComparable) ||
248         (rhs instanceof TimeSelectorComparable && lhs instanceof TimeLiteralComparable)) {
249         // comparisons between selectors and literals are redundant,
250         // as they are contained in the inferred bounds
251         continue;
252       } else {
253         newComparisons.add(comp);
254       }
255     }
256     return newComparisons;
257   }

259   /**
260    * Checks for contradictions between the inferred bounds and all relevant != constraints
261    *
262    * @param cNeq        != constraints as relations
263    * @param lowerBounds inferred lower bounds
264    * @param upperBounds inferred upper bounds
265    * @throws QueryContradictoryException if a contradiction is encountered
266    */
267   private void checkNeqContradictions(HashMap<ComparableExpression, HashSet<ComparableExpression>> cNeq,
268                                       HashMap<TimeSelector, Long> lowerBounds,
269                                       HashMap<TimeSelector, Long> upperBounds)
270     throws QueryContradictoryException {
271     ArrayList<ArrayList<ComparableExpression>> neq = relationToTuples(cNeq);
272     for (ArrayList<ComparableExpression> tuple : neq) {
273       if (tuple.get(0) instanceof TimeSelector) {
274         TimeSelector sel = (TimeSelector) tuple.get(0);
275         // if lower(sel)!=upper(sel), no constraint sel!=x is contradictory
276         if (!lowerBounds.get(sel).equals(upperBounds.get(sel))) {
277           return;
278         }
279         // selector1 != selector2 only contradictory if both selectors have the same fixed value
280         // (lower(selector1)=upper(selector1)=lower(selector2)=upper(selector2)
281         if (tuple.get(1) instanceof TimeSelector) {
282           TimeSelector sel2 = (TimeSelector) tuple.get(1);
283           if (lowerBounds.get(sel).equals(lowerBounds.get(sel2)) &&
284             lowerBounds.get(sel2).equals(upperBounds.get(sel2))) {
285             throw new QueryContradictoryException();
286           } // selector != literal only contradictory if lower(selector)=upper(selector)=literal
287         } else if (tuple.get(1) instanceof TimeLiteral) {
288           Long literalValue = ((TimeLiteral) tuple.get(1)).getMilliseconds();
289           if (lowerBounds.get(sel).equals(literalValue)) {
290             throw new QueryContradictoryException();
291           }
292         }
293       } else if (tuple.get(0) instanceof TimeLiteral && tuple.get(1) instanceof TimeSelector) {
294         // selector != literal only contradictory if lower(selector)=upper(selector)=literal
295         TimeSelector sel = (TimeSelector) tuple.get(1);
296         Long literalValue = ((TimeLiteral) tuple.get(0)).getMilliseconds();
297         if (lowerBounds.get(sel).equals(upperBounds.get(sel)) && lowerBounds.get(sel).equals(literalValue)) {
298           throw new QueryContradictoryException();
299         }
300       }
301     }
302   }

304   /**
305    * Updates lower and upper bounds with the lower-than relations
306    *
307    * @param rLt         lower-than relations
308    * @param lowerBounds lower bounds
309    * @param upperBounds upper bounds
310    * @return updated lower and upper bounds as a list [lower, upper]
311    * @throws QueryContradictoryException if a contradiction is encountered
312    */
313   private List<HashMap<TimeSelector, Long>> updateLt(
314     HashMap<ComparableExpression, HashSet<ComparableExpression>> rLt,
315     HashMap<TimeSelector, Long> lowerBounds, HashMap<TimeSelector, Long> upperBounds)
316     throws QueryContradictoryException {
317 
318     ArrayList<ArrayList<ComparableExpression>> lt = relationToTuples(rLt);
319     for (ArrayList<ComparableExpression> tuple : lt) {
320       TimeSelector selector = null;
321       Long literalValue = null;
322       // only comparisons between a selector and a literal are interesting here
323       // selector < literal  => upper(selector) = min(upper(selector), literal-1)
324       if (tuple.get(0) instanceof TimeSelector && tuple.get(1) instanceof TimeLiteral) {
325         selector = (TimeSelector) tuple.get(0);
326         literalValue = ((TimeLiteral) tuple.get(1)).getMilliseconds() - 1;
327         upperBounds.put(selector, Math.min(upperBounds.get(selector), literalValue));
328       } else if (tuple.get(1) instanceof TimeSelector &&
329         tuple.get(0) instanceof TimeLiteral) {
330         // selector > literal  => lower(selector) = max(lower(selector), literal+1)
331         selector = (TimeSelector) tuple.get(1);
332         literalValue = ((TimeLiteral) tuple.get(0)).getMilliseconds() + 1;
333         lowerBounds.put(selector, Math.max(lowerBounds.get(selector), literalValue));
334       } else {
335         continue;
336       }
337 
338       if (lowerBounds.get(selector) > upperBounds.get(selector)) {
339         throw new QueryContradictoryException();
340       }
341     }
342 
343     return Arrays.asList(lowerBounds, upperBounds);
344   }

347   /**
348    * Updates lower and upper bounds with the lighter-or-equal relations
349    *
350    * @param rLeq        leq relations
351    * @param lowerBounds lower bounds
352    * @param upperBounds upper bounds
353    * @return updated lower and upper bounds as a list [lower, upper]
354    * @throws QueryContradictoryException if a contradiction is encountered
355    */
356   private List<HashMap<TimeSelector, Long>> updateLeq(
357     HashMap<ComparableExpression, HashSet<ComparableExpression>> rLeq,
358     HashMap<TimeSelector, Long> lowerBounds, HashMap<TimeSelector, Long> upperBounds)
359     throws QueryContradictoryException {
360 
361     ArrayList<ArrayList<ComparableExpression>> leq = relationToTuples(rLeq);
362     for (ArrayList<ComparableExpression> tuple : leq) {
363       TimeSelector selector = null;
364       Long literalValue = null;
365       // only comparisons between a selector and a literal are interesting here
366       // selector <= literal  => upper(selector) = min(upper(selector), literal)
367       if (tuple.get(0) instanceof TimeSelector && tuple.get(1) instanceof TimeLiteral) {
368         selector = (TimeSelector) tuple.get(0);
369         literalValue = ((TimeLiteral) tuple.get(1)).getMilliseconds();
370         upperBounds.put(selector, Math.min(upperBounds.get(selector), literalValue));
371       } else if (tuple.get(1) instanceof TimeSelector &&
372         tuple.get(0) instanceof TimeLiteral) {
373         // selector >= literal  => lower(selector) = max(lower(selector), literal)
374         selector = (TimeSelector) tuple.get(1);
375         literalValue = ((TimeLiteral) tuple.get(0)).getMilliseconds();
376         lowerBounds.put(selector, Math.max(lowerBounds.get(selector), literalValue));
377       } else {
378         continue;
379       }
380 
381       if (lowerBounds.get(selector) > upperBounds.get(selector)) {
382         throw new QueryContradictoryException();
383       }
384     }
385 
386     return Arrays.asList(lowerBounds, upperBounds);
387   }

389   /**
390    * Updates lower and upper bounds with the equality relations
391    *
392    * @param rEq         equality relations
393    * @param lowerBounds lower bounds
394    * @param upperBounds upper bounds
395    * @return updated lower and upper bounds as a list [lower, upper]
396    * @throws QueryContradictoryException if a contradiction is encountered
397    */
398   private List<HashMap<TimeSelector, Long>> updateEq(
399     HashMap<ComparableExpression, HashSet<ComparableExpression>> rEq,
400     HashMap<TimeSelector, Long> lowerBounds, HashMap<TimeSelector, Long> upperBounds)
401     throws QueryContradictoryException {
402 
403     ArrayList<ArrayList<ComparableExpression>> eq = relationToTuples(rEq);
404     for (ArrayList<ComparableExpression> tuple : eq) {
405       TimeSelector selector = null;
406       TimeLiteral literal = null;
407       // only comparisons between a selector and a literal are interesting here
408       if (tuple.get(0) instanceof TimeSelector && tuple.get(1) instanceof TimeLiteral) {
409         selector = (TimeSelector) tuple.get(0);
410         literal = (TimeLiteral) tuple.get(1);
411       } else if (tuple.get(1) instanceof TimeSelector && tuple.get(0) instanceof TimeLiteral) {
412         selector = (TimeSelector) tuple.get(1);
413         literal = (TimeLiteral) tuple.get(0);
414       } else {
415         continue;
416       }
417       // for selector=literal, lowerBound(selector) = literal and upperBound(selector) = literal
418       Long literalValue = literal.getMilliseconds();
419       if (lowerBounds.get(selector) > literalValue || upperBounds.get(selector) < literalValue) {
420         throw new QueryContradictoryException();
421       } else {
422         lowerBounds.put(selector, literalValue);
423         upperBounds.put(selector, literalValue);
424       }
425     }
426     return Arrays.asList(lowerBounds, upperBounds);
427   }

429   /**
430    * Flattens the map representation of a relation to a list of list of ComparableExpressions.
431    * Each list has the length 2, representing one tuple in the relation.
432    *
433    * @param rel relation to flatten
434    * @return flattened relation
435    */
436   private ArrayList<ArrayList<ComparableExpression>> relationToTuples(HashMap<ComparableExpression,
437     HashSet<ComparableExpression>> rel) {
438     ArrayList<ArrayList<ComparableExpression>> list = new ArrayList<>();
439     for (Map.Entry<ComparableExpression, HashSet<ComparableExpression>> mapEntry : rel.entrySet()) {
440       ComparableExpression c1 = mapEntry.getKey();
441       for (ComparableExpression c2 : mapEntry.getValue()) {
442         list.add(new ArrayList<>(Arrays.asList(c1, c2)));
443       }
444     }
445     return list;
446   }

448   /**
449    * Checks the lighter-than relations for tuples (a,a) and throws an exception if such a tuple is
450    * encountered. Other contradictions in this relation are checked later
451    *
452    * @param ltRelation lighter-than relation
453    * @throws QueryContradictoryException if a tuple (a,a) is encountered
454    */
455   private void checkLtContradictions(HashMap<ComparableExpression, HashSet<ComparableExpression>> ltRelation)
456     throws QueryContradictoryException {
457     for (Map.Entry<ComparableExpression, HashSet<ComparableExpression>> tuple : ltRelation.entrySet()) {
458       ComparableExpression key = tuple.getKey();
459       for (ComparableExpression value : tuple.getValue()) {
460         if (key.equals(value)) {
461           throw new QueryContradictoryException();
462         }
463       }
464     }
465   }

467   /**
468    * Computes relations rEq, rLeq, rLt so that
469    * rEq contains (a,b) iff relevant comparisons imply a=b
470    * rLeq contains (a,b) iff relevant comparisons imply a<=b, but not a=b or a < b
471    * rLt contains (a,b) iff relevant comparisons imply a < b
472    *
473    * @param cEq  relation containing (a,b) iff relevant comparisons contain a=b
474    * @param cLeq relation containing (a,b) iff relevant comparisons contain a<=b
475    * @param cLt  relation containing (a,b) iff relevant comparisons contain a < b
476    * @return list [rEq, rLeq, rLt]
477    */
478   private List<HashMap<ComparableExpression, HashSet<ComparableExpression>>> computeClosureRelations(
479     HashMap<ComparableExpression, HashSet<ComparableExpression>> cEq,
480     HashMap<ComparableExpression, HashSet<ComparableExpression>> cLeq,
481     HashMap<ComparableExpression, HashSet<ComparableExpression>> cLt) {
482     // init relations
483     HashMap<ComparableExpression, HashSet<ComparableExpression>> rEq;
484     HashMap<ComparableExpression, HashSet<ComparableExpression>> rLeq;
485     HashMap<ComparableExpression, HashSet<ComparableExpression>> rLt;
486     // compute them
487     rEq = union(transitiveClosure(cEq), selfInverses(transitiveClosure(union(cEq, cLeq))));
488     rLeq = subtract(transitiveClosure(union(rEq, cLeq)), rEq);
489     rLt = transitiveClosure(union(
490       composition(union(rEq, rLeq), cLt),
491       composition(cLt, union(rEq, rLeq)),
492       cLt
493     ));
494     return Arrays.asList(rEq, rLeq, rLt);
495   }

497   /**
498    * Returns a subrelation containing all tuples whose inverses are also contained in the relation
499    * I.e., a tuple (a,b) is included in the return value, if (a,b) in rel and (b,a) in rel.
500    *
501    * @param rel relation
502    * @return subrelation containing all tuples whose inverses are also contained @code{rel}
503    */
504   private HashMap<ComparableExpression, HashSet<ComparableExpression>> selfInverses(
505     HashMap<ComparableExpression, HashSet<ComparableExpression>> rel) {
506 
507     HashMap<ComparableExpression, HashSet<ComparableExpression>> newRel = new HashMap<>();
508     for (Map.Entry<ComparableExpression, HashSet<ComparableExpression>> r : rel.entrySet()) {
509       ComparableExpression key = r.getKey();
510       for (ComparableExpression v : r.getValue()) {
511         if (rel.containsKey(v)) {
512           if (rel.get(v).contains(key)) {
513             newRel.putIfAbsent(key, new HashSet<>());
514             newRel.get(key).add(v);
515           }
516         }
517       }
518     }
519 
520     return newRel;
521   }

523   /**
524    * Subtracts one relation from another
525    *
526    * @param rel1 relation to subtract from
527    * @param rel2 relation to be subtracted
528    * @return @code{rel1} - @code{rel2}
529    */
530   private HashMap<ComparableExpression, HashSet<ComparableExpression>> subtract(
531     HashMap<ComparableExpression, HashSet<ComparableExpression>> rel1,
532     HashMap<ComparableExpression, HashSet<ComparableExpression>> rel2) {
533 
534     HashMap<ComparableExpression, HashSet<ComparableExpression>> newRel = new HashMap<>();
535 
536     for (Map.Entry<ComparableExpression, HashSet<ComparableExpression>> r1 : rel1.entrySet()) {
537       ComparableExpression key = r1.getKey();
538       HashSet<ComparableExpression> values1 = r1.getValue();
539       HashSet<ComparableExpression> values2 = rel2.getOrDefault(key, new HashSet<>());
540       HashSet<ComparableExpression> newValues = new HashSet<>(values1);
541       newValues.removeAll(values2);
542       newRel.put(key, new HashSet<>(newValues));
543     }
544     return newRel;
545   }

548   /**
549    * Compute the transitive closure of a relation
550    *
551    * @param rel relation
552    * @return transitive closure of relation
553    */
554   private HashMap<ComparableExpression, HashSet<ComparableExpression>> transitiveClosure(
555     HashMap<ComparableExpression, HashSet<ComparableExpression>> rel) {
556 
557     HashMap<ComparableExpression, HashSet<ComparableExpression>> newRel = rel;
558     boolean changed = true;
559     // apply transitivity until closure is complete: (a,b) in rel, (b,c) in rel => (a,c) in rel
560     while (changed) {
561       HashMap<ComparableExpression, HashSet<ComparableExpression>> oldRel =
562         (HashMap<ComparableExpression, HashSet<ComparableExpression>>) SerializationUtils.clone(newRel);
563       changed = false;
564       for (Map.Entry<ComparableExpression, HashSet<ComparableExpression>> r : oldRel.entrySet()) {
565         ComparableExpression key = r.getKey();
566         for (ComparableExpression value : r.getValue()) {
567           newRel.putIfAbsent(key, new HashSet<>());
568           changed = newRel.get(key).add(value) || changed;
569           if (oldRel.containsKey(value)) {
570             changed = newRel.get(key).addAll(oldRel.get(value)) || changed;
571           }
572         }
573       }
574     }
575     return newRel;
576   }

579    * Composition of two relations
580    *
581    * @param rel1 first relation
582    * @param rel2 second relation
583    * @return composition of both
584    */
585   private HashMap<ComparableExpression, HashSet<ComparableExpression>> composition(
586     HashMap<ComparableExpression, HashSet<ComparableExpression>> rel1,
587     HashMap<ComparableExpression, HashSet<ComparableExpression>> rel2) {
588 
589     HashMap<ComparableExpression, HashSet<ComparableExpression>> newRel = new HashMap<>();
590     // (a,b) in rel1, (b,c) in rel2 => put (a,c) into newRel
591     for (Map.Entry<ComparableExpression, HashSet<ComparableExpression>> r1 : rel1.entrySet()) {
592       ComparableExpression key = r1.getKey();
593       for (ComparableExpression oldValue : r1.getValue()) {
594         // transitive relation?
595         if (rel2.containsKey(oldValue)) {
596           HashSet<ComparableExpression> newValues = rel2.get(oldValue);
597           newRel.putIfAbsent(key, new HashSet<>());
598           newRel.get(key).addAll(newValues);
599         }
600       }
601     }
602     return newRel;
603   }

613     HashMap<ComparableExpression, HashSet<ComparableExpression>> newRel = new HashMap<>();
614     // add all relations from the array to newRel
615     for (HashMap<ComparableExpression, HashSet<ComparableExpression>> relation : relations) {
616       for (Map.Entry<ComparableExpression, HashSet<ComparableExpression>> entry : relation.entrySet()) {
617         ComparableExpression key = entry.getKey();

619         newRel.putIfAbsent(key, new HashSet<>());
620         newRel.get(key).addAll(values);
621       }
622     }
