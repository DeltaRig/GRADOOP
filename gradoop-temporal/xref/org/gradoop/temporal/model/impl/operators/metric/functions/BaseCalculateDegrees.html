16  package org.gradoop.temporal.model.impl.operators.metric.functions;
17  
18  import org.apache.flink.api.java.tuple.Tuple4;
19  import org.apache.flink.util.Collector;
20  import org.gradoop.common.model.impl.id.GradoopId;
21  
22  import java.util.Map;
23  import java.util.TreeMap;
24  
25  /**
26   * Base class for calculating the degree evolution by the given degree tree.
27   */
28  public abstract class BaseCalculateDegrees {
29  
30    /**
31     * Exception message string.
32     */
33    private static final String TEMPORAL_VIOLATION_MSG = "Last timestamp [%d] is not smaller than the " +
34      "current [%d] for vertex with id [%s]. A chronological order of timestamps is mandatory. Please " +
35      "check the temporal integrity of your graph in the given time domain. The operator " +
36      "TemporalGraph#updateEdgeValidity() can be used to update an edges validity to ensure its integrity.";
37  
38    /**
39     * Function calculates and collects the degree evolution for a given vertex and degree tree. It uses the
40     * from and to times of the vertex. If these times are not known, default timestamps can be used.
41     *
42     * @param vertexId the vertex id
43     * @param degreeTree the degree tree of that vertex
44     * @param vertexFromTime the from time of that vertex
45     * @param vertexToTime the to time of that vertex
46     * @param collector the collector that collects the resulting degree tuples
47     */
48    protected void calculateDegreeAndCollect(GradoopId vertexId, TreeMap<Long, Integer> degreeTree,
49      Long vertexFromTime, Long vertexToTime, Collector<Tuple4<GradoopId, Long, Long, Integer>> collector) {
50  
51      // we store for each timestamp the current degree
52      int degree = 0;
53  
54      // first degree 0 is from t_from(v) to the first occurrence of a start timestamp
55      Long lastTimestamp = vertexFromTime;
56  
57      for (Map.Entry<Long, Integer> entry : degreeTree.entrySet()) {
58        // check integrity
59        if (lastTimestamp > entry.getKey()) {
60          // This should not happen, seems that a temporal constraint is violated
61          throw new IllegalArgumentException(String.format(TEMPORAL_VIOLATION_MSG, lastTimestamp,
62            entry.getKey(), vertexId));
63        }
64  
65        if (lastTimestamp.equals(entry.getKey())) {
66          // First timestamp in tree is equal to the lower interval bound of the vertex
67          degree += entry.getValue();
68          continue;
69        }
70  
71        // The payload is 0, means the degree does not change and the intervals can be merged
72        if (entry.getValue() != 0) {
73          collector.collect(new Tuple4<>(vertexId, lastTimestamp, entry.getKey(), degree));
74          degree += entry.getValue();
75          // remember the last timestamp since it is the first one of the next interval
76          lastTimestamp = entry.getKey();
77        }
78      }
79  
80      // last degree is 0 from last occurence of timestamp to t_to(v)
81      if (lastTimestamp < vertexToTime) {
82        collector.collect(new Tuple4<>(vertexId, lastTimestamp, vertexToTime, degree));
83      } else if (lastTimestamp > vertexToTime) {
84        // This should not happen, seems that a temporal constraint is violated
85        throw new IllegalArgumentException(String.format(TEMPORAL_VIOLATION_MSG, lastTimestamp, vertexToTime,
86          vertexId));
87      } // else, the ending bound of the vertex interval equals the last timestamp of the edges
88    }
89  }
