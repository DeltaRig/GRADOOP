213   /**
214    * Initializes or updates all temporal estimations (dependent on the current reservoir
215    * sample)
216    */
217   private void computeTemporalEstimations() {
218     long now = new TimeLiteral("now").getMilliseconds();
219 
220     List<TemporalElement> reservoirSample = sampler.getReservoirSample();
221     int sampleSize = reservoirSample.size();
222     if (sampleSize == 0) {
223       return;
224     }
225 
226     int numBins = Math.min(DEFAULT_NUM_BINS, sampleSize);
227 
228     // sample size must be a multiple of the number of bins
229     if (sampleSize % numBins != 0) {
230       reservoirSample = reservoirSample.subList(0,
231         sampleSize - sampleSize % numBins);
232     }
233 
234     // lists collecting all values for each time property, i.e.
235     // TemporalElement.DEFAULT_TIME_FROM/MAX_VALUE, too
236     ArrayList<Long> allTxFroms = new ArrayList<Long>();
237     ArrayList<Long> allTxTos = new ArrayList<Long>();
238     ArrayList<Long> allValFroms = new ArrayList<Long>();
239     ArrayList<Long> allValTos = new ArrayList<Long>();
240 
241     // lists collecting only those values for each time property unequal to
242     // TemporalElement.DEFAULT_TIME_FROM/MAX_VALUE, too
243     ArrayList<Long> txFroms = new ArrayList<Long>();
244     ArrayList<Long> txTos = new ArrayList<Long>();
245     ArrayList<Long> valFroms = new ArrayList<Long>();
246     ArrayList<Long> valTos = new ArrayList<Long>();
247 
248     List<Long> valDurations = new ArrayList<Long>();
249     List<Long> txDurations = new ArrayList<Long>();
250 
251     // count, how often to values are TemporalElement.DEFAULT_TIME_TO
252     int txToMaxCount = 0;
253     int valToMaxCount = 0;
254     // count, how often from values are TemporalElement.DEFAULT_TIME_FROM
255     int txFromMinCount = 0;
256     int valFromMinCount = 0;
257 
258     for (TemporalElement element : reservoirSample) {
259       // collect all relevant temporal properties from the reservoir
260       long txFrom = element.getTxFrom();
261       long txTo = element.getTxTo();
262       allTxFroms.add(txFrom);
263       allTxTos.add(txTo);
264       if (txFrom > TemporalElement.DEFAULT_TIME_FROM) {
265         txFroms.add(txFrom);
266       } else {
267         txFromMinCount++;
268       }
269       if (txTo == TemporalElement.DEFAULT_TIME_TO) {
270         txToMaxCount += 1;
271         txDurations.add(now - txFrom);
272       } else {
273         txTos.add(txTo);
274         txDurations.add(txTo - txFrom);
275       }
276 
277       long valFrom = element.getValidFrom();
278       long valTo = element.getValidTo();
279       allValFroms.add(valFrom);
280       allValTos.add(valTo);
281       if (valFrom > TemporalElement.DEFAULT_TIME_FROM) {
282         valFroms.add(valFrom);
283       } else {
284         valFromMinCount++;
285       }
286       if (valTo == TemporalElement.DEFAULT_TIME_TO) {
287         valToMaxCount += 1;
288         valDurations.add(now - valFrom);
289       } else {
290         valTos.add(valTo);
291         valDurations.add(valTo - valFrom);
292       }
293 
294     }
295 
296     // statistics for temporal properties (without binning)
297     double txFromMean = mean(txFroms);
298     double txFromVariance = variance(txFroms, txFromMean);
299     double txFromMinEstimation = (double) txFromMinCount / (double) sampleSize;
300     txFromStats = new double[]{txFromMean, txFromVariance, txFromMinEstimation};
301 
302     double valFromMean = mean(valFroms);
303     double valFromVariance = variance(valFroms, valFromMean);
304     double valFromMinEstimation = (double) valFromMinCount / (double) sampleSize;
305     valFromStats = new double[]{valFromMean, valFromVariance, valFromMinEstimation};
306 
307     double txToMean = mean(txTos);
308     double txToVariance = variance(txTos, txToMean);
309     double txToMaxEstimation = (double) txToMaxCount / (double) sampleSize;
310     txToStats = new double[]{txToMean, txToVariance, txToMaxEstimation};
311 
312     double valToMean = mean(valTos);
313     double valToVariance = variance(valTos, valToMean);
314     double valToMaxEstimation = (double) valToMaxCount / (double) sampleSize;
315     valToStats = new double[]{valToMean, valToVariance, valToMaxEstimation};
316 
317     // create binnings for the temporal properties
318     estimatedTimeBins = new Binning[]{
319       new Binning(allTxFroms, numBins),
320       new Binning(allTxTos, numBins),
321       new Binning(allValFroms, numBins),
322       new Binning(allValTos, numBins)
323     };
324 
325     // determine means and variances of the durations
326     double txDurationMean = mean(txDurations);
327     double txDurationVar = variance(txDurations, txDurationMean);
328     txDurationStats = new double[]{txDurationMean, txDurationVar};
329 
330     double valDurationMean = mean(valDurations);
331     double valDurationVar = variance(valDurations, valDurationMean);
332     valDurationStats = new double[]{valDurationMean, valDurationVar};
333 
334     recomputeTemporalDataFlag = false;
335   }

337   /**
338    * Calculates the mean of a list of doubles
339    *
340    * @param ls list of doubles to calculate the mean of
341    * @return mean
342    */
343   private double mean(List<Long> ls) {
344     double mean = 0.;
345     for (Long l : ls) {
346       mean += (double) l / ls.size();
347     }
348     return mean;
349   }
350 
351   /**
352    * Calculates the variance of a list of doubles, given its mean
353    *
354    * @param ls   list to calculate variance of
355    * @param mean mean of the list
356    * @return variance
357    */
358   private double variance(List<Long> ls, double mean) {
359     double var = 0.;
360     for (Long l : ls) {
361       var += (l - mean) * (l - mean) * (1. / ls.size());
362     }
363     return var;
364   }

367   /**
368    * Initializes or updates all property related estimations (dependent on the current reservoir
369    * sample)
370    */
371   private void computePropertyEstimations() {
372     List<TemporalElement> reservoirSample = sampler.getReservoirSample();
373     int sampleSize = reservoirSample.size();
374     if (sampleSize == 0) {
375       return;
376     }
377 
378     // classify all property values, if not already done
379     if (categoricalProperties == null || numericalProperties == null) {
380       detectPropertyTypes(reservoirSample);
381     }
382     // these maps collect all property values in the sample
383     HashMap<String, List<PropertyValue>> categoricalData = new HashMap<>();
384     HashMap<String, List<PropertyValue>> numericalData = new HashMap<>();
385 
386     // add element data to the corresponding map
387     for (TemporalElement element : reservoirSample) {
388       if (element.getPropertyKeys() == null) {
389         continue;
390       }
391       for (String key : element.getPropertyKeys()) {
392         PropertyValue value = element.getPropertyValue(key);
393         HashMap<String, List<PropertyValue>> data;
394         // no outlier, i.e. NULL, NaN etc.?
395         if (numericalProperties.contains(key) && value.isNumber()) {
396           data = numericalData;
397         } else if (categoricalProperties.contains(key)) {
398           data = categoricalData;
399         } else {
400           continue;
401         }
402 
403         data.putIfAbsent(key, new ArrayList<>());
404         data.get(key).add(value);
405       }
406     }
407     computeCategoricalEstimations(categoricalData, sampleSize);
408     computeNumericalEstimations(numericalData, sampleSize);
409     recomputePropertyDataFlag = false;
410   }

412   /**
413    * Tries to automatically detect which properties are numerical and categorical.
414    * Sets {@code numericalProperties} and {@code categoricalProperties} accordingly.
415    *
416    * @param reservoirSample the list of elements to use for the detection
417    */
418   private void detectPropertyTypes(List<TemporalElement> reservoirSample) {
419     numericalProperties = new HashSet<>();
420     categoricalProperties = new HashSet<>();
421     // maps count which property is considered numerical/categorical how often
422     HashMap<String, Integer> numerical = new HashMap<>();
423     HashMap<String, Integer> categorical = new HashMap<>();
424     for (TemporalElement element : reservoirSample) {
425       if (element.getPropertyKeys() == null) {
426         continue;
427       }
428       for (String key : element.getPropertyKeys()) {
429         PropertyValue value = element.getPropertyValue(key);
430         if (value.isNumber()) {
431           numerical.put(key, numerical.getOrDefault(key, 0) + 1);
432         } else {
433           categorical.put(key, categorical.getOrDefault(key, 0) + 1);
434         }
435       }
436     }
437     // check which property is of what type using a threshold
438     double thresh = 0.9;
439     HashSet<String> allProperties = new HashSet<>();
440     allProperties.addAll(numerical.keySet());
441     allProperties.addAll(categorical.keySet());
442     for (String property : allProperties) {
443       int countNumerical = numerical.getOrDefault(property, 0);
444       int countCategorical = categorical.getOrDefault(property, 0);
445       int countOverall = countCategorical + countNumerical;
446       // which type?
447       if ((double) countNumerical / (double) countOverall >= thresh) {
448         numericalProperties.add(property);
449       } else if ((double) countCategorical / (double) countOverall >= thresh) {
450         categoricalProperties.add(property);
451       }
452       // else, the property is too ambiguous and simply ignored
453     }
454   }

474   /**
475    * Initializes or updates the selectivity estimation for categorical properties
476    *
477    * @param data       map from categorical property names to a list of their respective
478    *                   values in the sample
479    * @param sampleSize size of the sample
480    */
481   private void computeCategoricalEstimations(Map<String, List<PropertyValue>> data, int sampleSize) {
482     categoricalSelectivityEstimation = new HashMap<>();
483 
484     for (Map.Entry<String, List<PropertyValue>> entry : data.entrySet()) {
485       Map<PropertyValue, Double> selectivityMap = new HashMap<>();
486 
487       for (PropertyValue value : entry.getValue()) {
488         selectivityMap.put(value,
489           selectivityMap.getOrDefault(value, 0.) + 1. / sampleSize);
490       }
491 
492       categoricalSelectivityEstimation.put(entry.getKey(), selectivityMap);
493     }
494 
495   }

497   /**
498    * Initializes or updates the estimations for numerical properties
499    * All of them are considered to be normally distributed
500    *
501    * @param data       map from numerical property names to a list of their respective
502    *                   values in the sample
503    * @param sampleSize size of the sample
504    */
505   private void computeNumericalEstimations(Map<String, List<PropertyValue>> data, int sampleSize) {
506     numericalOccurrenceEstimation = new HashMap<>();
507     numericalPropertyStatsEstimation = new HashMap<>();
508     for (Map.Entry<String, List<PropertyValue>> entry : data.entrySet()) {
509       List<PropertyValue> values = data.get(entry.getKey());
510 
511       // estimate how often the property occurs
512       numericalOccurrenceEstimation.put(entry.getKey(),
513         (double) values.size() / sampleSize);
514 
515       // cast all values to doubles
516       List<Double> doubleValues = values.stream()
517         .map(this::propertyValueToDouble)
518         .collect(Collectors.toList());
519 
520       // compute mean and variance for the property
521       Double sum = doubleValues.stream().reduce(0., Double::sum);
522       Double mean = sum / values.size();
523       Double variance = doubleValues.stream()
524         .reduce(0., (i, j) -> i + ((j - mean) * (j - mean) * (1. / values.size())));
525 
526       numericalPropertyStatsEstimation.put(entry.getKey(), new Double[]{mean, variance});
527     }
528   }
