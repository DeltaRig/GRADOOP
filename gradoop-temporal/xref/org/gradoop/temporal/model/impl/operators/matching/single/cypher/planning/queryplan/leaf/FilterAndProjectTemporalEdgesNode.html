28  import java.util.ArrayList;
29  import java.util.List;

32  /**
33   * Leaf node that wraps a {@link FilterAndProjectTemporalEdges} operator.
34   */
35  public class FilterAndProjectTemporalEdgesNode extends LeafNode
36    implements FilterNode, ProjectionNode {
37    /**
38     * Input data set
39     */
40    private final DataSet<TemporalEdge> edges;
41    /**
42     * Query variable of the source vertex
43     */
44    private final String sourceVariable;
45    /**
46     * Query variable of the edge
47     */
48    private final String edgeVariable;
49    /**
50     * Query variable of the target vertex
51     */
52    private final String targetVariable;
53    /**
54     * Filter predicate that is applied on the input data set
55     */
56    private final CNF filterPredicate;
57    /**
58     * Property keys used for projection
59     */
60    private final List<String> projectionKeys;
61    /**
62     * Indicates if the edges is actually a path
63     */
64    private final boolean isPath;
65  
66    /**
67     * Creates a new node.
68     *
69     * @param edges           input edges
70     * @param sourceVariable  query variable of the source vertex
71     * @param edgeVariable    query variable of the edge
72     * @param targetVariable  query variable of the target vertex
73     * @param filterPredicate filter predicate to be applied on edges
74     * @param projectionKeys  property keys whose associated values are projected to the output
75     * @param isPath          indicates if the edges is actually a path
76     */
77    public FilterAndProjectTemporalEdgesNode(DataSet<TemporalEdge> edges,
78                                             String sourceVariable, String edgeVariable, String targetVariable,
79                                             CNF filterPredicate, Set<String> projectionKeys,
80                                             boolean isPath) {
81      this.edges = edges;
82      this.sourceVariable = sourceVariable;
83      this.edgeVariable = edgeVariable;
84      this.targetVariable = targetVariable;
85      this.filterPredicate = filterPredicate;
86      this.projectionKeys = new ArrayList<>(projectionKeys);
87      this.isPath = isPath;
88    }
89  
90    @Override
91    public DataSet<Embedding> execute() {
92      FilterAndProjectTemporalEdges op = new FilterAndProjectTemporalEdges(
93        edges,
94        filterPredicate,
95        projectionKeys,
96        isLoop()
97      );
98      op.setName(toString());
99      return op.evaluate();
100   }
101 
102   /**
103    * Returns a copy of the filter predicate attached to this node.
104    *
105    * @return filter predicate
106    */
107   public CNF getFilterPredicate() {
108     return new CNF(filterPredicate);
109   }
110 
111   /**
112    * Returns a copy of the projection keys attached to this node.
113    *
114    * @return projection keys
115    */
116   public List<String> getProjectionKeys() {
117     return new ArrayList<>(projectionKeys);
118   }
119 
120   /**
121    * Returns the name of the edge variable
122    *
123    * @return name of the edge variable
124    */
125   public String getEdgeVariable() {
126     return edgeVariable;
127   }
128 
129   /**
130    * checks whether edge is a self-loop
131    *
132    * @return true iff edge is a self-loop
133    */
134   public boolean isLoop() {
135     return sourceVariable.equals(targetVariable) && !isPath;
136   }
137 
138   @Override
139   protected EmbeddingMetaData computeEmbeddingMetaData() {
140     EmbeddingMetaData embeddingMetaData = new EmbeddingMetaData();
141     embeddingMetaData.setEntryColumn(sourceVariable, EmbeddingMetaData.EntryType.VERTEX, 0);
142     embeddingMetaData.setEntryColumn(edgeVariable, EmbeddingMetaData.EntryType.EDGE, 1);
143     if (!isLoop()) {
144       embeddingMetaData.setEntryColumn(targetVariable, EmbeddingMetaData.EntryType.VERTEX, 2);
145     }
146 
147     embeddingMetaData = setPropertyColumns(embeddingMetaData, edgeVariable, projectionKeys);
148     return embeddingMetaData;
149   }
150 
151   @Override
152   public String toString() {
153     return String.format("FilterAndProjectTemporalEdgesNode{" +
154         "sourceVariable='%s', " +
155         "edgeVariable='%s', " +
156         "targetVariable='%s', " +
157         "filterPredicate=%s, " +
158         "projectionKeys=%s}",
159       sourceVariable, edgeVariable, targetVariable, filterPredicate, projectionKeys);
160   }
