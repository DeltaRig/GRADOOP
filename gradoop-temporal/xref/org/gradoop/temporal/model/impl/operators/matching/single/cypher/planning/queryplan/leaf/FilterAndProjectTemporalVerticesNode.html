28  import java.util.ArrayList;
29  import java.util.List;

32  /**
33   * Leaf node that wraps a {@link FilterAndProjectTemporalVertices} operator.
34   */
35  public class FilterAndProjectTemporalVerticesNode extends LeafNode
36    implements FilterNode, ProjectionNode {
37  
38    /**
39     * Input data set
40     */
41    private final DataSet<TemporalVertex> vertices;
42    /**
43     * Query variable of the vertex
44     */
45    private final String vertexVariable;
46    /**
47     * Filter predicate that is applied on the input data set
48     */
49    private final CNF filterPredicate;
50    /**
51     * Property keys used for projection
52     */
53    private final List<String> projectionKeys;
54  
55    /**
56     * Creates a new node.
57     *
58     * @param vertices        input vertices
59     * @param vertexVariable  query variable of the vertex
60     * @param filterPredicate filter predicate to be applied on edges
61     * @param projectionKeys  property keys whose associated values are projected to the output
62     */
63    public FilterAndProjectTemporalVerticesNode(DataSet<TemporalVertex> vertices, String vertexVariable,
64                                                CNF filterPredicate, Set<String> projectionKeys) {
65      this.vertices = vertices;
66      this.vertexVariable = vertexVariable;
67      this.filterPredicate = filterPredicate;
68      this.projectionKeys = new ArrayList<>(projectionKeys);
69    }
70  
71    @Override
72    public DataSet<Embedding> execute() {
73      FilterAndProjectTemporalVertices op =
74        new FilterAndProjectTemporalVertices(vertices, filterPredicate, projectionKeys);
75      op.setName(toString());
76      return op.evaluate();
77    }
78  
79    /**
80     * Returns a copy of the filter predicate attached to this node.
81     *
82     * @return filter predicate
83     */
84    public CNF getFilterPredicate() {
85      return new CNF(filterPredicate);
86    }
87  
88    /**
89     * Returns a copy of the projection keys attached to this node.
90     *
91     * @return projection keys
92     */
93    public List<String> getProjectionKeys() {
94      return new ArrayList<>(projectionKeys);
95    }
96  
97    @Override
98    protected EmbeddingMetaData computeEmbeddingMetaData() {
99      EmbeddingMetaData embeddingMetaData = new EmbeddingMetaData();
100     embeddingMetaData.setEntryColumn(vertexVariable, EmbeddingMetaData.EntryType.VERTEX, 0);
101     embeddingMetaData = setPropertyColumns(embeddingMetaData, vertexVariable, projectionKeys);
102 
103     return embeddingMetaData;
104   }
105 
106   @Override
107   public String toString() {
108     return String.format("FilterAndProjectVerticesNode{" +
109         "vertexVariable=%s, " +
110         "filterPredicate=%s, " +
111         "projectionKeys=%s}",
112       vertexVariable, filterPredicate, projectionKeys);
113   }
