36  /**
37   * Looks for trivial contradictions like {@code a.tx_from > a.tx_to} or (b.prop!=b.prop).
38   * Every contradictory comparison is removed from its disjunctive clause.
39   * If every comparison in a clause is contradictory, the whole CNF is a contradiction.
40   * In this case, the transformation throws a {@link QueryContradictoryException}
41   * !!! This class assumes input CNFs to be normalized, i.e. not to contain < or <= !!!
42   */
43  public class TrivialContradictions implements QueryTransformation {
44  
45    @Override
46    public CNF transformCNF(CNF cnf) throws QueryContradictoryException {
47      ArrayList<CNFElement> newClauses = new ArrayList<>();
48      // check every clause for contradictions
49      for (CNFElement clause : cnf.getPredicates()) {
50        CNFElement newClause = transformDisjunction(clause);
51        newClauses.add(newClause);
52      }
53      return new CNF(newClauses);
54    }
55  
56    /**
57     * Checks a disjunctive clause for trivial contradictions and removes them.
58     * If all comparisons are contradictory, an exception is thrown.
59     *
60     * @param clause clause to check
61     * @return the clause iff it does not contain a trivial contradiction
62     * @throws QueryContradictoryException iff the clause contains a trivial contradiction
63     */
64    private CNFElement transformDisjunction(CNFElement clause) throws QueryContradictoryException {
65      List<ComparisonExpression> oldComparisons = clause.getPredicates();
66      ArrayList<ComparisonExpression> newComparisons = new ArrayList<>();
67      for (ComparisonExpression comparison : oldComparisons) {
68        if (!isContradictory(comparison)) {
69          // contradictory comparisons are omitted in the resulting clause
70          newComparisons.add(comparison);
71        }
72      }
73      // all comparisons contradictory?
74      if (newComparisons.isEmpty()) {
75        throw new QueryContradictoryException();
76      } else {
77        return new CNFElement(newComparisons);
78      }
79    }
80  
81  
82    /**
83     * Checks whether a comparison is a trivial contradiction (x < x, x!=x, !(a.tx_from <= a.tx_to,
84     * contradictory comparison between two time literals)
85     *
86     * @param comp comparison to check
87     * @return true iff comparison is a trivial contradiction
88     */
89    private boolean isContradictory(ComparisonExpression comp) {
90      // unwrap the comparison
91      ComparableExpression lhs = comp.getLhs().getWrappedComparable();
92      Comparator comparator = comp.getComparator();
93      ComparableExpression rhs = comp.getRhs().getWrappedComparable();
94  
95      // x<x, x!=x
96      if (lhs.equals(rhs)) {
97        if (!(comparator.equals(Comparator.EQ) || comparator.equals(LTE))) {
98          return true;
99        }
100     }
101     // a.tx_from > a.tx_to
102     if (lhs instanceof TimeSelector && rhs instanceof TimeSelector &&
103       lhs.getVariable().equals(rhs.getVariable())) {
104       if (((TimeSelector) lhs).getTimeProp().equals(TimeSelector.TimeField.TX_FROM) &&
105         ((TimeSelector) rhs).getTimeProp().equals(TimeSelector.TimeField.TX_TO) ||
106         ((TimeSelector) lhs).getTimeProp().equals(TimeSelector.TimeField.VAL_FROM) &&
107           ((TimeSelector) rhs).getTimeProp().equals(TimeSelector.TimeField.VAL_TO)) {
108         return comparator.equals(GT);
109       }
110     } else if ((lhs instanceof TimeLiteral && rhs instanceof TimeLiteral) ||
111       (lhs instanceof Literal && rhs instanceof Literal)) {
112       // comparison of two (time) literals is tautological iff the comparison holds
113       // true iff the comparison holds
114       return !comp.evaluate(new TemporalVertex());
115     }
116     return false;
117   }
