38  /**
39   * A TPGM operator calculating the evolution of vertex degrees for all vertices of the graph. The result is a
40   * dataset of tuples {@code {id,t-from,t-to,degree}} where {@code id} is the vertex id, {@code t-from} is the
41   * lower interval bound, {@code t-to} the upper interval bound and {@code degree} the degree value.
42   * <p>
43   * The type of the degree (IN, OUT, BOTH) as well as the time dimension to consider, can be chosen by the
44   * arguments.
45   * <p>
46   * If the vertex degree evolution of a single vertex is needed, use the edge induced subgraph operator
47   * {@link org.gradoop.flink.model.api.epgm.BaseGraphOperators#edgeInducedSubgraph(FilterFunction)} with filter
48   * {@link org.gradoop.flink.model.impl.functions.epgm.BySourceId} and/or
49   * {@link org.gradoop.flink.model.impl.functions.epgm.ByTargetId} before you apply this operator.
50   */
51  public class TemporalVertexDegree
52    implements UnaryBaseGraphToValueOperator<TemporalGraph, DataSet<Tuple4<GradoopId, Long, Long, Integer>>> {
53  
54    /**
55     * The time dimension that will be considered.
56     */
57    private final TimeDimension dimension;
58  
59    /**
60     * The degree type (IN, OUT, BOTH);
61     */
62    private final VertexDegree degreeType;
63  
64    /**
65     * A flag to decide whether to include the vertex interval as lower and upper bounds for the first and
66     * last degree period. If set to false, the first degree period starts at
67     * {@link org.gradoop.temporal.model.impl.pojo.TemporalElement#DEFAULT_TIME_FROM} and the last degree period
68     * ends at {@link org.gradoop.temporal.model.impl.pojo.TemporalElement#DEFAULT_TIME_TO}.
69     */
70    private boolean includeVertexTime = false;
71  
72    /**
73     * Creates an instance of this temporal vertex degree operator.
74     *
75     * @param degreeType the degree type to consider
76     * @param dimension the time dimension to consider
77     */
78    public TemporalVertexDegree(VertexDegree degreeType, TimeDimension dimension) {
79      this.degreeType = Objects.requireNonNull(degreeType);
80      this.dimension = Objects.requireNonNull(dimension);
81    }
82  
83    /**
84     * Configure the operator to consider or not the vertex time by calculating the degree evolution.
85     *
86     * @param includeVertexTime true, iff the vertex time is used as lower and upper bounds for the first and
87     *                          last degree period. Note that this requires a join.
88     */
89    public void setIncludeVertexTime(boolean includeVertexTime) {
90      this.includeVertexTime = includeVertexTime;
91    }
92  
93    @Override
94    public DataSet<Tuple4<GradoopId, Long, Long, Integer>> execute(TemporalGraph graph) {
95  
96      DataSet<Tuple2<GradoopId, TreeMap<Long, Integer>>> edgesWithTrees = graph.getEdges()
97        // 1) Extract vertex id(s) and corresponding time intervals
98        .flatMap(new FlatMapVertexIdEdgeInterval(dimension, degreeType))
99        // 2) Group them by the vertex id
100       .groupBy(0)
101       // 3) For each vertex id, build a degree tree data structure
102       .reduceGroup(new BuildTemporalDegreeTree());
103 
104     if (includeVertexTime) {
105       // The time information of the vertex must be included, therefore we need to join.
106       DataSet<Tuple3<GradoopId, Long, Long>> vertexIdInterval = graph.getVertices()
107         .map(new ExtractIdIntervalMap(dimension));
108 
109       return edgesWithTrees
110         // 4) Join the vertices to get each vertex interval
111         .join(vertexIdInterval)
112         .where(0).equalTo(0)
113         .with(new ExtendVertexDataWithInterval())
114         // 5) Since join leads to possible divided groups, we need to group again
115         .groupBy(0)
116         .reduce(new BuildTemporalDegreeTree())
117         // 6) For each vertex, calculate the degree evolution and output a tuple {v_id, t_from, t_to, degree}
118         .flatMap(new CalculateDegreesFlatMap());
119     } else {
120       return edgesWithTrees
121         // 4/7) For each vertex, calculate the degree evolution and output a tuple
122         // {v_id, t_from, t_to, degree}
123         .flatMap(new CalculateDegreesDefaultTimesFlatMap());
124     }
125   }
126 }
