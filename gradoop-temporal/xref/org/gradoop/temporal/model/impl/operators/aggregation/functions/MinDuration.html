28  /**
29   * Calculates the minimum duration of a {@link TimeDimension} of temporal elements.
30   *
31   * Time intervals with either the start or end time set to the respective default value are evaluated as zero.
32   */
33  public class MinDuration extends AbstractDurationAggregateFunction implements Min, TemporalAggregateFunction {
34  
35    /**
36     * Creates a new instance of this aggregate function.
37     *
38     * @param aggregatePropertyKey the property key of the new property where the aggregated value is stored
39     * @param dimension the time dimension to consider
40     * @param unit the temporal unit into which the result is converted. The supported units are specified in
41     *             {@link AbstractDurationAggregateFunction#SUPPORTED_UNITS}.
42     */
43    public MinDuration(String aggregatePropertyKey, TimeDimension dimension, TemporalUnit unit) {
44      super(aggregatePropertyKey, dimension, unit);
45    }
46  
47    /**
48     * Creates a new instance of this aggregate function.
49     *
50     * @param aggregatePropertyKey the property key of the new property where the aggregated value is stored
51     * @param dimension the time dimension to consider
52     */
53    public MinDuration(String aggregatePropertyKey, TimeDimension dimension) {
54      super(aggregatePropertyKey, dimension, AbstractDurationAggregateFunction.DEFAULT_UNIT);
55    }
56  
57    /**
58     * Creates a new instance of this aggregate function. The the property key of the new property, where the
59     * aggregated value is stored, will be defined as "minDuration_" + {dimension} + "_" + {unit}. Use
60     * constructor {@link MinDuration#MinDuration(String, TimeDimension, TemporalUnit)} to specify a
61     * user-defined key.
62     *
63     * @param dimension the time dimension to consider
64     * @param unit the temporal unit into which the result is converted. The supported units are specified in
65     *             {@link AbstractDurationAggregateFunction#SUPPORTED_UNITS}.
66     */
67    public MinDuration(TimeDimension dimension, TemporalUnit unit) {
68      this("minDuration_" + dimension + "_" + unit, dimension, unit);
69    }
70  
71    /**
72     * Calculates the duration of a given element depending on the given {@link TimeDimension}.
73     * Returns {@link Long#MAX_VALUE} if either the start or end time of the duration are default values.
74     *
75     * @param element the temporal element
76     * @return the duration of the time interval
77     */
78    @Override
79    public PropertyValue getIncrement(TemporalElement element) {
80      PropertyValue duration = getDuration(element);
81      if (duration.getLong() == -1L) {
82        return PropertyValue.create(TemporalElement.DEFAULT_TIME_TO);
83      }
84      return duration;
85    }
86  
87    /**
88     * Method to check whether all aggregated durations had been default values or need to be transformed to
89     * another unit.
90     *
91     * @param result the result of the MinDuration aggregation as {@link Long} in milliseconds
92     * @return By default (no time unit or the default unit [millis] is specified), the result of the
93     *         MinDuration Aggregation as {@link Long} in milliseconds is returned. If a different time unit is
94     *         given in {@link AbstractDurationAggregateFunction#timeUnit}, a {@link Double} of
95     *         the desired unit is returned. If the minimum duration is {@link TemporalElement#DEFAULT_TIME_TO},
96     *         the value {@link PropertyValue#NULL_VALUE}is returned.
97     * @throws UnsupportedTypeException if the type of the given property value is different from {@link Long}.
98     */
99    @Override
100   public PropertyValue postAggregate(PropertyValue result) throws UnsupportedTypeException {
101     // First check if the aggregated result is of type long.
102     if (!result.isLong()) {
103       throw new UnsupportedTypeException("The result type of the min duration aggregation must be [" +
104         Type.LONG + "], but is [" + result.getType() + "].");
105     }
106     // If the result is a default value, set it as null.
107     if (result.getLong() == TemporalElement.DEFAULT_TIME_TO) {
108       return PropertyValue.NULL_VALUE;
109     }
110     // If a time unit is specified, convert it.
111     if (timeUnit != AbstractDurationAggregateFunction.DEFAULT_UNIT) {
112       result.setDouble((double) result.getLong() / timeUnit.getDuration().toMillis());
113     }
114     return result;
115   }
