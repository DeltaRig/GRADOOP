32  /**
33   * Postprocessing pipeline for CNFs.
34   * Contains a list of {@link QueryTransformation} objects, that are applied to a CNF
35   */
36  public class CNFPostProcessing {
37  
38    /**
39     * List of transformations. They are applied in the order of the list
40     */
41    private final List<QueryTransformation> transformations;
42  
43    /**
44     * Creates a new custom postprocessing pipeline
45     *
46     * @param transformations list of transformations to apply
47     */
48    public CNFPostProcessing(List<QueryTransformation> transformations) {
49      this.transformations = transformations;
50    }
51  
52    /**
53     * Creates a default postprocessing pipeline:
54     * 1. Normalization
55     * 2. "Translate" suitable Min/Max expressions to simple predicates
56     * 3. subsumption on a purely syntactical basis
57     * 4. checking for trivial tautologies, maybe simplifying the query
58     * 5. checking for trivial contradictions, maybe simplifying the query
59     * 6. checking for less trivial contradictions in temporal comparisons
60     * 7. inferring implicit temporal information
61     * 8. subsumption with temporal information
62     * 9. deduplication of clauses
63     */
64    public CNFPostProcessing() {
65      this(Arrays.asList(
66        new Normalization(),
67        new MinMaxUnfolding(),
68        new SyntacticSubsumption(),
69        new TrivialTautologies(),
70        new TrivialContradictions(),
71        new AddTrivialConstraints(),
72        //new CheckForCircles(),
73        new BoundsInference(),
74        new TemporalSubsumption(),
75        new TrivialTautologies()
76      ));
77    }
78  
79    /**
80     * Executes the postprocessing pipeline on a CNF
81     *
82     * @param cnf CNF to postprocess
83     * @return postprocessed CNF
84     * @throws QueryContradictoryException if the CNF is found to be contradictory
85     *                                     during the postprocessing
86     */
87    public CNF postprocess(CNF cnf) throws QueryContradictoryException {
88      for (QueryTransformation transformation : transformations) {
89        cnf = transformation.transformCNF(cnf);
90      }
91      return cnf;
92    }
93  
94    public List<QueryTransformation> getTransformations() {
95      return transformations;
96    }
97  }
