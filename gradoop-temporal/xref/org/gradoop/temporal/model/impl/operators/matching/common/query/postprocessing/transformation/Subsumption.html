31  /**
32   * Abstract base class for subsumptions of clauses.
33   * A simplifying assumption is made: smaller clauses are not subsumed by larger ones.
34   * It is not guaranteed that all subsumptions or the optimal sequence of subsumptions are applied
35   */
36  public abstract class Subsumption implements QueryTransformation {
37  
38    @Override
39    public CNF transformCNF(CNF cnf) {
40      // separate singleton clauses from disjunctive clauses
41      ArrayList<CNFElement> necessaryClauses = cnf.getPredicates().stream()
42        .filter(clause -> clause.size() == 1).collect(Collectors.toCollection(ArrayList::new));
43      List<CNFElement> disjunctiveClauses = cnf.getPredicates().stream()
44        .filter(clause -> clause.size() > 1)
45        .collect(Collectors.toList());
46      // apply subsumption within disjunctive clauses while keeping the singletons and ...
47      cnf = new CNF(necessaryClauses).and(
48        subsumeDisjunctiveClauses(disjunctiveClauses));
49      List<CNFElement> oldClauses = cnf.getPredicates();
50      // ... apply subsumption to the result
51      return new CNF(subsumeClauses(oldClauses));
52    }
53  
54    /**
55     * Sort clauses: clauses that subsume most other clauses are sorted at the beginning of
56     * the list
57     *
58     * @param clauses clauses to sort
59     * @return sorted clauses
60     */
61    protected ArrayList<CNFElement> sortClauses(List<CNFElement> clauses) {
62      // count for every clause how much clauses it subsumes
63      HashMap<CNFElement, Integer> countSubsumes = new HashMap<>();
64      for (int i = 0; i < clauses.size(); i++) {
65        CNFElement clause1 = clauses.get(i);
66        countSubsumes.putIfAbsent(clause1, 0);
67        for (int j = i; j < clauses.size(); j++) {
68          CNFElement clause2 = clauses.get(j);
69          if (subsumes(clause1, clause2)) {
70            countSubsumes.put(clause1, countSubsumes.get(clause1) + 1);
71          }
72          if (subsumes(clause2, clause1)) {
73            countSubsumes.putIfAbsent(clause2, 0);
74            countSubsumes.put(clause2, countSubsumes.get(clause2) + 1);
75          }
76        }
77      }
78      // sort by these counts. If two clauses subsume the same number of clauses,
79      // the shorter clause is sorted first
80      ArrayList<CNFElement> clausesSorted = new ArrayList<>(clauses);
81      clausesSorted.sort(new Comparator<CNFElement>() {
82        @Override
83        public int compare(CNFElement t1, CNFElement t2) {
84          int subsumedDifference = countSubsumes.get(t2) -
85            countSubsumes.get(t1);
86          return subsumedDifference != 0 ? subsumedDifference :
87            t1.size() - t2.size();
88        }
89      });
90      return clausesSorted;
91    }

93    /**
94     * Performs the subsumption of a set of clauses
95     *
96     * @param clauses clauses for subsumption
97     * @return sublist of the clauses after subsumption
98     */
99    protected ArrayList<CNFElement> subsumeClauses(List<CNFElement> clauses) {
100     ArrayList<CNFElement> oldClauses = sortClauses(clauses);
101     // points to clauses that are already subsumed (their index in the clauses list)
102     HashSet<Integer> subsumed = new HashSet<>();
103 
104     // check for every clause if it subsumes other clauses
105     for (int i = 0; i < oldClauses.size(); i++) {
106       // clause c_i is already subsumed, i.e. actually not there anymore
107       if (subsumed.contains(i)) {
108         continue;
109       }
110       // clause c_i is not subsumed yet, i.e. may subsume other clauses
111       CNFElement c1 = oldClauses.get(i);
112       for (int j = i + 1; j < oldClauses.size(); j++) {
113         // clause c_j is already subsumed, i.e. actually not there anymore
114         if (subsumed.contains(j)) {
115           continue;
116         }
117         // clause c_j is not subsumed yet, i.e. may be subsumed
118         CNFElement c2 = oldClauses.get(j);
119         if (subsumes(c1, c2)) {
120           subsumed.add(j);
121         }
122       }
123     }
124     // only clauses that are not subsumed by other clauses are contained in the new CNF
125     ArrayList<CNFElement> newClauses = new ArrayList<>();
126     for (int i = 0; i < oldClauses.size(); i++) {
127       if (subsumed.contains(i)) {
128         continue;
129       }
130       newClauses.add(oldClauses.get(i));
131     }
132     return newClauses;
133   }

135   /**
136    * Applies {@link this::subsumeDisjunctiveClause} to all disjunctive clauses
137    * (CNF clauses of size > 1)
138    *
139    * @param disjClauses clauses
140    * @return CNF conjoining the subsumed clauses
141    */
142   protected CNF subsumeDisjunctiveClauses(List<CNFElement> disjClauses) {
143     List<CNFElement> newClauses = disjClauses.stream()
144       .map(this::subsumeDisjunctiveClause)
145       .collect(Collectors.toList());
146 
147     return new CNF(newClauses);
148   }
149 
150   /**
151    * Applies subsumption to a disjunctive clause (might simplify the clause)
152    *
153    * @param clause clause to apply subsumption to.
154    * @return subsumed clause
155    */
156   protected CNFElement subsumeDisjunctiveClause(CNFElement clause) {
157     return new CNFElement(subsumeClauses(
158       clause.getPredicates().stream()
159         .map(comparison -> new CNFElement(Collections.singletonList(comparison)))
160         .collect(Collectors.toList()))
161       .stream()
162       .map(c -> c.getPredicates().get(0))
163       .collect(Collectors.toList()));
164   }

166   /**
167    * Checks for two disjunctive clauses c1 and c2 whether c1 subsumes c2.
168    * Here, c1 subsumes c2 iff c1's comparisons form a subset of c2's comparisons
169    *
170    * @param c1 first clause
171    * @param c2 second clause
172    * @return true iff c1 subsumes c2
173    */
174   protected boolean subsumes(CNFElement c1, CNFElement c2) {
175     for (ComparisonExpression comp1 : c1) {
176       boolean foundMatch = false;
177       for (ComparisonExpression comp2 : c2) {
178         if (subsumes(comp1, comp2)) {
179           foundMatch = true;
180           break;
181         }
182       }
183       if (!foundMatch) {
184         return false;
185       }
186     }
187     return true;
188   }

190   /**
191    * Checks whether a ComparisonExpression c1 subsumes c2
192    *
193    * @param c1 potentially subsuming comparison
194    * @param c2 potentially subsumed comparison
195    * @return true iff c1 subsumes c2
196    */
197   public abstract boolean subsumes(ComparisonExpression c1,
198                                    ComparisonExpression c2);
