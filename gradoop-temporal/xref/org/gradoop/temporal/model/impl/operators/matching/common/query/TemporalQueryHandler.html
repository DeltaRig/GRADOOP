40  /**
41   * Wraps a {@link GDLHandler} and adds functionality needed for query
42   * processing during graph pattern matching.
43   * Extension of the flink QueryHandler for temporal queries
44   */
45  public class TemporalQueryHandler extends QueryHandler {
46    /**
47     * Time Literal representing the system time at the start of query processing
48     */
49    private final TimeLiteral now;
50    /**
51     * Transformations to apply to the query cnf
52     */
53    private final CNFPostProcessing cnfPostProcessing;
54  
55    /**
56     * The query CNF
57     */
58    private CNF cnf;
59  
60    /**
61     * Creates a new query handler that postprocesses the CNF with a default
62     * {@link CNFPostProcessing}
63     *
64     * @param gdlString GDL query string
65     * @throws QueryContradictoryException if a contradiction in the query is encountered
66     */
67    public TemporalQueryHandler(String gdlString) throws QueryContradictoryException {
68      this(gdlString, new CNFPostProcessing());
69    }
70  
71    /**
72     * Creates a new query handler.
73     *
74     * @param gdlString      GDL query string
75     * @param postProcessing transformations to apply to the CNF
76     * @throws QueryContradictoryException if a contradiction in the query is encountered
77     */
78    public TemporalQueryHandler(String gdlString, CNFPostProcessing postProcessing)
79      throws QueryContradictoryException {
80      super(gdlString);
81      this.cnfPostProcessing = postProcessing;
82      now = new TimeLiteral("now");
83      initCNF();
84    }
85  
86    /**
87     * Returns all available predicates in Conjunctive Normal Form {@link CNF}. If there are no
88     * predicates defined in the query, a CNF containing zero predicates is returned.
89     *
90     * @return predicates
91     */
92    @Override
93    public CNF getPredicates() {
94      return cnf;
95    }
96  
97    /**
98     * Initiates the query CNF from the query, including postprocessing
99     *
100    * @throws QueryContradictoryException if query is found to be contradictory
101    */
102   private void initCNF() throws QueryContradictoryException {
103     if (getGdlHandler().getPredicates().isPresent()) {
104       Predicate predicate = getGdlHandler().getPredicates().get();
105       // some transformations on the GDL query before conversion to CNF
106       predicate = preprocessPredicate(predicate);
107       CNF rawCNF = QueryPredicate.createFrom(predicate, new ComparableTPGMFactory(now)).asCNF();
108       cnf = cnfPostProcessing.postprocess(rawCNF);
109     } else {
110       // empty CNF
111       cnf = new CNF();
112     }
113   }
114 
115   /**
116    * Pipeline for preprocessing GDL query predicates. Currently, only predicates for
117    * valid interval overlap of edges and their vertices are added.
118    *
119    * @param predicate the GDL predicate to preprocess
120    * @return preprocessed GDL predicate
121    */
122   private Predicate preprocessPredicate(Predicate predicate) {
123     return addEdgeOverlapPredicates(predicate);
124   }
125 
126   /**
127    * Augments a predicate by constraints ensuring that valid times of
128    * edges and their adjacent nodes overlap
129    *
130    * @param predicate the predicate to augment
131    * @return predicate augmented by edge overlaps constraints for every edge
132    */
133   private Predicate addEdgeOverlapPredicates(Predicate predicate) {
134     GDLHandler handler = getGdlHandler();
135     Map<String, Edge> edgeCache = handler.getEdgeCache();
136 
137     // maps vertex ids to vertex variables
138     Map<Long, String> vertexMap = new HashMap<>();
139     for (Vertex vertex : handler.getVertexCache(true, true).values()) {
140       vertexMap.put(vertex.getId(), vertex.getVariable());
141     }
142 
143     // determine all (vertex, edge, vertex) triples
144     for (String edgeVar : handler.getEdgeCache().keySet()) {
145       Edge edge = edgeCache.get(edgeVar);
146       // no temporal predicates for paths
147       if (edge.hasVariableLength()) {
148         continue;
149       }
150       String sourceVar = vertexMap.get(edge.getSourceVertexId());
151       String targetVar = vertexMap.get(edge.getTargetVertexId());
152 
153 
154       // all relevant val-selectors
155       TimeSelector eFrom = new TimeSelector(edgeVar, TimeSelector.TimeField.VAL_FROM);
156       TimeSelector eTo = new TimeSelector(edgeVar, TimeSelector.TimeField.VAL_TO);
157 
158       TimeSelector sFrom = new TimeSelector(sourceVar, TimeSelector.TimeField.VAL_FROM);
159       TimeSelector sTo = new TimeSelector(sourceVar, TimeSelector.TimeField.VAL_TO);
160 
161       TimeSelector tFrom = new TimeSelector(targetVar, TimeSelector.TimeField.VAL_FROM);
162       TimeSelector tTo = new TimeSelector(targetVar, TimeSelector.TimeField.VAL_TO);
163 
164       // create predicates edgeVar.val.overlaps(sourceVar.val) AND
165       // edgeVar.val.overlaps(targetVar.val)...
166       Predicate overlaps = new Comparison(
167         new MaxTimePoint(eFrom, sFrom, tFrom), LTE, new MinTimePoint(eTo, sTo, tTo)
168       );
169       // ...and add them to the resulting predicate
170       predicate = new And(predicate, overlaps);
171     }
172     return predicate;
173   }

184   /**
185    * Checks if the graph returns a single vertex and no edges (no loops).
186    *
187    * @return true, if single vertex graph
188    */
189   public boolean isSingleVertexGraph() {
190     return getVertexCount() == 1 && getEdgeCount() == 0;
191   }
