107   /**
108    * Initializes typeMap, labelMap and cache from a {@link TemporalQueryHandler}
109    *
110    * @param queryHandler QueryHandler to use for initialization
111    */
112   private void initializeWithHandler(TemporalQueryHandler queryHandler) {
113     typeMap = new HashMap<>();
114     for (String edgeVar : queryHandler.getEdgeVariables()) {
115       typeMap.put(edgeVar, TemporalGraphStatistics.ElementType.EDGE);
116     }
117     for (String vertexVar : queryHandler.getVertexVariables()) {
118       typeMap.put(vertexVar, TemporalGraphStatistics.ElementType.VERTEX);
119     }
120 
121     labelMap = new HashMap<>();
122     queryHandler.getLabelsForVariables(queryHandler.getAllVariables())
123       .forEach((key, value) -> {
124         if (value.length() > 0) {
125           labelMap.put(key, value);
126         }
127       });
128     // pre-compute all estimations
129     cache = new HashMap<>();
130     for (CNFElement clause : queryHandler.getPredicates()) {
131       for (ComparisonExpression comp : clause.getPredicates()) {
132         cache.put(comp, estimateComparison(comp));
133       }
134     }
135   }

137   /**
138    * Estimates the probability that a CNF is true, based on the graph statistics.
139    * Naive assumption: all contained comparisons are pair-wise independent.
140    *
141    * @param cnf cnf
142    * @return estimation of the probability that the CNF evaluates to true
143    */
144   public double estimateCNF(CNF cnf) {
145     return cnf.getPredicates().stream()
146       .map(this::estimateCNFElement)
147       .collect(Collectors.toList())
148       .stream()
149       .reduce((i, j) -> i * j)
150       .orElse(1.);
151   }

153   /**
154    * Estimates the probability that a {@link CNFElement}, i.e. a disjunction of
155    * comparisons, is true (based on the graph statistics).
156    *
157    * @param element CNFElement
158    * @return estimation of probability that the CNFElement evaluates to true
159    */
160   private double estimateCNFElement(CNFElement element) {
161     if (element.getPredicates().size() == 1) {
162       return estimateComparison(element.getPredicates().get(0));
163     } else {
164       // recursively estimate P(a or b or c or...) as P(a) + P(b or c or...) - P(a)*P(b or c or...)
165       List<ComparisonExpression> comparisons = element.getPredicates();
166       double firstEst = estimateComparison(comparisons.get(0));
167       CNFElement rest = new CNFElement(comparisons.subList(1, comparisons.size()));
168       double restEst = estimateCNF(new CNF(Collections.singletonList(rest)));
169       return firstEst + restEst - (firstEst * restEst);
170     }
171   }

173   /**
174    * Estimates the probability that a comparison evaluates to true, based on
175    * the graph statistics.
176    *
177    * @param comparisonExpression comparison
178    * @return estimation of the probability that the comparison evaluates to true
179    */
180   private double estimateComparison(ComparisonExpression comparisonExpression) {
181     if (cache.containsKey(comparisonExpression)) {
182       return cache.get(comparisonExpression);
183     }
184     double result = 1.;
185     // probability of labels is not estimated here, belongs to the structural estimations
186     if (isLabelComp(comparisonExpression)) {
187       return 1.;
188     } else {
189       // more than one element in the comparison?
190       if (comparisonExpression.getVariables().size() > 1) {
191         result = estimateComparisonOnDifferent(comparisonExpression);
192       } else {
193         // exactly one element in the comparison
194         result = estimateCompOnSameTPGM(comparisonExpression);
195       }
196     }
197     cache.put(comparisonExpression, result);
198     return result;
199   }

201   /**
202    * Checks if a comparison describes the label of an element
203    *
204    * @param comparison comparison to check
205    * @return true iff the comparison describes the label of an element
206    */
207   private boolean isLabelComp(ComparisonExpression comparison) {
208     if (comparison.getComparator() != EQ) {
209       return false;
210     }
211     if (comparison.getLhs() instanceof PropertySelectorComparable) {
212       return ((PropertySelectorComparable) comparison.getLhs())
213         .getPropertyKey().equals("__label__");
214     } else if (comparison.getRhs() instanceof PropertySelectorComparable) {
215       return ((PropertySelectorComparable) comparison.getRhs())
216         .getPropertyKey().equals("__label__");
217     }
218     return false;
219   }

222    * Estimates the probability that a comparison involving only one element holds.
223    *
224    * @param comparisonExpression comparison
225    * @return estimation of probability that the comparison holds
226    */
227   private double estimateCompOnSameTPGM(ComparisonExpression comparisonExpression) {
228     QueryComparable lhs = comparisonExpression.getLhs();
229     QueryComparable rhs = comparisonExpression.getRhs();
230 
231     if (ComparisonUtil.isTemporal(comparisonExpression)) {
232       if (lhs instanceof TimeSelectorComparable && rhs instanceof TimeSelectorComparable) {
233         // TODO implement?
234         return 1.;
235       } else if (lhs instanceof TimeSelectorComparable && rhs instanceof TimeLiteralComparable ||
236         lhs instanceof TimeLiteralComparable && rhs instanceof TimeSelectorComparable) {
237         return simpleTemporalEstimation(comparisonExpression);
238       } else if (lhs instanceof MinTimePointComparable || lhs instanceof MaxTimePointComparable ||
239         rhs instanceof MinTimePointComparable || rhs instanceof MaxTimePointComparable) {
240         return minMaxTemporalEstimation(comparisonExpression);
241       } else if ((lhs instanceof DurationComparable && rhs instanceof TimeConstantComparable) ||
242         (lhs instanceof TimeConstantComparable && rhs instanceof DurationComparable)) {
243         return simpleDurationComparisonEstimation(comparisonExpression);
244       } else {
245         return 1.;
246       }
247     } else {
248       if ((lhs instanceof PropertySelectorComparable && rhs instanceof LiteralComparable) ||
249         (lhs instanceof LiteralComparable && rhs instanceof PropertySelectorComparable)) {
250         return simplePropertyEstimation(comparisonExpression);
251       } else if (lhs instanceof PropertySelectorComparable &&
252         rhs instanceof PropertySelectorComparable) {
253         return complexPropertyEstimation(comparisonExpression);
254       } else {
255         return 1.;
256       }
257     }

260   /**
261    * Estimates the probability that a comparison between a tx or val duration and
262    * a time constant evaluates to true.
263    * All other possible comparisons involving a duration are estimated 1.0
264    *
265    * @param comparisonExpression comparison
266    * @return estimation of the probability that the comparison evaluates to true
267    */
268   private double simpleDurationComparisonEstimation(ComparisonExpression comparisonExpression) {
269     QueryComparable lhs = comparisonExpression.getLhs();
270     QueryComparable rhs = comparisonExpression.getRhs();
271     Comparator comp = comparisonExpression.getComparator();
272     // ensure that the duration is always on the left side
273     if (rhs instanceof DurationComparable) {
274       QueryComparable t = lhs;
275       lhs = rhs;
276       rhs = t;
277       comp = switchComparator(comp);
278     }
279     if (!(rhs instanceof TimeConstantComparable)) {
280       return 1.;
281     }
282     // only valid and tx durations can be estimated
283     Duration duration = (Duration) ((DurationComparable) lhs).getWrappedComparable();
284     // complex durations not supported yet, only transaction and valid interval durations
285     if (!checkSimpleDuration(duration)) {
286       return 1.;
287     }
288 
289     String variable = new ArrayList<>(comparisonExpression.getVariables()).get(0);
290     TemporalGraphStatistics.ElementType type = typeMap.get(variable);
291     Optional<String> label = labelMap.containsKey(variable) ?
292       Optional.of(labelMap.get(variable)) : Optional.empty();
293     TimeSelector from = (TimeSelector) duration.getFrom();
294     // tx or valid interval?
295     boolean transaction = from.getTimeProp() == TimeSelector.TimeField.TX_FROM;
296     long rhsValue = ((TimeConstant) ((TimeConstantComparable) rhs)
297       .getWrappedComparable()).getMillis();
298 
299     return stats.estimateDurationProb(type, label, comp, transaction, rhsValue);
300   }

302   /**
303    * Checks if a duration is of the simple form [a.tx_from, a.tx_to] or
304    * [a.val_from, a.val_to]
305    *
306    * @param duration duration to check
307    * @return true iff the duration has this simple form
308    */
309   private boolean checkSimpleDuration(Duration duration) {
310     // two time selectors?
311     if (!(duration.getFrom() instanceof TimeSelector) || !(duration.getTo() instanceof TimeSelector)) {
312       return false;
313     }
314     TimeSelector from = (TimeSelector) duration.getFrom();
315     TimeSelector to = (TimeSelector) duration.getTo();
316     // val_from, val_to or tx_from, tx_to?
317     if (!
318       ((from.getTimeProp() == TimeSelector.TimeField.VAL_FROM &&
319         to.getTimeProp() == TimeSelector.TimeField.VAL_TO) ||
320         (from.getTimeProp() == TimeSelector.TimeField.TX_FROM &&
321           to.getTimeProp() == TimeSelector.TimeField.TX_TO))) {
322       return false;
323     }
324     // same variable for both selectors?
325     return from.getVariable().equals(to.getVariable());
326   }

423   /**
424    * Used to switch lhs and rhs of a comparison. Switches the comparator.
425    * This is not the inversion function!
426    * E.g., for <, the return value is > (while the inverse of < is >=)
427    *
428    * @param comp comparator to switch
429    * @return switched comparator
430    */
431   private Comparator switchComparator(Comparator comp) {
432     if (comp == EQ || comp == NEQ) {
433       return comp;
434     } else if (comp == LT) {
435       return GT;
436     } else if (comp == LTE) {
437       return GTE;
438     } else if (comp == GT) {
439       return LT;
440     } else {
441       return LTE;
442     }
443   }

544   /**
545    * Reorders the clauses of the CNF: the most selective clauses are moved to the beginning
546    * of the CNF (exception: label constraint is always the first constraint).
547    * Within a clause, the least selective comparisons are moved to the beginning of the clause.
548    * The motivation is to minimize the number of comparisons to check when processing the CNF
549    *
550    * @param cnf the CNF to reorder
551    * @return reordered CNF
552    */
553   public CNF reorderCNF(CNF cnf) {
554     ArrayList<CNFElement> clauses = new ArrayList<>(cnf.getPredicates());
555     // resort the clauses: labels and then selective clauses first
556     clauses.sort((clause1, clause2) -> {
557       if (clause1.getPredicates().size() == 1 && isLabelComp(clause1.getPredicates().get(0))) {
558         return 100;
559       } else if (clause2.getPredicates().size() == 1 &&
560         isLabelComp(clause2.getPredicates().get(0))) {
561         return -100;
562       } else {
563         return (int) (100. * (estimateCNFElement(clause1) - estimateCNFElement(clause2)));
564       }
565     });
566 
567     // resort comparisons within clauses: labels and then non-selective comparisons first
568     clauses = clauses.stream().map(clause -> {
569       ArrayList<ComparisonExpression> comps = new ArrayList<>(
570         clause.getPredicates());
571       comps.sort((c1, c2) -> {
572         if (isLabelComp(c1)) {
573           return 100;
574         } else if (isLabelComp(c2)) {
575           return -100;
576         } else {
577           return (int) (100. * (estimateComparison(c2) - estimateComparison(c1)));
578         }
579       });
580       return new CNFElement(comps);
581     }).collect(Collectors.toCollection(ArrayList::new));
582 
583     return new CNF(clauses);
584   }
