16  package org.gradoop.temporal.model.impl.operators.aggregation.functions;
17  
18  import org.apache.flink.api.java.tuple.Tuple2;
19  import org.gradoop.common.model.impl.properties.PropertyValue;
20  import org.gradoop.flink.model.impl.operators.aggregation.functions.BaseAggregateFunction;
21  import org.gradoop.temporal.model.api.TimeDimension;
22  import org.gradoop.temporal.model.impl.pojo.TemporalElement;
23  
24  import java.time.temporal.ChronoUnit;
25  import java.time.temporal.TemporalUnit;
26  import java.util.Arrays;
27  import java.util.List;
28  import java.util.Objects;
29  import java.util.stream.Collectors;
30  
31  /**
32   * Abstract base class for functions aggregating the duration of temporal elements for a certain {@link TimeDimension}.
33   */
34  public abstract class AbstractDurationAggregateFunction extends BaseAggregateFunction {
35  
36    /**
37     * The default temporal unit.
38     */
39    public static final TemporalUnit DEFAULT_UNIT = ChronoUnit.MILLIS;
40  
41    /**
42     * A list of supported temporal units.
43     */
44    static final List<TemporalUnit> SUPPORTED_UNITS = Arrays.asList(ChronoUnit.MILLIS, ChronoUnit.SECONDS,
45      ChronoUnit.MINUTES, ChronoUnit.HOURS, ChronoUnit.DAYS);
46  
47    /**
48     * Selects which time dimension is considered by this aggregate function.
49     */
50    protected final TimeDimension dimension;
51  
52    /**
53     * The unit to parse the duration after the aggregation.
54     */
55    protected final TemporalUnit timeUnit;
56  
57    /**
58     * Creates a new instance of this base aggregate function.
59     *
60     * @param aggregatePropertyKey the property key of the new property where the aggregated value is stored
61     * @param dimension the given TimeDimension
62     * @param unit the temporal unit into which the result is converted. The supported units are specified in
63     *             {@link AbstractDurationAggregateFunction#SUPPORTED_UNITS}.
64     */
65    AbstractDurationAggregateFunction(String aggregatePropertyKey, TimeDimension dimension, TemporalUnit unit) {
66      super(aggregatePropertyKey);
67      this.dimension = Objects.requireNonNull(dimension);
68      Objects.requireNonNull(unit);
69      if (!SUPPORTED_UNITS.contains(unit)) {
70        throw new IllegalArgumentException("The given unit [" + unit +
71          "] is not supported. Supported temporal units are [" +
72          SUPPORTED_UNITS.stream().map(TemporalUnit::toString).collect(Collectors.joining(",")) + "]");
73      }
74      this.timeUnit = unit;
75    }
76  
77    /**
78     * Returns the duration of an element based on the given {@link TimeDimension}.
79     * If start or end of the interval are `null` or set to a default value, {@code -1} is returned.
80     *
81     * @param element the given TemporalElement
82     * @return a {@link PropertyValue} object holding the duration as long value [ms] or {@value -1} in case of a null or default value
83     */
84    PropertyValue getDuration(TemporalElement element) {
85      Tuple2<Long, Long> timeInterval;
86      switch (dimension) {
87      case TRANSACTION_TIME:
88        timeInterval = element.getTransactionTime();
89        break;
90      case VALID_TIME:
91        timeInterval = element.getValidTime();
92        break;
93      default:
94        throw new IllegalArgumentException("Unknown dimension [" + dimension + "].");
95      }
96      if (timeInterval.f0 == null || timeInterval.f1 == null ||
97        timeInterval.f0.equals(TemporalElement.DEFAULT_TIME_FROM) ||
98        timeInterval.f0.equals(TemporalElement.DEFAULT_TIME_TO) ||
99        timeInterval.f1.equals(TemporalElement.DEFAULT_TIME_FROM) ||
100       timeInterval.f1.equals(TemporalElement.DEFAULT_TIME_TO)) {
101       return PropertyValue.create(-1L);
102     } else {
103       return PropertyValue.create(timeInterval.f1 - timeInterval.f0);
104     }
105   }
106 }
