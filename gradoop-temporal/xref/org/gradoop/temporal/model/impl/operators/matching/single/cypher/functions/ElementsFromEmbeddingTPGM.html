43  /**
44   * Extracts elements from an {@link Embedding}.
45   * Is almost identical to
46   * {@link org.gradoop.flink.model.impl.operators.matching.single.cypher.functions.ElementsFromEmbedding}
47   * extends it for TPGM embeddings.
48   * Could be further extended to specify temporal properties like global val/tx in the graph head
49   *
50   * @param <G> The graph head type.
51   * @param <V> The vertex type.
52   * @param <E> The edge type.
53   */
54  public class ElementsFromEmbeddingTPGM<
55    G extends TemporalGraphHead,
56    V extends TemporalVertex,
57    E extends TemporalEdge> implements FlatMapFunction<Embedding, Element> {
58    /**
59     * Constructs temporal graph heads
60     */
61    private final GraphHeadFactory<G> graphHeadFactory;
62    /**
63     * Constructs temporal vertices
64     */
65    private final VertexFactory<V> vertexFactory;
66    /**
67     * Constructs temporal edges
68     */
69    private final EdgeFactory<E> edgeFactory;
70    /**
71     * Describes the embedding content
72     */
73    private final EmbeddingMetaData metaData;
74    /**
75     * Source vertex variables by edge id
76     */
77    private final Map<String, Pair<String, String>> sourceTargetVariables;
78    /**
79     * Stores the mapping between query variable and element id. The mapping is added as a property
80     * to the graph head representing an embedding.
81     */
82    private final Map<PropertyValue, PropertyValue> variableMapping;
83    /**
84     * Stores the identifiers that have already been processed.
85     */
86    private final Set<GradoopId> processedIds;
87    /**
88     * Stores the mapping between return pattern variables and its labels
89     */
90    private final Map<String, String> labelMapping;
91  
92    /**
93     * Constructor.
94     *
95     * @param graphHeadFactory      temporal graph head factory
96     * @param vertexFactory         temporal vertex factory
97     * @param edgeFactory           temporal edge factory
98     * @param embeddingMetaData     meta data for the TPGM embedding
99     * @param sourceTargetVariables source and target vertex variables by edge variable
100    */
101   public ElementsFromEmbeddingTPGM(GraphHeadFactory<G> graphHeadFactory,
102                                    VertexFactory<V> vertexFactory, EdgeFactory<E> edgeFactory,
103                                    EmbeddingMetaData embeddingMetaData,
104                                    Map<String, Pair<String, String>> sourceTargetVariables) {
105     this(graphHeadFactory, vertexFactory, edgeFactory, embeddingMetaData,
106       sourceTargetVariables, Maps.newHashMapWithExpectedSize(0));

109   /**
110    * Constructor.
111    *
112    * @param graphHeadFactory      temporal graph head factory
113    * @param vertexFactory         temporal vertex factory
114    * @param edgeFactory           temporal edge factory
115    * @param embeddingMetaData     meta data for the TPGM embedding
116    * @param sourceTargetVariables source and target vertex variables by edge variable
117    * @param labelMapping          mapping between newElementVariables and its labels
118    */
119   public ElementsFromEmbeddingTPGM(GraphHeadFactory<G> graphHeadFactory,
120                                    VertexFactory<V> vertexFactory,
121                                    EdgeFactory<E> edgeFactory,
122                                    EmbeddingMetaData embeddingMetaData,
123                                    Map<String, Pair<String, String>> sourceTargetVariables,
124                                    Map<String, String> labelMapping) {
125     this.graphHeadFactory = graphHeadFactory;
126     this.vertexFactory = vertexFactory;
127     this.edgeFactory = edgeFactory;
128     this.metaData = embeddingMetaData;
129     this.sourceTargetVariables = sourceTargetVariables;
130     this.labelMapping = labelMapping;
131     this.variableMapping = new HashMap<>(embeddingMetaData.getEntryCount());
132     this.processedIds = new HashSet<>(embeddingMetaData.getEntryCount());
133   }
134 
135   @Override
136   public void flatMap(Embedding embedding, Collector<Element> out) throws Exception {
137     // clear for each embedding
138     processedIds.clear();
139 
140     // create graph head for this embedding
141     TemporalGraphHead graphHead = graphHeadFactory.createGraphHead();
142 
143     // vertices
144     for (String vertexVariable : metaData.getVertexVariables()) {
145       GradoopId id = embedding.getId(metaData.getEntryColumn(vertexVariable));
146 
147       if (labelMapping.containsKey(vertexVariable)) {
148         String label = labelMapping.get(vertexVariable);
149         initVertexWithData(out, graphHead, id, label);
150       } else {
151         initVertex(out, graphHead, id);
152       }
153       variableMapping.put(PropertyValue.create(vertexVariable), PropertyValue.create(id));
154     }
155 
156     // edges
157     GradoopId edgeId;
158     GradoopId sourceId;
159     GradoopId targetId;
160     // Long[] timeData;
161     for (String edgeVariable : metaData.getEdgeVariables()) {
162       edgeId = embedding.getId(metaData.getEntryColumn(edgeVariable));
163       sourceId = embedding.getId(
164         metaData.getEntryColumn(sourceTargetVariables.get(edgeVariable).getLeft()));
165       targetId = embedding.getId(
166         metaData.getEntryColumn(sourceTargetVariables.get(edgeVariable).getRight()));
167 
168       if (labelMapping.containsKey(edgeVariable)) {
169         String label = labelMapping.get(edgeVariable);
170         initEdgeWithData(out, graphHead, edgeId, sourceId, targetId, label);
171       } else {
172         initEdge(out, graphHead, edgeId, sourceId, targetId);
173       }
174       variableMapping.put(PropertyValue.create(edgeVariable), PropertyValue.create(edgeId));
175     }
176 
177     // paths (copied from EPGM)
178 
179     for (String pathVariable : metaData.getPathVariables()) {
180       ExpandDirection direction = metaData.getDirection(pathVariable);
181       List<GradoopId> path = embedding.getIdList(metaData.getEntryColumn(pathVariable));
182       List<PropertyValue> mappingValue = new ArrayList<>(path.size());
183       for (int i = 0; i < path.size(); i += 2) {
184         edgeId = path.get(i);
185         mappingValue.add(PropertyValue.create(edgeId));
186 
187         if (direction == ExpandDirection.OUT) {
188           sourceId = i > 0 ?
189             path.get(i - 1) :
190             embedding.getId(
191               metaData.getEntryColumn(sourceTargetVariables.get(pathVariable).getLeft()));
192 
193 
194           targetId = i < path.size() - 1 ?
195             path.get(i + 1) :
196             embedding.getId(
197               metaData.getEntryColumn(sourceTargetVariables.get(pathVariable).getRight()));
198 
199           if (i + 2 < path.size()) {
200             mappingValue.add(PropertyValue.create(targetId));
201           }
202         } else {
203           sourceId = i < path.size() - 1 ?
204             path.get(i + 1) :
205             embedding.getId(
206               metaData.getEntryColumn(sourceTargetVariables.get(pathVariable).getLeft()));
207 
208           targetId = i > 0 ?
209             path.get(i - 1) :
210             embedding.getId(metaData.getEntryColumn(
211               sourceTargetVariables.get(pathVariable).getRight()));
212 
213           if (i > 0) {
214             mappingValue.add(PropertyValue.create(sourceId));
215           }
216         }
217 
218         initVertex(out, graphHead, sourceId);
219         initVertex(out, graphHead, targetId);
220         initEdge(out, graphHead, edgeId, sourceId, targetId);
221       }
222       variableMapping.put(PropertyValue.create(pathVariable), PropertyValue.create(mappingValue));
223     }
224 
225 
226     graphHead.setProperty(PatternMatching.VARIABLE_MAPPING_KEY, variableMapping);
227     out.collect(graphHead);
228 
229 
230   }

232   /**
233    * Initializes an vertex using the specified parameters
234    *
235    * @param out       flat map collector
236    * @param graphHead temporal graph head to assign vertex to
237    * @param vertexId  vertex identifier
238    */
239   private void initVertex(Collector<Element> out, TemporalGraphHead graphHead,
240                           GradoopId vertexId) {
241     initVertexWithData(out, graphHead, vertexId, null);
242   }
243 
244   /**
245    * Initializes an vertex using the specified parameters and adds its label
246    * if the given vertex was created for the return pattern.
247    *
248    * @param out       flat map collector
249    * @param graphHead temporal graph head to assign vertex to
250    * @param vertexId  vertex identifier
251    * @param label     label associated with vertex
252    */
253   private void initVertexWithData(Collector<Element> out, TemporalGraphHead graphHead,
254                                   GradoopId vertexId,
255                                   String label) {
256     if (!processedIds.contains(vertexId)) {
257       TemporalVertex v = vertexFactory.initVertex(vertexId);
258       v.addGraphId(graphHead.getId());
259       v.setLabel(label);
260       out.collect(v);
261       processedIds.add(vertexId);
262     }
263   }
264 
265   /**
266    * Initializes an edge using the specified parameters.
267    *
268    * @param out       flat map collector
269    * @param graphHead temporal graph head to assign edge to
270    * @param edgeId    edge identifier
271    * @param sourceId  source vertex identifier
272    * @param targetId  target vertex identifier
273    */
274   private void initEdge(Collector<Element> out, TemporalGraphHead graphHead,
275                         GradoopId edgeId, GradoopId sourceId, GradoopId targetId) {
276     initEdgeWithData(out, graphHead, edgeId, sourceId, targetId, null);
277   }
278 
279   /**
280    * Initializes an edge using the specified parameters and adds its label
281    * if the given edge was created for return pattern
282    *
283    * @param out       flat map collector
284    * @param graphHead temporal graph head to assign edge to
285    * @param edgeId    edge identifier
286    * @param sourceId  source vertex identifier
287    * @param targetId  target vertex identifier
288    * @param label     label associated with edge
289    */
290   private void initEdgeWithData(Collector<Element> out, TemporalGraphHead graphHead,
291                                 GradoopId edgeId, GradoopId sourceId, GradoopId targetId,
292                                 String label) {
293     if (!processedIds.contains(edgeId)) {
294       TemporalEdge e = edgeFactory.initEdge(edgeId, sourceId, targetId);
295       e.addGraphId(graphHead.getId());
296       e.setLabel(label);
297       out.collect(e);
298       processedIds.add(edgeId);
299     }
300   }
