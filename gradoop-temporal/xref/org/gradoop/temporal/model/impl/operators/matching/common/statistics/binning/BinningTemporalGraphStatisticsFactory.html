26  import java.util.HashSet;

30  /**
31   * Factory for {@link BinningTemporalGraphStatistics}
32   */
33  public class BinningTemporalGraphStatisticsFactory implements
34    TemporalGraphStatisticsFactory<BinningTemporalGraphStatistics> {
35  
36    /**
37     * Default size for a sample
38     */
39    static final int DEFAULT_SAMPLE_SIZE = 5000;
40  
41    @Override
42    public BinningTemporalGraphStatistics fromGraph(TemporalGraph g) throws Exception {
43      return fromGraphWithSampling(g, DEFAULT_SAMPLE_SIZE);
44    }
45  
46    @Override
47    public BinningTemporalGraphStatistics fromGraph(TemporalGraph g, Set<String> numericalProperties,
48                                                    Set<String> categoricalProperties) throws Exception {
49      return fromGraphWithSampling(g, DEFAULT_SAMPLE_SIZE, numericalProperties, categoricalProperties);
50    }
51  
52    @Override
53    public BinningTemporalGraphStatistics fromGraphWithSampling(TemporalGraph g, int sampleSize)
54      throws Exception {
55      return fromGraphWithSampling(g, sampleSize, null, null);
56    }
57  
58    @Override
59    public BinningTemporalGraphStatistics fromGraphWithSampling(TemporalGraph g, int sampleSize,
60                                                                Set<String> numericalProperties,
61                                                                Set<String> categoricalProperties)
62      throws Exception {
63  
64      List<TemporalElementStats> vertexStats = Lists.newArrayList();
65      g.getVertices()
66        .groupBy(EPGMElement::getLabel)
67        .reduceGroup(new ElementsToStats<>(numericalProperties, categoricalProperties))
68        .output(new LocalCollectionOutputFormat<>(vertexStats));
69  
70      List<TemporalElementStats> edgeStats = Lists.newArrayList();
71      g.getEdges()
72        // do not replace this with the method reference!!!
73        .groupBy(edge -> edge.getLabel())
74        .reduceGroup(new ElementsToStats<>(numericalProperties, categoricalProperties))
75        .output(new LocalCollectionOutputFormat<>(edgeStats));
76  
77      g.getConfig().getExecutionEnvironment().execute();
78  
79      HashSet<String> relevantProperties = null;
80      // both only null, if all properties should be considered
81      // (use empty lists to ignore all properties)
82      if (numericalProperties != null && categoricalProperties != null) {
83        relevantProperties = new HashSet<>();
84        relevantProperties.addAll(numericalProperties);
85        relevantProperties.addAll(categoricalProperties);
86      }
87  
88      return new BinningTemporalGraphStatistics(vertexStats, edgeStats, relevantProperties);
89    }
90  }
