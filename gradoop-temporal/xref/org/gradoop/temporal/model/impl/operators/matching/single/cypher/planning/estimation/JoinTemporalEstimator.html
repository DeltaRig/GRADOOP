33  /**
34   * Keeps track of the joined leaf nodes in a query plan and computes a total estimated cardinality
35   * for the plan.
36   */
37  public class JoinTemporalEstimator extends TemporalEstimator {
38    /**
39     * Maps vertex and edge variables to their estimated cardinality
40     */
41    private final Map<String, Long> cardinalities;
42    /**
43     * Maps vertex variables to their distinct counts per embedding
44     */
45    private final Map<String, List<Long>> distinctValues;
46  
47    /**
48     * Creates a new estimator.
49     *
50     * @param queryHandler    query handler
51     * @param graphStatistics graph statistics
52     */
53    JoinTemporalEstimator(TemporalQueryHandler queryHandler, TemporalGraphStatistics graphStatistics) {
54      super(queryHandler, graphStatistics);
55      this.cardinalities = new HashMap<>();
56      this.distinctValues = new HashMap<>();
57    }
58  
59    /**
60     * Updates the cardinalities according to the given join node.
61     *
62     * @param node join node
63     */
64    void visit(JoinNode node) {
65      if (node instanceof BinaryNode) {
66        BinaryNode binaryNode = (BinaryNode) node;
67        if (binaryNode.getLeftChild() instanceof LeafNode) {
68          process(binaryNode.getLeftChild().getEmbeddingMetaData());
69        }
70        if (binaryNode.getRightChild() instanceof LeafNode) {
71          process(binaryNode.getRightChild().getEmbeddingMetaData());
72        }
73      }
74    }
75  
76    /**
77     * Computes the final cardinality according to the visited nodes.
78     *
79     * @return estimated cardinality
80     */
81    long getCardinality() {
82      long numerator = cardinalities.values().stream().reduce((i, j) -> i * j).orElse(0L);
83  
84      long denominator = distinctValues.values().stream()
85        .map(list -> list.stream().sorted().collect(Collectors.toList()))
86        .map(list -> list.subList(1, list.size()))
87        .flatMap(Collection::stream)
88        .reduce((i, j) -> i * j)
89        .orElse(1L);
90  
91      return Math.round(1.0 * numerator / denominator);
92    }
93  
94    /**
95     * Updates the state using the information stored in the given meta data.
96     *
97     * @param metaData meta data from leaf node
98     */
99    private void process(EmbeddingMetaData metaData) {
100     int entryCount = metaData.getEntryCount();
101     List<String> variables = metaData.getVariables();
102     if (entryCount == 1) {
103       processVertex(variables.get(0));
104     } else {
105       String edgeVariable = variables.get(1);
106       String sourceVariable = getQueryHandler().getVertexById(
107         getQueryHandler().getEdgeByVariable(edgeVariable).getSourceVertexId()).getVariable();
108       String targetVariable = getQueryHandler().getVertexById(
109         getQueryHandler().getEdgeByVariable(edgeVariable).getTargetVertexId()).getVariable();
110       processEdge(sourceVariable, variables.get(1), targetVariable);
111     }
112   }
113 
114   /**
115    * Updates the state according to vertex statistics.
116    *
117    * @param vertexVariable vertex variable
118    */
119   private void processVertex(String vertexVariable) {
120     String label = getLabel(vertexVariable, true);
121     long cardinality = getCardinality(label, true);
122     updateCardinality(vertexVariable, cardinality);
123     updateDistinctValues(vertexVariable, cardinality);
124   }
125 
126   /**
127    * Updates the state according to the edge statistics.
128    *
129    * @param sourceVariable source vertex variable
130    * @param edgeVariable   edge variable
131    * @param targetVariable target vertex variable
132    */
133   private void processEdge(String sourceVariable, String edgeVariable, String targetVariable) {
134     String edgeLabel = getLabel(edgeVariable, false);
135 
136     long distinctSourceCount = getGraphStatistics().getDistinctSourceVertexCount(edgeLabel);
137     if (distinctSourceCount == 0L) {
138       distinctSourceCount = getGraphStatistics().getDistinctSourceVertexCount();
139     }
140     long distinctTargetCount = getGraphStatistics().getDistinctTargetVertexCount(edgeLabel);
141     if (distinctTargetCount == 0L) {
142       distinctTargetCount = getGraphStatistics().getDistinctTargetVertexCount();
143     }
144 
145     Edge queryEdge = getQueryHandler().getEdgeByVariable(edgeVariable);
146     if (queryEdge.getUpperBound() > 1) {
147       // variable case: n-hop edge
148       updateCardinality(edgeVariable, getPathCardinality(getCardinality(edgeLabel, false),
149         queryEdge.getLowerBound(), queryEdge.getUpperBound(),
150         distinctSourceCount, distinctTargetCount));
151     } else {
152       // static case: 1-hop edge
153       updateCardinality(edgeVariable, getCardinality(edgeLabel, false));
154     }
155     updateDistinctValues(sourceVariable, distinctSourceCount);
156     updateDistinctValues(targetVariable, distinctTargetCount);
157   }
158 
159   /**
160    * Estimated the total number of paths whose length is between the specified bounds.
161    *
162    * @param edgeCardinality     cardinality of the traversed edge
163    * @param lowerBound          minimum path length
164    * @param upperBound          maximum path length
165    * @param distinctSourceCount number of distinct source vertices
166    * @param distinctTargetCount number of distinct target vertices
167    * @return estimated number of paths with a length in the given range
168    */
169   private long getPathCardinality(long edgeCardinality, int lowerBound, int upperBound,
170                                   long distinctSourceCount, long distinctTargetCount) {
171 
172     double totalCardinality = 0L;
173     long probability = distinctSourceCount * distinctTargetCount;
174 
175     for (int i = lowerBound; i <= upperBound; i++) {
176       totalCardinality += Math.pow(edgeCardinality, i) / Math.pow(probability, i - 1);
177     }
178 
179     return Math.round(totalCardinality);
180   }
181 
182   /**
183    * Updates the cardinality of the variable.
184    *
185    * @param variable    query variable
186    * @param cardinality cardinality
187    */
188   private void updateCardinality(String variable, long cardinality) {
189     cardinalities.put(variable, cardinalities.getOrDefault(variable, 1L) * cardinality);
190   }
191 
192   /**
193    * Updates the distinct value list of a vertex with the given count.
194    *
195    * @param variable vertex variable
196    * @param count    distinct count
197    */
198   private void updateDistinctValues(String variable, long count) {
199     if (distinctValues.containsKey(variable)) {
200       distinctValues.get(variable).add(count);
201     } else {
202       distinctValues.put(variable, Lists.newArrayList(count));
203     }
204   }
