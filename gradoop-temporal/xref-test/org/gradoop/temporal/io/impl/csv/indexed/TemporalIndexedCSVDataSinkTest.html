62          return current;

66    /**
67     * Test writing a temporal graph to an indexed csv sink.
68     *
69     * @throws Exception in case of a write error
70     */
71    @Test
72    public void testWrite() throws Exception {
73      File file = testFolder.newFolder();
74      String tempFolderPath = file.getPath();
75  
76      testGraph.writeTo(new TemporalIndexedCSVDataSink(tempFolderPath, getConfig()));
77      getExecutionEnvironment().execute();
78  
79      final File[] subdirs = file.listFiles(File::isDirectory);
80  
81      assertNotNull(subdirs);
82      assertEquals(3, subdirs.length);
83      final List<File> files = Arrays.asList(subdirs);
84      final List<String> subDirList = files.stream().map(File::getName).collect(Collectors.toList());
85      assertTrue(subDirList.contains("vertices"));
86      assertTrue(subDirList.contains("edges"));
87      assertTrue(subDirList.contains("graphs"));
88  
89      for (File elementDir : files) {
90        if (elementDir.getName().equals("vertices")) {
91          final File[] vertexLabelArray = elementDir.listFiles(File::isDirectory);
92          assertNotNull(vertexLabelArray);
93          final List<String> vertexLabelDirs = Arrays.stream(vertexLabelArray).map(File::getName)
94            .collect(Collectors.toList());
95          assertNotNull(vertexLabelDirs);
96          assertEquals(3, vertexLabelDirs.size());
97          assertTrue(vertexLabelDirs.contains("person"));
98          assertTrue(vertexLabelDirs.contains("forum"));
99          assertTrue(vertexLabelDirs.contains("tag"));
100 
101       } else if (elementDir.getName().equals("edges")) {
102         final File[] edgeLabelArray = elementDir.listFiles(File::isDirectory);
103         assertNotNull(edgeLabelArray);
104         final List<String> edgeLabelDirs = Arrays.stream(edgeLabelArray).map(File::getName)
105           .collect(Collectors.toList());
106         assertNotNull(edgeLabelDirs);
107         assertEquals(5, edgeLabelDirs.size());
108         assertTrue(edgeLabelDirs.contains("hasmember"));
109         assertTrue(edgeLabelDirs.contains("hasinterest"));
110         assertTrue(edgeLabelDirs.contains("knows"));
111         assertTrue(edgeLabelDirs.contains("hasmoderator"));
112         assertTrue(edgeLabelDirs.contains("hastag"));
113 
114       } else if (elementDir.getName().equals("graphs")) {
115         final File[] graphLabelArray = elementDir.listFiles(File::isDirectory);
116         assertNotNull(graphLabelArray);
117         final List<String> graphLabelDirs = Arrays.stream(graphLabelArray).map(File::getName)
118           .collect(Collectors.toList());
119         assertNotNull(graphLabelDirs);
120         assertEquals(1, graphLabelDirs.size());
121         assertTrue(graphLabelDirs.contains("_db"));
122       }
123     }
124   }
