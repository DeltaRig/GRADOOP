320   /**
321    * Create a temporal edge with temporal attributes set.
322    *
323    * @param label     The label of the edge.
324    * @param source    The element used as a source for the edge.
325    * @param target    The element used as a target for the edge.
326    * @param txFrom    The start of the transaction time.
327    * @param txTo      The end of the transaction time.
328    * @param validFrom The start of the valid time.
329    * @param validTo   The end of the valid time.
330    * @return A temporal edge with those times set.
331    */
332   private TemporalEdge createEdge(String label, Identifiable source, Identifiable target, long txFrom,
333                                   long txTo, long validFrom, long validTo) {
334     TemporalEdge edge = getConfig().getTemporalGraphFactory().getEdgeFactory()
335       .createEdge(source.getId(), target.getId());
336     edge.setLabel(label);
337     edge.setTransactionTime(Tuple2.of(txFrom, txTo));
338     edge.setValidTime(Tuple2.of(validFrom, validTo));
339     return edge;
340   }

389   /**
390    * Given a logical graph, this method transforms it to a temporal graph.
391    * {@code start} and {@code end} values are extracted from the edges (= "trips")
392    * and used to set {@code valid_from}/{@code tx_from} and {@code valid_to}/{@code tx_to}
393    * values.
394    *
395    * @param logicalGraph the logical graph to transform
396    * @return logical graph transformed to temporal graph
397    */
398   protected TemporalGraph transformToTemporalGraph(LogicalGraph logicalGraph) {
399     TemporalGraph tg = toTemporalGraph(logicalGraph);
400     return tg.getFactory().fromDataSets(
401       tg.getVertices().map(vertexTransform),
402       tg.getEdges().map(edgeTransform));
403   }

405   /**
406    * Set the edge's {@code valid_from} and {@code tx_from} according to the {@code start}
407    * property and the edge's {@code valid_to} and {@code tx_to} according to the
408    * {@code end} property. Both properties are retained.
409    */
410   protected static final MapFunction<TemporalEdge, TemporalEdge> edgeTransform = value -> {
411     long start = value.getPropertyValue("start").getLong();
412     long end = value.getPropertyValue("end").getLong();
413     value.setValidTime(new Tuple2<>(start, end));
414     value.setTransactionTime(new Tuple2<>(start, end));
415     return value;
416   };

423   protected static final MapFunction<TemporalVertex, TemporalVertex> vertexTransform = value -> {
