206   /**
207    * Test the {@code fromNonTemporalDataSets} method using time interval extractor functions.
208    *
209    * @throws Exception when the execution in Flink fails
210    */
211   @Test
212   public void testFromNonTemporalDataSetsWithExtractors() throws Exception {
213     final String from = "f";
214     final String to = "t";
215     EPGMVertex vertex = getConfig().getLogicalGraphFactory().getVertexFactory().createVertex();
216     EPGMEdge edge = getConfig().getLogicalGraphFactory().getEdgeFactory()
217       .createEdge(vertex.getId(), vertex.getId());
218     EPGMGraphHead graphHead = getConfig().getLogicalGraphFactory().getGraphHeadFactory().createGraphHead();
219     vertex.addGraphId(graphHead.getId());
220     edge.addGraphId(graphHead.getId());
221     vertex.setProperty(from, PropertyValue.create(1L));
222     vertex.setProperty(to, PropertyValue.create(2L));
223     edge.setProperty(from, PropertyValue.create(3L));
224     edge.setProperty(to, PropertyValue.create(4L));
225     graphHead.setProperty(from, PropertyValue.create(5L));
226     graphHead.setProperty(to, PropertyValue.create(6L));
227 
228     List<EPGMGraphHead> heads = Collections.singletonList(graphHead);
229     List<EPGMVertex> vertices = Collections.singletonList(vertex);
230     List<EPGMEdge> edges = Collections.singletonList(edge);
231 
232     TemporalGraphCollection temporalCollection = getConfig().getTemporalGraphCollectionFactory()
233       .fromNonTemporalDataSets(getExecutionEnvironment().fromCollection(heads),
234         e -> Tuple2.of(e.getPropertyValue(from).getLong(), e.getPropertyValue(to).getLong()),
235         getExecutionEnvironment().fromCollection(vertices),
236         e -> Tuple2.of(e.getPropertyValue(from).getLong(), e.getPropertyValue(to).getLong()),
237         getExecutionEnvironment().fromCollection(edges),
238         e -> Tuple2.of(e.getPropertyValue(from).getLong(), e.getPropertyValue(to).getLong()));
239 
240     List<TemporalGraphHead> temporalHeads = new ArrayList<>();
241     List<TemporalVertex> temporalVertices = new ArrayList<>();
242     List<TemporalEdge> temporalEdges = new ArrayList<>();
243     temporalCollection.getGraphHeads().output(new LocalCollectionOutputFormat<>(temporalHeads));
244     temporalCollection.getVertices().output(new LocalCollectionOutputFormat<>(temporalVertices));
245     temporalCollection.getEdges().output(new LocalCollectionOutputFormat<>(temporalEdges));
246 
247     getExecutionEnvironment().execute();
248 
249     assertEquals(1, temporalHeads.size());
250     assertEquals(1, temporalVertices.size());
251     assertEquals(1, temporalEdges.size());
252     validateElementCollections(heads, temporalHeads);
253     validateElementCollections(vertices, temporalVertices);
254     validateElementCollections(edges, temporalEdges);
255     assertEquals(Tuple2.of(1L, 2L), temporalVertices.get(0).getValidTime());
256     assertEquals(Tuple2.of(3L, 4L), temporalEdges.get(0).getValidTime());
257     assertEquals(Tuple2.of(5L, 6L), temporalHeads.get(0).getValidTime());
258   }

263    * @param expected the expected temporal graph collection
264    * @param result the resulting temporal graph collection

267   private void compareCollections(TemporalGraphCollection expected, TemporalGraphCollection result)
268     throws Exception {
269

271     List<TemporalVertex> temporalVerticesExpected = new ArrayList<>();
272     List<TemporalEdge> temporalEdgesExpected = new ArrayList<>();
273
