228   /**
229    * Test the {@link TemporalGraphCollection#fromGraphCollection(BaseGraphCollection)} method.
230    */
231   @Test
232   public void testFromGraphCollection() throws Exception {
233     TemporalGraphCollection temporalCollection = TemporalGraphCollection.fromGraphCollection(testCollection);
234 
235     Collection<TemporalGraphHead> loadedGraphHeads = new ArrayList<>();
236     Collection<TemporalVertex> loadedVertices = new ArrayList<>();
237     Collection<TemporalEdge> loadedEdges = new ArrayList<>();
238 
239     temporalCollection.getGraphHeads().output(new LocalCollectionOutputFormat<>(loadedGraphHeads));
240     temporalCollection.getVertices().output(new LocalCollectionOutputFormat<>(loadedVertices));
241     temporalCollection.getEdges().output(new LocalCollectionOutputFormat<>(loadedEdges));
242 
243     Collection<EPGMGraphHead> epgmGraphHeads = new ArrayList<>();
244     Collection<EPGMVertex> epgmVertices = new ArrayList<>();
245     Collection<EPGMEdge> epgmEdges = new ArrayList<>();
246 
247     testCollection.getGraphHeads().output(new LocalCollectionOutputFormat<>(epgmGraphHeads));
248     testCollection.getVertices().output(new LocalCollectionOutputFormat<>(epgmVertices));
249     testCollection.getEdges().output(new LocalCollectionOutputFormat<>(epgmEdges));
250 
251     getExecutionEnvironment().execute();
252 
253     assertFalse(loadedGraphHeads.isEmpty());
254     assertFalse(loadedVertices.isEmpty());
255     assertFalse(loadedEdges.isEmpty());
256 
257     validateElementCollections(epgmGraphHeads, loadedGraphHeads);
258     validateElementCollections(epgmVertices, loadedVertices);
259     validateElementCollections(epgmEdges, loadedEdges);
260     validateGraphElementCollections(epgmVertices, loadedVertices);
261     validateGraphElementCollections(epgmEdges, loadedEdges);
262 
263     loadedGraphHeads.forEach(this::checkDefaultTemporalElement);
264     loadedVertices.forEach(this::checkDefaultTemporalElement);
265     loadedEdges.forEach(this::checkDefaultTemporalElement);
266   }
