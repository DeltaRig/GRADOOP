106   @Test
107   public void testFixedPattern() throws Exception {
108     // 2 matches for ISO
109     String query = "MATCH (n)-[e]->(m)-[f]->(o) WHERE n.tx_to>Timestamp(1970-01-01)";
110     TemporalQueryHandler queryHandler = new TemporalQueryHandler(query,
111       new CNFPostProcessing(new ArrayList<>()));
112     est = new CNFEstimation(stats, queryHandler);
113 
114     LeafNode nNode = new FilterAndProjectTemporalVerticesNode(null, "n",
115       queryHandler.getPredicates().getSubCNF("n"), Sets.newHashSet());
116     LeafNode mNode = new FilterAndProjectTemporalVerticesNode(null, "m",
117       queryHandler.getPredicates().getSubCNF("m"), Sets.newHashSet());
118     LeafNode oNode = new FilterAndProjectTemporalVerticesNode(null, "o",
119       queryHandler.getPredicates().getSubCNF("o"), Sets.newHashSet());
120     LeafNode eNode = new FilterAndProjectTemporalEdgesNode(null, "n", "e", "m",
121       queryHandler.getPredicates().getSubCNF("e"), Sets.newHashSet(), false);
122     LeafNode fNode = new FilterAndProjectTemporalEdgesNode(null, "m", "f", "o",
123       queryHandler.getPredicates().getSubCNF("f"), Sets.newHashSet(), false);
124 
125     // (n)-[e]->
126     JoinEmbeddingsNode neJoin = new JoinEmbeddingsNode(nNode, eNode, Lists.newArrayList("n"),
127       MatchStrategy.ISOMORPHISM, MatchStrategy.ISOMORPHISM);
128 
129     QueryPlan plan = new QueryPlan(neJoin);
130     TemporalQueryPlanEstimator estimator = new TemporalQueryPlanEstimator(plan, queryHandler, stats, est);
131 
132     assertTrue(18 <= estimator.getCardinality());
133     assertTrue(estimator.getCardinality() <= 20);
134 
135     // (n)-[e]->(m)
136     JoinEmbeddingsNode nemJoin = new JoinEmbeddingsNode(neJoin, mNode, Lists.newArrayList("m"),
137       MatchStrategy.ISOMORPHISM, MatchStrategy.ISOMORPHISM);
138 
139     plan = new QueryPlan(nemJoin);
140     estimator = new TemporalQueryPlanEstimator(plan, queryHandler, stats, est);
141 
142     assertTrue(18 <= estimator.getCardinality());
143     assertTrue(estimator.getCardinality() <= 20);
144 
145     // (n)-[e]->(m)-[f]->
146     JoinEmbeddingsNode nemfJoin = new JoinEmbeddingsNode(nemJoin, fNode, Lists.newArrayList("m"),
147       MatchStrategy.ISOMORPHISM, MatchStrategy.ISOMORPHISM);
148 
149     plan = new QueryPlan(nemfJoin);
150     estimator = new TemporalQueryPlanEstimator(plan, queryHandler, stats, est);
151 
152     // 20*30*20*30 / (30*17*30)  rather inaccurate,...
153     assertTrue(22 <= estimator.getCardinality());
154     assertTrue(estimator.getCardinality() <= 24);
155 
156     // (n)-[e]->(m)-[f]->(o)
157     JoinEmbeddingsNode nemfoJoin = new JoinEmbeddingsNode(nemfJoin, oNode, Lists.newArrayList("o"),
158       MatchStrategy.ISOMORPHISM, MatchStrategy.ISOMORPHISM);
159 
160     plan = new QueryPlan(nemfoJoin);
161     estimator = new TemporalQueryPlanEstimator(plan, queryHandler, stats, est);
162 
163     assertTrue(20 <= estimator.getCardinality());
164     assertTrue(estimator.getCardinality() <= 30);
165   }

192   @Test
193   public void testComplexCartesianProduct() throws Exception {
194     String query = "MATCH (a)-[e1:trip]->(b:station),(c:station)-[e2:trip]->(d:station)" +
195       "WHERE a.tx_from > c.val_from AND a.tx_to>Timestamp(1970-01-01)";
196     TemporalQueryHandler queryHandler = new TemporalQueryHandler(query,
197       new CNFPostProcessing(new ArrayList<>()));
198     est = new CNFEstimation(stats, queryHandler);
199 
200     LeafNode aNode = new FilterAndProjectTemporalVerticesNode(null, "a",
201       queryHandler.getPredicates().getSubCNF("a"), Sets.newHashSet());
202     TemporalQueryPlanEstimator
203       aEstimator = new TemporalQueryPlanEstimator(new QueryPlan(aNode), queryHandler, stats, est);
204     assertTrue(28 <= aEstimator.getCardinality() && aEstimator.getCardinality() <= 30);
205 
206     LeafNode bNode = new FilterAndProjectTemporalVerticesNode(null, "b",
207       queryHandler.getPredicates().getSubCNF("b"), Sets.newHashSet());
208     TemporalQueryPlanEstimator
209       bEstimator = new TemporalQueryPlanEstimator(new QueryPlan(bNode), queryHandler, stats, est);
210     assertEquals(30, bEstimator.getCardinality());
211 
212     LeafNode cNode = new FilterAndProjectTemporalVerticesNode(null, "c",
213       queryHandler.getPredicates().getSubCNF("c"), Sets.newHashSet());
214     TemporalQueryPlanEstimator
215       cEstimator = new TemporalQueryPlanEstimator(new QueryPlan(cNode), queryHandler, stats, est);
216     assertEquals(30, cEstimator.getCardinality());
217 
218     LeafNode dNode = new FilterAndProjectTemporalVerticesNode(null, "d",
219       queryHandler.getPredicates().getSubCNF("d"), Sets.newHashSet());
220     TemporalQueryPlanEstimator
221       dEstimator = new TemporalQueryPlanEstimator(new QueryPlan(dNode), queryHandler, stats, est);
222     assertEquals(30, dEstimator.getCardinality());
223 
224     LeafNode e1Node = new FilterAndProjectTemporalEdgesNode(null, "a", "e1", "b",
225       queryHandler.getPredicates().getSubCNF("e1"), Sets.newHashSet(), false);
226     TemporalQueryPlanEstimator
227       e1Estimator = new TemporalQueryPlanEstimator(new QueryPlan(e1Node), queryHandler, stats, est);
228     assertTrue(18 <= e1Estimator.getCardinality() && e1Estimator.getCardinality() <= 20);
229 
230     LeafNode e2Node = new FilterAndProjectTemporalEdgesNode(null, "c", "e2", "d",
231       queryHandler.getPredicates().getSubCNF("e2"), Sets.newHashSet(), false);
232     TemporalQueryPlanEstimator
233       e2Estimator = new TemporalQueryPlanEstimator(new QueryPlan(e2Node), queryHandler, stats, est);
234     assertTrue(18 <= e2Estimator.getCardinality() && e2Estimator.getCardinality() <= 20);
235 
236     BinaryNode ae1 = new JoinEmbeddingsNode(aNode, e1Node, Lists.newArrayList("a"),
237       MatchStrategy.HOMOMORPHISM, MatchStrategy.HOMOMORPHISM);
238     TemporalQueryPlanEstimator ae1Estimator =
239       new TemporalQueryPlanEstimator(new QueryPlan(ae1), queryHandler, stats, est);
240     assertTrue(18 <= ae1Estimator.getCardinality() && ae1Estimator.getCardinality() <= 20);
241 
242     BinaryNode ae1b = new JoinEmbeddingsNode(ae1, bNode, Lists.newArrayList("b"),
243       MatchStrategy.HOMOMORPHISM, MatchStrategy.HOMOMORPHISM);
244     TemporalQueryPlanEstimator
245       ae1bEstimator = new TemporalQueryPlanEstimator(new QueryPlan(ae1b), queryHandler, stats, est);
246     assertTrue(18 <= ae1bEstimator.getCardinality() &&
247       ae1bEstimator.getCardinality() <= 20);
248 
249     BinaryNode ce2 = new JoinEmbeddingsNode(cNode, e2Node, Lists.newArrayList("c"),
250       MatchStrategy.HOMOMORPHISM, MatchStrategy.HOMOMORPHISM);
251     BinaryNode ce2d = new JoinEmbeddingsNode(ce2, dNode, Lists.newArrayList("d"),
252       MatchStrategy.HOMOMORPHISM, MatchStrategy.HOMOMORPHISM);
253 
254     CartesianProductNode crossNode = new CartesianProductNode(ae1b, ce2d, MatchStrategy
255       .HOMOMORPHISM, MatchStrategy.HOMOMORPHISM);
256 
257     QueryPlan ce2dPlan = new QueryPlan(ce2d);
258     TemporalQueryPlanEstimator
259       ce2dEstimator = new TemporalQueryPlanEstimator(ce2dPlan, queryHandler, stats, est);
260     assertTrue(18 <= ce2dEstimator.getCardinality() &&
261       ce2dEstimator.getCardinality() <= 20);
262 
263     QueryPlan crossPlan = new QueryPlan(crossNode);
264     TemporalQueryPlanEstimator
265       crossEstimator = new TemporalQueryPlanEstimator(crossPlan, queryHandler, stats, est);
266 
267     // join predicate "a.tx_from > c.val_from" has selectivity ~50%
268     long withoutPredicate = ce2dEstimator.getCardinality() * ae1bEstimator.getCardinality();
269     assertTrue(0.45 * withoutPredicate <= crossEstimator.getCardinality() &&
270       crossEstimator.getCardinality() <= 0.55 * withoutPredicate);
271   }
