138     Properties properties = Properties.create();
139     properties.set("name", "Alice");
140     TemporalEdge
141       edge = new TemporalEdgeFactory().createEdge("Label", GradoopId.get(), GradoopId.get(), properties);
142     Tuple2<Long, Long> transactionTime = new Tuple2<>(9876L, 98765L);
143     Tuple2<Long, Long> validTime = new Tuple2<>(123L, 54321L);
144     edge.setTransactionTime(transactionTime);
145     edge.setValidTime(validTime);
146 
147     DataSet<TemporalEdge> edges = getExecutionEnvironment().fromElements(edge);
148 
149     List<Embedding> result =
150       new FilterAndProjectTemporalEdges(edges, predicates, new ArrayList<>(), false)
151         .evaluate()
152         .collect();
153 
154     assertEquals(edge.getSourceId(), result.get(0).getId(0));
155     assertEquals(edge.getId(), result.get(0).getId(1));
156     assertEquals(edge.getTargetId(), result.get(0).getId(2));
157     assertEquals(edge.getValidTime(), validTime);
158     assertEquals(edge.getTransactionTime(), transactionTime);
