48    @Test
49    public void testLabelFree() throws Exception {
50      String query = "MATCH (n)-[e]->(m) WHERE n.tx_from.before(m.tx_from)";
51  
52      TemporalQueryHandler queryHandler = new TemporalQueryHandler(query);
53  
54      LeafNode nNode = new FilterAndProjectTemporalVerticesNode(null, "n",
55        queryHandler.getPredicates().getSubCNF("n"), Sets.newHashSet());
56      LeafNode mNode = new FilterAndProjectTemporalVerticesNode(null, "m",
57        queryHandler.getPredicates().getSubCNF("m"), Sets.newHashSet());
58      LeafNode eNode = new FilterAndProjectTemporalEdgesNode(null, "n", "e", "m",
59        queryHandler.getPredicates().getSubCNF("e"), Sets.newHashSet(), false);
60  
61      JoinEmbeddingsNode neJoin = new JoinEmbeddingsNode(nNode, eNode, Lists.newArrayList("n"),
62        MatchStrategy.ISOMORPHISM, MatchStrategy.ISOMORPHISM);
63      JoinEmbeddingsNode nemJoin = new JoinEmbeddingsNode(neJoin, mNode, Lists.newArrayList("m"),
64        MatchStrategy.ISOMORPHISM, MatchStrategy.ISOMORPHISM);
65  
66      JoinTemporalEstimator estimator = new JoinTemporalEstimator(queryHandler, stats);
67      estimator.visit(neJoin);
68      estimator.visit(nemJoin);
69  
70      // there are 20 "trips" in the citibike sample graph
71      assertEquals(20L, estimator.getCardinality());
72    }

180   @Test
181   public void testPathVariableLength() throws Exception {
182     // 20+2 such paths (condition always holds)
183     String query = "MATCH (n)-[e*1..2]->(m) WHERE n.tx.overlaps(m.val)";
184 
185     TemporalQueryHandler queryHandler = new TemporalQueryHandler(query);
186     LeafNode nNode = new FilterAndProjectTemporalVerticesNode(null, "n",
187       queryHandler.getPredicates().getSubCNF("n"), Sets.newHashSet());
188     LeafNode mNode = new FilterAndProjectTemporalVerticesNode(null, "m",
189       queryHandler.getPredicates().getSubCNF("m"), Sets.newHashSet());
190     LeafNode eNode = new FilterAndProjectTemporalEdgesNode(null, "n", "e", "m",
191       queryHandler.getPredicates().getSubCNF("e"), Sets.newHashSet(), false);
192 
193     ExpandEmbeddingsNode neJoin = new ExpandEmbeddingsNode(nNode, eNode, "n", "e", "m", 1, 10,
194       ExpandDirection.OUT, MatchStrategy.ISOMORPHISM, MatchStrategy.ISOMORPHISM);
195     JoinEmbeddingsNode nemJoin = new JoinEmbeddingsNode(neJoin, mNode, Lists.newArrayList("m"),
196       MatchStrategy.ISOMORPHISM, MatchStrategy.ISOMORPHISM);
197 
198     JoinTemporalEstimator estimator = new JoinTemporalEstimator(queryHandler, stats);
199     estimator.visit(neJoin);
200     estimator.visit(nemJoin);
201     // 24 1-edge paths + 10 2-edge paths
202     assertEquals(22L, estimator.getCardinality());
203   }
