39    @Test
40    public void testEvaluationReturnsLongValue() {
41      //------------------------
42      // test selector for each time field
43      //------------------------
44      TimeSelector
45        selector1 = new TimeSelector("a", "tx_from");
46      TimeSelectorComparable wrapper1 = new TimeSelectorComparable(selector1);
47  
48      TimeSelector
49        selector2 = new TimeSelector("a", "tx_to");
50      TimeSelectorComparable wrapper2 = new TimeSelectorComparable(selector2);
51  
52      TimeSelector
53        selector3 = new TimeSelector("a", "val_from");
54      TimeSelectorComparable wrapper3 = new TimeSelectorComparable(selector3);
55  
56      TimeSelector
57        selector4 = new TimeSelector("a", "val_to");
58      TimeSelectorComparable wrapper4 = new TimeSelectorComparable(selector4);
59  
60      //-----------------------------------------
61      // evaluate on an embedding
62      //-----------------------------------------
63      Embedding embedding = new Embedding();
64      Long txFrom = 1234L;
65      PropertyValue txFromValue = PropertyValue.create(txFrom);
66      Long txTo = 1234567L;
67      PropertyValue txToValue = PropertyValue.create(txTo);
68      Long validFrom = 987L;
69      PropertyValue valFromValue = PropertyValue.create(validFrom);
70      Long validTo = 98765L;
71      PropertyValue valToValue = PropertyValue.create(validTo);
72      embedding.add(GradoopId.get(), txFromValue, txToValue, valFromValue, valToValue);
73  
74      EmbeddingMetaData metaData = new EmbeddingMetaData();
75      metaData.setEntryColumn("a", EmbeddingMetaData.EntryType.VERTEX, 0);
76      metaData.setPropertyColumn("a", TX_FROM.toString(), 0);
77      metaData.setPropertyColumn("a", TX_TO.toString(), 1);
78      metaData.setPropertyColumn("a", VAL_FROM.toString(), 2);
79      metaData.setPropertyColumn("a", VAL_TO.toString(), 3);
80  
81      assertEquals(PropertyValue.create(txFrom), wrapper1.evaluate(embedding, metaData));
82      assertNotEquals(PropertyValue.create(String.valueOf(txFrom)),
83        wrapper1.evaluate(embedding, metaData));
84  
85      assertEquals(PropertyValue.create(txTo), wrapper2.evaluate(embedding, metaData));
86      assertEquals(PropertyValue.create(validFrom), wrapper3.evaluate(embedding, metaData));
87      assertEquals(PropertyValue.create(validTo), wrapper4.evaluate(embedding, metaData));
88  
89      //----------------------------------------------------
90      // evaluate on a GraphElement
91      //----------------------------------------------------
92      TemporalVertex vertex = new TemporalVertexFactory().createVertex();
93      vertex.setTransactionTime(new Tuple2<>(txFrom, txTo));
94      vertex.setValidTime(new Tuple2<>(validFrom, validTo));
95  
96      assertEquals(PropertyValue.create(txFrom), wrapper1.evaluate(vertex));
97      assertEquals(PropertyValue.create(txTo), wrapper2.evaluate(vertex));
98      assertEquals(PropertyValue.create(validFrom), wrapper3.evaluate(vertex));
99      assertEquals(PropertyValue.create(validTo), wrapper4.evaluate(vertex));
100   }
