48    /**
49     * Test grouping using the {@link TimeIntervalKeyFunction} key function.
50     *
51     * @throws Exception when the execution in Flink fails
52     */
53    @Test
54    public void testTimeIntervalKeyFunctionOnGraph() throws Exception {
55      final long testTimeFrom1 = asMillis("2019.01.01 12:00:00.000");
56      final long testTimeTo1 = asMillis("2019.02.01 12:00:00.000");
57      final long testTimeFrom2 = asMillis("2019.01.02 12:00:00.000");
58      final long testTimeTo2 = asMillis("2019.03.01 12:00:00.000");
59      FlinkAsciiGraphLoader loader = getLoaderFromString("input[" +
60        "(:t {__valFrom:" + testTimeFrom1 + "L, __valTo: " + testTimeTo1 + "L, a: 1L})" +
61        "(:t {__valFrom:" + testTimeFrom1 + "L, __valTo: " + testTimeTo1 + "L, a: 2L})" +
62        "(:t {__valFrom:" + testTimeFrom1 + "L, __valTo: " + testTimeTo2 + "L, a: 3L})" +
63        "(:t {__valFrom:" + testTimeFrom1 + "L, __valTo: " + testTimeTo2 + "L, a: 4L})" +
64        "(:t {__valFrom:" + testTimeFrom2 + "L, __valTo: " + testTimeTo1 + "L, a: 5L})" +
65        "(:t {__valFrom:" + testTimeFrom2 + "L, __valTo: " + testTimeTo1 + "L, a: 6L})" +
66        "(:t {__valFrom:" + testTimeFrom2 + "L, __valTo: " + testTimeTo2 + "L, a: 7L})" +
67        "(:t {__valFrom:" + testTimeFrom2 + "L, __valTo: " + testTimeTo2 + "L, a: 8L})" +
68        "]" +
69        "expected [" +
70        "({__valFrom:" + testTimeFrom1 + "L, __valTo: " + testTimeTo1 + "L, min_a: 1L, max_a: 2L, count: 2L})" +
71        "({__valFrom:" + testTimeFrom1 + "L, __valTo: " + testTimeTo2 + "L, min_a: 3L, max_a: 4L, count: 2L})" +
72        "({__valFrom:" + testTimeFrom2 + "L, __valTo: " + testTimeTo1 + "L, min_a: 5L, max_a: 6L, count: 2L})" +
73        "({__valFrom:" + testTimeFrom2 + "L, __valTo: " + testTimeTo2 + "L, min_a: 7L, max_a: 8L, count: 2L})" +
74        "]");
75      TemporalGraph input = toTemporalGraphWithDefaultExtractors(loader.getLogicalGraphByVariable("input"));
76      List<KeyFunction<TemporalVertex, ?>> vertexGroupingKeys = Collections.singletonList(
77        TemporalGroupingKeys.timeInterval(VALID_TIME));
78      List<AggregateFunction> vertexAggregateFunctions = Arrays.asList(
79        new MaxVertexProperty("a", "max_a"),
80        new MinVertexProperty("a", "min_a"),
81        new VertexCount("count"));
82      TemporalGraph result = input.callForGraph(new KeyedGrouping<>(vertexGroupingKeys, vertexAggregateFunctions,
83        Collections.emptyList(), Collections.emptyList()));
84      TemporalGraph expected = toTemporalGraphWithDefaultExtractors(
85        loader.getLogicalGraphByVariable("expected"))
86        .transformVertices((current, transformed) -> {
87          // These properties are not automatically removed when the graph is converted to a temporal graph,
88          // but they are during grouping, since no property grouping keys are given.
89          current.removeProperty("__valFrom");
90          current.removeProperty("__valTo");
91          return current;
92        });
93      collectAndAssertTrue(expected.toLogicalGraph().equalsByElementData(result.toLogicalGraph()));
94    }

96    /**
97     * Test grouping using the {@link TimeStampKeyFunction} key function.
98     *
99     * @throws Exception when the execution in Flink fails
100    */
101   @Test
102   public void testTimeStampKeyFunctionOnGraph() throws Exception {
103     final long testTime1 = asMillis("2019.01.01 01:00:00.000");
104     final long testTime2 = asMillis("2019.01.01 01:00:00.001");
105     final long testTime3 = asMillis("2018.01.01 01:00:00.000");
106     final long testTime4 = asMillis("2019.02.01 01:00:00.000");
107     FlinkAsciiGraphLoader loader = getLoaderFromString("input[" +
108       "(:t {__valFrom: " + testTime1 + "L, __valTo: " + testTime2 + "L, a: 1L})" +
109       "(:t {__valFrom: " + testTime1 + "L, __valTo: " + testTime2 + "L, a: 2L})" +
110       "(:t {__valFrom: " + testTime2 + "L, __valTo: " + testTime2 + "L, a: 3L})" +
111       "(:t {__valFrom: " + testTime2 + "L, __valTo: " + testTime2 + "L, a: 4L})" +
112       "(:t {__valFrom: " + testTime3 + "L, __valTo: " + testTime4 + "L, a: 5L})" +
113       "(:t {__valFrom: " + testTime3 + "L, __valTo: " + testTime4 + "L, a: 6L})" +
114       "(:t {__valFrom: " + testTime4 + "L, __valTo: " + testTime4 + "L, a: 7L})" +
115       "(:t {__valFrom: " + testTime4 + "L, __valTo: " + testTime4 + "L, a: 8L})" +
116       "] expected1 [" +
117       "({time_VALID_TIME_FROM: " + testTime1 + "L, min_a: 1L, max_a: 2L, count: 2L})" +
118       "({time_VALID_TIME_FROM: " + testTime2 + "L, min_a: 3L, max_a: 4L, count: 2L})" +
119       "({time_VALID_TIME_FROM: " + testTime3 + "L, min_a: 5L, max_a: 6L, count: 2L})" +
120       "({time_VALID_TIME_FROM: " + testTime4 + "L, min_a: 7L, max_a: 8L, count: 2L})" +
121       "] expected2 [" +
122       "({time_VALID_TIME_FROM_MonthOfYear: 1L, time_VALID_TIME_FROM_Year: 2019L," +
123       " min_a: 1L, max_a: 4L, count: 4L})" +
124       "({time_VALID_TIME_FROM_MonthOfYear: 1L, time_VALID_TIME_FROM_Year: 2018L," +
125       " min_a: 5L, max_a: 6L, count: 2L})" +
126       "({time_VALID_TIME_FROM_MonthOfYear: 2L, time_VALID_TIME_FROM_Year: 2019L," +
127       " min_a: 7L, max_a: 8L, count: 2L})" +
128       "] expected3 [" +
129       "({time_VALID_TIME_TO_MilliOfSecond: 1L, min_a: 1L, max_a: 4L, count: 4L})" +
130       "({time_VALID_TIME_TO_MilliOfSecond: 0L, min_a: 5L, max_a: 8L, count: 4L})" +
131       "]");
132     List<AggregateFunction> vertexAggregateFunctions = Arrays.asList(
133       new MaxVertexProperty("a", "max_a"),
134       new MinVertexProperty("a", "min_a"),
135       new VertexCount("count"));
136     TemporalGraph input = toTemporalGraphWithDefaultExtractors(loader.getLogicalGraphByVariable("input"));
137     // Test with no TemporalField calculated
138     List<KeyFunction<TemporalVertex, ?>> vertexKeysValidFrom = Collections.singletonList(
139       TemporalGroupingKeys.timeStamp(VALID_TIME, FROM));
140     TemporalGraph expected1 = toTemporalGraph(loader.getLogicalGraphByVariable("expected1"));
141     TemporalGraph result1 = input.callForGraph(new KeyedGrouping<>(vertexKeysValidFrom, vertexAggregateFunctions,
142       Collections.emptyList(), Collections.emptyList()));
143     collectAndAssertTrue(result1.toLogicalGraph().equalsByElementData(expected1.toLogicalGraph()));
144     // Test with two TemporalFields calculated
145     List<KeyFunction<TemporalVertex, ?>> vertexKeysValidFrom2 = Arrays.asList(
146       TemporalGroupingKeys.timeStamp(VALID_TIME, FROM, ChronoField.MONTH_OF_YEAR),
147       TemporalGroupingKeys.timeStamp(VALID_TIME, FROM, ChronoField.YEAR));
148     TemporalGraph expected2 = toTemporalGraph(loader.getLogicalGraphByVariable("expected2"));
149     TemporalGraph result2 = input.callForGraph(new KeyedGrouping<>(vertexKeysValidFrom2,
150       vertexAggregateFunctions, Collections.emptyList(), Collections.emptyList()));
151     collectAndAssertTrue(result2.toLogicalGraph().equalsByElementData(expected2.toLogicalGraph()));
152     // Test with validTo time
153     List<KeyFunction<TemporalVertex, ?>> vertexKeysValidTo = Collections.singletonList(
154       TemporalGroupingKeys.timeStamp(VALID_TIME, TO, ChronoField.MILLI_OF_SECOND));
155     TemporalGraph expected3 = toTemporalGraph(loader.getLogicalGraphByVariable("expected3"));
156     TemporalGraph result3 = input.callForGraph(new KeyedGrouping<>(vertexKeysValidTo, vertexAggregateFunctions,
157       Collections.emptyList(), Collections.emptyList()));
158     collectAndAssertTrue(result3.toLogicalGraph().equalsByElementData(expected3.toLogicalGraph()));
159   }

167   public void testDurationKeyFunctionOnGraph() throws Exception {

169     final long testTimeTo1 = asMillis("2019.07.20 00:00:00.000");
170     final long testTimeFrom2 = asMillis("2020.08.01 12:00:00.000");

172     final long testTimeFrom3 = asMillis("2019.01.01 12:00:00.000");
173     final long testTimeTo3 = asMillis("2019.01.01 12:30:00.000");
174     final long testTimeFrom4 = asMillis("2019.01.01 18:10:10.100");
175     final long testTimeTo4 = asMillis("2019.01.01 18:40:10.110");
176     FlinkAsciiGraphLoader loader = getLoaderFromString("input[" +
177       "(:t {__valFrom: " + testTimeFrom1 + "L, __valTo: " + testTimeTo1 + "L, a: 1L})" +
178       "(:t {__valFrom: " + testTimeFrom2 + "L, __valTo: " + testTimeTo2 + "L, a: 2L})" +
179       "] expected1 [" +
180       "({duration_VALID_TIME_Months: 3L, min_a: 1L, max_a: 2L, count: 2L})" +
181       "] expected2 [" +
182       "({duration_VALID_TIME_Days: 91L, min_a: 1L, max_a: 1L, count: 1L})" +
183       "({duration_VALID_TIME_Days: 93L, min_a: 2L, max_a: 2L, count: 1L})" +
184       "]" +
185       "input2 [" +
186       "(:t {__valFrom: " + testTimeFrom3 + "L, __valTo: " + testTimeTo3 + "L, a: 1L})" +
187       "(:t {__valFrom: " + testTimeFrom4 + "L, __valTo: " + testTimeTo4 + "L, a: 2L})" +
188       "] expected3 [" +
189       "({duration_VALID_TIME_Minutes: 30L, min_a: 1L, max_a: 2L, count: 2L})" +
190       "] expected4 [" +
191       "({duration_VALID_TIME_Millis: 1800000L, min_a: 1L, max_a: 1L, count: 1L})" +
192       "({duration_VALID_TIME_Millis: 1800010L, min_a: 2L, max_a: 2L, count: 1L})" +
193       "]");
194     List<AggregateFunction> vertexAggregateFunctions = Arrays.asList(
195       new MaxVertexProperty("a", "max_a"),
196       new MinVertexProperty("a", "min_a"),
197       new VertexCount("count"));
198     TemporalGraph input = toTemporalGraphWithDefaultExtractors(loader.getLogicalGraphByVariable("input"));
199     List<KeyFunction<TemporalVertex, ?>> vertexKeysMonths = Collections.singletonList(
200       TemporalGroupingKeys.duration(VALID_TIME, ChronoUnit.MONTHS));
201     List<KeyFunction<TemporalVertex, ?>> vertexKeysDays = Collections.singletonList(
