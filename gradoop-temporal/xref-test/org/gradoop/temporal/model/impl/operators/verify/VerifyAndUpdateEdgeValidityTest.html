36  /**
37   * Test for the {@link VerifyAndUpdateEdgeValidity} operator.
38   */
39  public class VerifyAndUpdateEdgeValidityTest extends TemporalGradoopTestBase {
40  
41    /**
42     * Test the operator on a graph.
43     *
44     * @throws Exception when the execution in Flink fails.
45     */
46    @Test
47    public void testVerify() throws Exception {
48      final TemporalVertexFactory vf = getVertexFactory();
49      final TemporalEdgeFactory ef = getEdgeFactory();
50      final long validFromMax = asMillis("2019.03.01 12:00:00.000");
51      final long validToMin = asMillis("2019.06.01 12:00:00.000");
52      // Some vertices.
53      TemporalVertex v1 = vf.createVertex();
54      v1.setValidFrom(asMillis("2019.01.01 12:00:00.000"));
55      v1.setValidTo(validToMin);
56      TemporalVertex v2 = vf.createVertex();
57      v2.setValidFrom(validFromMax);
58      v2.setValidTo(asMillis("2019.09.01 12:00:00.000"));
59      // Some edges.
60      TemporalEdge e1 = ef.createEdge(v1.getId(), v2.getId());
61      TemporalEdge e2 = ef.createEdge(v1.getId(), v1.getId());
62      e2.setValidFrom(asMillis("2019.04.01 12:00:00.000"));
63      e2.setValidTo(asMillis("2019.10.01 12:00:00.000"));
64      TemporalEdge e3 = ef.createEdge(v2.getId(), v2.getId());
65      e3.setValidFrom(asMillis("2018.01.01 12:00:00.000"));
66      e3.setValidTo(asMillis("2019.05.01 12:00:00.000"));
67      // Some dangling edges.
68      TemporalEdge ed1 = ef.createEdge(v1.getId(), GradoopId.get());
69      TemporalEdge ed2 = ef.createEdge(GradoopId.get(), v2.getId());
70      TemporalEdge ed3 = ef.createEdge(GradoopId.get(), GradoopId.get());
71  
72      // Create graph and verify.
73      DataSet<TemporalVertex> vertexSet = getExecutionEnvironment().fromElements(v1, v2);
74      DataSet<TemporalEdge> edgeSet = getExecutionEnvironment().fromElements(e1, e2, e3, ed1, ed2, ed3);
75      TemporalGraph graph = getConfig().getTemporalGraphFactory().fromDataSets(vertexSet, edgeSet);
76      TemporalGraph result = graph.callForGraph(new VerifyAndUpdateEdgeValidity());
77  
78      // Get result.
79      List<TemporalVertex> resultVertices = new ArrayList<>();
80      List<TemporalEdge> resultEdges = new ArrayList<>();
81      result.getVertices().output(new LocalCollectionOutputFormat<>(resultVertices));
82      result.getEdges().output(new LocalCollectionOutputFormat<>(resultEdges));
83      getExecutionEnvironment().execute();
84  
85      assertEquals(resultVertices.size(), 2);
86      assertEquals(resultEdges.size(), 3);
87      // Check if correct edges were removed.
88      assertEquals(resultEdges.stream().map(TemporalEdge::getId).sorted().toArray(),
89        Stream.of(e1, e2, e3).map(TemporalEdge::getId).sorted().toArray());
90      // Check if times were updated correctly.
91      for (TemporalEdge resultEdge : resultEdges) {
92        if (resultEdge.getId().equals(e1.getId())) {
93          assertEquals((long) resultEdge.getValidFrom(), validFromMax);
94          assertEquals((long) resultEdge.getValidTo(), validToMin);
95        } else if (resultEdge.getId().equals(e2.getId())) {
96          assertEquals(resultEdge.getValidFrom(), e2.getValidFrom());
97          assertEquals((long) resultEdge.getValidTo(), validToMin);
98        } else if (resultEdge.getId().equals(e3.getId())) {
99          assertEquals((long) resultEdge.getValidFrom(), validFromMax);
100         assertEquals(resultEdge.getValidTo(), e3.getValidTo());
101       } else {
102         fail();
103       }
104     }
105   }
