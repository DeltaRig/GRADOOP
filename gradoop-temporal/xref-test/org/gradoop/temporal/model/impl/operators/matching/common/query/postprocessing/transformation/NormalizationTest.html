35  public class NormalizationTest {
36  
37    TimeSelector ts1 = new TimeSelector("a", TimeSelector.TimeField.TX_FROM);
38    TimeSelector ts2 = new TimeSelector("b", TimeSelector.TimeField.TX_TO);
39  
40    Comparison lte = new Comparison(ts1, LTE, ts2);
41    Comparison lt = new Comparison(ts1, LT, ts2);
42    Comparison eq = new Comparison(ts1, EQ, ts2);
43    Comparison neq = new Comparison(ts1, NEQ, ts2);
44    Comparison gte = new Comparison(ts1, GTE, ts2);
45    Comparison gt = new Comparison(ts1, GT, ts2);
46  
47    Normalization normalization = new Normalization();
48  
49    @Test
50    public void singleClauseTest() {
51      // for GT or GTE, sides should be switched
52      CNF gteCNF = Util.cnfFromLists(Collections.singletonList(gte));
53      CNF gteExpected = Util.cnfFromLists(Collections.singletonList(gte.switchSides()));
54      assertEquals(gteExpected, normalization.transformCNF(gteCNF));
55  
56      CNF gtCNF = Util.cnfFromLists(Collections.singletonList(gt));
57      CNF gtExpected = Util.cnfFromLists(Collections.singletonList(gt.switchSides()));
58      assertEquals(gtExpected, normalization.transformCNF(gtCNF));
59  
60      // all other comparisons should be left unchanged
61      CNF eqCNF = Util.cnfFromLists(Collections.singletonList(eq));
62      assertEquals(eqCNF, normalization.transformCNF(eqCNF));
63  
64      CNF neqCNF = Util.cnfFromLists(Collections.singletonList(neq));
65      assertEquals(neqCNF, normalization.transformCNF(neqCNF));
66  
67      CNF ltCNF = Util.cnfFromLists(Collections.singletonList(lt));
68      assertEquals(ltCNF, normalization.transformCNF(ltCNF));
69  
70      CNF lteCNF = Util.cnfFromLists(Collections.singletonList(lte));
71      assertEquals(lteCNF, normalization.transformCNF(lteCNF));
72    }
73  
74    @Test
75    public void complexCNFTest() {
76      // input
77      ArrayList<Comparison> clause1 = new ArrayList<>(Arrays.asList(eq, gt, lte));
78      ArrayList<Comparison> clause2 = new ArrayList<>(Arrays.asList(neq, lt));
79      ArrayList<Comparison> clause3 = new ArrayList<>(Collections.singletonList(gte));
80      CNF input = Util.cnfFromLists(clause1, clause2, clause3);
81  
82      // expected
83      ArrayList<Comparison> expectedClause1 = new ArrayList<>(Arrays.asList(eq, gt.switchSides(), lte));
84      ArrayList<Comparison> expectedClause2 = new ArrayList<>(Arrays.asList(neq, lt));
85      ArrayList<Comparison> expectedClause3 = new ArrayList<>(Collections.singletonList(gte.switchSides()));
86      CNF expected = Util.cnfFromLists(expectedClause1, expectedClause2, expectedClause3);
87  
88      assertEquals(expected, normalization.transformCNF(input));
89    }
