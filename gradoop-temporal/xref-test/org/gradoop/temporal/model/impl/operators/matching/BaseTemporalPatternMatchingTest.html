36  /**
37   * Base class for all temporal pattern matching tests that read Temporal-GDL as input db.
38   */
39  @RunWith(Parameterized.class)
40  public abstract class BaseTemporalPatternMatchingTest extends TemporalGradoopTestBase {
41  
42    /**
43     * Name of the test.
44     */
45    protected final String testName;
46  
47    /**
48     * The query as Temporal-GDL string.
49     */
50    protected final String queryGraph;
51  
52    /**
53     * Expected graph variables (names).
54     */
55    protected final String[] expectedGraphVariables;
56  
57    /**
58     * Expected graph collection as comma-separated Temporal-GDLs.
59     */
60    protected final String expectedCollection;
61  
62    private static TemporalGraphStatistics temporalGraphStatistics;
63  
64    @BeforeClass
65    public static void setUp() throws Exception {
66      ExecutionEnvironment environment = ExecutionEnvironment.getExecutionEnvironment();
67      TemporalGradoopConfig config = TemporalGradoopConfig.createConfig(environment);
68      FlinkAsciiGraphLoader loader = new FlinkAsciiGraphLoader(config);
69      loader.initDatabaseFromFile(TEMPORAL_GDL_CITIBIKE_PATH);
70      TemporalGraph temporalGraph = TemporalGraph.fromGraph(loader.getLogicalGraph());
71      temporalGraph = config.getTemporalGraphFactory()
72        .fromDataSets(
73          temporalGraph.getVertices().map(vertexTransform),
74          temporalGraph.getEdges().map(edgeTransform));
75      temporalGraphStatistics = new BinningTemporalGraphStatisticsFactory()
76        .fromGraph(temporalGraph);
77    }
78  
79    /**
80     * Initializes a test with a data graph.
81     *
82     * @param testName               name of the test
83     * @param queryGraph             the query graph as Temporal-GDL string
84     * @param expectedGraphVariables expected graph variables (names) as comma-separated string
85     * @param expectedCollection     expected graph collection as comma-separated GDLs
86     */
87    public BaseTemporalPatternMatchingTest(String testName, String queryGraph, String expectedGraphVariables,
88      String expectedCollection) {
89      this.testName = testName;
90      this.queryGraph = queryGraph;
91      this.expectedGraphVariables = expectedGraphVariables.split(",");
92      this.expectedCollection = expectedCollection;
93    }
94  
95    /**
96     * Given a graph collection, this method transforms it to a temporal graph collection.
97     * {@code start} and {@code end} values are extracted from the edges (= "trips")
98     * and used to set {@code valid_from}/{@code tx_from} and {@code valid_to}/{@code tx_to} values.
99     *
100    * @param graphCollection the collection to transform
101    * @return the temporal representation
102    */
103   protected TemporalGraphCollection transformExpectedToTemporal(GraphCollection graphCollection) {
104     //transform edges
105     TemporalGraphCollection tgc = toTemporalGraphCollection(graphCollection);
106     DataSet<TemporalEdge> newEdges = tgc.getEdges().map(edgeTransform);
107     DataSet<TemporalVertex> newVertices = tgc.getVertices().map(vertexTransform);
108     tgc = tgc.getFactory().fromDataSets(tgc.getGraphHeads(), newVertices, newEdges);
109     return tgc;
110   }
111 
112   /**
113    * Yields a pattern matching implementation.
114    *
115    * @param queryGraph query graph as GDL string
116    * @param stats graph statistics
117    * @return a pattern matching implementation
118    */
119   public abstract CypherTemporalPatternMatching getImplementation(String queryGraph, TemporalGraphStatistics stats) throws Exception;
120 
121   @Test
122   public void testPatternMatching() throws Exception {
123     FlinkAsciiGraphLoader loader = new FlinkAsciiGraphLoader(getConfig());
124 
125     loader.initDatabaseFromFile(TEMPORAL_GDL_CITIBIKE_PATH);
126 
127     TemporalGraph db = transformToTemporalGraph(loader.getLogicalGraph());
128 
129     loader.appendToDatabaseFromString(expectedCollection);
130 
131     TemporalGraphCollection result = getImplementation(queryGraph, temporalGraphStatistics)
132       .execute(db);
133 
134     TemporalGraphCollection expectedByID = toTemporalGraphCollection(
135       loader.getGraphCollectionByVariables(expectedGraphVariables));
136 
137     TemporalGraphCollection expectedByData = transformExpectedToTemporal(
138       loader.getGraphCollectionByVariables(expectedGraphVariables));
139 
140     // element id equality
141     collectAndAssertTrue(result.equalsByGraphElementIds(expectedByID));
142     // graph element equality
143     collectAndAssertTrue(result.equalsByGraphElementData(expectedByData));
144   }
