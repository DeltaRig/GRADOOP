30  /**
31   * Test class to check the unit converting functionality of the {@link MinDuration} and {@link MaxDuration}
32   * aggregation classes.
33   */
34  public class MinMaxDurationWithTemporalUnitTest extends TemporalGradoopTestBase {
35  
36    /**
37     * Provides test data to check several unit conversions.
38     *
39     * @return Object[][] with several postAggregate inputs, units and expected outputs.
40     */
41    @DataProvider(name = "units")
42    public static Object[][] parameters() {
43      return new Object[][] {
44        new Object[] {PropertyValue.create(0L), ChronoUnit.MILLIS, PropertyValue.create(0L)},
45        new Object[] {PropertyValue.create(123456789L), ChronoUnit.MILLIS, PropertyValue.create(123456789L)},
46        new Object[] {PropertyValue.create(123456789L), ChronoUnit.SECONDS, PropertyValue.create(123456.789)},
47        new Object[] {PropertyValue.create(123456789L), ChronoUnit.MINUTES, PropertyValue.create((double) 123456789 / (1000 * 60))},
48        new Object[] {PropertyValue.create(123456789L), ChronoUnit.HOURS, PropertyValue.create((double) 123456789 / (1000 * 60 * 60))},
49        new Object[] {PropertyValue.create(123456789L), ChronoUnit.DAYS, PropertyValue.create((double) 123456789 / (1000 * 60 * 60 * 24))}
50      };
51    }

53    /**
54     * Tests the {@link MinDuration#postAggregate(PropertyValue)} function.
55     *
56     * @param aggregatedValue the aggregated value as input of the postAggregate function
57     * @param temporalUnit the temporal unit to test
58     * @param expectedOutputValue the expected output of the postAggregate function
59     */
60    @Test(dataProvider = "units")
61    public void testMinDurationPostAggregate(PropertyValue aggregatedValue, TemporalUnit temporalUnit,
62      PropertyValue expectedOutputValue) {
63      MinDuration minDuration = new MinDuration("myKey", VALID_TIME, temporalUnit);
64      PropertyValue resultValue = minDuration.postAggregate(aggregatedValue);
65      assertEquals(expectedOutputValue, resultValue);
66    }
67  
68    /**
69     * Test the expected exception by providing a unsupported property value type as input.
70     */
71    @Test(expectedExceptions = { UnsupportedTypeException.class }, expectedExceptionsMessageRegExp =
72      "The result type of the min duration aggregation must be \\[long\\], but is \\[null\\].")
73    public void testMinDurationPostAggregateWrongType() {
74      MinDuration minDuration = new MinDuration("myKey", VALID_TIME, ChronoUnit.DAYS);
75      minDuration.postAggregate(PropertyValue.NULL_VALUE);
76    }
77  
78    /**
79     * Test if the postAggregate returns a null property if a default timestamp is given.
80     */
81    @Test
82    public void testMinDurationPostAggregateDefaultResult() {
83      MinDuration minDuration = new MinDuration("myKey", VALID_TIME, ChronoUnit.DAYS);
84      PropertyValue resultValue = minDuration.postAggregate(PropertyValue.create(
85        TemporalElement.DEFAULT_TIME_TO));
86      assertEquals(PropertyValue.NULL_VALUE, resultValue);
87    }
