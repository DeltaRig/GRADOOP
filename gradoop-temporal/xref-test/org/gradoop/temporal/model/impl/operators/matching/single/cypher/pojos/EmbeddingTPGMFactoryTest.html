38    @Test
39    public void testFromVertex() {
40      Properties properties = new Properties();
41      properties.set("foo", 1);
42      properties.set("bar", "42");
43      properties.set("baz", false);
44      TemporalVertex vertex = new TemporalVertexFactory().createVertex("TestVertex", properties);
45      Long valFrom = 123456L;
46      Long txTo = 124L;
47      Tuple2<Long, Long> tx = new Tuple2<>(123L, txTo);
48      vertex.setTransactionTime(tx);
49      Tuple2<Long, Long> val = new Tuple2<>(valFrom, 12456787L);
50      vertex.setValidTime(val);
51  
52      Embedding embedding =
53        EmbeddingTPGMFactory.fromVertex(vertex, Lists.newArrayList("foo", "bar", VAL_FROM.toString(),
54          TX_TO.toString()));
55  
56      assertEquals(1, embedding.size());
57      assertEquals(vertex.getId(), embedding.getId(0));
58      assertEquals(PropertyValue.create(1), embedding.getProperty(0));
59      assertEquals(PropertyValue.create("42"), embedding.getProperty(1));
60      assertEquals(PropertyValue.create(valFrom), embedding.getProperty(2));
61      assertEquals(PropertyValue.create(txTo), embedding.getProperty(3));
62    }

64    @Test
65    public void testFromEdge() {
66      Properties properties = new Properties();
67      properties.set("foo", 1);
68      properties.set("bar", "42");
69      properties.set("baz", false);
70      TemporalEdge edge = new TemporalEdgeFactory().createEdge(
71        "TestVertex", GradoopId.get(), GradoopId.get(), properties);
72      Long txFrom = 1253453L;
73      Long valTo = 127834487L;
74      Tuple2<Long, Long> tx = new Tuple2<>(txFrom, 124346557L);
75      edge.setTransactionTime(tx);
76      Tuple2<Long, Long> val = new Tuple2<>(13456L, valTo);
77      edge.setValidTime(val);
78  
79      Embedding embedding =
80        EmbeddingTPGMFactory.fromEdge(edge, Lists.newArrayList(
81          TX_FROM.toString(), "foo", VAL_TO.toString(), "bar"), false);
82  
83      assertEquals(3, embedding.size());
84      assertEquals(edge.getSourceId(), embedding.getId(0));
85      assertEquals(edge.getId(), embedding.getId(1));
86      assertEquals(edge.getTargetId(), embedding.getId(2));
87      assertEquals(PropertyValue.create(txFrom), embedding.getProperty(0));
88      assertEquals(PropertyValue.create(1), embedding.getProperty(1));
89      assertEquals(PropertyValue.create(valTo), embedding.getProperty(2));
90      assertEquals(PropertyValue.create("42"), embedding.getProperty(3));
