55    Comparison tlCont3 = new Comparison(tl2, NEQ, tl2);
56    Comparison tlCont4 = new Comparison(tl1, EQ, tl2);

64    Comparison lCont3 = new Comparison(l2, NEQ, l2);

66    Comparison lNonCont1 = new Comparison(l1, NEQ, l2);

79        Arrays.asList(tlNonCont2, tlCont2),

81        Arrays.asList(tlCont4, tlNonCont1, tsCont2),
82        Arrays.asList(tsCont2, tsCont3, tsNonCont2),
83        Arrays.asList(tsCont3, tsNonCont1),
84        Arrays.asList(tlNonCont2, lNonCont2),

94        Collections.singletonList(lNonCont2),
95        Collections.singletonList(lNonCont2),
96        Collections.singletonList(tlNonCont2),
97        Collections.singletonList(tsNonCont1),
98        Collections.singletonList(tlNonCont1),
99        Collections.singletonList(tsNonCont2),
100       Collections.singletonList(tsNonCont1),
101       Arrays.asList(tlNonCont2, lNonCont2),
102       Collections.singletonList(tlNonCont2),
103       Collections.singletonList(tlNonCont2),
104       Arrays.asList(tlNonCont1, lNonCont2)
105     );
