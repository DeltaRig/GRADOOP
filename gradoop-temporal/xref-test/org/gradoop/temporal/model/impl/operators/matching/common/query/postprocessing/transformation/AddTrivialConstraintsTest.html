39    final TimeSelector aTxTo = new TimeSelector("a", TimeSelector.TimeField.TX_TO);
40    final TimeSelector bValFrom = new TimeSelector("b", TimeSelector.TimeField.VAL_FROM);
41    final TimeSelector bValTo = new TimeSelector("b", TimeSelector.TimeField.VAL_TO);
42    final TimeSelector bTxFrom = new TimeSelector("b", TimeSelector.TimeField.TX_FROM);
43    final TimeSelector bTxTo = new TimeSelector("b", TimeSelector.TimeField.TX_TO);

45    final TimeLiteral lit1 = new TimeLiteral("1970-01-01T23:23:23");

50    final AddTrivialConstraints constraintAdder = new AddTrivialConstraints();
51  
52    @Test
53    public void addTrivialTest1() throws QueryContradictoryException {
54      CNF cnf = Util.cnfFromLists(
55        Collections.singletonList(new Comparison(aTxFrom, NEQ, bTxTo)),
56        Collections.singletonList(new Comparison(aTxTo, LTE, bValFrom)),
57        Collections.singletonList(new Comparison(aTxTo, LTE, lit3)),
58        Collections.singletonList(new Comparison(lit1, LTE, bValFrom)),
59        Collections.singletonList(new Comparison(lit2, LTE, bValFrom))
60      );
61      CNF expected = new CNF(cnf).and(Util.cnfFromLists(
62        Collections.singletonList(new Comparison(aTxFrom, LTE, aTxTo)),
63        Collections.singletonList(new Comparison(bTxFrom, LTE, bTxTo)),
64        Collections.singletonList(new Comparison(bValFrom, LTE, bValTo)),
65        Collections.singletonList(new Comparison(lit1, LT, lit2)),
66        Collections.singletonList(new Comparison(lit1, LT, lit3)),
67        Collections.singletonList(new Comparison(lit2, LT, lit3))
68      ));
69  
70      assertEquals(expected, constraintAdder.transformCNF(cnf));
71    }
