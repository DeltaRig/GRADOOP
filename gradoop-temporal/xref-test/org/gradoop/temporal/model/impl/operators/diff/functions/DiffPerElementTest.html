113   /**
114    * Run a {@link FlatMapFunction} on a single elements, store it's results in a {@link List}.
115    * This is a helper function used get the result of a flat map operation.
116    *
117    * @param function The function to test.
118    * @param input    The value to be used as the input for that function.
119    * @param <E>      The argument and result type of the function.
120    * @return         A list of results of the flat map operation.
121    * @throws Exception when the flat map operation throws an exception.
122    */
123   private <E> List<E> runFlatMapFunction(FlatMapFunction<E, E> function, E input) throws Exception {
124     List<E> results = new ArrayList<>();
125     function.flatMap(input, new ListCollector<>(results));
126     return results;
127   }

129   /**
130    * Test the map function on some test elements.
131    * This will try all possible outcomes of the diff and check if the property value is set accordingly.
132    *
133    * @param elementFactory A supplier used to create the test elements.
134    * @param <E> The temporal element type to test the map function on.
135    * @param dimension The {@link TimeDimension} to compute the diff for.
136    * @throws Exception When the flat map operation throws an exception.
137    */
138   private <E extends TemporalElement> void runTestForElement(Supplier<E> elementFactory, TimeDimension dimension)
139     throws Exception {
140     // The element will be present in both snapshots, it is "equal".
141     E testElement = elementFactory.get();
142     testElement.setValidTime(TEST_TIME);
143     List<E> results = runFlatMapFunction(new DiffPerElement<>(ALL, ALL, dimension), testElement);
144     assertEquals(1, results.size());
145     E result = results.get(0);
146     assertSame(testElement, result); // The function should return the same instance.
147     assertEquals(Diff.VALUE_EQUAL, result.getPropertyValue(Diff.PROPERTY_KEY));
148 
149     // The element will only be present in the first snapshot, it is "removed".
150     testElement = elementFactory.get();
151     testElement.setValidTime(TEST_TIME);
152     results = runFlatMapFunction(new DiffPerElement<>(ALL, NONE, dimension), testElement);
153     assertEquals(1, results.size());
154     result = results.get(0);
155     assertSame(testElement, result);
156     assertEquals(Diff.VALUE_REMOVED, result.getPropertyValue(Diff.PROPERTY_KEY));
157 
158     // The element will only be present in the second snapshot, it is "added".
159     testElement = elementFactory.get();
160     testElement.setValidTime(TEST_TIME);
161     results = runFlatMapFunction(new DiffPerElement<>(NONE, ALL, dimension), testElement);
162     assertEquals(1, results.size());
163     result = results.get(0);
164     assertSame(testElement, result);
165     assertEquals(Diff.VALUE_ADDED, result.getPropertyValue(Diff.PROPERTY_KEY));
166 
167     // The element will be present in neither snapshot, it should be removed.
168     testElement = elementFactory.get();
169     testElement.setValidTime(TEST_TIME);
170     results = runFlatMapFunction(new DiffPerElement<>(NONE, NONE, dimension), testElement);
171     assertEquals(0, results.size());
172   }

174   /**
175    * Test the map function by applying it on a dataset.
176    *
177    * @param elementFactory A supplier used to create the test elements.
178    * @param <E> The temporal element type to test the map function on.
179    * @param dimension The {@link TimeDimension} to base the computation on.
180    * @throws Exception when the execution in Flink fails.
181    */
182   private <E extends TemporalElement> void runTestOnDataSet(Supplier<E> elementFactory, TimeDimension dimension)
183     throws Exception {
184     // A predicate used to check if a range was valid before 0 (unix timestamp).
185     TemporalPredicate beforeEpoch = (from, to) -> from < 0;
186     // A predicate used to check if a range was valid after 0 (unix timestamp).
187     TemporalPredicate afterEpoch = (from, to) -> to > 0;
188     // Create some test elements. Those elements are either
189     // 1. never
190     // 2. only before epoch
191     // 3. only after epoch
192     // 4. before and after epoch
193     // A label is set on each of those elements to distinguish it later.
194     E neverValid = elementFactory.get();
195     neverValid.setLabel("never");
196     neverValid.setValidTime(Tuple2.of(0L, 0L));
197     neverValid.setTransactionTime(Tuple2.of(0L, 0L));
198     E validBeforeEpoch = elementFactory.get();
199     validBeforeEpoch.setLabel("before");
200     validBeforeEpoch.setValidTime(Tuple2.of(-1L, 0L));
201     validBeforeEpoch.setTransactionTime(Tuple2.of(-1L, 0L));
202     E validAfterEpoch = elementFactory.get();
203     validAfterEpoch.setLabel("after");
204     validAfterEpoch.setValidTime(Tuple2.of(0L, 1L));
205     validAfterEpoch.setTransactionTime(Tuple2.of(0L, 1L));
206     E validInBoth = elementFactory.get();
207     validInBoth.setLabel("both");
208     validInBoth.setValidTime(Tuple2.of(-1L, 1L));
209     validInBoth.setTransactionTime(Tuple2.of(-1L, 1L));
210     // A diff is called, comparing the dataset before and after epoch.
211     List<E> result = getExecutionEnvironment()
212       .fromElements(neverValid, validAfterEpoch, validBeforeEpoch, validInBoth)
213       .flatMap(new DiffPerElement<>(beforeEpoch, afterEpoch, dimension)).collect();
214     assertEquals(3, result.size());
215     assertEquals(3, result.stream().map(Element::getLabel).distinct().count());
216     // Check if each of the test elements has the correct property value set.
217     for (E resultElement : result) {
218       switch (resultElement.getLabel()) {
219       case "never":
220         fail("The element not matching either predicate should have been removed.");
221         break;
222       case "both":
223         assertEquals(Diff.VALUE_EQUAL, resultElement.getPropertyValue(Diff.PROPERTY_KEY));
224         break;
225       case "before":
226         assertEquals(Diff.VALUE_REMOVED, resultElement.getPropertyValue(Diff.PROPERTY_KEY));
227         break;
228       case "after":
229         assertEquals(Diff.VALUE_ADDED, resultElement.getPropertyValue(Diff.PROPERTY_KEY));
230         break;
231       default:
232         fail("Unknown label.");
233       }
234     }
235   }
