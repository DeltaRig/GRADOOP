63      MaxDuration function = new MaxDuration("", VALID_TIME);
64      TemporalElement testElement = getVertexFactory().createVertex();
65      testElement.setValidTime(Tuple2.of(DEFAULT_TIME_FROM, DEFAULT_TIME_TO));
66      assertEquals(Long.MIN_VALUE, function.getIncrement(testElement).getLong());
67      testElement.setValidTime(Tuple2.of(DEFAULT_TIME_FROM, 6L));

69      testElement.setValidTime(Tuple2.of(3L, DEFAULT_TIME_TO));
70      assertEquals(Long.MIN_VALUE, function.getIncrement(testElement).getLong());
71      testElement.setValidTime(Tuple2.of(2L, 7L));
72      assertEquals(5L, function.getIncrement(testElement).getLong());

76     * Test the {@link MaxDuration} aggregate function and its subclasses in an aggregation.
77     *
78     * @throws Exception when the execution in Flink fails.
79     */
80    @Test
81    public void testWithMaxAggregation() throws Exception {
82  
83      TemporalGraphFactory graphFactory = getConfig().getTemporalGraphFactory();
84      VertexFactory<TemporalVertex> vertexFactory = graphFactory.getVertexFactory();
85      EdgeFactory<TemporalEdge> edgeFactory = graphFactory.getEdgeFactory();
86      TemporalVertex v1 = vertexFactory.createVertex();
87      v1.setTransactionTime(Tuple2.of(0L, 2000L));
88      v1.setValidTime(Tuple2.of(DEFAULT_TIME_FROM, DEFAULT_TIME_TO));
89      TemporalVertex v2 = vertexFactory.createVertex();
90      v2.setTransactionTime(Tuple2.of(DEFAULT_TIME_FROM, 5L));
91      v2.setValidTime(Tuple2.of(-3L, DEFAULT_TIME_TO));
92      TemporalVertex v3 = vertexFactory.createVertex();
93      v3.setTransactionTime(Tuple2.of(0L, DEFAULT_TIME_TO));
94      v3.setValidTime(Tuple2.of(DEFAULT_TIME_FROM, 0L));
95      TemporalVertex v4 = vertexFactory.createVertex();
96      v4.setTransactionTime(Tuple2.of(DEFAULT_TIME_FROM, DEFAULT_TIME_TO));
97      v4.setValidTime(Tuple2.of(-5L, -2L));
98      TemporalVertex v5 = vertexFactory.createVertex();
99      v5.setTransactionTime(Tuple2.of(1L, 4L));
100     v5.setValidTime(Tuple2.of(-100L, -3L));
101 
102     TemporalEdge e1 = edgeFactory.createEdge(v1.getId(), v2.getId());
103     e1.setTransactionTime(Tuple2.of(0L, 300L));
104     e1.setValidTime(Tuple2.of(DEFAULT_TIME_FROM, DEFAULT_TIME_TO));
105     TemporalEdge e2 = edgeFactory.createEdge(v2.getId(), v3.getId());
106     e2.setTransactionTime(Tuple2.of(DEFAULT_TIME_FROM, 7L));
107     e2.setValidTime(Tuple2.of(-100L, DEFAULT_TIME_TO));
108     TemporalEdge e3 = edgeFactory.createEdge(v3.getId(), v1.getId());
109     e3.setTransactionTime(Tuple2.of(-1L, DEFAULT_TIME_TO));
110     e3.setValidTime(Tuple2.of(DEFAULT_TIME_FROM, -80L));
111     TemporalEdge e4 = edgeFactory.createEdge(v1.getId(), v3.getId());
112     e4.setTransactionTime(Tuple2.of(DEFAULT_TIME_FROM, DEFAULT_TIME_TO));
113     e4.setValidTime(Tuple2.of(-120L, 100L));
114     TemporalEdge e5 = edgeFactory.createEdge(v2.getId(), v4.getId());
115     e5.setTransactionTime(Tuple2.of(10L, 50L));
116     e5.setValidTime(Tuple2.of(-301L, -276L));
117 
118     DataSet<TemporalVertex> vertices = getExecutionEnvironment().fromElements(v1, v2, v3, v4, v5);
119     DataSet<TemporalEdge> edges = getExecutionEnvironment().fromElements(e1, e2, e3, e4, e5);
120     TemporalGraph temporalResult = graphFactory.fromDataSets(vertices, edges)
121       .aggregate(new MaxDuration("maxDurTx", TRANSACTION_TIME),
122         new MaxDuration("maxDurValid", VALID_TIME),
123         new MaxVertexDuration("maxVertexDurTx", TRANSACTION_TIME),
124         new MaxVertexDuration("maxVertexDurValid", VALID_TIME),
125         new MaxEdgeDuration("maxEdgeDurTx", TRANSACTION_TIME),
126         new MaxEdgeDuration("maxEdgeDurValid", VALID_TIME));
127 
128     LogicalGraph result = temporalResult.toLogicalGraph();
129     List<EPGMGraphHead> graphHead = new ArrayList<>();
130     result.getGraphHead().output(new LocalCollectionOutputFormat<>(graphHead));
131     getExecutionEnvironment().execute();
132 
133     Properties headProperties = graphHead.get(0).getProperties();
134     assertEquals(2000L, headProperties.get("maxDurTx").getLong());
135     assertEquals(220L, headProperties.get("maxDurValid").getLong());
136     assertEquals(2000L, headProperties.get("maxVertexDurTx").getLong());
137     assertEquals(97L, headProperties.get("maxVertexDurValid").getLong());
138     assertEquals(300L, headProperties.get("maxEdgeDurTx").getLong());
139     assertEquals(220L, headProperties.get("maxEdgeDurValid").getLong());
140   }

173     v2.setTransactionTime(Tuple2.of(DEFAULT_TIME_FROM, 5L));
174     v2.setValidTime(Tuple2.of(-3L, DEFAULT_TIME_TO));
175     TemporalVertex v3 = vertexFactory.createVertex();
176     v3.setTransactionTime(Tuple2.of(0L, DEFAULT_TIME_TO));
177     v3.setValidTime(Tuple2.of(DEFAULT_TIME_FROM, 0L));
178     TemporalVertex v4 = vertexFactory.createVertex();
179     v4.setTransactionTime(Tuple2.of(DEFAULT_TIME_FROM, DEFAULT_TIME_TO));
180     v4.setValidTime(Tuple2.of(-5L, -2L));
181     TemporalVertex v5 = vertexFactory.createVertex();
182     v5.setTransactionTime(Tuple2.of(1L, 4L));
183     v5.setValidTime(Tuple2.of(-100L, -3L));
184 
185     TemporalEdge e1 = edgeFactory.createEdge(v1.getId(), v2.getId());
186     e1.setTransactionTime(Tuple2.of(0L, 10L));
187     e1.setValidTime(Tuple2.of(DEFAULT_TIME_FROM, DEFAULT_TIME_TO));
188     TemporalEdge e2 = edgeFactory.createEdge(v2.getId(), v3.getId());
189     e2.setTransactionTime(Tuple2.of(DEFAULT_TIME_FROM, 7L));
190     e2.setValidTime(Tuple2.of(-100L, DEFAULT_TIME_TO));
191     TemporalEdge e3 = edgeFactory.createEdge(v3.getId(), v1.getId());
192     e3.setTransactionTime(Tuple2.of(-1L, DEFAULT_TIME_TO));
193     e3.setValidTime(Tuple2.of(DEFAULT_TIME_FROM, -80L));
194     TemporalEdge e4 = edgeFactory.createEdge(v1.getId(), v3.getId());
195     e4.setTransactionTime(Tuple2.of(DEFAULT_TIME_FROM, DEFAULT_TIME_TO));
196     e4.setValidTime(Tuple2.of(-120L, 100L));
197     TemporalEdge e5 = edgeFactory.createEdge(v2.getId(), v4.getId());
198     e5.setTransactionTime(Tuple2.of(10L, 50L));
199     e5.setValidTime(Tuple2.of(-302L, -298L));
200 
201     DataSet<TemporalVertex> vertices = getExecutionEnvironment().fromElements(v1, v2, v3, v4, v5);
202     DataSet<TemporalEdge> edges = getExecutionEnvironment().fromElements(e1, e2, e3, e4, e5);
203     TemporalGraph temporalResult = graphFactory.fromDataSets(vertices, edges)
204       .aggregate(
205         new MinDuration("minDurTx", TRANSACTION_TIME),
206         new MinDuration("minDurValid", VALID_TIME),
207         new MinVertexDuration("minVertexDurTx", TRANSACTION_TIME),
208         new MinVertexDuration("minVertexDurValid", VALID_TIME),
209         new MinEdgeDuration("minEdgeDurTx", TRANSACTION_TIME),
210         new MinEdgeDuration("minEdgeDurValid", VALID_TIME));
211     LogicalGraph result = temporalResult.toLogicalGraph();
212

214     result.getGraphHead().output(new LocalCollectionOutputFormat<>(graphHead));
215     getExecutionEnvironment().execute();
216

218     assertEquals(2L, headProperties.get("minDurTx").getLong());
219     assertEquals(3L, headProperties.get("minDurValid").getLong());
220     assertEquals(2L, headProperties.get("minVertexDurTx").getLong());
221     assertEquals(3L, headProperties.get("minVertexDurValid").getLong());
222     assertEquals(10L, headProperties.get("minEdgeDurTx").getLong());
223     assertEquals(4L, headProperties.get("minEdgeDurValid").getLong());
224   }
