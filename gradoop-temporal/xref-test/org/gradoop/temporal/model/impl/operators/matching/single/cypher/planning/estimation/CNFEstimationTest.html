70    @Test
71    public void timeSelectorComparisonTest() throws Exception {
72      CNFEstimation estimator = getEstimator();
73  
74      //tx_from of v1 equally distributed from 100L to 200L
75      // => 175L is part of the 76th bin
76      // => 24 bins are greater => should yield 0.24
77      Comparison comp1 = new Comparison(aTxFrom, GT, new TimeLiteral(175L));
78      CNFElement e1 =
79        new CNFElement(Collections.singletonList(new ComparisonExpression(comp1, new ComparableTPGMFactory())));
80      CNF cnf1 = new CNF(Collections.singletonList(e1));
81      double estimation1 = estimator.estimateCNF(cnf1);
82      assertEquals(0.24, estimation1, 0.01);
83      // does the cache work?
84      double estimation1Cached = estimator.estimateCNF(cnf1);
85      assertEquals(estimation1, estimation1Cached, .0);
86  
87      // switch sides
88      Comparison comp2 = new Comparison(new TimeLiteral(175L), LT, aTxFrom);
89      CNFElement e2 =
90        new CNFElement(Collections.singletonList(new ComparisonExpression(comp2, new ComparableTPGMFactory())));
91      CNF cnf2 = new CNF(Collections.singletonList(e2));
92      double estimation2 = estimator.estimateCNF(cnf2);
93      assertEquals(estimation1, estimation2, 0.);
94  
95      // comparisons where lhs and rhs are both selectors are estimated to 1.
96      // (no matter how much sense they make)
97      Comparison comp3 = new Comparison(aTxTo, EQ, aTxFrom);
98      CNFElement e3 =
99        new CNFElement(Collections.singletonList(new ComparisonExpression(comp3, new ComparableTPGMFactory())));
100     CNF cnf3 = new CNF(Collections.singletonList(e3));
101     double estimation3 = estimator.estimateCNF(cnf3);
102     assertEquals(1., estimation3, 0.00001);
103   }

336   @Test
337   public void disjunctionTest() throws Exception {
338     CNFEstimation estimator = getEstimator();
339     // ~0.9     (a)
340     Comparison comp1 = new Comparison(eTxDuration, GT, new TimeConstant(10));
341     ComparisonExpression c1 = new ComparisonExpression(comp1, new ComparableTPGMFactory());
342     // ~0.125      (b)
343     Comparison comp2 = new Comparison(
344       new PropertySelector("a", "numProp"),
345       GTE,
346       new PropertySelector("a", "numProp")
347     );
348     ComparisonExpression c2 = new ComparisonExpression(comp2, new ComparableTPGMFactory());
349     // 1.       (c)
350     Comparison comp3 = new Comparison(eTxDuration, LT, eValDuration);
351     ComparisonExpression c3 = new ComparisonExpression(comp3, new ComparableTPGMFactory());
352     // ~0.15    (d)
353     Comparison comp4 = new Comparison(
354       new Literal(50),
355       GTE,
356       new PropertySelector("c", "numProp")
357     );
358     ComparisonExpression c4 = new ComparisonExpression(comp4, new ComparableTPGMFactory());
359 
360     // (a) OR (b) = P(a)+P(b)-P(a)*P(b) = 0.9+0.125 - 0.9*0.125 = 0.9125
361     CNFElement e1 = new CNFElement(Arrays.asList(c1, c2));
362     CNF cnf1 = new CNF(Collections.singletonList(e1));
363     double estimation1 = estimator.estimateCNF(cnf1);
364     assertEquals(0.9125, estimation1, 0.02);
365 
366     // (a) OR (b) OR (d)
367     // = P(a) + P(b) + P(d) - P(a)*P(b) - P(a)*P(d) - P(b)*P(d) + P(a)*P(b)*P(d)
368     // = 0.9+0.125+0.15 - 0.9*0.125 - 0.9*0.15 - 0.125*0.15 + 0.9*0.125*0.15
369     // = 0.925625
370     CNFElement e2 = new CNFElement(Arrays.asList(c1, c2, c4));
371     CNF cnf2 = new CNF(Collections.singletonList(e2));
372     double estimation2 = estimator.estimateCNF(cnf2);
373     assertEquals(0.925, estimation2, 0.02);
374 
375     // (a) OR (b) OR (c) = 1
376     CNFElement e3 = new CNFElement(Arrays.asList(c1, c2, c3));
377     CNF cnf3 = new CNF(Collections.singletonList(e3));
378     double estimation3 = estimator.estimateCNF(cnf3);
379     assertEquals(1., estimation3, 0.01);
380 
381     // conjunction of disjunctions
382     // independence assumption, should thus be ~ 0.9125*0.925*1. = 0.844
383     CNF cnf4 = new CNF(Arrays.asList(e2, e1, e3));
384     double estimation4 = estimator.estimateCNF(cnf4);
385     assertEquals(0.84, estimation4, 0.03);
386 
387     // check reordering of cnf4: sequence should be e1, e2, e3
388     CNFElement e2Reordered = new CNFElement(Arrays.asList(c1, c4, c2));
389     CNFElement e3Reordered = new CNFElement(Arrays.asList(c3, c1, c2));
390     CNF cnf4Reordered = new CNF(Arrays.asList(e1, e2Reordered, e3Reordered));
391     assertEquals(cnf4Reordered, estimator.reorderCNF(cnf4));
392   }

394   /**
395    * Creates a CNFEstimation over a test graph (see {@link this::getDummyStats}.
396    * Variable to label mapping: "a"->"v1", "b"->"v2", e->"edge"
397    * Variable to type mapping "a"->VERTEX, "b"->VERTEX, "c"->VERTEX, "e"->EDGE
398    * ("c" has no label specified)
399    *
400    * @return CNFEstimation object
401    */
402   private CNFEstimation getEstimator() throws Exception {
403     BinningTemporalGraphStatistics stats = getDummyStats();
404 
405     HashMap<String, String> labelMap = new HashMap<>();
406     labelMap.put("a", "v1");
407     labelMap.put("b", "v2");
408     labelMap.put("e", "edge");
409 
410     HashMap<String, TemporalGraphStatistics.ElementType> typeMap = new HashMap<>();
411     typeMap.put("a", TemporalGraphStatistics.ElementType.VERTEX);
412     typeMap.put("b", TemporalGraphStatistics.ElementType.VERTEX);
413     typeMap.put("c", TemporalGraphStatistics.ElementType.VERTEX);
414     typeMap.put("e", TemporalGraphStatistics.ElementType.EDGE);
415 
416     return new CNFEstimation(stats, typeMap, labelMap);
417   }

419   /**
420    * Creates a BinningTemporalGraphStatistics object
421    * Vertices:
422    * 100 "v1" vertices:
423    * "catProp"   = "x" (6 vertices)
424    * = "y" (34 vertices)
425    * "numProp"   = 0,2,...,100 (50 vertices)
426    * "numProp2"  = 0,3,6,...,297 (100 vertices)
427    * tx_from goes from 100L to 200L, val_from from 150L to 250L (100 vertices)
428    * tx_to goes from 300L to 350L for half of the vertices, other half is unbounded
429    * val_to goes from 350L to 450L for half of the vertices, other half is unbounded
430    * <p>
431    * 100 "v2" vertices:
432    * "catProp"   = "y" (20 vertices)
433    * "numProp"   = 0,10,...,90 (10 vertices)
434    * "gender"    = "m" (34 vertices)
435    * = "f" (66 vertices)
436    * tx_from goes from 1000L to 2000L, val_from from 3000L to 4000L (100 vertices)
437    * tx_to goes from 1500L to 2500L (step 20) for half of the vertices, other half is unbounded
438    * val_to goes from 3500L to 4500L for half of the vertices (step 20), other half is unbounded
439    * <p>
440    * Edges: identical tx and val times, their length equally distributed
441    * from 0 to 100L
442    *
443    * @return dummy statistics
444    * @throws Exception if anything flink-related goes wrong
445    */
446   private BinningTemporalGraphStatistics getDummyStats() throws Exception {
447     ArrayList<TemporalVertex> vertexList = new ArrayList<>();
448 
449     // first type of vertex has label1
450     String vLabel1 = "v1";
451     // tx_from goes from 100L to 200L, val_from from 150L to 250L (100 vertices)
452     // tx_to goes from 300L to 350L for half of the vertices, other half is unbounded
453     // val_to goes from 350L to 450L for half of the vertices, other half is unbounded
454     int numL1Vertices = 100;
455     for (int i = 0; i < numL1Vertices; i++) {
456       TemporalVertex vertex = new TemporalVertex();
457       vertex.setId(GradoopId.get());
458       vertex.setLabel(vLabel1);
459 
460       vertex.setTxFrom(100L + i);
461       long txTo = i % 2 == 0 ? 300L + i : Long.MAX_VALUE;
462       vertex.setTxTo(txTo);
463 
464       vertex.setValidFrom(150L + i);
465       long valTo = i % 2 == 0 ? 350L + i : Long.MAX_VALUE;
466       vertex.setValidTo(valTo);
467 
468       // 6 nodes with property value x
469       if (i % 10 == 0) {
470         vertex.setProperty("catProp1", PropertyValue.create("x"));
471       }
472       // 34 nodes with property value y
473       if (i % 3 == 0) {
474         vertex.setProperty("catProp1", PropertyValue.create("y"));
475       }
476 
477       // every second node has i as a property
478       if (i % 2 == 0) {
479         vertex.setProperty("numProp", PropertyValue.create(i));
480       }
481 
482       vertex.setProperty("numProp2", PropertyValue.create(3 * i));
483 
484       vertexList.add(vertex);
485     }
486 
487     // first type of vertex has label1
488     String vLabel2 = "v2";
489     // tx_from goes from 1000L to 2000L, val_from from 3000L to 4000L (100 vertices)
490     // tx_to goes from 1500L to 2500L (step 20) for half of the vertices, other half is unbounded
491     // val_to goes from 3500L to 4500L for half of the vertices (step 20), other half is unbounded
492     int numL2Vertices = 100;
493     for (int i = 0; i < numL2Vertices; i++) {
494       TemporalVertex vertex = new TemporalVertex();
495       vertex.setId(GradoopId.get());
496       vertex.setLabel(vLabel2);
497 
498       vertex.setTxFrom(1000L + i * 10);
499       long txTo = i % 2 == 0 ? 1500L + i * 20 : Long.MAX_VALUE;
500       vertex.setTxTo(txTo);
501 
502       vertex.setValidFrom(3000L + i * 10);
503       long valTo = i % 2 == 0 ? 3500L + i * 20 : Long.MAX_VALUE;
504       vertex.setValidTo(valTo);
505 
506       if (i % 5 == 0) {
507         vertex.setProperty("catProp1", "y");
508       }
509 
510       // every 10th node has i as property
511       if (i % 10 == 0) {
512         vertex.setProperty("numProp", PropertyValue.create(i));
513       }
514       vertexList.add(vertex);
515 
516       if (i % 3 == 0) {
517         vertex.setProperty("gender", PropertyValue.create("m"));
518       } else {
519         vertex.setProperty("gender", PropertyValue.create("f"));
520       }
521     }
522 
523     //edges (only one type)
524     ArrayList<TemporalEdge> edgeList = new ArrayList<>();
525     String edgeLabel = "edge";
526     int numEdges = 100;
527     // identical tx and val times.
528     // lengths are equally distributed from 0 to 100
529     for (int i = 0; i < numEdges; i++) {
530       TemporalEdge edge = new TemporalEdge();
531       edge.setId(GradoopId.get());
