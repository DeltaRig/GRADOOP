76    @Test
77    public void testExecute() throws Exception {
78      GradoopId sourceId = GradoopId.get();
79      GradoopId targetId = GradoopId.get();
80  
81      GradoopId edge1Id = GradoopId.get();
82      Map<String, Object> edge1Props = new HashMap<>();
83      edge1Props.put("foo", 23);
84      Long[] edge1TimeData = new Long[] {123L, 1234L, 345L, 456L};
85  
86      GradoopId edge2Id = GradoopId.get();
87      Map<String, Object> edge2Props = new HashMap<>();
88      edge2Props.put("foo", 42);
89      Long[] edge2TimeData = new Long[] {123L, 1234L, 345L, 456L};
90  
91      GradoopId edge3Id = GradoopId.get();
92      Map<String, Object> edge3Props = new HashMap<>();
93      edge3Props.put("foo", 23);
94      Long[] edge3TimeData = new Long[] {12345L, 1234567L, 3452L, 456789L};
95  
96  
97      TemporalEdge e1 =
98        factory.initEdge(edge1Id, "a", sourceId, targetId, Properties.createFromMap(edge1Props));
99      e1.setTransactionTime(new Tuple2<>(edge1TimeData[0], edge1TimeData[1]));
100     e1.setValidTime(new Tuple2<>(edge1TimeData[2], edge1TimeData[3]));
101 
102     TemporalEdge e2 =
103       factory.initEdge(edge2Id, "b", sourceId, targetId, Properties.createFromMap(edge2Props));
104     e2.setTransactionTime(new Tuple2<>(edge2TimeData[0], edge2TimeData[1]));
105     e2.setValidTime(new Tuple2<>(edge2TimeData[2], edge2TimeData[3]));
106 
107     TemporalEdge e3 =
108       factory.initEdge(edge3Id, "c", sourceId, targetId, Properties.createFromMap(edge3Props));
109     e3.setTransactionTime(new Tuple2<>(edge3TimeData[0], edge3TimeData[1]));
110     e3.setValidTime(new Tuple2<>(edge3TimeData[2], edge3TimeData[3]));
111 
112     DataSet<TemporalEdge> edges = getExecutionEnvironment().fromElements(e1, e2, e3);
113 
114     //only matched by e1
115     String query = "MATCH (a)-[e]->(b) WHERE e.foo = 23 " +
116       "AND a.val_from <= e.val_from";
117 
118     TemporalQueryHandler queryHandler = new TemporalQueryHandler(query);
119     CNF filterPredicate = queryHandler.getPredicates().getSubCNF(Sets.newHashSet("e"));
120     Set<String> projectionKeys = queryHandler.getPredicates().getPropertyKeys("e");
121 
122     FilterAndProjectTemporalEdgesNode node = new FilterAndProjectTemporalEdgesNode(
123       edges, "a", "e", "b", filterPredicate, projectionKeys, false);
124 
125     List<Embedding> filteredEdges = new ArrayList<>();
126     node.execute().output(new LocalCollectionOutputFormat<>(filteredEdges));
127 
128     getExecutionEnvironment().execute();
129 
130     assertEquals(2, filteredEdges.size());
131     assertEquals(sourceId, filteredEdges.get(0).getId(0));
132     assertEquals(edge1Id, filteredEdges.get(0).getId(1));
133     assertEquals(targetId, filteredEdges.get(0).getId(2));
134 
135 
136     EmbeddingMetaData metaData = node.getEmbeddingMetaData();
137     assertEquals(0, metaData.getEntryColumn("a"));
138     assertEquals(1, metaData.getEntryColumn("e"));
139     assertEquals(2, metaData.getEntryColumn("b"));
140     assertEquals(3, metaData.getPropertyKeys("e").size());
141 
142     assertEquals(1, metaData.getPropertyColumn("e", "foo"));
143     assertEquals(2, metaData.getPropertyColumn("e", TimeSelector.TimeField.VAL_FROM.toString()));
144     assertEquals(0, metaData.getPropertyColumn("e", TimeSelector.TimeField.VAL_TO.toString()));
145   }
