54    @Test
55    public void durationStatsTest() {
56      ArrayList<TemporalVertex> input = getInputVertices();
57      for (int i = 0; i < input.size(); i++) {
58        // add some distortion
59        // => mean of tx duration now 2.5, variance (0.5)²=0.25
60        long newTxFrom = i % 2 == 0 ? 0L : 2L;
61        long newTxTo = i % 2 == 0 ? 3L : 4L;
62        // => mean of val duration now 7.5, variance (2.5²) = 6.75
63        long newValFrom = i % 2 == 0 ? 0L : 5L;
64        long newValTo = 10L;
65        input.get(i).setTransactionTime(new Tuple2<>(newTxFrom, newTxTo));
66        input.get(i).setValidTime(new Tuple2<>(newValFrom, newValTo));
67      }
68  
69      TemporalElementStats stats = new TemporalElementStats();
70      for (TemporalVertex vertex : input) {
71        stats.addElement(vertex);
72      }
73  
74      double txDurationMean = stats.getTxDurationStats()[0];
75      double txDurationVariance = stats.getTxDurationStats()[1];
76      assertEquals(2.5, txDurationMean, 0.001);
77      assertEquals(0.25, txDurationVariance, 0.001);
78  
79      double valDurationMean = stats.getValDurationStats()[0];
80      double valDurationVariance = stats.getValDurationStats()[1];
81      assertEquals(7.5, valDurationMean, 0.001);
82      assertEquals(6.25, valDurationVariance, 0.001);
83    }

85    @Test
86    public void txToValToStatsTest() {
87      ArrayList<TemporalVertex> input = getInputVertices();
88      for (int i = 0; i < input.size(); i++) {
89        input.get(i).setTxFrom(0L);
90        input.get(i).setValidFrom(0L);
91        /*
92         * 1/10th of the vertices: both TOs unbounded
93         * 1/10th of the vertices: tx to unbounded
94         * 1/5th : val to unbounded
95         * if val to bounded, val to = 8
96         * half of the unbounded tx to = 8, other half = 16
97         * => mean(tx_to - val_to)(both unbounded) = 4
98         * => variance(tx_to - val_to)(both unbounded) = 16
99         * => P(both unbounded) = 0.8
100        */
101       long newTxTo = i % 10 == 0 ? Long.MAX_VALUE :
102         i % 2 == 0 ? 8L : 16L;
103       long newValTo = i % 5 == 0 ? Long.MAX_VALUE : 8L;
104       input.get(i).setTxTo(newTxTo);
105       input.get(i).setValidTo(newValTo);
106     }
107     TemporalElementStats stats = new TemporalElementStats();
108     for (TemporalVertex vertex : input) {
109       stats.addElement(vertex);
110     }
111   }

113   @Test
114   public void numericalPropertyTest() {
115     ArrayList<TemporalVertex> input = getInputVertices();
116     //----------------------------------------
117     // ----------------  DATA ----------------
118     //-----------------------------------------
119 
120     //----------------------------------------------------
121     // Float values
122     //----------------------------------------------------
123 
124     // 1% with float property
125     int numFloatDecimals = 50;
126     String decimalKeyFloat = "float";
127 
128     // mean = 1.3f: half of the values have value 1.2, the other half 1.4
129     // variance is then (0.1)*(0.1) = 0.01
130     for (int i = 0; i < numFloatDecimals; i++) {
131       float value = i % 2 == 0 ? 1.2f : 1.4f;
132       input.get(i).setProperty(decimalKeyFloat, value);
133     }
134     double meanFloat = 1.3;
135     double varianceFloat = 0.01;
136 
137     //----------------------------------------------------
138     // Double values
139     //----------------------------------------------------
140 
141     //3 % with double property
142     int numDoubleDecimals = 150;
143     String decimalKeyDouble = "double";
144 
145     // mean = 1.0, variance = ((0.5)^2)*2/3 = 1/6
146     for (int i = 0; i < numDoubleDecimals; i = i + 3) {
147       input.get(i).setProperty(decimalKeyDouble, 1.0);
148       input.get(i + 1).setProperty(decimalKeyDouble, 1.5);
149       input.get(i + 2).setProperty(decimalKeyDouble, 0.5);
150     }
151     double meanDouble = 1.0;
152     double varianceDouble = 0.166666;
153 
154     //----------------------------------------------------
155     // Integer values
156     //----------------------------------------------------
157 
158     //10% with integer property
159     int numIntegers = 500;
160     String integerKey = "int";
161     int[] integerValues = new int[]{1, 2, 3, 4, 5};
162 
163     // mean = 3, variance = 0.4*(2^2) + 0.4*(1^2) = 2
164     for (int i = 0; i < numIntegers; i += integerValues.length) {
165       for (int j = 0; j < integerValues.length; j++) {
166         input.get(i + j).setProperty(integerKey, integerValues[j]);
167       }
168     }
169     double meanInteger = 3.;
170     double varianceInteger = 2.;
171 
172     //----------------------------------------------------
173     // Short values
174     //----------------------------------------------------
175 
176     //10% with short property
177     int numShorts = 500;
178     String shortKey = "short";
179     short[] shortValues = new short[]{1, 2, 3, 4, 5};
180 
181     // mean = 3, variance = 0.4*(2^2) + 0.4*(1^2) = 2
182     for (int i = 0; i < numShorts; i += shortValues.length) {
183       for (int j = 0; j < shortValues.length; j++) {
184         input.get(i + j).setProperty(shortKey, shortValues[j]);
185       }
186     }
187     double meanShort = 3.;
188     double varianceShort = 2.;
189 
190     //----------------------------------------------------
191     // Long values
192     //----------------------------------------------------
193     // 20% with long property
194     int numLongs = 1000;
195     String longKey = "long";
196 
197     // mean = 262.5, variance = 62*5^2 = 3906.25
198     for (int i = 0; i < numLongs; i++) {
199       long value = i % 2 == 0 ? 200 : 325;
200       input.get(i).setProperty(longKey, value);
201     }
202     double meanLong = 262.5;
203     double varianceLong = 3906.25;
204 
205 
206     TemporalElementStats stats = new TemporalElementStats();
207     for (TemporalVertex vertex : input) {
208       stats.addElement(vertex);
209     }
210 
211     //-------------------------------------------
212     //--------------RESULTS--------------------
213     //------------------------------------------
214     Map<String, Double> occurence = stats.getNumericalOccurrenceEstimation();
215     Map<String, Double[]> meanAndVariance = stats.getNumericalPropertyStatsEstimation();
216 
217     assertEquals(0.01, occurence.get(decimalKeyFloat), 0.);
218     assertEquals(meanFloat, meanAndVariance.get(decimalKeyFloat)[0], 0.00001);
219     assertEquals(varianceFloat, meanAndVariance.get(decimalKeyFloat)[1], 0.00001);
220 
221     assertEquals(0.03, occurence.get(decimalKeyDouble), 0.);
222     assertEquals(meanDouble, meanAndVariance.get(decimalKeyDouble)[0], 0.);
223     assertEquals(varianceDouble, meanAndVariance.get(decimalKeyDouble)[1], 0.001);
224 
225     assertEquals(0.1, occurence.get(integerKey), 0.);
226     assertEquals(meanInteger, meanAndVariance.get(integerKey)[0], 0.);
227     assertEquals(varianceInteger, meanAndVariance.get(integerKey)[1], 0.000001);
228 
229     assertEquals(0.1, occurence.get(shortKey), 0.);
230     assertEquals(meanShort, meanAndVariance.get(shortKey)[0], 0.);
231     assertEquals(varianceShort, meanAndVariance.get(shortKey)[1], 0.000001);
232 
233     assertEquals(0.2, occurence.get(longKey), 0.);
234     assertEquals(meanLong, meanAndVariance.get(longKey)[0], 0.);
235     assertEquals(varianceLong, meanAndVariance.get(longKey)[1], 0.);

239   @Test
240   public void categoricalPropertiesTest() {
241     ArrayList<TemporalVertex> input = getInputVertices();
242 
243     //first property occurs in 5 nodes (0.1%)
244     String prop1 = "prop1";
245     // one value 3 times (0.06% of all nodes), 2 further values only once (0.2% of all nodes)
246     PropertyValue[] prop1Values = new PropertyValue[]{PropertyValue.create("a"),
247       PropertyValue.create("b"), PropertyValue.create("c")};
248     input.get(0).setProperty(prop1, prop1Values[0]);
249     input.get(1).setProperty(prop1, prop1Values[0]);
250     input.get(2).setProperty(prop1, prop1Values[0]);
251     input.get(3).setProperty(prop1, prop1Values[1]);
252     input.get(4).setProperty(prop1, prop1Values[2]);
253     double[] prop1Selectivity = new double[]{0.0006, 0.0002, 0.0002};
254 
255     // second property occurs in 10 nodes (0.2%)
256     String prop2 = "prop2";
257     // two values, 5 times each
258     PropertyValue[] prop2Values = new PropertyValue[]{PropertyValue.create("x"),
259       PropertyValue.create("y")};
260     for (int i = 0; i < 10; i++) {
261       String value = i % 2 == 0 ? prop2Values[0].getString() : prop2Values[1].getString();
262       input.get(i).setProperty(prop2, value);
263     }
264     double[] prop2Selectivity = new double[]{0.001, 0.001};
265 
266 
267     TemporalElementStats stats = new TemporalElementStats();
268     for (TemporalVertex vertex : input) {
269       stats.addElement(vertex);
270     }
271     Map<String, Map<PropertyValue, Double>> categoricalEstimations =
272       stats.getCategoricalSelectivityEstimation();
273     assertEquals(2, categoricalEstimations.keySet().size());
274 
275     // first property
276     Map<PropertyValue, Double> firstProperty = categoricalEstimations.get(prop1);
277     assertEquals(firstProperty.keySet().size(), 3);
278     for (int i = 0; i < prop1Values.length; i++) {
279       double expected = prop1Selectivity[i];
280       double actual = firstProperty.get(prop1Values[i]);
281       assertEquals(expected, actual, 0.0001);
282     }
283 
284     // second
285     Map<PropertyValue, Double> secondProperty = categoricalEstimations.get(prop2);
286     assertEquals(secondProperty.keySet().size(), 2);
287     for (int i = 0; i < prop2Values.length; i++) {
288       double expected = prop2Selectivity[i];
289       double actual = secondProperty.get(prop2Values[i]);
290       assertEquals(expected, actual, 0.0001);
291     }
292   }

294   /**
295    * Generates 5000 temporal vertices. All of them have tx duration 10000L and val duration
296    * 1000000L
297    *
298    * @return list of temporal vertices
299    */
300   private ArrayList<TemporalVertex> getInputVertices() {
301     ArrayList<TemporalVertex> input = new ArrayList<>();
302     long txFrom = 0L;
303     long txTo = 10000L;
304     long validFrom = 1000000L;
305     long validTo = 2000000L;
306     for (int i = 0; i < 5000; i++) {
307       TemporalVertex vertex = new TemporalVertex();
308       vertex.setLabel("test");
309       vertex.setTransactionTime(new Tuple2<>(txFrom, txTo));
310       vertex.setValidTime(new Tuple2<>(validFrom, validTo));
311       txFrom++;
312       txTo++;
313       validFrom++;
314       validTo++;
315       input.add(vertex);
316     }
317     return input;
318   }
