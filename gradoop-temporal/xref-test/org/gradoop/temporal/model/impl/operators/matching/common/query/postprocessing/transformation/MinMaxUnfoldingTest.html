39  public class MinMaxUnfoldingTest {
40  
41    TimeSelector ts1 = new TimeSelector("a", TX_FROM);
42    TimeSelector ts2 = new TimeSelector("a", TX_TO);
43    TimeSelector ts3 = new TimeSelector("a", VAL_FROM);
44    TimeLiteral l1 = new TimeLiteral("1970-01-01");
45    TimeLiteral l2 = new TimeLiteral("2020-05-24");
46  
47    MinTimePoint min1 = new MinTimePoint(ts1, ts2, ts3);
48    MinTimePoint min2 = new MinTimePoint(ts3, l1);
49  
50    MaxTimePoint max1 = new MaxTimePoint(l2, ts1, ts2);
51    MaxTimePoint max2 = new MaxTimePoint(ts2, l1);
52  
53    MinMaxUnfolding unfolder = new MinMaxUnfolding();
54  
55    @Test
56    public void minMaxUnfoldingTest() {
57      Comparison c1 = new Comparison(min1, Comparator.LT, l1);
58      CNF cnf1 = Util.cnfFromLists(
59        Collections.singletonList(c1)
60      );
61      CNF expected1 = Util.cnfFromLists(
62        Arrays.asList(
63          new Comparison(min1.getArgs().get(0), Comparator.LT, l1),
64          new Comparison(min1.getArgs().get(1), Comparator.LT, l1),
65          new Comparison(min1.getArgs().get(2), Comparator.LT, l1)
66        )
67      );
68      assertEquals(expected1, unfolder.transformCNF(cnf1));
69  
70  
71      Comparison c2 = new Comparison(max1, LTE, l1);
72      CNF cnf2 = Util.cnfFromLists(
73        Collections.singletonList(c2)
74      );
75      CNF expected2 = Util.cnfFromLists(
76        Collections.singletonList(new Comparison(max1.getArgs().get(0), LTE, l1)),
77        Collections.singletonList(new Comparison(max1.getArgs().get(1), LTE, l1)),
78        Collections.singletonList(new Comparison(max1.getArgs().get(2), LTE, l1))
79  
80      );
81      assertEquals(expected2, unfolder.transformCNF(cnf2));
82  
83  
84      Comparison c3 = new Comparison(max1, Comparator.EQ, l1);
85      CNF cnf3 = Util.cnfFromLists(
86        Collections.singletonList(c3)
87      );
88      CNF expected3 = Util.cnfFromLists(
89        Collections.singletonList(new Comparison(max1, EQ, l1)));
90      assertEquals(expected3, unfolder.transformCNF(cnf3));
91  
92  
93      Comparison c4 = new Comparison(l1, LTE, min1);
94      CNF cnf4 = Util.cnfFromLists(
95        Collections.singletonList(c4)
96      );
97      CNF expected4 = Util.cnfFromLists(
98        Collections.singletonList(new Comparison(l1, LTE, min1.getArgs().get(0))),
99        Collections.singletonList(new Comparison(l1, LTE, min1.getArgs().get(1))),
100       Collections.singletonList(new Comparison(l1, LTE, min1.getArgs().get(2)))
101     );
102     assertEquals(expected4, unfolder.transformCNF(cnf4));
103 
104 
105     Comparison c5 = new Comparison(l1, Comparator.LT, max1);
106     CNF cnf5 = Util.cnfFromLists(
107       Collections.singletonList(c5)
108     );
109     CNF expected5 = Util.cnfFromLists(
110       Arrays.asList(
111         new Comparison(l1, Comparator.LT, max1.getArgs().get(0)),
112         new Comparison(l1, Comparator.LT, max1.getArgs().get(1)),
113         new Comparison(l1, Comparator.LT, max1.getArgs().get(2))
