32  /**
33   * Test for the {@link AbstractTimeAggregateFunction}. This test will check if
34   * the correct temporal attribute is read from an element.
35   */
36  public class AbstractTimeAggregateFunctionTest extends TemporalGradoopTestBase {
37  
38    /**
39     * The element used as an input to the aggregate function in this test.
40     */
41    private TemporalElement testElement;
42  
43    /**
44     * Set up this test.
45     */
46    @BeforeClass
47    public void setUp() {
48      testElement = getConfig().getTemporalGraphFactory().getVertexFactory().createVertex();
49      testElement.setTransactionTime(Tuple2.of(1L, 2L));
50      testElement.setValidTime(Tuple2.of(3L, 4L));
51    }
52  
53    /**
54     * Test if the {@link AbstractTimeAggregateFunction#getIncrement(TemporalElement)} returns
55     * the correct field.
56     */
57    @Test(dataProvider = "timeIntervals")
58    public void testGetIncrement(TimeDimension interval, TimeDimension.Field field, long expectedValue) {
59      // Create a mock of the abstract function. (The first constructor parameter, the property key,
60      // is irrelevant for this test.
61      TemporalAggregateFunction mock = mock(AbstractTimeAggregateFunction.class, withSettings()
62        .useConstructor("", interval, field).defaultAnswer(CALLS_REAL_METHODS));
63      PropertyValue increment = mock.getIncrement(testElement);
64      assertTrue(increment.isLong());
65      assertEquals(expectedValue, increment.getLong());
66    }
67  
68    /**
69     * Parameters for this test.
70     * Those are:
71     * <ol start="0">
72     * <li>The time interval used.</li>
73     * <li>The field of the time interval used.</li>
74     * <li>The expected value.</li>
75     * </ol>
76     *
77     * @return Parameters in the given format.
78     */
79    @DataProvider(name = "timeIntervals")
80    public static Object[][] timeIntervals() {
81      return new Object[][] {
82        {TimeDimension.TRANSACTION_TIME, TimeDimension.Field.FROM, 1L},
83        {TimeDimension.TRANSACTION_TIME, TimeDimension.Field.TO, 2L},
84        {TimeDimension.VALID_TIME, TimeDimension.Field.FROM, 3L},
85        {TimeDimension.VALID_TIME, TimeDimension.Field.TO, 4L}
86      };
87    }
