41    @Before
42    public void setUp() throws Exception {
43      stats = new BinningTemporalGraphStatisticsFactory().fromGraph(loadCitibikeSample());

46    @Test
47    public void testVertex() throws Exception {
48      String query = "MATCH (n) WHERE n.tx_to.after(Timestamp(2013-07-20))";
49      TemporalQueryHandler queryHandler = new TemporalQueryHandler(query,
50        new CNFPostProcessing(new ArrayList<>()));
51      est = new CNFEstimation(stats, queryHandler);
52  
53      FilterAndProjectTemporalVerticesNode node = new FilterAndProjectTemporalVerticesNode(null,
54        "n", queryHandler.getPredicates().getSubCNF("n"), Sets.newHashSet());
55  
56      FilterTemporalEstimator elementEstimator = new FilterTemporalEstimator(queryHandler, stats, est);
57      elementEstimator.visit(node);
58  
59      assertEquals(30L, elementEstimator.getCardinality());
60      assertEquals(
61        elementEstimator.getCnfEstimation().estimateCNF(queryHandler.getPredicates().getSubCNF("n")),
62        elementEstimator.getSelectivity(), 0.001);
63    }
