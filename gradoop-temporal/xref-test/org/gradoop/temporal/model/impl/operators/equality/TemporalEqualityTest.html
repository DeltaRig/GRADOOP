50    public void testCollectionEqualityByGraphIds() throws Exception {
51      FlinkAsciiGraphLoader loader = getTestGraphLoader();
52  
53      CollectionEqualityByGraphIds<TemporalGraphHead, TemporalVertex, TemporalEdge, TemporalGraph,
54        TemporalGraphCollection> equality = new CollectionEqualityByGraphIds<>();
55  
56      TemporalGraphCollection gRef1 = toTemporalGraphCollectionWithDefaultExtractors(
57        loader.getGraphCollectionByVariables("gRef")).apply(removeTimeProperties);
58      TemporalGraphCollection gRef2 = toTemporalGraphCollectionWithDefaultExtractors(
59        loader.getGraphCollectionByVariables("gRef")).apply(removeTimeProperties);
60      TemporalGraphCollection gClone = toTemporalGraphCollectionWithDefaultExtractors(
61        loader.getGraphCollectionByVariables("gClone")).apply(removeTimeProperties);
62      TemporalGraphCollection gEmpty = getConfig().getTemporalGraphCollectionFactory()
63        .createEmptyCollection();
64  
65      // direct operator call
66      collectAndAssertTrue(equality.execute(gRef1, gRef2));
67      collectAndAssertFalse(equality.execute(gRef1, gClone));
68      collectAndAssertFalse(equality.execute(gRef1, gEmpty));
69  
70      // convenience method
71      collectAndAssertTrue(gRef1.equalsByGraphIds(gRef2));
72      collectAndAssertFalse(gRef1.equalsByGraphIds(gClone));
73      collectAndAssertFalse(gRef1.equalsByGraphIds(gEmpty));

355   private FlinkAsciiGraphLoader
356   getTestGraphLoader() {
357     String asciiGraphs = "gEmpty[]" +
358 
359       "gRef:G{dataDiff : false}[" +
360       // loop around a1 and edge from a1 to a2
361       "(a1:A{x : 1})-[loop:a{x : 1, __valFrom : 1543400000000L, __valTo : 1543900000000L}]->" +
362       "(a1)-[aa:a{x : 1}]->(a2:A{x : 2, __valFrom : 1543800000000L})" +
363       // parallel edge from a1 to b1
364       "(a1)-[par1:p]->(b1:B),(a1)-[par2:p]->(b1)" +
365       // cycle of bs
366       "(b1)-[cyc1:c]->(b2:B)-[cyc2:c]->(b3:B)-[cyc3:c]->(b1)]" +
367 
368       // element id copy of gRef
369       "gClone:G{dataDiff : true}[" +
370       "(a1)-[loop]->(a1)-[aa]->(a2)" +
371       "(a1)-[par1]->(b1),(a1)-[par2]->(b1)" +
372       "(b1)-[cyc1]->(b2)-[cyc2]->(b3)-[cyc3]->(b1)]" +
373 
374       // element id copy of gRef with one different edge id
375       "gDiffId:G{dataDiff : false}[" +
376       "(a1)-[loop]->(a1)-[aa]->(a2)" +
377       "(a1)-[par1]->(b1),(a1)-[par2]->(b1)" +
378       "(b1)-[cyc1]->(b2)-[:c]->(b3)-[cyc3]->(b1)]" +
379 
380       // element id copy of gRef
381       // with each one different vertex and edge valid time
382       "gDiffTime:G[" +
383       "(a1)-[loop]->(a1)-[:a{x : 1, __valFrom : 1543400000000L, __valTo : 1546900000000L}]->" +
384       "(:A{x : 2, __valFrom : 1543300000000L})" +
385       "(a1)-[par1]->(b1),(a1)-[par2]->(b1)" +
386       "(b1)-[cyc1]->(b2)-[cyc2]->(b3)-[cyc3]->(b1)]" +
387 
388       // copy of gRef with partially reverse edges
389       "gRev:G{dataDiff : false}[" +
390       "(a1)-[loop]->(a1)<-[:a{x : 1}]-(a2)" +
391       "(a1)<-[:p]-(b1),(a1)-[:p]->(b1)" +
392       "(b1)-[cyc1]->(b2)-[cyc2]->(b3)-[cyc3]->(b1)]";
393 
394     return getLoaderFromString(asciiGraphs);
395   }
