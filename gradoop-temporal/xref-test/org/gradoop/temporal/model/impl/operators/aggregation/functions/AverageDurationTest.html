150   /**
151    * Test the {@link AverageDuration} aggregate function and its subclasses in an aggregation.
152    *
153    * @throws Exception when the execution in Flink fails.
154    */
155   @Test
156   public void testWithAggregation() throws Exception {
157     TemporalGraphFactory graphFactory = getConfig().getTemporalGraphFactory();
158     VertexFactory<TemporalVertex> vertexFactory = graphFactory.getVertexFactory();
159     EdgeFactory<TemporalEdge> edgeFactory = graphFactory.getEdgeFactory();
160     TemporalVertex v1 = vertexFactory.createVertex();
161     v1.setTransactionTime(Tuple2.of(1L, 2L));
162     v1.setValidTime(Tuple2.of(DEFAULT_TIME_FROM, DEFAULT_TIME_TO));
163     TemporalVertex v2 = vertexFactory.createVertex();
164     v2.setTransactionTime(Tuple2.of(DEFAULT_TIME_FROM, 5L));
165     v2.setValidTime(Tuple2.of(-3L, DEFAULT_TIME_TO));
166     TemporalVertex v3 = vertexFactory.createVertex();
167     v3.setTransactionTime(Tuple2.of(0L, DEFAULT_TIME_TO));
168     v3.setValidTime(Tuple2.of(DEFAULT_TIME_FROM, 0L));
169     TemporalVertex v4 = vertexFactory.createVertex();
170     v4.setTransactionTime(Tuple2.of(DEFAULT_TIME_FROM, DEFAULT_TIME_TO));
171     v4.setValidTime(Tuple2.of(-5L, -2L));
172     TemporalVertex v5 = vertexFactory.createVertex();
173     v5.setTransactionTime(Tuple2.of(1L, 4L));
174     v5.setValidTime(Tuple2.of(-10L, -3L));
175 
176     TemporalEdge e1 = edgeFactory.createEdge(v1.getId(), v2.getId());
177     e1.setTransactionTime(Tuple2.of(0L, 30L));
178     e1.setValidTime(Tuple2.of(DEFAULT_TIME_FROM, DEFAULT_TIME_TO));
179     TemporalEdge e2 = edgeFactory.createEdge(v2.getId(), v3.getId());
180     e2.setTransactionTime(Tuple2.of(DEFAULT_TIME_FROM, 7L));
181     e2.setValidTime(Tuple2.of(-100L, DEFAULT_TIME_TO));
182     TemporalEdge e3 = edgeFactory.createEdge(v3.getId(), v1.getId());
183     e3.setTransactionTime(Tuple2.of(-1L, DEFAULT_TIME_TO));
184     e3.setValidTime(Tuple2.of(DEFAULT_TIME_FROM, -80L));
185     TemporalEdge e4 = edgeFactory.createEdge(v1.getId(), v3.getId());
186     e4.setTransactionTime(Tuple2.of(DEFAULT_TIME_FROM, DEFAULT_TIME_TO));
187     e4.setValidTime(Tuple2.of(-120L, -107L));
188     TemporalEdge e5 = edgeFactory.createEdge(v2.getId(), v4.getId());
189     e5.setTransactionTime(Tuple2.of(10L, 50L));
190     e5.setValidTime(Tuple2.of(-301L, -276L));
191 
192     DataSet<TemporalVertex> vertices = getExecutionEnvironment().fromElements(v1, v2, v3, v4, v5);
193     DataSet<TemporalEdge> edges = getExecutionEnvironment().fromElements(e1, e2, e3, e4, e5);
194     TemporalGraph result = graphFactory.fromDataSets(vertices, edges)
195       .aggregate(
196         new AverageDuration("avgDurTx", TRANSACTION_TIME),
197         new AverageDuration("avgDurVal", VALID_TIME),
198         new AverageVertexDuration("avgVertexDurTx", TRANSACTION_TIME),
199         new AverageVertexDuration("avgVertexDurVal", VALID_TIME),
200         new AverageEdgeDuration("avgEdgeDurTx", TRANSACTION_TIME),
201         new AverageEdgeDuration("avgEdgeDurVal", VALID_TIME),
202         new AverageDuration("avgDurTxMillis", TRANSACTION_TIME, ChronoUnit.MILLIS),
203         new AverageDuration("avgDurValMillis", VALID_TIME, ChronoUnit.MILLIS),
204         new AverageDuration("avgDurTxSec", TRANSACTION_TIME, ChronoUnit.SECONDS),
205         new AverageDuration("avgDurValSec", VALID_TIME, ChronoUnit.SECONDS),
206         new AverageDuration("avgDurTxMin", TRANSACTION_TIME, ChronoUnit.MINUTES),
207         new AverageDuration("avgDurValMin", VALID_TIME, ChronoUnit.MINUTES),
208         new AverageDuration("avgDurTxHr", TRANSACTION_TIME, ChronoUnit.HOURS),
209         new AverageDuration("avgDurValHr", VALID_TIME, ChronoUnit.HOURS),
210         new AverageDuration("avgDurTxDays", TRANSACTION_TIME, ChronoUnit.DAYS),
211         new AverageDuration("avgDurValDays", VALID_TIME, ChronoUnit.DAYS),
212         new AverageDuration(TRANSACTION_TIME, ChronoUnit.SECONDS),
213         new AverageDuration(VALID_TIME, ChronoUnit.SECONDS));
214     Properties headProperties = result.getGraphHead().collect().get(0).getProperties();
215     assertNotNull(headProperties);
216     assertEquals(18.5d, headProperties.get("avgDurTx").getDouble(), 1e-7d);
217     assertEquals(12.d, headProperties.get("avgDurVal").getDouble(), 1e-7d);
218     assertEquals(2.d, headProperties.get("avgVertexDurTx").getDouble(), 1e-7d);
219     assertEquals(5.d, headProperties.get("avgVertexDurVal").getDouble(), 1e-7d);
220     assertEquals(35.d, headProperties.get("avgEdgeDurTx").getDouble(), 1e-7d);
221     assertEquals(19.d, headProperties.get("avgEdgeDurVal").getDouble(), 1e-7d);
222 
223     assertEquals(18.5d, headProperties.get("avgDurTxMillis").getDouble(), 1e-7d);
224     assertEquals(12.d, headProperties.get("avgDurValMillis").getDouble(), 1e-7d);
225     assertEquals(18.5e-3, headProperties.get("avgDurTxSec").getDouble(), 1e-7d);
226     assertEquals(12e-3d, headProperties.get("avgDurValSec").getDouble(), 1e-7d);
227     assertEquals(18.5e-3 / 60, headProperties.get("avgDurTxMin").getDouble(), 1e-7d);
228     assertEquals(12e-3d / 60, headProperties.get("avgDurValMin").getDouble(), 1e-7d);
229     assertEquals(18.5e-3 / 3600, headProperties.get("avgDurTxHr").getDouble(), 1e-7d);
230     assertEquals(12e-3d / 3600, headProperties.get("avgDurValHr").getDouble(), 1e-7d);
231     assertEquals(18.5e-3 / (3600 * 24), headProperties.get("avgDurTxDays").getDouble(), 1e-7d);
232     assertEquals(12e-3d / (3600 * 24), headProperties.get("avgDurValDays").getDouble(), 1e-7d);
233 
234     assertTrue(headProperties.containsKey("avgDuration_VALID_TIME_Seconds"));
235     assertEquals(12e-3d, headProperties.get("avgDuration_VALID_TIME_Seconds").getDouble(), 1e-7d);
236     assertTrue(headProperties.containsKey("avgDuration_TRANSACTION_TIME_Seconds"));
237     assertEquals(18.5e-3, headProperties.get("avgDuration_TRANSACTION_TIME_Seconds").getDouble(), 1e-7d);
238   }
