16  package org.gradoop.flink.algorithms.fsm.dimspan.model;
17  
18  import org.apache.commons.lang3.ArrayUtils;
19  
20  /**
21   * Util methods to interpret and manipulate int-array encoded patterns represented by DFS codes.
22   */
23  public class DFSCodeUtils extends GraphUtilsBase {
24  
25    /**
26     * Extracts the branch of a give DFS code multiplex.
27     *
28     * @param mux DFS code multiplex
29     *
30     * @return mux of its first extension (branch)
31     */
32    public int[] getBranch(int[] mux) {
33      return ArrayUtils.subarray(mux, 0, EDGE_LENGTH);
34    }
35  
36    /**
37     * Extends a parent DFS code's multiplex
38     *
39     * @param parentMux multiplex
40     * @param fromTime from time
41     * @param fromLabel from label
42     * @param outgoing outgoing traversal
43     * @param edgeLabel traversed edge label
44     * @param toTime to time
45     * @param toLabel to label
46     *
47     * @return child DFS code's multiplex
48     */
49    public int[] addExtension(int[] parentMux,
50      int fromTime, int fromLabel, boolean outgoing, int edgeLabel, int toTime, int toLabel) {
51  
52      return ArrayUtils.addAll(
53        parentMux, multiplex(fromTime, fromLabel, outgoing, edgeLabel, toTime, toLabel));
54    }
55  
56    /**
57     * Checks if a DFS code is child of another one.
58     *
59     * @param childMux multiplex of the child's DFS code
60     * @param parentMux mulitplex of the parent't DFS code
61     *
62     * @return true, if child is actually a child of the parent
63     */
64    public boolean isChildOf(int[] childMux, int[] parentMux) {
65      boolean isChild = parentMux.length >= childMux.length;
66  
67      if (isChild) {
68        for (int i = 0; i < childMux.length; i++) {
69          if (childMux[i] != parentMux[i]) {
70            isChild = false;
71            break;
72          }
73        }
74      }
75  
76      return isChild;
77    }
78  }
