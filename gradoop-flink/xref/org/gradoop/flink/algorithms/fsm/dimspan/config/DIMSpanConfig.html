27  public class DIMSpanConfig implements Serializable {
28  
29    /**
30     * support threshold for subgraphs to be considered to be frequenct
31     */
32    private float minSupport = 1.0f;
33  
34    /**
35     * Direction mode (true=directed; false=undirected)
36     */
37    private boolean directed = true;
38  
39    /**
40     * Dictionary type used for dictionary coding.
41     */
42    private DictionaryType dictionaryType = DictionaryType.INVERSE_PROPORTIONAL;
43  
44    /**
45     * Flag to enable embedding compression (true=enabled).
46     */
47    private boolean embeddingCompressionEnabled = true;
48  
49    /**
50     * Flag to enable graph compression (true=enabled).
51     */
52    private boolean graphCompressionEnabled = true;
53  
54    /**
55     * Flag to set dataflow position of pattern compression.
56     */
57    private DataflowStep patternCompressionInStep = DataflowStep.MAP;
58  
59    /**
60     * Flag to set dataflow position of pattern validation.
61     */
62    private DataflowStep patternVerificationInStep = DataflowStep.COMBINE;
63  
64    /**
65     * Flag to enable branch constraint in pattern growth (true=enabled).
66     */
67    private boolean branchConstraintEnabled = true;
68  
69    /**
70     * valued constructor
71     * @param minSupport minimum relative support of a subgraph
72     * @param directed direction mode
73     */
74    public DIMSpanConfig(float minSupport, boolean directed) {
75      this.minSupport = minSupport;
76      this.directed = directed;
77    }
78  
79    @Override
80    public String toString() {
81      List<String> parameters = Lists.newArrayList();
82  
83      parameters.add("s_min : " + minSupport);
84      parameters.add((directed ? "directed" : "undirected") + " mode");
85      parameters.add("dictionary type : " + dictionaryType.toString());
86  
87      parameters.add(getParameterEnabled("branch constraint", branchConstraintEnabled));
88      parameters.add(getParameterEnabled("graph compression", graphCompressionEnabled));
89      parameters.add(
90        getParameterEnabled("embedding compression", embeddingCompressionEnabled));
91      parameters.add("pattern compression @ " + patternCompressionInStep.toString());
92      parameters.add("pattern validation @ " + patternVerificationInStep.toString());
93  
94      return StringUtils.join(parameters, "|");
95    }
96  
97    /**
98     * Convenience method for string formatting of enum parameters.
99     *
100    * @param parameter parameter name
101    * @param enabled parameter value
102    *
103    * @return string representation of the parameter and its value
104    */
105   private String getParameterEnabled(String parameter, boolean enabled) {
106     return parameter + " " + (enabled ? "enabled" : "disabled");
107   }
