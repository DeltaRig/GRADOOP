60  /**
61   * Abstract superclass of different implementations of the gSpan frequent
62   * subgraph mining algorithm as Gradoop operator
63   */
64  public class DIMSpan {
65  
66    /**
67     * Maximum number of iterations if set of k-edge frequent patterns is not running empty before.
68     */
69    private static final int MAX_ITERATIONS = 100;
70  
71    /**
72     * FSM configuration
73     */
74    protected final DIMSpanConfig fsmConfig;
75  
76    /**
77     * input graph collection cardinality
78     */
79    protected DataSet<Long> graphCount;
80  
81    /**
82     * minimum frequency for patterns to be considered to be frequent
83     */
84    protected DataSet<Long> minFrequency;

86    /**
87     * Pattern growth and verification logic derived from gSpan.
88     * See <a href="https://www.cs.ucsb.edu/~xyan/software/gSpan.htm">gSpan</a>
89     */
90    protected final GSpanLogic gSpan;

95    private DataSet<String[]> vertexDictionary;

98     * Edge label dictionary for dictionary coding.

105   private final LabelComparator comparator;

107   /**
108    * Constructor.
109    *
110    * @param fsmConfig FSM configuration
111    */
112   public DIMSpan(DIMSpanConfig fsmConfig) {
113     this.fsmConfig = fsmConfig;
114 
115     // set gSpan implementation depending on direction mode
116     gSpan = fsmConfig.isDirected() ?
117       new DirectedGSpanLogic(fsmConfig) :
118       new UndirectedGSpanLogic(fsmConfig);
119 
120     // set comparator based on dictionary type
121     if (fsmConfig.getDictionaryType() == DictionaryType.PROPORTIONAL) {
122       comparator =  new ProportionalLabelComparator();
123     } else if (fsmConfig.getDictionaryType() == DictionaryType.INVERSE_PROPORTIONAL) {
124       comparator =  new InverseProportionalLabelComparator();
125     } else {
126       comparator = new AlphabeticalLabelComparator();
127     }
128   }

130   /**
131    * Executes the DIMSpan algorithm.
132    * Orchestration of preprocessing, mining and postprocessing.
133    *
134    * @param input input graph collection
135    * @return frequent patterns
136    */
137   public DataSet<GraphTransaction> execute(DataSet<LabeledGraphStringString> input) {
138 
139     DataSet<int[]> encodedInput = preProcess(input);
140     DataSet<WithCount<int[]>> encodedOutput = mine(encodedInput);
141 
142     return postProcess(encodedOutput);
143   }

145   /**
146    * Triggers the label-frequency base preprocessing
147    *
148    * @param graphs input
149    * @return preprocessed input
150    */
151   private DataSet<int[]> preProcess(DataSet<LabeledGraphStringString> graphs) {
152 
153     // Determine cardinality of input graph collection
154     this.graphCount = Count
155       .count(graphs);
156 
157     // Calculate minimum frequency
158     this.minFrequency = graphCount
159       .map(new MinFrequency(fsmConfig));
160 
161     // Execute vertex label pruning and dictionary coding
162     DataSet<LabeledGraphIntString> graphsWithEncodedVertices = encodeVertices(graphs);
163 
164     // Execute edge label pruning and dictionary coding
165     DataSet<int[]> encodedGraphs = encodeEdges(graphsWithEncodedVertices);
166 
167     // return all non-obsolete encoded graphs
168     return encodedGraphs
169       .filter(new NotEmpty());
170   }

172   /**
173    * Triggers the iterative mining process.
174    *
175    * @param graphs preprocessed input graph collection
176    * @return frequent patterns
177    */
178   protected DataSet<WithCount<int[]>> mine(DataSet<int[]> graphs) {
179 
180     DataSet<GraphWithPatternEmbeddingsMap> searchSpace = graphs
181       .map(new InitSingleEdgePatternEmbeddingsMap(gSpan, fsmConfig));
182 
183     // Workaround to support multiple data sinks: create pseudo-graph (collector),
184     // which embedding map will be used to union all k-edge frequent patterns
185     DataSet<GraphWithPatternEmbeddingsMap> collector = graphs
186       .getExecutionEnvironment()
187       .fromElements(true)
188       .map(new CreateCollector());
189 
190     searchSpace = searchSpace.union(collector);
191 
192     // ITERATION HEAD
193 
194     IterativeDataSet<GraphWithPatternEmbeddingsMap> iterative = searchSpace
195       .iterate(MAX_ITERATIONS);
196 
197     // ITERATION BODY
198 
199     DataSet<WithCount<int[]>> reports = iterative
200       .flatMap(new ReportSupportedPatterns());
201 
202     DataSet<WithCount<int[]>> frequentPatterns = getFrequentPatterns(reports);
203 
204     DataSet<GraphWithPatternEmbeddingsMap> grownEmbeddings = iterative
205       .map(new GrowFrequentPatterns(gSpan, fsmConfig))
206       .withBroadcastSet(frequentPatterns, DIMSpanConstants.FREQUENT_PATTERNS)
207       .filter(new NotObsolete());
208 
209     // ITERATION FOOTER
210 
211     return iterative
212       .closeWith(grownEmbeddings, frequentPatterns)
213       // keep only collector and expand embedding map keys
214       .filter(new IsFrequentPatternCollector())
215       .flatMap(new ExpandFrequentPatterns());
216   }

218   /**
219    * Triggers the postprocessing.
220    *
221    * @param encodedOutput frequent patterns represented by multiplexed int-arrays
222    * @return Gradoop graph transactions
223    */
224   private DataSet<GraphTransaction> postProcess(DataSet<WithCount<int[]>> encodedOutput) {
225     return encodedOutput
226       .map(new DFSCodeToEPGMGraphTransaction(fsmConfig))
227       .withBroadcastSet(vertexDictionary, DIMSpanConstants.VERTEX_DICTIONARY)
228       .withBroadcastSet(edgeDictionary, DIMSpanConstants.EDGE_DICTIONARY)
229       .withBroadcastSet(graphCount, DIMSpanConstants.GRAPH_COUNT);
230   }

232   /**
233    * Executes pruning and dictionary coding of vertex labels.
234    *
235    * @param graphs graphs with string-labels
236    * @return graphs with dictionary-encoded vertex labels
237    */
238   private DataSet<LabeledGraphIntString> encodeVertices(DataSet<LabeledGraphStringString> graphs) {
239 
240     // LABEL PRUNING
241 
242     DataSet<WithCount<String>> vertexLabels = graphs
243       .flatMap(new ReportVertexLabels());
244 
245     vertexLabels = getFrequentLabels(vertexLabels);
246 
247     // DICTIONARY ENCODING
248 
249     vertexDictionary = vertexLabels
250       .reduceGroup(new CreateDictionary(comparator));
251 
252     return graphs
253       .map(new EncodeAndPruneVertices())
254       .withBroadcastSet(vertexDictionary, DIMSpanConstants.VERTEX_DICTIONARY);
255   }
256 
257   /**
258    * Executes pruning and dictionary coding of edge labels.
259    *
260    * @param graphs graphs with dictionary-encoded vertex labels
261    * @return graphs with dictionary-encoded vertex and edge labels
262    */
263   private DataSet<int[]> encodeEdges(DataSet<LabeledGraphIntString> graphs) {
264 
265     DataSet<WithCount<String>> edgeLabels = graphs
266       .flatMap(new ReportEdgeLabels());
267 
268     edgeLabels = getFrequentLabels(edgeLabels);
269 
270     edgeDictionary = edgeLabels
271       .reduceGroup(new CreateDictionary(comparator));
272 
273     return graphs
274       .map(new EncodeAndPruneEdges(fsmConfig))
275       .withBroadcastSet(edgeDictionary, DIMSpanConstants.EDGE_DICTIONARY);
276   }

278   /**
279    * Determines frequent labels.
280    *
281    * @param labels dataset of labels
282    *
283    * @return dataset of frequent labels
284    */
285   private DataSet<WithCount<String>> getFrequentLabels(DataSet<WithCount<String>> labels) {
286     // enabled
287     if (fsmConfig.getDictionaryType() != DictionaryType.RANDOM) {
288 
289       labels = labels
290         .groupBy(0)
291         .sum(1)
292         .filter(new Frequent<>())
293         .withBroadcastSet(minFrequency, DIMSpanConstants.MIN_FREQUENCY);
294 
295       // disabled
296     } else {
297       labels = labels
298         .distinct();
299     }
300 
301     return labels;
302   }

304   /**
305    * Identifies valid frequent patterns from a dataset of reported patterns.
306    *
307    * @param patterns reported patterns
308    * @return valid frequent patterns
309    */
310   private DataSet<WithCount<int[]>> getFrequentPatterns(DataSet<WithCount<int[]>> patterns) {
311     // COMBINE
312 
313     patterns = patterns
314       .groupBy(0)
315       .combineGroup(sumPartition());
316 
317     if (fsmConfig.getPatternVerificationInStep() == DataflowStep.COMBINE) {
318       patterns = patterns
319         .filter(new VerifyPattern(gSpan, fsmConfig));
320     }
321 
322     if (fsmConfig.getPatternCompressionInStep() == DataflowStep.COMBINE) {
323       patterns = patterns
324         .map(new CompressPattern());
325     }
326 
327     // REDUCE
328 
329     patterns = patterns
330       .groupBy(0)
331       .sum(1);
332 
333     // FILTER
334 
335     patterns = patterns
336       .filter(new Frequent<>())
337       .withBroadcastSet(minFrequency, DIMSpanConstants.MIN_FREQUENCY);
338 
339     if (fsmConfig.getPatternVerificationInStep() == DataflowStep.FILTER) {
340       patterns = patterns
341         .filter(new VerifyPattern(gSpan, fsmConfig));
342     }
343 
344     if (fsmConfig.getPatternCompressionInStep() == DataflowStep.FILTER) {
345       patterns = patterns
346         .map(new CompressPattern());
347     }
348 
349     return patterns;
350   }

352   /**
353    * Creates a Flink sum aggregate function that can be applied in group combine operations.
354    *
355    * @return sum group combine function
356    */
357   private GroupCombineFunction<WithCount<int[]>, WithCount<int[]>> sumPartition() {
358 
359     @SuppressWarnings("unchecked")
360     AggregationFunction<Long>[] sum = new AggregationFunction[] {
361       new SumAggregationFunction.SumAggregationFunctionFactory()
362         .createAggregationFunction(Long.class)
363     };
364 
365     int[] fields = { 1 };
366 
367     return new AggregateMultipleFunctions(sum, fields);
368   }
