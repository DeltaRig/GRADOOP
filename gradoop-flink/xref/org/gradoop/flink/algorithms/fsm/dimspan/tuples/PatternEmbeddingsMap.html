26  /**
27   * f0 : [pattern1,..,patternN]
28   * f1 : [[vertexCount,edgeCount,v11,..,vj1,e11,..,ek1,v21,..]1,..]
29   */
30  public class PatternEmbeddingsMap extends Tuple2<int[][], int[][]> {
31  
32    /**
33     * Default Constructor.
34     */
35    public PatternEmbeddingsMap() {
36    }

38    /**
39     * Valued constructor.
40     *
41     * @param patternMuxes multiplexes of patterns
42     * @param embeddingMuxes multiplexes of embeddings
43     */
44    public PatternEmbeddingsMap(
45      int[][] patternMuxes, int[][] embeddingMuxes) {
46      super(patternMuxes, embeddingMuxes);
47    }

49    /**
50     * Convenience method.
51     *
52     * @return number of contained patterns
53     */
54    public int getPatternCount() {
55      return getKeys().length;
56    }
57  
58  
59    /**
60     * Convenience method to check is an embeddings map is empty
61     *
62     * @return true, if empty
63     */
64    public boolean isEmpty() {
65      return getPatternCount() == 0;
66    }

68    /**
69     * Returns a pattern at a given index.
70     *
71     * @param index index
72     *
73     * @return pattern multiplex
74     */
75    public int[] getPattern(int index) {
76      return getKeys()[index];
77    }
78  
79    /**
80     * Find index of a given pattern
81     *
82     * @param  patternMux search pattern
83     *
84     * @return index if contained, -1 otherwise
85     */
86    public int getIndex(int[] patternMux) {
87      int index = -1;
88      int i = 0;
89      for (int[] mux : getKeys()) {
90        if (Objects.deepEquals(mux, patternMux)) {
91          index = i;
92          break;
93        }
94        i++;
95      }
96  
97      return index;
98    }
99  
100   /**
101    * Creates an entry for a pattern including an initial embedding or just adds an embedding.
102    *
103    * @param patternMux pattern
104    * @param vertexIds the embedding's vertex ids
105    * @param edgeIds the embedding's edge ids
106    */
107   public void put(int[] patternMux, int[] vertexIds, int[] edgeIds) {
108     int patternIndex = getIndex(patternMux);
109     int[] vertexEdgeIds = ArrayUtils.addAll(vertexIds, edgeIds);
110 
111     if (patternIndex < 0) {
112       // insert
113       setKeys(ArrayUtils.add(getKeys(), patternMux));
114 
115       int[] embeddingData = new int[] {vertexIds.length, edgeIds.length};
116       embeddingData = ArrayUtils.addAll(embeddingData, vertexEdgeIds);
117 
118       setValues(ArrayUtils.add(getValues(), embeddingData));
119     } else {
120       // update
121       getValues()[patternIndex] =
122         ArrayUtils.addAll(getValues()[patternIndex], vertexEdgeIds);
123     }
124   }
125 
126   /**
127    * Adds a pattern and existing embeddings.
128    *
129    * @param patternMux pattern
130    * @param embeddingsMux embeddings
131    */
132   public void put(int[] patternMux, int[] embeddingsMux) {
133     setKeys(ArrayUtils.add(getKeys(), patternMux));
134     setValues(ArrayUtils.add(getValues(), embeddingsMux));
135   }

137   /**
138    * Adds all patterns and embeddings of another map.
139    *
140    * @param that other map
141    */
142   public void append(PatternEmbeddingsMap that) {
143     setKeys(ArrayUtils.addAll(this.getKeys(), that.getKeys()));
144     setValues(ArrayUtils.addAll(this.getValues(), that.getValues()));
145   }

148   /**
149    * Adds a pattern but no embeddings.
150    *
151    * @param muxWithCount pattern
152    */
153   public void collect(WithCount<int[]> muxWithCount) {
154     setKeys(ArrayUtils.add(getKeys(), muxWithCount.getObject()));
155     setValues(ArrayUtils.add(getValues(), new int[] {(int) muxWithCount.getCount()}));
156   }

158   /**
159    * Returns all embeddings for a give pattern index.
160    *
161    * @param index pattern index
162    * @param uncompress true, if embeddings need to be uncompressed
163    *
164    * @return array of embeddings
165    */
166   public int[][] getEmbeddings(int index, boolean uncompress) {
167     int[] embeddingData = getValues()[index];
168 
169     if (uncompress) {
170       embeddingData = Simple16Compressor.uncompress(embeddingData);
171     }
172 
173     int vertexCount = embeddingData[0];
174     int edgeCount = embeddingData[1];
175     int embeddingLength = vertexCount + edgeCount;
176     int embeddingCount = (embeddingData.length - 2) / embeddingLength;
177 
178     int[][] embeddings = new int[2 * embeddingCount][];
179 
180     int i = 2;
181     for (int m = 0; m < embeddingCount; m++) {
182 
183       int[] vertexIds = new int[vertexCount];
184       for (int v = 0; v < vertexCount; v++) {
185         vertexIds[v] = embeddingData[i];
186         i++;
187       }
188       embeddings[2 * m] = vertexIds;
189 
190       int[] edgeIds = new int[edgeCount];
191       for (int e = 0; e < edgeCount; e++) {
192         edgeIds[e] = embeddingData[i];
193         i++;
194       }
195       embeddings[2 * m + 1] = edgeIds;
196     }
197 
198     return embeddings;
199   }

219   @Override
220   public String toString() {
221     StringBuilder builder = new StringBuilder();
222 
223     for (int i = 0; i < getPatternCount(); i++) {
224       builder.append(new Tuple1<>(getPattern(i)));
225       builder.append("\t");
226       builder.append(new Tuple1<>(getValues()[i]));
227       builder.append("\n");
228     }
229 
230     return builder.toString();
231   }
