37  /**
38   * Superclass of frequent subgraph implementations in the graph transaction setting.
39   */
40  public abstract class TransactionalFSMBase implements UnaryCollectionToCollectionOperator {
41  
42    /**
43     * FSM configuration
44     */
45    protected final FSMConfig fsmConfig;
46  
47    /**
48     * search space size
49     */
50    protected DataSet<Long> graphCount;
51  
52    /**
53     * minimum frequency for patterns to be considered to be frequent
54     */
55    protected DataSet<Long> minFrequency;
56  
57    /**
58     * Gradoop configuration
59     */
60    protected GradoopFlinkConfig config;
61  
62    /**
63     * Constructor.
64     *
65     * @param fsmConfig FSM configuration
66     */
67    public TransactionalFSMBase(FSMConfig fsmConfig) {
68      this.fsmConfig = fsmConfig;
69    }
70  
71    @Override
72    public GraphCollection execute(GraphCollection collection)  {
73      config = collection.getConfig();
74  
75      DataSet<GraphTransaction> input = collection
76        .getGraphTransactions();
77  
78      DataSet<GraphTransaction> output = execute(input);
79  
80      return collection.getFactory()
81        .fromTransactions(output);
82    }
83  
84    /**
85     * Executes the algorithm for a dataset of graphs in transactional representation.
86     *
87     * @param transactions dataset of graphs
88     *
89     * @return frequent patterns as dataset of graphs
90     */
91    protected abstract DataSet<GraphTransaction> execute(DataSet<GraphTransaction> transactions);
92  
93    /**
94     * Triggers the label-frequency base preprocessing
95     *
96     * @param transactions input
97     * @return preprocessed input
98     */
99    protected DataSet<GraphTransaction> preProcess(DataSet<GraphTransaction> transactions) {
100     transactions = transactions
101       .map(new DropPropertiesAndGraphContainment());
102 
103     this.graphCount = Count
104       .count(transactions);
105 
106     this.minFrequency = graphCount
107       .map(new MinFrequency(fsmConfig));
108 
109     DataSet<String> frequentVertexLabels = transactions
110       .flatMap(new VertexLabels())
111       .groupBy(0)
112       .sum(1)
113       .filter(new Frequent<>())
114       .withBroadcastSet(minFrequency, DIMSpanConstants.MIN_FREQUENCY)
115       .map(new ValueOfWithCount<>());
116 
117     transactions = transactions
118       .map(new FilterVerticesByLabel())
119       .withBroadcastSet(frequentVertexLabels, TFSMConstants.FREQUENT_VERTEX_LABELS);
120 
121     DataSet<String> frequentEdgeLabels = transactions
122       .flatMap(new EdgeLabels())
123       .groupBy(0)
124       .sum(1)
125       .filter(new Frequent<>())
126       .withBroadcastSet(minFrequency, DIMSpanConstants.MIN_FREQUENCY)
127       .map(new ValueOfWithCount<>());
128 
129     transactions = transactions
130       .map(new FilterEdgesByLabel())
131       .withBroadcastSet(frequentEdgeLabels, TFSMConstants.FREQUENT_EDGE_LABELS);
132 
133     transactions = transactions
134       .filter(new NotEmpty());
135 
136     return transactions;
137   }
