31  /**
32   * Creates a canonical label for a given graph embedding that represents its
33   * adjacency matrix.
34   */
35  public class CanonicalLabeler implements Serializable {
36  
37    /**
38     * separator among adjacency lists
39     */
40    private static final char NEW_LIST = '|';
41    /**
42     * separator among vertex label and adjacency list entries
43     */
44    private static final char LIST_START = ':';
45    /**
46     * separator among adjacency list entries
47     */
48    private static final char NEW_ENTRY = ',';
49    /**
50     * edge separator indicating an outgoing edge
51     */
52    private static final char OUTGOING_EDGE = '>';
53    /**
54     * edge separator indicating an incoming edge
55     */
56    private static final char INCOMING_EDGE = '<';
57    /**
58     * edge separator indicating an undirected edge
59     */
60    private static final char UNDIRECTED_EDGE = '-';
61  
62    /**
63     * true, if the label should represent directed graphs,
64     * false, otherwise
65     */
66    private final boolean directed;
67  
68    /**
69     * Constructor.
70     *
71     * @param directed true for directed mode
72     */
73    public CanonicalLabeler(boolean directed) {
74      this.directed = directed;
75    }
76  
77    /**
78     * Creates a canonical subgraph graph label
79     *
80     * @param embedding subgraph
81     *
82     * @return canonical label
83     */
84    public String label(Embedding embedding) {
85  
86      Map<Integer, String> vertices = embedding.getVertices();
87      Map<Integer, FSMEdge> edges = embedding.getEdges();
88  
89      Map<Integer, Map<Integer, Set<Integer>>> adjacencyMatrix =
90        createAdjacencyMatrix(vertices, edges);
91  
92      List<String> adjacencyListLabels =
93        Lists.newArrayListWithCapacity(vertices.size());
94  
95      // for each vertex
96      for (Map.Entry<Integer, String> vertex : vertices.entrySet()) {
97  
98        int vertexId = vertex.getKey();
99        String adjacencyListLabel = vertex.getValue() + LIST_START;
100 
101       Map<Integer, Set<Integer>> adjacencyList = adjacencyMatrix.get(vertexId);
102 
103       List<String> entryLabels =
104         Lists.newArrayListWithCapacity(adjacencyList.size());
105 
106       // for each adjacent vertex
107       for (Map.Entry<Integer, Set<Integer>> entry : adjacencyList.entrySet()) {
108 
109         int adjacentVertexId = entry.getKey();
110         String entryLabel = vertices.get(adjacentVertexId);
111 
112         // for each edge
113         Set<Integer> incidentEdgeIds = entry.getValue();
114 
115         if (incidentEdgeIds.size() == 1) {
116           FSMEdge incidentEdge = edges.get(incidentEdgeIds.iterator().next());
117 
118           entryLabel += format(incidentEdge, vertexId);
119 
120         } else {
121 
122           List<String> incidentEdges =
123             Lists.newArrayListWithExpectedSize(incidentEdgeIds.size());
124 
125           for (int incidentEdgeId : incidentEdgeIds) {
126             incidentEdges.add(format(edges.get(incidentEdgeId), vertexId));
127           }
128 
129           Collections.sort(incidentEdges);
130           entryLabel += StringUtils.join(incidentEdges, "");
131         }
132 
133         entryLabels.add(entryLabel);
134       }
135 
136       Collections.sort(entryLabels);
137       adjacencyListLabel += StringUtils.join(entryLabels, NEW_ENTRY);
138       adjacencyListLabels.add(adjacencyListLabel);
139     }
140 
141     Collections.sort(adjacencyListLabels);
142     return StringUtils.join(adjacencyListLabels, NEW_LIST);
143   }

145   /**
146    * Creates an adjacency matrix.
147    *
148    * @param vertices vertices
149    * @param edges edges
150    * @return adjacency matrix
151    */
152   private Map<Integer, Map<Integer, Set<Integer>>> createAdjacencyMatrix(
153     Map<Integer, String> vertices, Map<Integer, FSMEdge> edges) {
154 
155     Map<Integer, Map<Integer, Set<Integer>>> adjacencyMatrix =
156       Maps.newHashMapWithExpectedSize(vertices.size());
157 
158     for (Map.Entry<Integer, FSMEdge> edgeEntry : edges.entrySet()) {
159 
160       int edgeId = edgeEntry.getKey();
161       FSMEdge edge = edgeEntry.getValue();
162 
163       int sourceId = edge.getSourceId();
164       int targetId = edge.getTargetId();
165 
166       addAdjacencyMatrixEntry(adjacencyMatrix, sourceId, edgeId, targetId);
167 
168       if (sourceId != targetId) {
169         addAdjacencyMatrixEntry(adjacencyMatrix, targetId, edgeId, sourceId);
170       }
171     }
172 
173     return adjacencyMatrix;
174   }
175 
176   /**
177    * Adds an entry to an adjacency matrix.
178    *
179    * @param adjacencyMatrix adjacency matrix
180    * @param vertexId id of the vertex the entry should be added to
181    * @param incidentEdgeId incident edge id
182    * @param adjacentVertexId adjacent vertex id
183    */
184   private void addAdjacencyMatrixEntry(
185     Map<Integer, Map<Integer, Set<Integer>>> adjacencyMatrix,
186     int vertexId, int incidentEdgeId, int adjacentVertexId) {
187 
188     // create entry for target to source
189     Map<Integer, Set<Integer>> adjacencyList = adjacencyMatrix.get(vertexId);
190 
191     // first visit of this vertex
192     if (adjacencyList == null) {
193 
194       // create target map
195       adjacencyList = Maps.newHashMap();
196       adjacencyMatrix.put(vertexId, adjacencyList);
197     }
198 
199     Set<Integer> incidentEdgeIds = adjacencyList.get(adjacentVertexId);
200 
201     // first edge connecting to other vertex
202     if (incidentEdgeIds == null) {
203       adjacencyList.put(adjacentVertexId, Sets.newHashSet(incidentEdgeId));
204     } else {
205       incidentEdgeIds.add(incidentEdgeId);
206     }
207   }
208 
209   /**
210    * Formats an edge in an adjacency list entry.
211    *
212    * @param edge edge
213    * @param adjacencyListVertexId vertex id of the related adjacency list
214    *
215    * @return string representation
216    */
217   private String format(FSMEdge edge, int adjacencyListVertexId) {
218 
219     char edgeChar = directed ?
220       (edge.getSourceId() == adjacencyListVertexId ?
221         OUTGOING_EDGE : INCOMING_EDGE
222       ) :
223       UNDIRECTED_EDGE;
224 
225     return edgeChar + edge.getLabel();
226   }
