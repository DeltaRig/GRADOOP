49  /**
50   * Part of the BIIIG approach.
51   * Vertex-centric implementation to isolate business transaction graphs.
52   */
53  public class BusinessTransactionGraphs implements
54    UnaryGraphToCollectionOperator {
55  
56    /**
57     * reserved property key referring to master or transactional data
58     */
59    public static final String SUPERTYPE_KEY = "superType";
60    /**
61     * reserved property value to mark master data
62     */
63    public static final String SUPERCLASS_VALUE_MASTER = "M";
64    /**
65     * reserved property value to mark transactional data
66     */
67    public static final String SUPERCLASS_VALUE_TRANSACTIONAL = "T";
68    /**
69     * reserved label to mark business transaction graphs
70     */
71    public static final String BTG_LABEL = "BusinessTransactionGraph";
72    /**
73     * reserved property key referring to the source identifier of vertices
74     */
75    public static final String SOURCEID_KEY = "sid";
76  
77    @Override
78    public GraphCollection execute(LogicalGraph iig) {
79  
80      DataSet<EPGMVertex> masterVertices = iig.getVertices()
81        .filter(new MasterData<>());
82  
83      LogicalGraph transGraph = iig
84        .vertexInducedSubgraph(new TransactionalData<>());
85  
86      DataSet<EPGMVertex> transVertices = transGraph
87        .getVertices();
88  
89      DataSet<org.apache.flink.graph.Edge<GradoopId, NullValue>> transEdges =
90        transGraph.getEdges().map(new ToGellyEdgeWithNullValue());
91  
92      Graph<GradoopId, GradoopId, NullValue> gellyTransGraph = Graph.fromDataSet(
93        transVertices.map(new VertexToGellyVertexWithGradoopId()),
94        transEdges,
95        iig.getConfig().getExecutionEnvironment()
96      );
97  
98      gellyTransGraph = gellyTransGraph
99        .getUndirected()
100       .runScatterGatherIteration(new BtgMessenger(), new BtgUpdater(), 100);
101 
102 
103     DataSet<Tuple2<GradoopId, GradoopIdSet>> btgVerticesMap = gellyTransGraph
104       .getVerticesAsTuple2()
105       .map(new SwitchPair<>())
106       .groupBy(0)
107       .reduceGroup(new CollectGradoopIds())
108       .map(new ComponentToNewBtgId());
109 
110     DataSet<Tuple2<GradoopId, GradoopId>> vertexBtgMap = btgVerticesMap
111       .flatMap(new ExpandGradoopIds<>())
112       .map(new SwitchPair<>());
113 
114     DataSet<EPGMGraphHead> graphHeads = btgVerticesMap
115       .map(new Value0Of2<>())
116       .map(new NewBtgGraphHead<>(iig.getFactory().getGraphHeadFactory()));
117 
118     // filter and update edges
119     DataSet<EPGMEdge> btgEdges = iig.getEdges()
120       .join(vertexBtgMap)
121       .where(new SourceId<>()).equalTo(0)
122       .with(new SetBtgId<>());
123 
124     // update transactional vertices
125     transVertices = transVertices
126       .join(vertexBtgMap)
127       .where(new Id<>()).equalTo(0)
128       .with(new SetBtgId<>());
129 
130     // create master data BTG map
131     vertexBtgMap = btgEdges
132       .map(new TargetIdBtgId<>())
133       .join(masterVertices)
134       .where(0).equalTo(new Id<>())
135       .with(new LeftSide<>())
136       .distinct();
137 
138     DataSet<Tuple2<GradoopId, GradoopIdSet>> vertexBtgsMap = vertexBtgMap
139       .groupBy(0)
140       //.combineGroup(new CollectGradoopIds())
141       .reduceGroup(new CollectGradoopIds());
142 
143     masterVertices = masterVertices.join(vertexBtgsMap)
144       .where(new Id<>()).equalTo(0)
145       .with(new SetBtgIds<>());
146 
147     return iig.getCollectionFactory()
148       .fromDataSets(graphHeads, transVertices.union(masterVertices), btgEdges);
149   }
