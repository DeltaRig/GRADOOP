34  /**
35   * A gradoop operator wrapping {@link org.apache.flink.graph.library.linkanalysis.HITS}
36   * <p>
37   * This algorithm can be configured to terminate either by a limit on the number of iterations, a
38   * convergence threshold, or both.
39   * <p>
40   * The Results are stored as properties of the vertices (with given keys).
41   *
42   * @param <G>  Gradoop graph head type.
43   * @param <V>  Gradoop vertex type.
44   * @param <E>  Gradoop edge type.
45   * @param <LG> Gradoop type of the graph.
46   * @param <GC> Gradoop type of the graph collection.
47   */
48  public class HITS<
49    G extends GraphHead,
50    V extends Vertex,
51    E extends Edge,
52    LG extends BaseGraph<G, V, E, LG, GC>,
53    GC extends BaseGraphCollection<G, V, E, LG, GC>>
54    extends GradoopGellyAlgorithm<G, V, E, LG, GC, NullValue, NullValue> {
55  
56    /**
57     * Property key to store the authority score.
58     */
59    private String authorityPropertyKey;
60  
61    /**
62     * Property key to store the hub score.
63     */
64    private String hubPropertyKey;
65  
66    /**
67     * Gelly HITS implementation
68     */
69    private org.apache.flink.graph.library.linkanalysis.HITS<GradoopId, NullValue, NullValue> hits;
70  
71    /**
72     * HITS with fixed number of iterations
73     *
74     * @param authorityPropertyKey Property key to store the authority score.
75     * @param hubPropertyKey       Property key to store the hub score.
76     * @param iterations           number of iterations
77     */
78    public HITS(String authorityPropertyKey, String hubPropertyKey, int iterations) {
79      super(new VertexToGellyVertexWithNullValue<>(), new EdgeToGellyEdgeWithNullValue<>());
80      this.authorityPropertyKey = authorityPropertyKey;
81      this.hubPropertyKey = hubPropertyKey;
82      hits = new org.apache.flink.graph.library.linkanalysis.HITS<>(iterations, Double.MAX_VALUE);
83    }
84  
85  
86    /**
87     * HITS with convergence threshold
88     *
89     * @param authorityPropertyKey Property key to store the authority score.
90     * @param hubPropertyKey       Property key to store the hub score.
91     * @param convergenceThreshold convergence threshold for sum of scores
92     */
93    public HITS(String authorityPropertyKey, String hubPropertyKey, double convergenceThreshold) {
94      super(new VertexToGellyVertexWithNullValue<>(), new EdgeToGellyEdgeWithNullValue<>());
95      this.authorityPropertyKey = authorityPropertyKey;
96      this.hubPropertyKey = hubPropertyKey;
97      hits = new org.apache.flink.graph.library.linkanalysis.HITS<>(Integer.MAX_VALUE,
98        convergenceThreshold);
99    }
100 
101   /**
102    * HITS with convergence threshold and maximum number of iterations
103    *
104    * @param authorityPropertyKey Property key to store the authority score.
105    * @param hubPropertyKey       Property key to store the hub score.
106    * @param convergenceThreshold convergence threshold for sum of scores
107    * @param maxIterations        maximum number of iterations
108    */
109   public HITS(String authorityPropertyKey, String hubPropertyKey, int maxIterations,
110     double convergenceThreshold) {
111     super(new VertexToGellyVertexWithNullValue<>(), new EdgeToGellyEdgeWithNullValue<>());
112     this.authorityPropertyKey = authorityPropertyKey;
113     this.hubPropertyKey = hubPropertyKey;
114     hits =
115       new org.apache.flink.graph.library.linkanalysis.HITS<>(maxIterations, convergenceThreshold);
116   }
117 
118 
119   @Override
120   public LG executeInGelly(Graph<GradoopId, NullValue, NullValue> gellyGraph) throws Exception {
121 
122     DataSet<V> newVertices = hits.runInternal(gellyGraph)
123       .join(currentGraph.getVertices())
124       .where(new HitsResultKeySelector()).equalTo(new Id<>())
125       .with(new HITSToAttributes<>(authorityPropertyKey, hubPropertyKey));
126 
127     return currentGraph.getFactory()
128       .fromDataSets(currentGraph.getGraphHead(), newVertices, currentGraph.getEdges());
129   }
