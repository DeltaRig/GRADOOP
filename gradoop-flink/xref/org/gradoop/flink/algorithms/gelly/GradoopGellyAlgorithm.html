30  /**
31   * Base class for Algorithms executed in Flink Gelly that returns a {@link BaseGraph}.
32   *
33   * @param <G>  Gradoop graph head type.
34   * @param <V>  Gradoop vertex type.
35   * @param <E>  Gradoop edge type.
36   * @param <LG> Gradoop type of the graph.
37   * @param <GC> Gradoop type of the graph collection.
38   * @param <VV> Value type for gelly vertices.
39   * @param <EV> Value type for gelly edges.
40   */
41  public abstract class GradoopGellyAlgorithm<
42    G extends GraphHead,
43    V extends Vertex,
44    E extends Edge,
45    LG extends BaseGraph<G, V, E, LG, GC>,
46    GC extends BaseGraphCollection<G, V, E, LG, GC>,
47    VV, EV> extends BaseGellyAlgorithm<G, V, E, LG, GC, GradoopId, VV, EV, LG>
48    implements UnaryBaseGraphToBaseGraphOperator<LG> {
49  
50    /**
51     * The graph used in {@link #execute}.
52     */
53    protected BaseGraph<G, V, E, LG, GC> currentGraph;
54  
55    /**
56     * Function mapping edge to gelly edge.
57     */
58    private final EdgeToGellyEdge<E, EV> toGellyEdge;
59  
60    /**
61     * Function mapping vertex to gelly vertex.
62     */
63    private final VertexToGellyVertex<V, VV> toGellyVertex;
64  
65    /**
66     * Base constructor, only setting the mapper functions.
67     *
68     * @param vertexValue Function mapping vertices from Gradoop to Gelly.
69     * @param edgeValue   function mapping edges from Gradoop to Gelly.
70     */
71    protected GradoopGellyAlgorithm(
72      VertexToGellyVertex<V, VV> vertexValue, EdgeToGellyEdge<E, EV> edgeValue) {
73      this.toGellyVertex = vertexValue;
74      this.toGellyEdge = edgeValue;
75    }
76  
77    @Override
78    public LG execute(LG graph) {
79      this.currentGraph = graph;
80      try {
81        return executeInGelly(transformToGelly(graph));
82      } catch (Exception e) {
83        throw new RuntimeException(e);
84      }
85    }
86  
87    /**
88     * Default transformation from a Gradoop Graph to a Gelly Graph.
89     *
90     * @param graph Gradoop Graph.
91     * @return Gelly Graph.
92     */
93    public Graph<GradoopId, VV, EV> transformToGelly(LG graph) {
94      DataSet<org.apache.flink.graph.Vertex<GradoopId, VV>> gellyVertices = graph.getVertices()
95        .map(toGellyVertex);
96      DataSet<org.apache.flink.graph.Edge<GradoopId, EV>> gellyEdges = graph.getEdges().map(toGellyEdge);
97      return Graph.fromDataSet(gellyVertices, gellyEdges, graph.getConfig().getExecutionEnvironment());
98    }
99  
100   /**
101    * Perform some operation in Gelly and transform the Gelly graph back to a Gradoop {@link BaseGraph}.
102    *
103    * @param gellyGraph The Gelly graph.
104    * @return The Gradoop graph.
105    * @throws Exception on failure
106    */
107   public abstract LG executeInGelly(Graph<GradoopId, VV, EV> gellyGraph) throws Exception;
108 }
