16  package org.gradoop.flink.algorithms.gelly.connectedcomponents;
17  
18  import org.gradoop.common.model.api.entities.Edge;
19  import org.gradoop.common.model.api.entities.GraphHead;
20  import org.gradoop.common.model.api.entities.Vertex;
21  import org.gradoop.flink.model.api.epgm.BaseGraph;
22  import org.gradoop.flink.model.api.epgm.BaseGraphCollection;
23  import org.gradoop.flink.model.api.operators.UnaryBaseGraphToBaseGraphCollectionOperator;
24  
25  /**
26   * Computes the weakly connected components of a graph. Uses the gradoop wrapper
27   * {@link AnnotateWeaklyConnectedComponents} of Flinks ConnectedComponents.
28   * Splits the resulting {@link BaseGraph} into a {@link BaseGraphCollection} of its weakly connected
29   * components.
30   *
31   * @param <G>  Gradoop graph head type.
32   * @param <V>  Gradoop vertex type.
33   * @param <E>  Gradoop edge type.
34   * @param <LG> Gradoop type of the graph.
35   * @param <GC> Gradoop type of the graph collection.
36   */
37  public class WeaklyConnectedComponentsAsCollection<
38    G extends GraphHead,
39    V extends Vertex,
40    E extends Edge,
41    LG extends BaseGraph<G, V, E, LG, GC>,
42    GC extends BaseGraphCollection<G, V, E, LG, GC>>
43    implements UnaryBaseGraphToBaseGraphCollectionOperator<LG, GC> {
44  
45    /**
46     * Default property key to temporarily store the component id.
47     */
48    private static final String DEFAULT_PROPERTY_KEY = "wcc_component_id";
49  
50    /**
51     * Property key to temporarily store the component id.
52     */
53    private final String propertyKey;
54  
55    /**
56     * Maximum number of iterations.
57     */
58    private final int maxIterations;
59  
60    /**
61     * Initialize the operator.
62     *
63     * @param maxIterations Maximum number of iterations for
64     *                      {@link AnnotateWeaklyConnectedComponents}.
65     */
66    public WeaklyConnectedComponentsAsCollection(int maxIterations) {
67      this(DEFAULT_PROPERTY_KEY, maxIterations);
68    }
69  
70    /**
71     * Initialize the operator.
72     *
73     * @param propertyKey   Property key to temporarily store the component id.
74     * @param maxIterations Maximum number of iteration for
75     *                      {@link AnnotateWeaklyConnectedComponents}.
76     */
77    public WeaklyConnectedComponentsAsCollection(String propertyKey, int maxIterations) {
78      this.propertyKey = propertyKey;
79      this.maxIterations = maxIterations;
80    }
81  
82    @Override
83    public GC execute(LG graph) {
84  
85      LG graphWithWccIds = graph.callForGraph(
86        new AnnotateWeaklyConnectedComponents<>(propertyKey, maxIterations));
87  
88      return graphWithWccIds.splitBy(propertyKey);
89    }
90  }
