36  /**
37   * Data source to create a {@link LogicalGraph} from an edge list.
38   *
39   * 0 1
40   * 2 0
41   *
42   * The example line denotes an edge between two vertices:
43   * First edge:
44   * source: with id = 0
45   * target: with id = 1
46   *
47   * Second edge:
48   * source: with id = 2
49   * target: with id = 0
50   */
51  public class EdgeListDataSource implements DataSource {
52    /**
53     * Path to edge list file
54     */
55    private String edgeListPath;
56    /**
57     * Separator token of the tsv file
58     */
59    private String tokenSeparator;
60    /**
61     * Gradoop Flink configuration
62     */
63    private GradoopFlinkConfig config;
64  
65    /**
66     * Creates a new data source. Paths can be local (file://) or HDFS (hdfs://).
67     *
68     * @param edgeListPath    Path to edge list file
69     * @param tokenSeparator  Id separator token
70     * @param config          Gradoop Flink configuration
71     */
72    public EdgeListDataSource(String edgeListPath, String tokenSeparator,
73      GradoopFlinkConfig config) {
74      this.edgeListPath = edgeListPath;
75      this.tokenSeparator = tokenSeparator;
76      this.config = config;
77    }
78  
79    @Override
80    public LogicalGraph getLogicalGraph() throws IOException {
81      return getGraphCollection().reduce(new ReduceCombination<>());
82    }
83  
84    @Override
85    public GraphCollection getGraphCollection() throws IOException {
86  
87      ExecutionEnvironment env = getConfig().getExecutionEnvironment();
88  
89      //--------------------------------------------------------------------------
90      // generate tuple that contains all information
91      //--------------------------------------------------------------------------
92  
93      DataSet<Tuple2<Long, Long>> lineTuples = env
94        .readCsvFile(getEdgeListPath())
95        .fieldDelimiter(getTokenSeparator())
96        .types(Long.class, Long.class);
97  
98      //--------------------------------------------------------------------------
99      // generate vertices
100     //--------------------------------------------------------------------------
101 
102     DataSet<ImportVertex<Long>> importVertices = lineTuples
103       .<Tuple1<Long>>project(0)
104       .union(lineTuples.project(1))
105       .distinct()
106       .map(new CreateImportVertex<>());
107 
108     //--------------------------------------------------------------------------
109     // generate edges
110     //--------------------------------------------------------------------------
111 
112     DataSet<ImportEdge<Long>> importEdges = DataSetUtils
113       .zipWithUniqueId(lineTuples)
114       .map(new CreateImportEdge<>());
115 
116     //--------------------------------------------------------------------------
117     // create graph data source
118     //--------------------------------------------------------------------------
119 
120     return new GraphDataSource<>(importVertices, importEdges, getConfig())
121       .getGraphCollection();
122   }
123 
124   GradoopFlinkConfig getConfig() {
125     return config;
126   }
127 
128   String getEdgeListPath() {
129     return edgeListPath;
130   }
131 
132   String getTokenSeparator() {
133     return tokenSeparator;
134   }
135 }
