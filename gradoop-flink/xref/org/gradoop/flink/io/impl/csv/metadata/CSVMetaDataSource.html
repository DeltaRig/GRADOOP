42  /**
43   * Implementation of a meta data factory that reads from csv files.
44   */
45  public class CSVMetaDataSource implements MetaDataSource<CSVMetaData> {
46  
47    @Override
48    public CSVMetaData fromTuples(List<Tuple3<String, String, String>> metaDataStrings) {
49      Map<String, List<PropertyMetaData>> graphMetaDataMap = new HashMap<>(metaDataStrings.size());
50      Map<String, List<PropertyMetaData>> vertexMetaDataMap = new HashMap<>(metaDataStrings.size());
51      Map<String, List<PropertyMetaData>> edgeMetaDataMap = new HashMap<>(metaDataStrings.size());
52  
53      for (Tuple3<String, String, String> tuple : metaDataStrings) {
54        List<PropertyMetaData> propertyMetaDataList;
55        if (tuple.f2.length() > 0) {
56          String[] propertyStrings = StringEscaper.split(
57            tuple.f2, PropertyMetaData.PROPERTY_DELIMITER);
58          propertyMetaDataList = new ArrayList<>(propertyStrings.length);
59          for (String propertyMetaData : propertyStrings) {
60            String[] propertyMetaDataTokens = StringEscaper.split(
61              propertyMetaData, PropertyMetaData.PROPERTY_TOKEN_DELIMITER, 2);
62            propertyMetaDataList.add(new PropertyMetaData(
63              StringEscaper.unescape(propertyMetaDataTokens[0]),
64              propertyMetaDataTokens[1],
65              CSVMetaDataParser.getPropertyValueParser(propertyMetaDataTokens[1])));
66          }
67        } else {
68          propertyMetaDataList = new ArrayList<>(0);
69        }
70        String label = StringEscaper.unescape(tuple.f1);
71        switch (tuple.f0) {
72        case GRAPH_TYPE:
73          graphMetaDataMap.put(label, propertyMetaDataList);
74          break;
75        case VERTEX_TYPE:
76          vertexMetaDataMap.put(label, propertyMetaDataList);
77          break;
78        case EDGE_TYPE:
79          edgeMetaDataMap.put(label, propertyMetaDataList);
80          break;
81        default:
82          throw new IllegalArgumentException("Type " + tuple.f0 + " is not a valid epgm type " +
83            "string. Valid strings are g, v and e.");
84        }
85      }
86  
87      return new CSVMetaData(graphMetaDataMap, vertexMetaDataMap, edgeMetaDataMap);
88    }
89  
90    @Override
91    public DataSet<Tuple3<String, String, String>> readDistributed(String path, GradoopFlinkConfig
92      config) {
93      return config.getExecutionEnvironment()
94        .readTextFile(path)
95        .map(line -> StringEscaper.split(line, CSVConstants.TOKEN_DELIMITER, 3))
96        .map(tokens -> Tuple3.of(tokens[0], tokens[1], tokens[2]))
97        .returns(new TypeHint<Tuple3<String, String, String>>() {
98        });
99    }
100 
101   @Override
102   public CSVMetaData readLocal(String path, Configuration hdfsConfig) throws IOException {
103     FileSystem fs = FileSystem.get(hdfsConfig);
104     Path file = new Path(path);
105     Charset charset = StandardCharsets.UTF_8;
106 
107     if (!fs.exists(file)) {
108       throw new FileNotFoundException(path);
109     } else {
110 
111       try (BufferedReader br = new BufferedReader(new InputStreamReader(fs.open(file), charset))) {
112         return fromTuples(br.lines()
113           .map(line -> StringEscaper.split(line, CSVConstants.TOKEN_DELIMITER, 3))
114           .map(tokens -> Tuple3.of(tokens[0], tokens[1], tokens[2]))
115           .collect(Collectors.toList()));
116       }
117     }
118   }
