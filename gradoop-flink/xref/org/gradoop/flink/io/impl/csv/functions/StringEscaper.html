16  package org.gradoop.flink.io.impl.csv.functions;
17  
18  import com.google.common.collect.BiMap;
19  import com.google.common.collect.ImmutableBiMap;
20  
21  import java.util.ArrayList;
22  import java.util.List;
23  import java.util.Set;
24  
25  /**
26   * Escapes characters in strings and allows to split escaped strings.
27   */
28  public class StringEscaper {
29    /**
30     * Escape character.
31     */
32    private static final char ESCAPE_CHARACTER = '\\';
33    /**
34     * Custom escape sequences to avoid disruptive behavior of the file reader (e.g. newline).
35     */
36    private static final BiMap<Character, CharSequence> CUSTOM_ESCAPE_SEQUENCES =
37      new ImmutableBiMap.Builder<Character, CharSequence>()
38        .put('\t', String.format("%c%c", ESCAPE_CHARACTER, 't'))
39        .put('\b', String.format("%c%c", ESCAPE_CHARACTER, 'b'))
40        .put('\n', String.format("%c%c", ESCAPE_CHARACTER, 'n'))
41        .put('\r', String.format("%c%c", ESCAPE_CHARACTER, 'r'))
42        .put('\f', String.format("%c%c", ESCAPE_CHARACTER, 'f'))
43        .build();
44  
45    /**
46     * Escapes the {@code escapedCharacters} in a string.
47     *
48     * @param string string to be escaped
49     * @param escapedCharacters characters to be escaped
50     * @return escaped string
51     */
52    public static String escape(String string, Set<Character> escapedCharacters) {
53      StringBuilder sb = new StringBuilder();
54      for (char c : string.toCharArray()) {
55        if (escapedCharacters.contains(c)) {
56          sb.append(escapeCharacter(c));
57        } else {
58          sb.append(c);
59        }
60      }
61      return sb.toString();
62    }
63  
64    /**
65     * Unescapes the escaped characters in a string.
66     *
67     * @param escapedString string to be unescaped
68     * @return unescaped string
69     */
70    public static String unescape(String escapedString) {
71      StringBuilder sb = new StringBuilder();
72      boolean escaped = false;
73      for (int i = 0; i < escapedString.length(); i++) {
74        if (escaped) {
75          escaped = false;
76          sb.append(unescapeSequence(escapedString.subSequence(i - 1, i + 1)));
77        } else if (escapedString.charAt(i) == ESCAPE_CHARACTER) {
78          escaped = true;
79        } else {
80          sb.append(escapedString.charAt(i));
81        }
82      }
83      return sb.toString();
84    }
85  
86    /**
87     * Splits an escaped string while ignoring escaped delimiters. Does not unescape the tokens.
88     *
89     * @param escapedString escaped string to be split
90     * @param delimiter delimiter string
91     * @return string array with still escaped strings split by the delimiter
92     * @throws IllegalArgumentException if the delimiter contains the escape character
93     */
94    public static String[] split(String escapedString, String delimiter)
95      throws IllegalArgumentException {
96      return split(escapedString, delimiter, 0);
97    }
98  
99    /**
100    * Splits an escaped string while ignoring escaped delimiters. Does not unescape the tokens.
101    *
102    * @param escapedString escaped string to be split
103    * @param delimiter delimiter string
104    * @param limit limits the size of the output
105    * @return string array with still escaped strings split by the delimiter
106    * @throws IllegalArgumentException if the delimiter contains the escape character
107    */
108   public static String[] split(String escapedString, String delimiter, int limit)
109     throws IllegalArgumentException {
110     if (delimiter.contains(Character.toString(ESCAPE_CHARACTER))) {
111       throw new IllegalArgumentException(String.format(
112         "Delimiter must not contain the escape character: '%c'", ESCAPE_CHARACTER));
113     }
114     if (limit <= 0) {
115       limit = escapedString.length() + 1;
116     }
117 
118     List<String> tokens = new ArrayList<>();
119     StringBuilder sb = new StringBuilder();
120     boolean escaped = false;
121     int delimiterIndex = 0;
122     for (char c : escapedString.toCharArray()) {
123       // parse and match delimiter
124       if (!escaped && c == delimiter.charAt(delimiterIndex)) {
125         delimiterIndex++;
126         if (delimiterIndex == delimiter.length()) {
127           if (tokens.size() < limit - 1) {
128             tokens.add(sb.toString());
129             sb.setLength(0);
130           } else {
131             sb.append(delimiter, 0, delimiterIndex);
132           }
133           delimiterIndex = 0;
134         }
135       } else {
136         // reset delimiter parsing
137         sb.append(delimiter, 0, delimiterIndex);
138         delimiterIndex = 0;
139 
140         // escape
141         if (escaped) {
142           escaped = false;
143         } else if (c == ESCAPE_CHARACTER) {
144           escaped = true;
145         }
146 
147         sb.append(c);
148       }
149     }
150     sb.append(delimiter, 0, delimiterIndex);
151     tokens.add(sb.toString());
152     return tokens.toArray(new String[0]);
153   }
154 
155   /**
156    * Returns the escape sequence of a given character.
157    *
158    * @param character character to be escaped
159    * @return escape sequence
160    */
161   private static CharSequence escapeCharacter(char character) {
162     if (CUSTOM_ESCAPE_SEQUENCES.containsKey(character)) {
163       return CUSTOM_ESCAPE_SEQUENCES.get(character);
164     }
165     return String.format("%c%c", ESCAPE_CHARACTER, character);
166   }
167 
168   /**
169    * Returns the character of a given escape sequence.
170    *
171    * @param sequence escape sequence
172    * @return escaped character
173    */
174   private static char unescapeSequence(CharSequence sequence) {
175     if (CUSTOM_ESCAPE_SEQUENCES.containsValue(sequence)) {
176       return CUSTOM_ESCAPE_SEQUENCES.inverse().get(sequence);
177     }
178     return sequence.charAt(1);
179   }
