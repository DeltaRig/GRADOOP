16  package org.gradoop.flink.io.impl.csv;
17  
18  import org.gradoop.flink.io.impl.csv.functions.StringEscaper;
19  import org.gradoop.flink.io.impl.csv.indexed.functions.MultipleFileOutputFormat;
20  import org.gradoop.flink.util.GradoopFlinkConfig;
21  
22  import java.io.File;
23  import java.util.Objects;
24  
25  /**
26   * Base class for CSV data source and data sink.
27   */
28  public abstract class CSVBase {
29    /**
30     * Broadcast set identifier for meta data.
31     */
32    public static final String BC_METADATA = "metadata";
33    /**
34     * File ending for CSV files.
35     */
36    private static final String CSV_FILE_SUFFIX = ".csv";
37    /**
38     * Path for indexed vertices
39     */
40    private static final String VERTEX_PATH = "vertices";
41    /**
42     * CSV file for vertices.
43     */
44    private static final String VERTEX_FILE = "vertices" + CSV_FILE_SUFFIX;
45    /**
46     * Path for indexed graph heads.
47     */
48    private static final String GRAPH_HEAD_PATH = "graphs";
49    /**
50     * CSV file containing the graph heads.
51     */
52    private static final String GRAPH_HEAD_FILE = "graphs" + CSV_FILE_SUFFIX;
53    /**
54     * Path for indexed edges
55     */
56    private static final String EDGE_PATH = "edges";
57    /**
58     * CSV file for edges.
59     */
60    private static final String EDGE_FILE = "edges" + CSV_FILE_SUFFIX;
61    /**
62     * CSV file for meta data.
63     */
64    private static final String METADATA_FILE = "metadata" + CSV_FILE_SUFFIX;
65    /**
66     * Root directory containing the CSV and metadata files.
67     */
68    private final String csvRoot;
69    /**
70     * Gradoop Flink configuration
71     */
72    private final GradoopFlinkConfig config;
73  
74    /**
75     * Constructor.
76     *
77     * @param csvPath directory to the CSV files
78     * @param config  Gradoop Flink configuration
79     */
80    protected CSVBase(String csvPath, GradoopFlinkConfig config) {
81      Objects.requireNonNull(csvPath);
82      Objects.requireNonNull(config);
83      this.csvRoot = csvPath.endsWith(File.separator) ? csvPath : csvPath + File.separator;
84      this.config = config;
85    }
86  
87    /**
88     * Returns the path to the graph head directory.
89     *
90     * @return graph head path
91     */
92    protected String getGraphHeadPath() {
93      return csvRoot + GRAPH_HEAD_PATH;
94    }
95  
96    /**
97     * Returns the path to the vertex directory.
98     *
99     * @return vertex path
100    */
101   protected String getVertexPath() {
102     return csvRoot + VERTEX_PATH;
103   }
104 
105   /**
106    * Returns the path to the edge directory.
107    *
108    * @return edge path
109    */
110   protected String getEdgePath() {
111     return csvRoot + EDGE_PATH;
112   }
113 
114   /**
115    * Returns the path to the graph head file.
116    *
117    * @return graph head file path
118    */
119   protected String getGraphHeadCSVPath() {
120     return csvRoot + GRAPH_HEAD_FILE;
121   }
122 
123   /**
124    * Returns the path to the vertex file.
125    *
126    * @return vertex file path
127    */
128   protected String getVertexCSVPath() {
129     return csvRoot + VERTEX_FILE;
130   }
131 
132   /**
133    * Returns the path to the edge file.
134    *
135    * @return edge file path
136    */
137   protected String getEdgeCSVPath() {
138     return csvRoot + EDGE_FILE;
139   }
140 
141   /**
142    * Returns the path to the graph head file containing only graph heads with the specified label.
143    *
144    * @param label graph head label
145    * @return path to csv file
146    */
147   protected String getGraphHeadCSVPath(String label) {
148     return getElementCSVPath(label, getGraphHeadPath());
149   }
150 
151   /**
152    * Returns the path to the vertex file containing only vertices with the specified label.
153    *
154    * @param label vertex label
155    * @return path to csv file
156    */
157   protected String getVertexCSVPath(String label) {
158     return getElementCSVPath(label, getVertexPath());
159   }
160 
161   /**
162    * Returns the path to the edge file containing only edges with the specified label.
163    *
164    * @param label edge label
165    * @return path to csv file
166    */
167   protected String getEdgeCSVPath(String label) {
168     return getElementCSVPath(label, getEdgePath());
169   }
170 
171   /**
172    * Returns the path to the element file containing only elements with the specified label.
173    *
174    * @param label       element label
175    * @param elementPath path of the element (e.g. "edge")
176    * @return path to csv file
177    */
178   private String getElementCSVPath(String label, String elementPath) {
179     Objects.requireNonNull(label);
180     if (label.isEmpty()) {
181       label = CSVConstants.DEFAULT_DIRECTORY;
182     } else {
183       label = MultipleFileOutputFormat
184         .cleanFilename(StringEscaper.escape(label, CSVConstants.ESCAPED_CHARACTERS));
185     }
186     return elementPath +
187       CSVConstants.DIRECTORY_SEPARATOR +
188       label +
189       CSVConstants.DIRECTORY_SEPARATOR +
190       CSVConstants.SIMPLE_FILE;
191   }
192 
193   /**
194    * Get the path to the metadata file.
195    *
196    * @return the path to the metadata file as string
197    */
198   protected String getMetaDataPath() {
199     return csvRoot + METADATA_FILE;
200   }
201 
202   /**
203    * Get the gradoop configuration.
204    *
205    * @return the gradoop configuration instance
206    */
207   protected GradoopFlinkConfig getConfig() {
208     return config;
209   }
210 }
