38  /**
39   * Responsible for creating a {@link MetaData} instance from its string representation.
40   */
41  public class CSVMetaDataParser {
42    /**
43     * Used to map a simple type string to its corresponding parsing function.
44     */
45    private static final Map<String, Function<String, Object>> SIMPLE_TYPE_PARSER_MAP =
46      getSimpleTypeParserMap();
47  
48    /**
49     * Creates the type - parser function mapping of static property SIMPLE_TYPE_PARSER_MAP
50     *
51     * @return a HashMap containing the mapping of a simple type string to its corresponding
52     * parsing function.
53     */
54    private static Map<String, Function<String, Object>> getSimpleTypeParserMap() {
55      Map<String, Function<String, Object>> map = new HashMap<>();
56      map.put(Type.SHORT.toString(), Short::parseShort);
57      map.put(Type.INTEGER.toString(), Integer::parseInt);
58      map.put(Type.LONG.toString(), Long::parseLong);
59      map.put(Type.FLOAT.toString(), Float::parseFloat);
60      map.put(Type.DOUBLE.toString(), Double::parseDouble);
61      map.put(Type.BOOLEAN.toString(), Boolean::parseBoolean);
62      map.put(Type.STRING.toString(), StringEscaper::unescape);
63      map.put(Type.BIG_DECIMAL.toString(), BigDecimal::new);
64      map.put(Type.GRADOOP_ID.toString(), GradoopId::fromString);
65      map.put(Type.DATE.toString(), LocalDate::parse);
66      map.put(Type.TIME.toString(), LocalTime::parse);
67      map.put(Type.DATE_TIME.toString(), LocalDateTime::parse);
68      map.put(Type.NULL.toString(), CSVMetaDataParser::parseNullProperty);
69      return Collections.unmodifiableMap(map);
70    }

72    /**
73     * Returns the property meta data string for the specified property. The string consists of the
74     * property key and the property value type, e.g. "foo:int".
75     *
76     * @param property property
77     * @return property meta data
78     */
79    public static String getPropertyMetaData(Property property) {
80      return String.format("%s%s%s",
81        StringEscaper.escape(property.getKey(), CSVConstants.ESCAPED_CHARACTERS),
82        MetaData.PROPERTY_TOKEN_DELIMITER,
83        MetaData.getTypeString(property.getValue())); // no need to escape
84    }

86    /**
87     * Creates a parsing function for the given property type.
88     *
89     * @param propertyType string specifying the property type
90     * @return parsing function for the specific type
91     */
92    static Function<String, Object> getPropertyValueParser(String propertyType) {
93      String[] typeTokens = StringEscaper.split(
94        propertyType.toLowerCase(), PropertyMetaData.PROPERTY_TOKEN_DELIMITER);
95      String mainType = typeTokens[0];
96  
97      if (mainType.equals(Type.LIST.toString())) {
98        return getListValueParser(typeTokens);
99      } else if (mainType.equals(Type.SET.toString())) {
100       return getSetValueParser(typeTokens);
101     } else if (mainType.equals(Type.MAP.toString())) {
102       return getMapValueParser(typeTokens);
103     } else if (SIMPLE_TYPE_PARSER_MAP.containsKey(mainType)) {
104       return SIMPLE_TYPE_PARSER_MAP.get(mainType);
105     } else {
106       throw new TypeNotPresentException(mainType, null);
107     }
108   }

110   /**
111    * Creates a parsing function for list property type.
112    *
113    * @param listTypeTokens string tokens of the list type and its items type, e.g.
114    *                       ["list", "string"]
115    * @return parsing function
116    */
117   private static Function<String, Object> getListValueParser(String[] listTypeTokens) {
118     // It's a list with one additional data type (type of list items).
119     if (listTypeTokens.length != 2) {
120       throw new IllegalArgumentException("Item type of List type is missing");
121     }
122     final String itemType = listTypeTokens[1];
123     // check the validity of the list item type
124     if (!SIMPLE_TYPE_PARSER_MAP.containsKey(itemType)) {
125       throw new TypeNotPresentException(itemType, null);
126     }
127 
128     return s -> parseListProperty(s, SIMPLE_TYPE_PARSER_MAP.get(itemType));
129   }
130 
131   /**
132    * Creates a parsing function for map property type.
133    *
134    * @param mapTypeTokens string tokens of the map type and its key type and value type, e.g.
135    *                      ["map", "string", "double"]
136    * @return parsing function
137    */
138   private static Function<String, Object> getMapValueParser(String[] mapTypeTokens) {
139     // It's a map with two additional data types (key type + value type).
140     if (mapTypeTokens.length != 3) {
141       throw new IllegalArgumentException("Key type or value type of Map type is missing");
142     }
143 
144     final String keyType = mapTypeTokens[1];
145     // check the validity of the map key type
146     if (!SIMPLE_TYPE_PARSER_MAP.containsKey(keyType)) {
147       throw new TypeNotPresentException(keyType, null);
148     }
149 
150     final String valueType = mapTypeTokens[2];
151     // check the validity of the map value type
152     if (!SIMPLE_TYPE_PARSER_MAP.containsKey(valueType)) {
153       throw new TypeNotPresentException(valueType, null);
154     }
155 
156     return s -> parseMapProperty(
157       s,
158       SIMPLE_TYPE_PARSER_MAP.get(keyType),
159       SIMPLE_TYPE_PARSER_MAP.get(valueType)
160     );
161   }

163   /**
164    * Creates a parsing function for set property type.
165    *
166    * @param setTypeTokens string tokens of the set type and its item type, e.g. ["set", "string"]
167    * @return parsing function
168    */
169   private static Function<String, Object> getSetValueParser(String[] setTypeTokens) {
170     // It's a set with one additional data type (type of set items).
171     if (setTypeTokens.length != 2) {
172       throw new IllegalArgumentException("Item type of Set type is missing");
173     }
174     final String itemType = setTypeTokens[1];
175     // check the validity of the set item type
176     if (!SIMPLE_TYPE_PARSER_MAP.containsKey(itemType)) {
177       throw new TypeNotPresentException(itemType, null);
178     }
179 
180     return s -> parseSetProperty(s, SIMPLE_TYPE_PARSER_MAP.get(itemType));
181   }

183   /**
184    * Parse function to translate string representation of a List to a list of PropertyValues
185    *
186    * @param s          the string to parse as list, e.g. "[myString1,myString2]"
187    * @param itemParser the function to parse the list items
188    * @return the list represented by the argument
189    */
190   private static Object parseListProperty(String s, Function<String, Object> itemParser) {
191     s = s.substring(1, s.length() - 1);
192     return Arrays.stream(StringEscaper.split(s, CSVConstants.LIST_DELIMITER))
193       .map(itemParser)
194       .map(PropertyValue::create)
195       .collect(Collectors.toList());
196   }
197 
198   /**
199    * Parse function to translate string representation of a Map to a Map with
200    * key and value of type PropertyValue.
201    *
202    * @param s           the string to parse as map, e.g. "{myString1=myValue1,myString2=myValue2}"
203    * @param keyParser   the function to parse the keys
204    * @param valueParser the function to parse the values
205    * @return the map represented by the argument
206    */
207   private static Object parseMapProperty(String s, Function<String, Object> keyParser,
208     Function<String, Object> valueParser) {
209     s = s.substring(1, s.length() - 1);
210     return Arrays.stream(StringEscaper.split(s, CSVConstants.LIST_DELIMITER))
211       .map(st -> StringEscaper.split(st, CSVConstants.MAP_SEPARATOR))
212       .map(strings -> new Object[]{keyParser.apply(strings[0]), valueParser.apply(strings[1])})
213       .collect(Collectors.toMap(e -> PropertyValue.create(e[0]), e -> PropertyValue.create(e[1])));
214   }
215 
216   /**
217    * Parse function to translate string representation of a Set to a set of PropertyValues.
218    *
219    * @param s          the string to parse as set, e.g. "[myString1,myString2]"
220    * @param itemParser the function to parse the set items
221    * @return the set represented by the argument
222    */
223   private static Object parseSetProperty(String s, Function<String, Object> itemParser) {
224     s = s.substring(1, s.length() - 1);
225     return Arrays.stream(StringEscaper.split(s, CSVConstants.LIST_DELIMITER))
226       .map(itemParser)
227       .map(PropertyValue::create)
228       .collect(Collectors.toSet());
229   }

231   /**
232    * Parse function to create null from the null string representation.
233    *
234    * @param nullString The string representing null.
235    * @return Returns null
236    * @throws IllegalArgumentException The string that is passed has to represent null.
237    */
238   private static Object parseNullProperty(String nullString) throws IllegalArgumentException {
239     if (nullString != null && nullString.equalsIgnoreCase(
240       Type.NULL.toString())) {
241       return null;
242     } else {
243       throw new IllegalArgumentException("Only null represents a null string.");
244     }
245   }
