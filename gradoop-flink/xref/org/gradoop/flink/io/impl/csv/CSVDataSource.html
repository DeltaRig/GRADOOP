37  /**
38   * A graph data source for CSV files.
39   * <p>
40   * The datasource expects files separated by vertices and edges, e.g. in the following directory
41   * structure:
42   * <p>
43   * csvRoot
44   * |- vertices.csv # all vertex data
45   * |- edges.csv    # all edge data
46   * |- graphs.csv   # all graph head data
47   * |- metadata.csv # Meta data for all data contained in the graph
48   */
49  public class CSVDataSource extends CSVBase implements DataSource {
50  
51    /**
52     * Creates a new CSV data source.
53     *
54     * @param csvPath path to the directory containing the CSV files
55     * @param config  Gradoop Flink configuration
56     */
57    public CSVDataSource(String csvPath, GradoopFlinkConfig config) {
58      super(csvPath, config);
59    }
60  
61    /**
62     * Will use a single graph head of the collection as final graph head for the graph.
63     * Issue #1217 (https://github.com/dbs-leipzig/gradoop/issues/1217) will optimize further.
64     *
65     * {@inheritDoc}
66     */
67    @Override
68    public LogicalGraph getLogicalGraph() {
69      GraphCollection collection = getGraphCollection();
70      return collection.getGraphFactory()
71        .fromDataSets(
72          collection.getGraphHeads().first(1), collection.getVertices(), collection.getEdges());
73    }
74  
75    @Override
76    public GraphCollection getGraphCollection() {
77      GraphCollectionFactory collectionFactory = getConfig().getGraphCollectionFactory();
78      return getCollection(
79        new CSVLineToGraphHead(collectionFactory.getGraphHeadFactory()),
80        new CSVLineToVertex(collectionFactory.getVertexFactory()),
81        new CSVLineToEdge(collectionFactory.getEdgeFactory()),
82        collectionFactory);
83    }
84  
85    /**
86     * Create a graph collection from CSV lines using {@link CSVLineToElement} mapper functions.
87     *
88     * @param csvToGraphHead    A function mapping a CSV line to a graph head.
89     * @param csvToVertex       A function mapping a CSV line to a vertex.
90     * @param csvToEdge         A function mapping a CSV line to an edge.
91     * @param collectionFactory A factory used to create the final graph collection.
92     * @param <G>  The graph head type.
93     * @param <V>  The vertex type.
94     * @param <E>  The edge type.
95     * @param <LG> The graph type.
96     * @param <GC> The graph collection type.
97     * @return A graph collection representing the graph data stored as CSV.
98     */
99    protected <
100     G extends GraphHead,
101     V extends Vertex,
102     E extends Edge,
103     LG extends BaseGraph<G, V, E, LG, GC>,
104     GC extends BaseGraphCollection<G, V, E, LG, GC>> GC getCollection(
105       CSVLineToElement<G> csvToGraphHead,
106       CSVLineToElement<V> csvToVertex,
107       CSVLineToElement<E> csvToEdge,
108       BaseGraphCollectionFactory<G, V, E, LG, GC> collectionFactory) {
109 
110     // Read the meta data
111     DataSet<Tuple3<String, String, String>> metaData =
112       new CSVMetaDataSource().readDistributed(getMetaDataPath(), getConfig());
113 
114     // Read the datasets of each graph element
115     DataSet<G> graphHeads = getConfig().getExecutionEnvironment()
116       .readTextFile(getGraphHeadCSVPath())
117       .map(csvToGraphHead).withBroadcastSet(metaData, BC_METADATA);
118 
119     DataSet<V> vertices = getConfig().getExecutionEnvironment()
120       .readTextFile(getVertexCSVPath())
121       .map(csvToVertex).withBroadcastSet(metaData, BC_METADATA);
122 
123     DataSet<E> edges = getConfig().getExecutionEnvironment()
124       .readTextFile(getEdgeCSVPath())
125       .map(csvToEdge).withBroadcastSet(metaData, BC_METADATA);
126 
127     // Create the graph
128     return collectionFactory.fromDataSets(graphHeads, vertices, edges);
129   }
