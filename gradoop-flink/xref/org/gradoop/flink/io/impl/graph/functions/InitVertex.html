30  /**
31   * Initializes an EPGM vertex from the given {@link ImportVertex}.
32   *
33   * @param <K> Import EPGMEdge/EPGMVertex identifier type
34   */
35  @FunctionAnnotation.ForwardedFields(
36    "f0;" + // vertex id
37    "f1->f2.label;" + // vertex label
38    "f2->f2.properties" // vertex properties
39  )
40  public class InitVertex<K extends Comparable<K>>
41    extends InitElement<EPGMVertex, K>
42    implements MapFunction<ImportVertex<K>, Tuple3<K, GradoopId, EPGMVertex>>,
43    ResultTypeQueryable<Tuple3<K, GradoopId, EPGMVertex>> {
44  
45    /**
46     * Used to create new EPGM vertex.
47     */
48    private final VertexFactory<EPGMVertex> vertexFactory;
49  
50    /**
51     * Reduce object instantiation.
52     */
53    private final Tuple3<K, GradoopId, EPGMVertex> reuseTuple;
54  
55    /**
56     * Creates a new map function
57     * @param epgmVertexFactory       vertex factory
58     * @param lineagePropertyKey  property key to store import identifier
59     *                            (can be {@code null})
60     * @param externalIdType         type info for the import vertex identifier
61     */
62    public InitVertex(VertexFactory<EPGMVertex> epgmVertexFactory,
63      String lineagePropertyKey, TypeInformation<K> externalIdType) {
64      super(lineagePropertyKey, externalIdType);
65      this.vertexFactory      = epgmVertexFactory;
66      this.reuseTuple         = new Tuple3<>();
67    }
68  
69    /**
70     * Outputs a triple containing of the import vertex identifier, the new EPGM
71     * vertex identifier and the EPGM vertex.
72     *
73     * @param importVertex import vertex
74     * @return triple containing import and EPGM id as well as the EPGM vertex
75     * @throws Exception on failure
76     */
77    @Override
78    public Tuple3<K, GradoopId, EPGMVertex> map(ImportVertex<K> importVertex) throws
79      Exception {
80      reuseTuple.f0 = importVertex.getId();
81  
82      EPGMVertex vertex = vertexFactory.createVertex(importVertex.getLabel(),
83        importVertex.getProperties());
84  
85      reuseTuple.f1 = vertex.getId();
86      reuseTuple.f2 = updateLineage(vertex, importVertex.getId());
87  
88      return reuseTuple;
89    }
90  
91    @Override
92    public TypeInformation<Tuple3<K, GradoopId, EPGMVertex>> getProducedType() {
93      return new TupleTypeInfo<>(getKeyTypeInfo(),
94        TypeExtractor.getForClass(GradoopId.class),
95        TypeExtractor.createTypeInfo(vertexFactory.getType()));
96    }
97  }
