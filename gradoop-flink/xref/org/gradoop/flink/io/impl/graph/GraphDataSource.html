43  /**
44   * Transforms an external graph into an EPGM database. The external graph needs
45   * to be represented by a data set of {@link ImportVertex} and a data set of
46   * {@link ImportEdge}. This class transforms the external graph into an EPGM
47   * {@link LogicalGraph}.
48   *
49   * @param <K> External vertex/edge identifier type
50   */
51  public class GraphDataSource<K extends Comparable<K>> implements DataSource {
52  
53    /**
54     * Vertices to import.
55     */
56    private final DataSet<ImportVertex<K>> importVertices;
57  
58    /**
59     * Edges to import.
60     */
61    private final DataSet<ImportEdge<K>> importEdges;
62  
63    /**
64     * Used to store the lineage information at the resulting EPGM element.
65     */
66    private final String lineagePropertyKey;
67  
68    /**
69     * Gradoop config
70     */
71    private final GradoopFlinkConfig config;
72  
73  
74    /**
75     * Creates a new graph reader with no lineage information stored at the
76     * resulting EPGM graph.
77     *
78     * @param importVertices  vertices to import
79     * @param importEdges     edges to import
80     * @param config          gradoop config
81     */
82    public GraphDataSource(DataSet<ImportVertex<K>> importVertices,
83      DataSet<ImportEdge<K>> importEdges, GradoopFlinkConfig config) {
84      this(importVertices, importEdges, null, config);
85    }
86  
87    /**
88     * Creates a new graph reader.
89     *
90     * If the given {@code lineagePropertyKey} is {@code null}, no lineage info
91     * is stored.
92     *
93     * @param importVertices      vertices to import
94     * @param importEdges         edges to import
95     * @param lineagePropertyKey  property key to store import identifiers
96     *                            (can be {@code null})
97     * @param config              gradoop config
98     */
99    public GraphDataSource(DataSet<ImportVertex<K>> importVertices,
100     DataSet<ImportEdge<K>> importEdges, String lineagePropertyKey,
101     GradoopFlinkConfig config) {
102     this.importVertices     = checkNotNull(importVertices);
103     this.importEdges        = checkNotNull(importEdges);
104     this.lineagePropertyKey = lineagePropertyKey;
105     this.config             = checkNotNull(config);
106   }
107 
108   /**
109    * Transforms the import graph into a logical graph.
110    *
111    * @return logical graph
112    */
113   @Override
114   public LogicalGraph getLogicalGraph() {
115 
116     TypeInformation<K> externalIdType = ((TupleTypeInfo<?>) importVertices
117       .getType()).getTypeAt(0);
118 
119     LogicalGraphFactory factory = config.getLogicalGraphFactory();
120 
121     DataSet<Tuple3<K, GradoopId, EPGMVertex>> vertexTriples = importVertices
122       .map(new InitVertex<K>(
123         factory.getVertexFactory(), lineagePropertyKey, externalIdType));
124 
125     DataSet<EPGMVertex> epgmVertices = vertexTriples
126       .map(new Value2Of3<K, GradoopId, EPGMVertex>());
127 
128     DataSet<Tuple2<K, GradoopId>> vertexIdPair = vertexTriples
129       .map(new Project3To0And1<K, GradoopId, EPGMVertex>());
130 
131     DataSet<EPGMEdge> epgmEdges = importEdges
132       .join(vertexIdPair)
133       .where(1).equalTo(0)
134       .with(new InitEdge<K>(
135         factory.getEdgeFactory(), lineagePropertyKey, externalIdType))
136       .join(vertexIdPair)
137       .where(0).equalTo(0)
138       .with(new UpdateEdge<EPGMEdge, K>());
139 
140     return factory.fromDataSets(epgmVertices, epgmEdges);
141   }
142 
143   @Override
144   public GraphCollection getGraphCollection() throws IOException {
145     LogicalGraph logicalGraph = getLogicalGraph();
146     return logicalGraph.getCollectionFactory().fromGraph(logicalGraph);
147   }
