34  /**
35   * TLFRecordReader class to read through a given TLF document to
36   * output graph blocks as records which are specified by the start tag and
37   * end tag.
38   */
39  public class TLFRecordReader extends RecordReader<LongWritable, Text> {
40  
41    /**
42     * The start position of the split.
43     */
44    private final long start;
45  
46    /**
47     * The end position of the split.
48     */
49    private final long end;
50  
51    /**
52     * Input stream which reads the data from the split file.
53     */
54    private final FSDataInputStream fsin;
55  
56    /**
57     * Output buffer which writes only needed content.
58     */
59    private final DataOutputBuffer buffer = new DataOutputBuffer();
60  
61    /**
62     * The current key.
63     */
64    private LongWritable currentKey;
65  
66    /**
67     * The current value.
68     */
69    private Text currentValue;
70  
71    /**
72     * The length of the buffer data to be set to the value.
73     */
74    private int valueLength;
75  
76    /**
77     * Constructor for the reader which handles TLF splits and
78     * initializes the file input stream.
79     *
80     * @param split the split of the file containing all TLF content
81     * @param conf the configuration of the task attempt context
82     * @throws IOException on failure
83     */
84    public TLFRecordReader(FileSplit split, Configuration conf) throws
85      IOException {
86      // open the file and seek to the start of the split
87      start = split.getStart();
88      end = start + split.getLength();
89      Path file = split.getPath();
90      FileSystem fs = file.getFileSystem(conf);
91      fsin = fs.open(split.getPath());
92      fsin.seek(start);
93    }
94  
95    /**
96     * Reads the next key/value pair from the input for processing.
97     *
98     * @param key the new key
99     * @param value the new value
100    * @return true if a key/value pair was found
101    * @throws IOException on failure
102    */
103   private boolean next(LongWritable key, Text value) throws IOException {
104     if (fsin.getPos() < end &&
105       readUntilMatch(TLFConstants.START_TAG.getBytes(Charsets.UTF_8), false)) {
106       try {
107         buffer.write(TLFConstants.START_TAG.getBytes(Charsets.UTF_8));
108         if (readUntilMatch(TLFConstants.END_TAG.getBytes(Charsets.UTF_8), true)) {
109           key.set(fsin.getPos());
110           if (fsin.getPos() != end) {
111             //- end tag because it is the new start tag and shall not be added
112             valueLength = buffer.getLength() - TLFConstants.END_TAG.getBytes(Charsets.UTF_8).length;
113           } else {
114             // in this case there is no new start tag
115             valueLength = buffer.getLength();
116           }
117           //- end tag because it is the new start tag and shall not be added
118           value.set(buffer.getData(), 0, valueLength);
119           //set the buffer to position before end tag of old graph which is
120           // start tag of the new one
121           fsin.seek(fsin.getPos() - TLFConstants.END_TAG.getBytes(Charsets.UTF_8).length);
122           return true;
123         }
124       } finally {
125         buffer.reset();
126       }
127     }
128     return false;
129   }
130 
131   /**
132    * Reads the split and searches for matches with given 'match byte array'.
133    *
134    * @param match the match byte to be found
135    * @param withinBlock specifies if match is within the graph block
136    * @return true if match was found or the end of file was reached, so
137    * that the current block can be closed
138    * @throws IOException on failure
139    */
140   private boolean readUntilMatch(byte[] match, boolean withinBlock) throws
141     IOException {
142     int i = 0;
143     while (true) {
144       int b = fsin.read();
145       // end of file:
146       if (b == -1) {
147         return true;
148       }
149       // save to buffer:
150       if (withinBlock) {
151         buffer.write(b);
152       }
153       // check if we are matching:
154       if (b == match[i]) {
155         i++;
156         if (i >= match.length) {
157           return true;
158         }
159       } else {
160         i = 0;
161       }
162       // see if we have passed the stop point:
163       if (!withinBlock && i == 0 && fsin.getPos() >= end) {
164         return false;
165       }
166     }
167   }
168 
169   /**
170    * Closes open buffers
171    *
172    * @throws IOException on failure
173    */
174   @Override
175   public void close() throws IOException {
176     fsin.close();
177     buffer.close();
178   }

180   /**
181    * Returns the current process of input streaming.
182    *
183    * @return percentage of the completion
184    * @throws IOException on failure
185    */
186   @Override
187   public float getProgress() throws IOException {
188     return (fsin.getPos() - start) / (float) (end - start);
189   }

224   @Override

227   }

229   /**
230    * Reads the next kex/value pair from the input for processing.
231    *
232    * @return true if a key/value pair was found
233    * @throws IOException on failure
234    * @throws InterruptedException if interrupted
235    */
236   @Override
237   public boolean nextKeyValue() throws IOException, InterruptedException {
238     currentKey = new LongWritable();
239     currentValue = new Text();
240     return next(currentKey, currentValue);
241   }
