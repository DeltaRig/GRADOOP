318   /**
319    * Prepare the given edges for drawing. Assign them start- and end-coordinates from their
320    * vertices.
321    *
322    * @param vertices The vertices to take the edge-coordinates from
323    * @param edges    The raw edges
324    * @return The prepared edges
325    */
326   protected DataSet<EPGMEdge> prepareEdges(DataSet<EPGMVertex> vertices, DataSet<EPGMEdge> edges) {
327     edges = edges
328       .join(vertices).where(new SourceId<>()).equalTo(new Id<>())
329       .with(new SourceCoordinateJoin())
330       .join(vertices).where(new TargetId<>()).equalTo(new Id<>())
331       .with(new TargetCoordinateJoin());
332     return edges;
333   }

335   /**
336    * Scale the coordinates of the graph so that the layout-space matches the requested drawing-size
337    *
338    * @param vertices original vertices
339    * @return vertices with scaled coordinates
340    */
341   protected DataSet<EPGMVertex> scaleLayout(DataSet<EPGMVertex> vertices) {
342 
343     if (zoom) {
344       final int imageWidthF = imageWidth - 2 * zoomBorder;
345       final int imageHeightF = imageHeight - 2 * zoomBorder;
346       final int zoomBorderF = zoomBorder;
347 
348       DataSet<Tuple4<Integer, Integer, Integer, Integer>> minMaxCoords = vertices
349         .map(new ToCoordsTuple())
350         .aggregate(Aggregations.MIN, 0)
351         .and(Aggregations.MIN, 1)
352         .and(Aggregations.MAX, 2)
353         .and(Aggregations.MAX, 3);
354 
355       return vertices
356         .map(new VertexZoomMap(imageWidthF, imageHeightF, zoomBorderF))
357         .withBroadcastSet(minMaxCoords, BORDER_BROADCAST);
358 
359     } else {
360 
361       final double widthScale = imageWidth / (double) layoutWidth;
362       final double heightScale = imageHeight / (double) layoutHeight;
363       return vertices.map(new VertexScaleMap(widthScale, heightScale));
364     }
365   }

411   @Override
412   public void write(LogicalGraph logicalGraph, boolean overwrite) throws IOException {
413 
414     ImageOutputFormat pof = new ImageOutputFormat(path);
415     FileSystem.WriteMode writeMode =
416       overwrite ? FileSystem.WriteMode.OVERWRITE : FileSystem.WriteMode.NO_OVERWRITE;
417     pof.setWriteMode(writeMode);
418 
419     DataSet<EPGMVertex> vertices = scaleLayout(logicalGraph.getVertices());
420     DataSet<EPGMEdge> edges = prepareEdges(vertices, logicalGraph.getEdges());
421 
422     ImageGenerator imgg = new ImageGenerator(this);
423     DataSet<byte[]> image = edges.combineGroup(imgg::combineEdges).reduce(imgg::mergeImages);
424     if (!ignoreVertices) {
425       DataSet<byte[]> vertexImage =
426         vertices.combineGroup(imgg::combineVertices).reduce(imgg::mergeImages);
427       image = image.map(new RichMapFunction<byte[], byte[]>() {
428         @Override
429         public byte[] map(byte[] bufferedImage) throws Exception {
430           List<byte[]> vertexImage = this.getRuntimeContext().getBroadcastVariable("vertexImage");
431           return imgg.mergeImages(bufferedImage, vertexImage.get(0));
432         }
433       }).withBroadcastSet(vertexImage, "vertexImage");
434     }
435     image = image.map(imgg::addBackground);
436 
437     image.output(pof).setParallelism(1);
438   }

446    * This class contains functionality to create images from graph-parts.
447    * For some strange reasons BufferedImage can not be used as DataSet-Type without crashing the
448    * JVM. Therefore byte[] is used as intermediate-representation.
449    */
450   protected static class ImageGenerator implements Serializable {

452     /**
453      * Contains all necessary parameters
454      */
455     private ImageDataSink plotter;

458      * Create new ImageGenerator

460      * @param p Contains all necessary parameters (cannot use non-static class du to flink-madness)
461      */
462     public ImageGenerator(ImageDataSink p) {
463       this.plotter = p;
464     }

467     /**
468      * Combine multiple edges into one Image
469      *
470      * @param iterable  The edges to combine
471      * @param collector The output-collector
472      */
473     public void combineEdges(Iterable<EPGMEdge> iterable, Collector<byte[]> collector) {
474       BufferedImage img =
475         new BufferedImage(plotter.imageWidth, plotter.imageHeight, BufferedImage.TYPE_INT_ARGB);
476       Graphics2D gfx = img.createGraphics();
477       gfx.setColor(plotter.edgeColor);
478       gfx.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
479       gfx.setStroke(new BasicStroke(plotter.edgeSize));
480       for (EPGMEdge e : iterable) {
481         drawEdge(gfx, e);
482       }
483       collector.collect(imgToArr(img));
484       gfx.dispose();
485     }
486 
487     /**
488      * Combine multiple vertices into one Image
489      *
490      * @param iterable  The vertices to combine
491      * @param collector The output-collector
492      */
493     public void combineVertices(Iterable<EPGMVertex> iterable, Collector<byte[]> collector) {
494       BufferedImage img =
495         new BufferedImage(plotter.imageWidth, plotter.imageHeight, BufferedImage.TYPE_INT_ARGB);
496       Graphics2D gfx = img.createGraphics();
497       gfx.setColor(plotter.vertexColor);
498       for (EPGMVertex v : iterable) {
499         drawVertex(gfx, v);
500       }
501       collector.collect(imgToArr(img));
502       gfx.dispose();
503     }
504 
505     /**
506      * Draw a single edge
507      *
508      * @param gfx The graphics-object to use for drawing
509      * @param e   The edge to draw
510      */
511     private void drawEdge(Graphics2D gfx, EPGMEdge e) {
512       gfx.setColor(plotter.edgeColor);
513       float edgeSize = plotter.edgeSize;
514       if (plotter.dynamicEdgeSize && e.getPropertyValue("SIZE") != null) {
515         edgeSize *= Math.sqrt((float) e.getPropertyValue("SIZE").getInt());
516       }
517       gfx.setStroke(new BasicStroke(edgeSize));
518       try {
519         int sourceX = e.getPropertyValue(SOURCE_X).getInt();
520         int sourceY = e.getPropertyValue(SOURCE_Y).getInt();
521 
522         int targetX = e.getPropertyValue(TARGET_X).getInt();
523         int targetY = e.getPropertyValue(TARGET_Y).getInt();
524 
525         gfx.drawLine(sourceX, sourceY, targetX, targetY);
526       } catch (NullPointerException ef) {
527 
528       }
529     }
530 
531     /**
532      * Draw a single vertex
533      *
534      * @param gfx The graphics-object to use for drawing
535      * @param v   The vertex to draw
536      */
537     private void drawVertex(Graphics2D gfx, EPGMVertex v) {
538       int x = v.getPropertyValue(LayoutingAlgorithm.X_COORDINATE_PROPERTY).getInt();
539       int y = v.getPropertyValue(LayoutingAlgorithm.Y_COORDINATE_PROPERTY).getInt();
540       int size = plotter.vertexSize;
541       if (plotter.dynamicVertexSize && v.getPropertyValue("SIZE") != null) {
542         size *= Math.sqrt((double) v.getPropertyValue("SIZE").getInt());
543       }
544       gfx.fillOval(x - size / 2, y - size / 2, size, size);
545       if (plotter.vertexLabel != null) {
546         String label = v.getPropertyValue(plotter.vertexLabel).getString();
547         gfx.drawString(label, x, y + (plotter.vertexSize) + 10 + (plotter.vertexLabelSize / 2));
548       }
549     }

551     /**
552      * Merge two intermediate Images into one
553      *
554      * @param arr1 Image 1
555      * @param arr2 Image 2
556      * @return Output-Image
557      */
558     public byte[] mergeImages(byte[] arr1, byte[] arr2) {
559       BufferedImage bufferedImage = arrToImg(arr1);
560       BufferedImage t1 = arrToImg(arr2);
561       Graphics2D g = bufferedImage.createGraphics();
562       g.drawImage(t1, 0, 0, plotter.imageWidth, plotter.imageHeight, null);
563       g.dispose();
564       return imgToArr(bufferedImage);
565     }
566 
567     /**
568      * Draw a background behind the image.
569      *
570      * @param arr Input image
571      * @return Input-image + black background
572      */
573     public byte[] addBackground(byte[] arr) {
574       BufferedImage bufferedImage = arrToImg(arr);
575       BufferedImage out =
576         new BufferedImage(plotter.imageWidth, plotter.imageHeight, BufferedImage.TYPE_INT_ARGB);
577       Graphics2D gfx = out.createGraphics();
578       gfx.setColor(plotter.backgroundColor);
579       gfx.fillRect(0, 0, plotter.imageWidth, plotter.imageHeight);
580       gfx.drawImage(bufferedImage, 0, 0, plotter.imageWidth, plotter.imageHeight, null);
581       gfx.dispose();
582       return imgToArr(out);
583     }

611     /**
612      * Get the file extension of a file
613      *
614      * @param path The name/path of the file
615      * @return The extension (without dot)
616      */
617     private String getFileExtension(String path) {
618       return path.substring(path.lastIndexOf('.') + 1);
619     }

621     @Override
622     public void writeRecord(byte[] img) throws IOException {
623       String outputFormat = getFileExtension(path);
624       if (!outputFormat.equals(INTERMEDIATE_ENCODING)) {
625         BufferedImage bimg = arrToImg(img);
626         ImageIO.write(bimg, outputFormat, this.stream);
627       } else {
628         this.stream.write(img);
629       }
630     }
