25  /**
26   * Traversal optimized representation of a graph transaction.
27   * <ul>
28   *   <li>{@code f0: graph head}</li>
29   *   <li>{@code f1: vertex/edge id -> label}</li>
30   *   <li>{@code f2: vertex/edge id -> properties}</li>
31   *   <li>{@code f3: vertex id -> outgoing adjacency rows}</li>
32   *   <li>{@code f4: vertex id -> incoming adjacency rows}</li>
33   * </ul>
34   *
35   * @param <ID> ID type
36   * @param <L> label type
37   * @param <ED> edge data type
38   * @param <VD> vertex data type
39   */
40  public class AdjacencyList<ID extends Comparable<ID>, L extends Comparable<L>, ED, VD>
41    extends Tuple5<
42      EPGMGraphHead,
43      Map<ID, L>,
44      Map<ID, Properties>,
45      Map<ID, AdjacencyListRow<ED, VD>>,
46      Map<ID, AdjacencyListRow<ED, VD>>> {
47  
48    /**
49     * Default constructor.
50     */
51    public AdjacencyList() {
52    }
53  
54    /**
55     * Constructor.
56     *
57     * @param graphHead graph id
58     * @param labels graph, vertex and edge labels
59     * @param properties graph, vertex and edge properties
60     * @param outgoingRows outgoing adjacency list rows
61     * @param incomingRows incoming adjacency list rows
62     */
63    public AdjacencyList(
64      EPGMGraphHead graphHead,
65      Map<ID, L> labels,
66      Map<ID, Properties> properties,
67      Map<ID, AdjacencyListRow<ED, VD>> outgoingRows,
68      Map<ID, AdjacencyListRow<ED, VD>> incomingRows) {
69      super(graphHead, labels, properties, outgoingRows, incomingRows);
70    }
71  
72    /**
73     * Label accessor.
74     *
75     * @param elementId graph / vertex / edge id
76     * @return label
77     */
78    public L getLabel(ID elementId) {
79      return f1.get(elementId);
80    }
81  
82    /**
83     * Property accessor.
84     *
85     * @param elementId graph / vertex / edge id
86     * @return property list
87     */
88    public Properties getProperties(ID elementId) {
89      return f2.get(elementId);
90    }
91  
92    public EPGMGraphHead getGraphHead() {
93      return f0;
94    }
95  
96    public void setGraphHead(EPGMGraphHead graphId) {
97      this.f0 = graphId;
98    }
99  
100   public Map<ID, AdjacencyListRow<ED, VD>> getOutgoingRows() {
101     return f3;
102   }
103 
104   public Map<ID, AdjacencyListRow<ED, VD>> getIncomingRows() {
105     return f4;
106   }
