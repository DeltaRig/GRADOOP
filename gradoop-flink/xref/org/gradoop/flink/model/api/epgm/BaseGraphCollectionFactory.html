31  /**
32   * Responsible for creating instances of graph collections with type {@link GC} based on a specific
33   * {@link org.gradoop.flink.model.api.layouts.GraphCollectionLayout}.
34   *
35   * @param <G> type of the graph head
36   * @param <V> the vertex type
37   * @param <E> the edge type
38   * @param <LG> the type of the logical graph
39   * @param <GC> the type of the graph collection that will be created with this factory
40   */
41  public interface BaseGraphCollectionFactory<
42    G extends GraphHead,
43    V extends Vertex,
44    E extends Edge,
45    LG extends BaseGraph<G, V, E, LG, GC>,
46    GC extends BaseGraphCollection<G, V, E, LG, GC>> extends ElementFactoryProvider<G, V, E> {
47  
48    /**
49     * Get the layout factory responsible for creating a graph collection layout.
50     *
51     * @return The graph collection layout factory.
52     */
53    GraphCollectionLayoutFactory<G, V, E> getLayoutFactory();
54  
55    /**
56     * Sets the layout factory that is responsible for creating a graph collection layout.
57     *
58     * @param layoutFactory graph collection layout factory
59     */
60    void setLayoutFactory(GraphCollectionLayoutFactory<G, V, E> layoutFactory);
61  
62    /**
63     * Creates a collection from the given datasets.
64     *
65     * @param graphHeads  GraphHead DataSet
66     * @param vertices    Vertex DataSet
67     * @return Graph collection
68     */
69    GC fromDataSets(DataSet<G> graphHeads, DataSet<V> vertices);
70  
71    /**
72     * Creates a collection layout from the given datasets.
73     *
74     * @param graphHeads  GraphHead DataSet
75     * @param vertices    Vertex DataSet
76     * @param edges       Edge DataSet
77     * @return Graph collection
78     */
79    GC fromDataSets(DataSet<G> graphHeads, DataSet<V> vertices, DataSet<E> edges);
80  
81    /**
82     * Creates a graph collection from the given datasets. The method assumes that all vertices and
83     * edges are already assigned to the specified graph heads.
84     *
85     * @param graphHeads label indexed graph head dataset
86     * @param vertices label indexed vertex datasets
87     * @param edges label indexed edge datasets
88     * @return graph collection
89     */
90    GC fromIndexedDataSets(Map<String, DataSet<G>> graphHeads, Map<String, DataSet<V>> vertices,
91      Map<String, DataSet<E>> edges);
92  
93    /**
94     * Creates a collection layout from the given collections.
95     *
96     * @param graphHeads  Graph Head collection
97     * @param vertices    Vertex collection
98     * @param edges       Edge collection
99     * @return Graph collection
100    */
101   GC fromCollections(Collection<G> graphHeads, Collection<V> vertices, Collection<E> edges);
102 
103   /**
104    * Creates a graph collection from a given logical graph.
105    *
106    * @param logicalGraphLayout  input graph
107    * @return 1-element graph collection
108    */
109   GC fromGraph(LogicalGraphLayout<G, V, E> logicalGraphLayout);
110 
111   /**
112    * Creates a graph collection from multiple given logical graphs.
113    *
114    * @param logicalGraphLayout  input graphs
115    * @return graph collection
116    */
117   GC fromGraphs(LogicalGraphLayout<G, V, E>... logicalGraphLayout);
118 
119   /**
120    * Creates a graph collection from a graph transaction dataset.
121    *
122    * Overlapping vertices and edge are merged by Id comparison only.
123    *
124    * @param transactions  transaction dataset
125    * @return graph collection
126    */
127   GC fromTransactions(DataSet<GraphTransaction> transactions);
128 
129   /**
130    * Creates a graph collection layout from graph transactions.
131    *
132    * Overlapping vertices and edge are merged using provided reduce functions.
133    *
134    * @param transactions        transaction dataset
135    * @param vertexMergeReducer  vertex merge function
136    * @param edgeMergeReducer    edge merge function
137    * @return graph collection
138    */
139   GC fromTransactions(
140     DataSet<GraphTransaction> transactions,
141     GroupReduceFunction<V, V> vertexMergeReducer,
142     GroupReduceFunction<E, E> edgeMergeReducer);
143 
144   /**
145    * Creates an empty graph collection.
146    *
147    * @return empty graph collection
148    */
149   GC createEmptyCollection();
