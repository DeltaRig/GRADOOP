61  /**
62   * Defines the operators that are available on a {@link BaseGraphCollection}.
63   *
64   * @param <G> type of the graph head
65   * @param <V> the vertex type
66   * @param <E> the edge type
67   * @param <LG> the type of the base graph
68   * @param <GC> the type of the graph collection
69   */
70  public interface BaseGraphCollectionOperators<
71    G extends GraphHead,
72    V extends Vertex,
73    E extends Edge,
74    LG extends BaseGraph<G, V, E, LG, GC>,
75    GC extends BaseGraphCollection<G, V, E, LG, GC>> {

86     * @return base graph with given id or an empty base graph
87     */
88    LG getGraph(final GradoopId graphID);

117   default GC select(final FilterFunction<G> predicateFunction) {

127   default GC limit(int n) {

129   }

131   /**
132    * Matches a given pattern on a graph collection.
133    * The boolean flag {@code returnEmbeddings} specifies, if the return shall be the input graphs with
134    * a new property {@link AddMatchesToProperties#DEFAULT_KEY}, or a new collection consisting of the
135    * constructed embeddings.
136    *
137    * @param query the query pattern in GDL syntax
138    * @param algorithm custom pattern matching algorithm, e.g., {@link DepthSearchMatching}
139    * @param returnEmbeddings if true it returns the embeddings as a new graph collection
140    *                         if false it returns the input collection with a new property with key
141    *                         {@link AddMatchesToProperties#DEFAULT_KEY} and value true/false if the pattern
142    *                         is contained in the respective graph
143    * @return a graph collection containing either the embeddings or the input
144    * graphs with a new property with name {@link AddMatchesToProperties#DEFAULT_KEY}
145    */
146   default GC query(String query, PatternMatchingAlgorithm algorithm, boolean returnEmbeddings) {
147     return callForCollection(new TransactionalPatternMatching<>(query, algorithm, returnEmbeddings));
148   }
149 
150   /**
151    * Verifies each graph of this collection, removing dangling edges, i.e. edges pointing to or from
152    * a vertex not contained in the graph.<br>
153    * This operator can be applied after an operator that has not checked these graphs validity.
154    * The graph heads of these base graphs remains unchanged.
155    *
156    * @return this graph collection with all dangling edges removed.
157    */
158   default GC verify() {
159     return callForCollection(new VerifyCollection<>());
160   }
161 
162   /**
163    * Verifies this graph collection, removing dangling graph ids from its elements,
164    * i.e. ids not contained in the collection.<br>
165    * This operator can be applied after an operator that has not checked the graphs validity.
166    * The graph head of this graph collection remains unchanged.
167    *
168    * @return this graph collection with all dangling graph ids removed.
169    */
170   default GC verifyGraphsContainment() {
171     return callForCollection(new VerifyGraphsContainment<>());
172   }
