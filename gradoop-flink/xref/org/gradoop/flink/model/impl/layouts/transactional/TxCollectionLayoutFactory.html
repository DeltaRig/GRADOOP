55    @Override
56    public GraphCollectionLayout<EPGMGraphHead, EPGMVertex, EPGMEdge> fromDataSets(
57      DataSet<EPGMGraphHead> graphHeads, DataSet<EPGMVertex> vertices) {
58  
59      Objects.requireNonNull(graphHeads);
60      Objects.requireNonNull(vertices);
61  
62      return fromDataSets(graphHeads, vertices,
63        createEdgeDataSet(Lists.newArrayListWithCapacity(0)));
64    }

66    @Override
67    public GraphCollectionLayout<EPGMGraphHead, EPGMVertex, EPGMEdge> fromDataSets(
68      DataSet<EPGMGraphHead> inGraphHeads,
69      DataSet<EPGMVertex> inVertices,
70      DataSet<EPGMEdge> inEdges) {
71      Objects.requireNonNull(inGraphHeads);
72      Objects.requireNonNull(inVertices);
73      Objects.requireNonNull(inEdges);
74  
75      // Add a dummy graph head for entities which have no assigned graph
76      DataSet<EPGMGraphHead> dbGraphHead = getConfig().getExecutionEnvironment().fromElements(
77        getGraphHeadFactory()
78          .initGraphHead(GradoopConstants.DB_GRAPH_ID, GradoopConstants.DB_GRAPH_LABEL)
79      );
80      inGraphHeads = inGraphHeads.union(dbGraphHead);
81  
82      DataSet<Tuple2<GradoopId, EPGMGraphElement>> vertices = inVertices
83        .map(new Cast<>(EPGMGraphElement.class))
84        .returns(TypeExtractor.getForClass(EPGMGraphElement.class))
85        .flatMap(new GraphElementExpander<>());
86  
87      DataSet<Tuple2<GradoopId, EPGMGraphElement>> edges = inEdges
88        .map(new Cast<>(EPGMGraphElement.class))
89        .returns(TypeExtractor.getForClass(EPGMGraphElement.class))
90        .flatMap(new GraphElementExpander<>());
91  
92      DataSet<Tuple3<GradoopId, Set<EPGMVertex>, Set<EPGMEdge>>> transactions = vertices
93        .union(edges)
94        .groupBy(0)
95        .combineGroup(new GraphVerticesEdges())
96        .groupBy(0)
97        .reduceGroup(new GraphVerticesEdges());
98  
99      DataSet<GraphTransaction> graphTransactions = inGraphHeads
100       .leftOuterJoin(transactions)
101       .where(new Id<>()).equalTo(0)
102       .with(new TransactionFromSets());
103 
104     return new TxCollectionLayout(graphTransactions);
105   }

107   @Override
108   public GraphCollectionLayout<EPGMGraphHead, EPGMVertex, EPGMEdge> fromIndexedDataSets(
109     Map<String, DataSet<EPGMGraphHead>> graphHeads,
110     Map<String, DataSet<EPGMVertex>> vertices,
111     Map<String, DataSet<EPGMEdge>> edges) {
112     Objects.requireNonNull(graphHeads);
113     Objects.requireNonNull(vertices);
114     Objects.requireNonNull(edges);
115     return fromDataSets(
116       graphHeads.values().stream().reduce(DataSet::union)
117         .orElseThrow(() -> new RuntimeException("Error during graph head union")),
118       vertices.values().stream().reduce(DataSet::union)
119         .orElseThrow(() -> new RuntimeException("Error during vertex union")),
120       edges.values().stream().reduce(DataSet::union)
121         .orElseThrow(() -> new RuntimeException("Error during edge union"))
122     );
123   }

164   /**
165    * Creates a dataset from a given (possibly empty) collection of graph transactions.
166    *
167    * @param transactions graph transactions
168    * @return a dataset containing the given transactions
169    */
170   private DataSet<GraphTransaction> createGraphTransactionDataSet(
171     Collection<GraphTransaction> transactions) {
172     ExecutionEnvironment env = getConfig().getExecutionEnvironment();
173 
174     DataSet<GraphTransaction> graphTransactionSet;
175     if (transactions.isEmpty()) {
176       graphTransactionSet = env.fromCollection(Lists.newArrayList(new GraphTransaction()),
177         new TypeHint<GraphTransaction>() { }.getTypeInfo())
178         .filter(new False<>());
179     } else {
180       graphTransactionSet = env.fromCollection(transactions);
181     }
182     return graphTransactionSet;
183   }
