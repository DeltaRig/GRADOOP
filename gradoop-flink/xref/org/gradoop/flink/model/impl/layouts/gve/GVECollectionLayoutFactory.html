38  /**
39   * Responsible for creating a {@link GVELayout} from given data.
40   */
41  public class GVECollectionLayoutFactory extends GVEBaseFactory
42    implements GraphCollectionLayoutFactory<EPGMGraphHead, EPGMVertex, EPGMEdge> {
43  
44    @Override
45    public GraphCollectionLayout<EPGMGraphHead, EPGMVertex, EPGMEdge> fromDataSets(
46      DataSet<EPGMGraphHead> graphHeads, DataSet<EPGMVertex> vertices) {
47      return fromDataSets(graphHeads, vertices,
48        createEdgeDataSet(new ArrayList<>(0)));
49    }
50  
51    @Override
52    public GraphCollectionLayout<EPGMGraphHead, EPGMVertex, EPGMEdge> fromDataSets(
53      DataSet<EPGMGraphHead> graphHeads,
54      DataSet<EPGMVertex> vertices,
55      DataSet<EPGMEdge> edges) {
56      return create(graphHeads, vertices, edges);
57    }
58  
59    @Override
60    public GraphCollectionLayout<EPGMGraphHead, EPGMVertex, EPGMEdge> fromIndexedDataSets(
61      Map<String, DataSet<EPGMGraphHead>> graphHeads,
62      Map<String, DataSet<EPGMVertex>> vertices,
63      Map<String, DataSet<EPGMEdge>> edges) {
64      return create(graphHeads, vertices, edges);
65    }
66  
67    @Override
68    public GraphCollectionLayout<EPGMGraphHead, EPGMVertex, EPGMEdge> fromCollections(
69      Collection<EPGMGraphHead> graphHeads,
70      Collection<EPGMVertex> vertices,
71      Collection<EPGMEdge> edges) {
72      return fromDataSets(
73        createGraphHeadDataSet(Objects.requireNonNull(graphHeads, "EPGMGraphHead collection was null")),
74        createVertexDataSet(Objects.requireNonNull(vertices, "EPGMVertex collection was null")),
75        createEdgeDataSet(Objects.requireNonNull(edges, "EPGMEdge collection was null")));
76    }
77  
78    @Override
79    public GraphCollectionLayout<EPGMGraphHead, EPGMVertex, EPGMEdge> fromGraphLayout(
80      LogicalGraphLayout<EPGMGraphHead, EPGMVertex, EPGMEdge> graph) {
81      return fromDataSets(graph.getGraphHead(), graph.getVertices(), graph.getEdges());
82    }
83  
84    @Override
85    public GraphCollectionLayout<EPGMGraphHead, EPGMVertex, EPGMEdge> fromTransactions(
86      DataSet<GraphTransaction> transactions) {
87      GroupReduceFunction<EPGMVertex, EPGMVertex> vertexReducer = new First<>();
88      GroupReduceFunction<EPGMEdge, EPGMEdge> edgeReducer = new First<>();
89  
90      return fromTransactions(transactions, vertexReducer, edgeReducer);
91    }
92  
93    @Override
94    public GraphCollectionLayout<EPGMGraphHead, EPGMVertex, EPGMEdge> fromTransactions(
95      DataSet<GraphTransaction> transactions,
96      GroupReduceFunction<EPGMVertex, EPGMVertex> vertexMergeReducer,
97      GroupReduceFunction<EPGMEdge, EPGMEdge> edgeMergeReducer) {
98  
99      DataSet<EPGMGraphHead> graphHeads = transactions.map(new TransactionGraphHead<>());
100 
101     DataSet<EPGMVertex> vertices = transactions
102       .flatMap(new TransactionVertices<>())
103       .groupBy(new Id<>())
104       .reduceGroup(vertexMergeReducer);
105 
106     DataSet<EPGMEdge> edges = transactions
107       .flatMap(new TransactionEdges<>())
108       .groupBy(new Id<>())
109       .reduceGroup(edgeMergeReducer);
110 
111     return fromDataSets(graphHeads, vertices, edges);
112   }
113 
114   @Override
115   public GraphCollectionLayout<EPGMGraphHead, EPGMVertex, EPGMEdge> createEmptyCollection() {
116     Collection<EPGMGraphHead> graphHeads = new ArrayList<>();
117     Collection<EPGMVertex> vertices = new ArrayList<>();
118     Collection<EPGMEdge> edges = new ArrayList<>();
119 
120     return fromCollections(graphHeads, vertices, edges);
121   }
