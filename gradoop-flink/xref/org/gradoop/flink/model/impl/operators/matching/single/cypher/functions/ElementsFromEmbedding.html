43  /**
44   * Extracts elements from an {@link Embedding}.
45   *
46   * @param <G> The graph head type.
47   * @param <V> The vertex type.
48   * @param <E> The edge type.
49   */
50  public class ElementsFromEmbedding<
51    G extends GraphHead,
52    V extends Vertex,
53    E extends Edge> implements FlatMapFunction<Embedding, Element> {
54    /**
55     * Constructs graph heads
56     */
57    private final GraphHeadFactory<G> graphHeadFactory;
58    /**
59     * Constructs vertices
60     */
61    private final VertexFactory<V> vertexFactory;
62    /**
63     * Constructs edges
64     */
65    private final EdgeFactory<E> edgeFactory;
66    /**
67     * Describes the embedding content
68     */
69    private final EmbeddingMetaData metaData;
70    /**
71     * Source vertex variables by edge id
72     */
73    private final Map<String, Pair<String, String>> sourceTargetVariables;
74    /**
75     * Stores the mapping between query variable and element id. The mapping is added as a property
76     * to the graph head representing an embedding.
77     */
78    private final Map<PropertyValue, PropertyValue> variableMapping;
79    /**
80     * Stores the identifiers that have already been processed.
81     */
82    private final Set<GradoopId> processedIds;
83    /**
84     * Stores the mapping between return pattern variables and its labels
85     */
86    private final Map<String, String> labelMapping;
87  
88    /**
89     * Constructor.
90     *
91     * @param graphHeadFactory      graph head factory
92     * @param vertexFactory         vertex factory
93     * @param edgeFactory           edge factory
94     * @param embeddingMetaData     meta data for the embedding
95     * @param sourceTargetVariables source and target vertex variables by edge variable
96     */
97    public ElementsFromEmbedding(GraphHeadFactory<G> graphHeadFactory,
98      VertexFactory<V> vertexFactory, EdgeFactory<E> edgeFactory, EmbeddingMetaData embeddingMetaData,
99      Map<String, Pair<String, String>> sourceTargetVariables) {
100     this(graphHeadFactory, vertexFactory, edgeFactory, embeddingMetaData,
101       sourceTargetVariables, Maps.newHashMapWithExpectedSize(0));
102   }
103   /**
104    * Constructor.
105    *
106    * @param graphHeadFactory      graph head factory
107    * @param vertexFactory         vertex factory
108    * @param edgeFactory           edge factory
109    * @param embeddingMetaData     meta data for the embedding
110    * @param sourceTargetVariables source and target vertex variables by edge variable
111    * @param labelMapping          mapping between newElementVariables and its labels
112    */
113   public ElementsFromEmbedding(GraphHeadFactory<G> graphHeadFactory, VertexFactory<V> vertexFactory,
114     EdgeFactory<E> edgeFactory, EmbeddingMetaData embeddingMetaData,
115     Map<String, Pair<String, String>> sourceTargetVariables, Map<String, String> labelMapping) {
116     this.graphHeadFactory = graphHeadFactory;
117     this.vertexFactory = vertexFactory;
118     this.edgeFactory = edgeFactory;
119     this.metaData = embeddingMetaData;
120     this.sourceTargetVariables = sourceTargetVariables;
121     this.labelMapping = labelMapping;
122     this.variableMapping = new HashMap<>(embeddingMetaData.getEntryCount());
123     this.processedIds = new HashSet<>(embeddingMetaData.getEntryCount());
124   }
125 
126   @Override
127   public void flatMap(Embedding embedding, Collector<Element> out) throws Exception {
128     // clear for each embedding
129     processedIds.clear();
130 
131     // create graph head for this embedding
132     G graphHead = graphHeadFactory.createGraphHead();
133 
134     // vertices
135     for (String vertexVariable : metaData.getVertexVariables()) {
136       GradoopId id = embedding.getId(metaData.getEntryColumn(vertexVariable));
137 
138       if (labelMapping.containsKey(vertexVariable)) {
139         String label = labelMapping.get(vertexVariable);
140         initVertexWithData(out, graphHead, id, label);
141       } else {
142         initVertex(out, graphHead, id);
143       }
144       variableMapping.put(PropertyValue.create(vertexVariable), PropertyValue.create(id));
145     }
146 
147     // edges
148     GradoopId edgeId;
149     GradoopId sourceId;
150     GradoopId targetId;
151     for (String edgeVariable : metaData.getEdgeVariables()) {
152       edgeId = embedding.getId(metaData.getEntryColumn(edgeVariable));
153       sourceId = embedding.getId(
154         metaData.getEntryColumn(sourceTargetVariables.get(edgeVariable).getLeft()));
155       targetId = embedding.getId(
156         metaData.getEntryColumn(sourceTargetVariables.get(edgeVariable).getRight()));
157 
158       if (labelMapping.containsKey(edgeVariable)) {
159         String label = labelMapping.get(edgeVariable);
160         initEdgeWithData(out, graphHead, edgeId, sourceId, targetId, label);
161       } else {
162         initEdge(out, graphHead, edgeId, sourceId, targetId);
163       }
164       variableMapping.put(PropertyValue.create(edgeVariable), PropertyValue.create(edgeId));
165     }
166 
167     // paths
168     for (String pathVariable : metaData.getPathVariables()) {
169       ExpandDirection direction = metaData.getDirection(pathVariable);
170       List<GradoopId> path = embedding.getIdList(metaData.getEntryColumn(pathVariable));
171       List<PropertyValue> mappingValue = new ArrayList<>(path.size());
172       for (int i = 0; i < path.size(); i += 2) {
173         edgeId = path.get(i);
174         mappingValue.add(PropertyValue.create(edgeId));
175 
176         if (direction == ExpandDirection.OUT) {
177           sourceId = i > 0 ?
178             path.get(i - 1) :
179             embedding.getId(
180               metaData.getEntryColumn(sourceTargetVariables.get(pathVariable).getLeft()));
181 
182           targetId = i < path.size() - 1 ?
183             path.get(i + 1) :
184             embedding.getId(
185               metaData.getEntryColumn(sourceTargetVariables.get(pathVariable).getRight()));
186 
187           if (i + 2 < path.size()) {
188             mappingValue.add(PropertyValue.create(targetId));
189           }
190         } else {
191           sourceId = i < path.size() - 1 ?
192             path.get(i + 1) :
193             embedding.getId(
194               metaData.getEntryColumn(sourceTargetVariables.get(pathVariable).getLeft()));
195 
196           targetId = i > 0 ?
197             path.get(i - 1) :
198             embedding.getId(
199               metaData.getEntryColumn(sourceTargetVariables.get(pathVariable).getRight()));
200 
201           if (i > 0) {
202             mappingValue.add(PropertyValue.create(sourceId));
203           }
204         }
205 
206         initVertex(out, graphHead, sourceId);
207         initVertex(out, graphHead, targetId);
208         initEdge(out, graphHead, edgeId, sourceId, targetId);
209       }
210       variableMapping.put(PropertyValue.create(pathVariable), PropertyValue.create(mappingValue));
211     }
212 
213     graphHead.setProperty(PatternMatching.VARIABLE_MAPPING_KEY, variableMapping);
214     out.collect(graphHead);
215   }

217   /**
218    * Initializes an vertex using the specified parameters
219    *
220    * @param out flat map collector
221    * @param graphHead graph head to assign vertex to
222    * @param vertexId vertex identifier
223    */
224   private void initVertex(Collector<Element> out, G graphHead, GradoopId vertexId) {
225     initVertexWithData(out, graphHead, vertexId, null);
226   }
227 
228   /**
229    * Initializes an vertex using the specified parameters and adds its label
230    * if the given vertex was created for the return pattern.
231    *
232    * @param out flat map collector
233    * @param graphHead graph head to assign vertex to
234    * @param vertexId vertex identifier
235    * @param label label associated with vertex
236    */
237   private void initVertexWithData(Collector<Element> out, G graphHead, GradoopId vertexId,
238                                   String label) {
239     if (!processedIds.contains(vertexId)) {
240       V v = vertexFactory.initVertex(vertexId);
241       v.addGraphId(graphHead.getId());
242       v.setLabel(label);
243       out.collect(v);
244       processedIds.add(vertexId);
245     }
246   }
247 
248   /**
249    * Initializes an edge using the specified parameters.
250    *
251    * @param out flat map collector
252    * @param graphHead graph head to assign edge to
253    * @param edgeId edge identifier
254    * @param sourceId source vertex identifier
255    * @param targetId target vertex identifier
256    */
257   private void initEdge(Collector<Element> out, G graphHead, GradoopId edgeId,
258     GradoopId sourceId, GradoopId targetId) {
259     initEdgeWithData(out, graphHead, edgeId, sourceId, targetId, null);
260   }
261 
262   /**
263    * Initializes an edge using the specified parameters and adds its label
264    * if the given edge was created for return pattern
265    *
266    * @param out flat map collector
267    * @param graphHead graph head to assign edge to
268    * @param edgeId edge identifier
269    * @param sourceId source vertex identifier
270    * @param targetId target vertex identifier
271    * @param label label associated with edge
272    */
273   private void initEdgeWithData(Collector<Element> out, G graphHead, GradoopId edgeId,
274                                 GradoopId sourceId, GradoopId targetId, String label) {
275     if (!processedIds.contains(edgeId)) {
276       E e = edgeFactory.initEdge(edgeId, sourceId, targetId);
277       e.addGraphId(graphHead.getId());
278       e.setLabel(label);
279       out.collect(e);
280       processedIds.add(edgeId);
281     }
282   }
