32  import static org.gradoop.common.model.impl.properties.PropertyValue.NULL_VALUE;

34  /**
35   * Build a tuple-based representation of elements for grouping.
36   * Tuples will contain some Gradoop IDs, all grouping keys followed by all properties to be
37   * aggregated.
38   * <p>
39   * <i>Note: </i> This function sets all grouping keys and aggregate values, make sure to set
40   * additional fields, if {@code tupleDataOffset} is not {@code 0}.
41   *
42   * @param <E> The element type.
43   */
44  public class BuildTuplesFromElements<E extends Element>
45    implements MapFunction<E, Tuple>, ResultTypeQueryable<Tuple> {
46  
47    /**
48     * The grouping key functions.
49     */
50    private final List<KeyFunction<E, ?>> keys;
51  
52    /**
53     * The aggregate functions.
54     */
55    private final List<AggregateFunction> aggregateFunctions;
56  
57    /**
58     * The number of fields to be reserved for IDs.
59     * Those fields will be of type {@link GradoopId}.
60     */
61    private final int tupleDataOffset;
62  
63    /**
64     * The types of the produced tuple.
65     */
66    private final TypeInformation<?>[] elementTypes;
67  
68    /**
69     * Reduce object instantiations.
70     */
71    private final Tuple reuseTuple;
72  
73    /**
74     * Initialize this function, setting the grouping keys and aggregate functions.
75     *
76     * @param tupleDataOffset    The number of tuple fields reserved for IDs.
77     * @param keys               The grouping keys.
78     * @param aggregateFunctions The aggregate functions used to determine the aggregate property
79     */
80    public BuildTuplesFromElements(int tupleDataOffset, List<KeyFunction<E, ?>> keys,
81      List<AggregateFunction> aggregateFunctions) {
82      this.tupleDataOffset = tupleDataOffset;
83      if (tupleDataOffset < 0) {
84        throw new IllegalArgumentException("The number of reserved tuple fields must not be negative.");
85      }
86      this.keys = Objects.requireNonNull(keys);
87      this.aggregateFunctions = Objects.requireNonNull(aggregateFunctions);
88      final int tupleSize = tupleDataOffset + keys.size() + aggregateFunctions.size();
89      if (tupleSize > Tuple.MAX_ARITY) {
90        throw new UnsupportedOperationException("Number of elements is too high for tuple: " + tupleSize +
91          " (max.: " + Tuple.MAX_ARITY + ")");
92      }
93      elementTypes = new TypeInformation[tupleSize];
94      for (int i = 0; i < tupleDataOffset; i++) {
95        elementTypes[i] = TypeInformation.of(GradoopId.class);
96      }
97      // Fill grouping key types.
98      for (int i = 0; i < keys.size(); i++) {
99        elementTypes[i + tupleDataOffset] = keys.get(i).getType();
100     }
101     // Fill remaining spots with property value types.
102     for (int i = 0; i < aggregateFunctions.size(); i++) {
103       elementTypes[i + keys.size() + tupleDataOffset] = TypeInformation.of(PropertyValue.class);
104     }
105     reuseTuple = Tuple.newInstance(tupleSize);
106     // Fill first fields with default ID values.
107     for (int i = 0; i < tupleDataOffset; i++) {
108       reuseTuple.setField(GradoopId.NULL_VALUE, i);
109     }
110   }
111 
112   @Override
113   public TypeInformation<Tuple> getProducedType() {
114     return new TupleTypeInfo<>(elementTypes);
115   }
116 
117   @Override
118   public Tuple map(E element) throws Exception {
119     int field = tupleDataOffset;
120     for (KeyFunction<? super E, ?> key : keys) {
121       reuseTuple.setField(key.getKey(element), field);
122       field++;
123     }
124     for (AggregateFunction aggregateFunction : aggregateFunctions) {
125       final PropertyValue increment = aggregateFunction.getIncrement(element);
126       reuseTuple.setField(increment == null ? NULL_VALUE : increment, field);
127       field++;
128     }
129     return reuseTuple;
130   }
