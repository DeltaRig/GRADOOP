137   @Override
138   protected GC executeForVertex(LG graph) {
139     BaseGraphFactory<G, V, E, LG, GC> graphFactory = graph.getFactory();
140     GraphHeadFactory<G> graphHeadFactory = graphFactory.getGraphHeadFactory();
141     VertexFactory<V> vertexFactory = graphFactory.getVertexFactory();
142     String variable = getQueryHandler().getVertices().iterator().next().getVariable();
143 
144     DataSet<V> matchingVertices = graph.getVertices()
145       .filter(new MatchingVertices<>(getQuery()));
146 
147     if (!doAttachData()) {
148       matchingVertices = matchingVertices
149         .map(new Id<>())
150         .map(new ObjectTo1<>())
151         .map(new VertexFromId<>(vertexFactory));
152     }
153 
154     DataSet<Tuple2<V, G>> pairs = matchingVertices
155       .map(new AddGraphElementToNewGraph<>(graphHeadFactory, variable))
156       .returns(new TupleTypeInfo<>(
157         TypeExtractor.getForClass(vertexFactory.getType()),
158         TypeExtractor.getForClass(graphHeadFactory.getType())));
159 
160     return graph.getCollectionFactory().fromDataSets(
161       pairs.map(new Value1Of2<>()),
162       pairs.map(new Value0Of2<>()));
163   }

165   @Override
166   protected GC executeForPattern(LG graph) {
167 
168     TraversalCode traversalCode = traverser.traverse();
169 
170     DataSet<Tuple1<Embedding<GradoopId>>> embeddings;
171 
172     if (traverserStrategy == TraverserStrategy.SET_PAIR_BULK_ITERATION ||
173       traverserStrategy == TraverserStrategy.SET_PAIR_FOR_LOOP_ITERATION) {
174 
175       //--------------------------------------------------------------------------
176       // Pre-processing (filter candidates)
177       //--------------------------------------------------------------------------
178 
179       DataSet<IdWithCandidates<GradoopId>> vertices = PreProcessor.filterVertices(
180         graph, getQuery());
181 
182       DataSet<TripleWithCandidates<GradoopId>> edges = PreProcessor.filterEdges(
183         graph, getQuery());
184 
185       //--------------------------------------------------------------------------
186       // Exploration via Traversal
187       //--------------------------------------------------------------------------
188 
189       SetPairTraverser<GradoopId> distributedTraverser;
190 
191       if (traverserStrategy == TraverserStrategy.SET_PAIR_BULK_ITERATION) {
192         distributedTraverser = new SetPairBulkTraverser<>(traversalCode, matchStrategy,
193           traverser.getQueryHandler().getVertexCount(), traverser.getQueryHandler().getEdgeCount(),
194           GradoopId.class, edgeStepJoinStrategy, vertexStepJoinStrategy, getVertexMapping(),
195           getEdgeMapping());
196       } else {
197         distributedTraverser = new SetPairForLoopTraverser<>(traversalCode, matchStrategy,
198           traverser.getQueryHandler().getVertexCount(), traverser.getQueryHandler().getEdgeCount(),
199           GradoopId.class, edgeStepJoinStrategy, vertexStepJoinStrategy, getVertexMapping(),
200           getEdgeMapping());
201       }
202 
203       embeddings = distributedTraverser.traverse(vertices, edges);
204     } else if (traverserStrategy == TraverserStrategy.TRIPLES_FOR_LOOP_ITERATION) {
205       DataSet<TripleWithCandidates<GradoopId>> triples = PreProcessor
206         .filterTriplets(graph, getQuery());
207 
208       TripleTraverser<GradoopId> distributedTraverser = new TripleForLoopTraverser<>(
209         traversalCode, matchStrategy,
210         traverser.getQueryHandler().getVertexCount(),
211         traverser.getQueryHandler().getEdgeCount(),
212         GradoopId.class, edgeStepJoinStrategy, getVertexMapping(), getEdgeMapping());
213 
214       embeddings = distributedTraverser.traverse(triples);
215 
216     } else {
217       throw new IllegalArgumentException("Unsupported traverser strategy: " + traverserStrategy);
218     }
219 
220     //--------------------------------------------------------------------------
221     // Post-Processing (build Graph Collection from embeddings)
222     //--------------------------------------------------------------------------
223 
224     DataSet<Element> elements = embeddings
225       .flatMap(new ElementsFromEmbedding(traversalCode,
226         graph.getFactory().getGraphHeadFactory(),
227         graph.getFactory().getVertexFactory(),
228         graph.getFactory().getEdgeFactory(),
229         getQueryHandler()
230       ));
231 
232     return doAttachData() ?
233       PostProcessor.extractGraphCollectionWithData(elements, graph, true) :
234       PostProcessor.extractGraphCollection(elements, graph.getCollectionFactory(), true);
235   }
