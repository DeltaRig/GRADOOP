33  /**
34   * Wraps a {@link Comparison}
35   */
36  public class ComparisonExpression extends QueryPredicate {
37    /**
38     * Holds the wrapped comparison
39     */
40    private final Comparison comparison;
41  
42    /**
43     * Optional factory for query comparables
44     */
45    private QueryComparableFactory comparableFactory = null;
46  
47    /**
48     * Creates a new comparison wrapped
49     *
50     * @param comparison the wrapped comparison
51     */
52    public ComparisonExpression(Comparison comparison) {
53      this(comparison, null);
54    }
55  
56    /**
57     * Creates a new wrapped comparison and sets the query comparable factory
58     *
59     * @param comparison the wrapped comparison
60     * @param comparableFactory QueryComparableFactory to use
61     */
62    public ComparisonExpression(Comparison comparison, QueryComparableFactory comparableFactory) {
63      this.comparison = comparison;
64      this.comparableFactory = comparableFactory;
65    }
66  
67    /**
68     * Returns the wrapped left hand side of the comparison
69     *
70     * @return wrapped left hand side
71     */
72    public QueryComparable getLhs() {
73      return comparableFactory == null ?
74        QueryComparable.createFrom(comparison.getComparableExpressions()[0]) :
75        comparableFactory.createFrom(comparison.getComparableExpressions()[0]);
76    }
77  
78    /**
79     * Returns the wrapped left hand side of the comparison
80     *
81     * @return wrapped left hand side
82     */
83    public QueryComparable getRhs() {
84      return comparableFactory == null ?
85        QueryComparable.createFrom(comparison.getComparableExpressions()[1]) :
86        comparableFactory.createFrom(comparison.getComparableExpressions()[1]);
87    }
88  
89    public Comparator getComparator() {
90      return comparison.getComparator();
91    }
92  
93    /**
94     * Evaluates the comparisson for the given embedding
95     *
96     * @param embedding the embedding record holding the data
97     * @param metaData the embedding meta data
98     * @return evaluation result
99     */
100   public boolean evaluate(Embedding embedding, EmbeddingMetaData metaData) {
101     PropertyValue lhsValue = getLhs().evaluate(embedding, metaData);
102     PropertyValue rhsValue = getRhs().evaluate(embedding, metaData);
103 
104     return compare(lhsValue, rhsValue);
105   }
106 
107   /**
108    * Evaluates the comparison for the given graph element
109    *
110    * @param element EPGMGraphElement under which the comparison will be evaluated
111    * @return evaluation result
112    */
113   public boolean evaluate(GraphElement element) {
114     PropertyValue lhsValue = getLhs().evaluate(element);
115     PropertyValue rhsValue = getRhs().evaluate(element);
116     return compare(lhsValue, rhsValue);
117   }
118 
119   /**
120    * Compares two property values with the given comparator
121    *
122    * @param lhsValue left property value
123    * @param rhsValue right property value
124    * @return comparison result
125    */
126   private boolean compare(PropertyValue lhsValue, PropertyValue rhsValue) {
127     try {
128       int result = lhsValue.compareTo(rhsValue);
129 
130       return
131         comparison.getComparator() == Comparator.EQ  && result ==  0 ||
132           comparison.getComparator() == Comparator.NEQ && result !=  0 ||
133           comparison.getComparator() == Comparator.LT  && result == -1 ||
134           comparison.getComparator() == Comparator.GT  && result ==  1 ||
135           comparison.getComparator() == Comparator.LTE && result <=  0 ||
136           comparison.getComparator() == Comparator.GTE && result >=  0;
137 
138     } catch (IllegalArgumentException e) {
139       return comparison.getComparator() == Comparator.NEQ;
140     }
141   }
142 
143   /**
144    * Returns the variables referenced by the expression
145    *
146    * @return set of variables
147    */
148   public Set<String> getVariables() {
149     return comparison.getVariables();
150   }
151 
152   /**
153    * Returns the properties referenced by the expression for a given variable
154    *
155    * @param variable the variable
156    * @return set of referenced properties
157    */
158   public Set<String> getPropertyKeys(String variable) {
159     Set<String> properties = getLhs().getPropertyKeys(variable);
160     properties.addAll(getRhs().getPropertyKeys(variable));
161 
162     return properties;
163   }
164 
165   /**
166    * Returns a ComparisonExpression that wraps the same comparison, but with lhs and
167    * rhs switched
168    *
169    * @return ComparisonExpression wrapping comparison with lhs and rhs switched
170    */
171   public ComparisonExpression switchSides() {
172     Comparison switchedComp = comparison.switchSides();
173     return new ComparisonExpression(switchedComp, comparableFactory);
174   }
175 
176   /**
177    * Converts the predicate into conjunctive normal form
178    *
179    * @return predicate in cnf
180    */
181   @Override
182   public CNF asCNF() {
183     CNF cnf = new CNF();
184     CNFElement cnfElement = new CNFElement();
185 
186     cnfElement.addPredicate(this);
187     cnf.addPredicate(cnfElement);
188 
189     return cnf;
190   }

192   @Override
193   public boolean equals(Object o) {
194     if (this == o) {
195       return true;
196     }
197     if (o == null || getClass() != o.getClass()) {
198       return false;
199     }
200 
201     ComparisonExpression that = (ComparisonExpression) o;
202 
203     return Objects.equals(comparison, that.comparison);
204 
205   }
206 
207   @Override
208   public int hashCode() {
209     return comparison != null ? comparison.hashCode() : 0;
210   }
211 
212   @Override
213   public String toString() {
214     return comparison.toString();
215   }
