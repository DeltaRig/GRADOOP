33  /**
34   * Reduce vertex neighborhood operator.
35   *
36   * @param <G>  The graph head type.
37   * @param <V>  The vertex type.
38   * @param <E>  The edge type.
39   * @param <LG> The type of the graph.
40   * @param <GC> The type of the graph collection.
41   */
42  public class ReduceVertexNeighborhood<
43    G extends GraphHead,
44    V extends Vertex,
45    E extends Edge,
46    LG extends BaseGraph<G, V, E, LG, GC>,
47    GC extends BaseGraphCollection<G, V, E, LG, GC>> extends VertexNeighborhood<LG> {
48  
49    /**
50     * Valued constructor.
51     *
52     * @param function  vertex aggregate function
53     * @param direction considered edge direction
54     */
55    public ReduceVertexNeighborhood(VertexAggregateFunction function, EdgeDirection direction) {
56      super(function, direction);
57    }
58  
59    @Override
60    public LG execute(LG graph) {
61      DataSet<V> vertices;
62      switch (getDirection()) {
63      case IN:
64        // takes edges and gets the corresponding vertices and applies the aggregate function for
65        // vertices of incoming edges
66        vertices = graph.getEdges()
67          // tuple of source id and target id
68          .map(new VertexIdsFromEdge<>())
69          .join(graph.getVertices())
70          .where(1).equalTo(new Id<>())
71          // replace the second id with the vertex
72          .with(new VertexToFieldOne<>())
73          .join(graph.getVertices())
74          // replace the first id with the vertex
75          .where(0).equalTo(new Id<>())
76          .with(new VertexToFieldZero<>())
77          // group by the target vertex
78          .groupBy(new IdInTuple<>(1))
79          // aggregate values
80          .reduceGroup(new NeighborVertexReduceFunction<>((VertexAggregateFunction) getFunction()));
81        break;
82      case OUT:
83        // takes edges and gets the corresponding vertices and applies the aggregate function for
84        // vertices of outgoing edges
85        vertices = graph.getEdges()
86          // tuple of target id and source id
87          .map(new VertexIdsFromEdge<>(true))
88          .join(graph.getVertices())
89          .where(1).equalTo(new Id<>())
90          // replace the second id with the vertex
91          .with(new VertexToFieldOne<>())
92          .join(graph.getVertices())
93          // replace the first id with the vertex
94          .where(0).equalTo(new Id<>())
95          .with(new VertexToFieldZero<>())
96          // group by the target vertex
97          .groupBy(new IdInTuple<>(1))
98          // aggregate values
99          .reduceGroup(new NeighborVertexReduceFunction<>((VertexAggregateFunction) getFunction()));
100       break;
101     case BOTH:
102       // takes edges and gets the corresponding vertices and applies the aggregate function for
103       // vertices of incoming and outgoing edges
104       vertices = graph.getEdges()
105         // maps source-target and target-source ids from the edge
106         .flatMap(new ShuffledVertexIdsFromEdge<>())
107         .join(graph.getVertices())
108         .where(1).equalTo(new Id<>())
109         // replace the second id with the vertex
110         .with(new VertexToFieldOne<>())
111         .join(graph.getVertices())
112         .where(0).equalTo(new Id<>())
113         // replace the first id with the vertex
114         .with(new VertexToFieldZero<>())
115         .groupBy(new IdInTuple<>(1))
116         // aggregate values
117         .reduceGroup(new NeighborVertexReduceFunction<>((VertexAggregateFunction) getFunction()));
118       break;
119     default:
120       vertices = null;
121     }
122     return graph.getFactory().fromDataSets(graph.getGraphHead(), vertices, graph.getEdges());
123   }
