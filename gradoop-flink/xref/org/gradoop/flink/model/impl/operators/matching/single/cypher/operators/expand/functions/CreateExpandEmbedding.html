16  package org.gradoop.flink.model.impl.operators.matching.single.cypher.operators.expand.functions;
17  
18  import org.apache.flink.api.common.functions.RichFlatJoinFunction;
19  import org.apache.flink.api.java.functions.FunctionAnnotation;
20  import org.apache.flink.util.Collector;
21  import org.gradoop.common.model.impl.id.GradoopId;
22  import org.gradoop.flink.model.impl.operators.matching.single.cypher.pojos.Embedding;
23  import org.gradoop.flink.model.impl.operators.matching.single.cypher.operators.expand.tuples.EdgeWithTiePoint;
24  import org.gradoop.flink.model.impl.operators.matching.single.cypher.operators.expand.tuples.ExpandEmbedding;
25  
26  import java.util.List;
27  
28  /**
29   * Creates the initial expand embeddings
30   */
31  @FunctionAnnotation.ReadFieldsSecond("f1; f2")
32  public class CreateExpandEmbedding
33    extends RichFlatJoinFunction<Embedding, EdgeWithTiePoint, ExpandEmbedding> {
34  
35    /**
36     * Holds the index of all base vertex columns that should be distinct
37     */
38    private final List<Integer> distinctVertices;
39    /**
40     * Holds the index of all base edge columns that should be distinct
41     */
42    private final List<Integer> distinctEdges;
43    /**
44     * Specifies a base column that should be equal to the paths end node
45     */
46    private final int closingColumn;
47  
48  
49    /**
50     * Create new FlatJoin Function
51     * @param distinctVertices indices of distinct vertex columns
52     * @param distinctEdges indices of distinct edge columns
53     * @param closingColumn base column that should be equal to a paths end node
54     */
55    public CreateExpandEmbedding(List<Integer> distinctVertices,
56      List<Integer> distinctEdges, int closingColumn) {
57  
58      this.distinctVertices = distinctVertices;
59      this.distinctEdges = distinctEdges;
60      this.closingColumn = closingColumn;
61    }
62  
63    @Override
64    public void join(Embedding input, EdgeWithTiePoint edge, Collector<ExpandEmbedding> out)
65        throws Exception {
66  
67      if (checkDistinctiveness(input, edge)) {
68        GradoopId[] path = new GradoopId[]{edge.getId(), edge.getTarget()};
69        out.collect(new ExpandEmbedding(input, path));
70      }
71    }
72  
73    /**
74     * Checks the distinct criteria for the expansion
75     * @param input the base part of the expansion
76     * @param edge edge along which we expand
77     * @return true if distinct criteria hold for the expansion
78     */
79    private boolean checkDistinctiveness(Embedding input, EdgeWithTiePoint edge) {
80      GradoopId edgeId = edge.getId();
81      GradoopId tgt = edge.getTarget();
82  
83      for (int i : distinctVertices) {
84        if (input.getIdAsList(i).contains(tgt) && i != closingColumn) {
85          return false;
86        }
87      }
88  
89      for (int i : distinctEdges) {
90        if (input.getIdAsList(i).contains(edgeId)) {
91          return false;
92        }
93      }
94  
95      return true;
96    }
97  }
