16  package org.gradoop.flink.model.impl.operators.matching.common.statistics;
17  
18  import java.io.File;
19  import java.io.IOException;
20  import java.nio.charset.Charset;
21  import java.nio.charset.StandardCharsets;
22  import java.nio.file.Files;
23  import java.nio.file.Path;
24  import java.nio.file.Paths;
25  import java.util.Map;
26  
27  /**
28   * Reads {@link GraphStatistics} from dedicated files in the local file system.
29   */
30  public class GraphStatisticsLocalFSReader extends GraphStatisticsReader {
31    /**
32     * Reads statistics from files contains in the specified directory and creates a
33     * {@link GraphStatistics} object from them.
34     *
35     * The method expects all files to be present and formatted according to the docs.
36     *
37     * @param inputPath path to directory containing statistics files
38     * @return graph statistics
39     * @throws IOException if an I/O error occurs opening the files
40     */
41    public static GraphStatistics read(String inputPath) throws IOException {
42  
43      Path statisticsDir = Paths.get(new File(inputPath).getAbsolutePath());
44      Charset charset = StandardCharsets.UTF_8;
45  
46      Path p = statisticsDir.resolve(Paths.get(GraphStatisticsReader.FILE_VERTEX_COUNT));
47      long vertexCount = readSingleValue(Files.lines(p, charset));
48  
49      p = statisticsDir.resolve(Paths.get(GraphStatisticsReader.FILE_EDGE_COUNT));
50      long edgeCount = readSingleValue(Files.lines(p, charset));
51  
52      p = statisticsDir.resolve(Paths.get(GraphStatisticsReader.FILE_VERTEX_COUNT_BY_LABEL));
53      Map<String, Long> vertexCountByLabel = readKeyValueMap(Files.lines(p, charset));
54  
55      p = statisticsDir.resolve(Paths.get(GraphStatisticsReader.FILE_EDGE_COUNT_BY_LABEL));
56      Map<String, Long> edgeCountByLabel = readKeyValueMap(Files.lines(p, charset));
57  
58      p = statisticsDir
59        .resolve(Paths.get(GraphStatisticsReader.FILE_EDGE_COUNT_BY_SOURCE_VERTEX_AND_EDGE_LABEL));
60      Map<String, Map<String, Long>> edgeCountBySourceVertexAndEdgeLabel =
61        readNestedKeyValueMap(Files.lines(p, charset));
62  
63      p = statisticsDir
64        .resolve(Paths.get(GraphStatisticsReader.FILE_EDGE_COUNT_BY_TARGET_VERTEX_AND_EDGE_LABEL));
65      Map<String, Map<String, Long>> edgeCountByTargetVertexAndEdgeLabel =
66        readNestedKeyValueMap(Files.lines(p, charset));
67  
68      p = statisticsDir.resolve(Paths.get(GraphStatisticsReader.FILE_DISTINCT_SOURCE_VERTEX_COUNT));
69      long distinctSourceVertexCount = readSingleValue(Files.lines(p, charset));
70  
71      p = statisticsDir.resolve(Paths.get(GraphStatisticsReader.FILE_DISTINCT_TARGET_VERTEX_COUNT));
72      long distinctTargetVertexCount = readSingleValue(Files.lines(p, charset));
73  
74      p = statisticsDir
75        .resolve(Paths.get(GraphStatisticsReader.FILE_DISTINCT_SOURCE_VERTEX_COUNT_BY_EDGE_LABEL));
76      Map<String, Long> distSourceVertexCountByEdgeLabel = readKeyValueMap(Files.lines(p, charset));
77  
78      p = statisticsDir
79        .resolve(Paths.get(GraphStatisticsReader.FILE_DISTINCT_TARGET_VERTEX_COUNT_BY_EDGE_LABEL));
80      Map<String, Long> distTargetVertexCountByEdgeLabel = readKeyValueMap(Files.lines(p, charset));
81  
82      p = statisticsDir.resolve(Paths.get(
83          GraphStatisticsReader.FILE_DISTINCT_EDGE_PROPERTIES_BY_LABEL));
84      Map<String, Map<String, Long>> distinctPropertyValuesByEdgeLabelAndPropertyName =
85        readNestedKeyValueMap(Files.lines(p, charset));
86  
87      p = statisticsDir.resolve(Paths.get(
88        GraphStatisticsReader.FILE_DISTINCT_VERTEX_PROPERTIES_BY_LABEL));
89      Map<String, Map<String, Long>> distinctPropertyValuesByVertexLabelAndPropertyName =
90        readNestedKeyValueMap(Files.lines(p, charset));
91  
92      p = statisticsDir.resolve(Paths.get(
93        GraphStatisticsReader.FILE_DISTINCT_EDGE_PROPERTIES));
94      Map<String, Long> distinctEdgePropertyValuesByPropertyName =
95        readKeyValueMap(Files.lines(p, charset));
96  
97      p = statisticsDir.resolve(Paths.get(
98        GraphStatisticsReader.FILE_DISTINCT_VERTEX_PROPERTIES));
99      Map<String, Long> distinctVertexPropertyValuesByPropertyName =
100       readKeyValueMap(Files.lines(p, charset));
101 
102     return new GraphStatistics(vertexCount, edgeCount, vertexCountByLabel, edgeCountByLabel,
103       edgeCountBySourceVertexAndEdgeLabel, edgeCountByTargetVertexAndEdgeLabel,
104       distinctSourceVertexCount, distinctTargetVertexCount,
105       distSourceVertexCountByEdgeLabel, distTargetVertexCountByEdgeLabel,
106       distinctPropertyValuesByEdgeLabelAndPropertyName,
107       distinctPropertyValuesByVertexLabelAndPropertyName,
108       distinctEdgePropertyValuesByPropertyName,
109       distinctVertexPropertyValuesByPropertyName);
110   }
