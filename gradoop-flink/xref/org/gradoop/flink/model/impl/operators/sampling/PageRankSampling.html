33  /**
34   * Computes a PageRank-Sampling of the graph (new graph head will be generated).
35   *
36   * Uses the Gradoop-Wrapper of Flinks PageRank-algorithm {@link PageRank} with a dampening factor
37   * and a number of maximum iterations. It computes a per-vertex score which is the sum of the
38   * PageRank-scores transmitted over all in-edges. The score of each vertex is divided evenly
39   * among its out-edges.
40   * The PageRank-algorithm is called with {@code setIncludeZeroDegreeVertices(true)}.
41   *
42   * If vertices got different PageRank-scores, all scores are scaled in a range between 0 and 1.
43   * Then it retains all vertices with a PageRank-score greater or equal/smaller than a given
44   * sampling threshold - depending on the Boolean set in {@code sampleGreaterThanThreshold}.
45   *
46   * If ALL vertices got the same PageRank-score, it can be decided whether to sample all vertices
47   * or none of them - depending on the Boolean set in {@code keepVerticesIfSameScore}.
48   *
49   * Retains all edges which source- and target-vertices were chosen. There may retain some
50   * unconnected vertices in the sampled graph.
51   *
52   * @param <G>  The graph head type.
53   * @param <V>  The vertex type.
54   * @param <E>  The edge type.
55   * @param <LG> The type of the graph.
56   * @param <GC> The type of the graph collection.
57   */
58  public class PageRankSampling<
59    G extends GraphHead,
60    V extends Vertex,
61    E extends Edge,
62    LG extends BaseGraph<G, V, E, LG, GC>,
63    GC extends BaseGraphCollection<G, V, E, LG, GC>> extends SamplingAlgorithm<G, V, E, LG, GC> {
64  
65    /**
66     * Dampening factor used by PageRank-algorithm
67     */
68    private final double dampeningFactor;
69    /**
70     * Number of iterations used by PageRank-algorithm
71     */
72    private final int maxIteration;
73    /**
74     * Sampling threshold for PageRankScore
75     */
76    private final double threshold;
77    /**
78     * Whether to sample vertices with PageRank-score greater (true) or equal/smaller (false)
79     * than the threshold
80     */
81    private final boolean sampleGreaterThanThreshold;
82    /**
83     * Whether to sample all vertices (true) or none of them (false), in case all vertices got the
84     * same PageRank-score.
85     */
86    private final boolean keepVerticesIfSameScore;
87  
88    /**
89     * Creates a new PageRankSampling instance.
90     *
91     * @param dampeningFactor The dampening factor used by PageRank-algorithm, e.g. 0.85
92     * @param maxIteration The number of iterations used by PageRank-algorithm, e.g. 40
93     * @param threshold The threshold for the PageRank-score (ranging between 0 and 1 when scaled),
94     *                  determining if a vertex is sampled, e.g. 0.5
95     * @param sampleGreaterThanThreshold Whether to sample vertices with a PageRank-score
96     *                                   greater (true) or equal/smaller (false) the threshold
97     * @param keepVerticesIfSameScore Whether to sample all vertices (true) or none of them (false)
98     *                                in case all vertices got the same PageRank-score.
99     */
100   public PageRankSampling(double dampeningFactor, int maxIteration, double threshold,
101     boolean sampleGreaterThanThreshold, boolean keepVerticesIfSameScore) {
102     this.dampeningFactor = dampeningFactor;
103     this.threshold = threshold;
104     this.maxIteration = maxIteration;
105     this.sampleGreaterThanThreshold = sampleGreaterThanThreshold;
106     this.keepVerticesIfSameScore = keepVerticesIfSameScore;
107   }
108 
109   @Override
110   public LG sample(LG graph) {
111 
112     LG pageRankGraph = graph.callForGraph(new PageRank<>(
113       SamplingConstants.PAGE_RANK_SCORE_PROPERTY_KEY,
114       dampeningFactor,
115       maxIteration,
116       true));
117 
118     graph = graph.getFactory().fromDataSets(
119       graph.getGraphHead(), pageRankGraph.getVertices(), pageRankGraph.getEdges());
120 
121     graph = graph
122       .aggregate(new MinVertexProperty(SamplingConstants.PAGE_RANK_SCORE_PROPERTY_KEY),
123         new MaxVertexProperty(SamplingConstants.PAGE_RANK_SCORE_PROPERTY_KEY),
124         new SumVertexProperty(SamplingConstants.PAGE_RANK_SCORE_PROPERTY_KEY),
125         new VertexCount());
126 
127     DataSet<V> scaledVertices = graph.getVertices()
128       .crossWithTiny(graph.getGraphHead().first(1))
129       .with(new AddPageRankScoresToVertexCrossFunction<>())
130       .filter(new PageRankResultVertexFilter<>(threshold, sampleGreaterThanThreshold,
131         keepVerticesIfSameScore));
132 
133     return graph.getFactory().fromDataSets(scaledVertices, graph.getEdges()).verify();
134   }
