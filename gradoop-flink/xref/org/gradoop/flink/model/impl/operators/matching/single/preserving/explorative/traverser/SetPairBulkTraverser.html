32  /**
33   * Extracts {@link Embedding}s iteratively from a given graph by traversing the
34   * graph according to a given {@link TraversalCode}.
35   *
36   * For the iteration the traverser uses Flink bulk iteration.
37   *
38   * @param <K> key type
39   */
40  public class SetPairBulkTraverser<K> extends SetPairTraverser<K> {
41    /**
42     * Creates a new distributed traverser.
43     *
44     * @param traversalCode describes the traversal through the graph
45     * @param vertexCount   number of query vertices
46     * @param edgeCount     number of query edges
47     * @param keyClazz      needed for embedding initialization
48     */
49    public SetPairBulkTraverser(TraversalCode traversalCode,
50      int vertexCount, int edgeCount, Class<K> keyClazz) {
51      this(traversalCode, MatchStrategy.ISOMORPHISM,
52        vertexCount, edgeCount,
53        keyClazz,
54        JoinOperatorBase.JoinHint.OPTIMIZER_CHOOSES,
55        JoinOperatorBase.JoinHint.OPTIMIZER_CHOOSES,
56        null, null); // debug mappings
57    }
58  
59    /**
60     * Creates a new distributed traverser.
61     *
62     * @param traversalCode          describes the graph traversal
63     * @param matchStrategy          matching strategy for vertex and edge mapping
64     * @param vertexCount            number of query vertices
65     * @param edgeCount              number of query edges
66     * @param keyClazz               key type for embedding initialization
67     * @param edgeStepJoinStrategy   Join strategy for edge extension
68     * @param vertexStepJoinStrategy Join strategy for vertex extension
69     * @param vertexMapping          used for debug
70     * @param edgeMapping            used for debug
71     */
72    public SetPairBulkTraverser(TraversalCode traversalCode, MatchStrategy matchStrategy,
73      int vertexCount, int edgeCount,
74      Class<K> keyClazz,
75      JoinOperatorBase.JoinHint edgeStepJoinStrategy,
76      JoinOperatorBase.JoinHint vertexStepJoinStrategy,
77      DataSet<Tuple2<K, PropertyValue>> vertexMapping,
78      DataSet<Tuple2<K, PropertyValue>> edgeMapping) {
79  
80      super(traversalCode, matchStrategy, vertexCount, edgeCount, keyClazz,
81        edgeStepJoinStrategy, vertexStepJoinStrategy, vertexMapping, edgeMapping);
82    }
83  
84    @Override
85    public DataSet<Tuple1<Embedding<K>>> traverse(
86      DataSet<IdWithCandidates<K>> vertices,
87      DataSet<TripleWithCandidates<K>> edges) {
88  
89      return iterate(vertices, edges, buildInitialEmbeddings(vertices))
90        .project(1);
91    }
92  
93    @Override
94    boolean isIterative() {
95      return true;
96    }
97  
98    /**
99     * Explores the data graph iteratively using the provided traversal code.
100    *
101    * @param vertices          vertex candidates
102    * @param edges             edge candidates
103    * @param initialEmbeddings initial embeddings which are extended in each
104    *                          iteration
105    * @return final embeddings
106    */
107   private DataSet<EmbeddingWithTiePoint<K>> iterate(
108     DataSet<IdWithCandidates<K>> vertices,
109     DataSet<TripleWithCandidates<K>> edges,
110     DataSet<EmbeddingWithTiePoint<K>> initialEmbeddings) {
111 
112     // ITERATION HEAD
113     IterativeDataSet<EmbeddingWithTiePoint<K>> embeddings = initialEmbeddings
114       .iterate(getTraversalCode().getSteps().size());
115 
116     // ITERATION BODY
117 
118     // get current superstep
119     DataSet<Integer> superstep = embeddings
120       .first(1)
121       .map(new Superstep<>());
122 
123     // traverse to outgoing/incoming edges
124     String[] forwardedFieldsEdgeSteps = new String[] {
125       "f0" // forward edge id
126     };
127     DataSet<EmbeddingWithTiePoint<K>> nextWorkSet = traverseEdges(edges,
128       embeddings, superstep, TraverserStrategy.SET_PAIR_BULK_ITERATION,
129       forwardedFieldsEdgeSteps);
130 
131     // traverse to vertices
132     nextWorkSet = traverseVertices(vertices, nextWorkSet, superstep,
133       TraverserStrategy.SET_PAIR_BULK_ITERATION);
134 
135     // ITERATION FOOTER
136     return embeddings.closeWith(nextWorkSet, nextWorkSet);
137   }
