45  /**
46   * Group a graph based on some key functions.<p>
47   * This operator is initialized with {@link List lists} of {@link KeyFunction key functions} and
48   * {@link AggregateFunction aggregate functions}. The output of this operator is the grouped graph
49   * (also called summarized graph or summary graph) which is calculated by reducing similar vertices and
50   * edges to single elements, called super-vertices and -edges respectively. Elements are considered similar
51   * if the values of the key functions are all equal.<p>
52   * The aggregate functions will summarize certain property values of similar elements and store the result
53   * on the super-elements.<p>
54   * This implementation will use tuples to represent elements during the execution of this operator.
55   * These tuples will contain IDs, the values of key functions and the aggregate values for each element
56   * and super-element.
57   *
58   * @param <G> The graph head type.
59   * @param <V> The vertex type.
60   * @param <E> The edge type.
61   * @param <LG> The graph type.
62   * @param <GC> The graph collection type.
63   */
64  public class KeyedGrouping<
65    G extends GraphHead,
66    V extends Vertex,
67    E extends Edge,
68    LG extends BaseGraph<G, V, E, LG, GC>,
69    GC extends BaseGraphCollection<G, V, E, LG, GC>> implements UnaryBaseGraphToBaseGraphOperator<LG> {
70  
71    /**
72     * The vertex grouping keys.
73     */
74    private final List<KeyFunction<V, ?>> vertexGroupingKeys;
75  
76    /**
77     * The vertex aggregate functions.
78     */
79    private final List<AggregateFunction> vertexAggregateFunctions;
80  
81    /**
82     * The edge grouping keys.
83     */
84    private final List<KeyFunction<E, ?>> edgeGroupingKeys;
85  
86    /**
87     * The edge aggregate functions.
88     */
89    private final List<AggregateFunction> edgeAggregateFunctions;
90  
91    /**
92     * Should a combine step be used before grouping? Note that this currently only affects edges.
93     */
94    private boolean useGroupCombine = true;
95  
96    /**
97     * Instantiate this grouping function.
98     *
99     * @param vertexGroupingKeys       The vertex grouping keys.
100    * @param vertexAggregateFunctions The vertex aggregate functions.
101    * @param edgeGroupingKeys         The edge grouping keys.
102    * @param edgeAggregateFunctions   The edge aggregate functions.
103    */
104   public KeyedGrouping(List<KeyFunction<V, ?>> vertexGroupingKeys,
105     List<AggregateFunction> vertexAggregateFunctions,
106     List<KeyFunction<E, ?>> edgeGroupingKeys,
107     List<AggregateFunction> edgeAggregateFunctions) {
108     if (vertexGroupingKeys == null || vertexGroupingKeys.isEmpty()) {
109       // Grouping with no keys is not supported, we therefore add a custom key function that returns a
110       // constant as a pseudo-key for every element.
111       this.vertexGroupingKeys = Collections.singletonList(GroupingKeys.nothing());
112     } else {
113       this.vertexGroupingKeys = vertexGroupingKeys;
114     }
115     this.vertexAggregateFunctions = vertexAggregateFunctions == null ? Collections.emptyList() :
116       vertexAggregateFunctions;
117     this.edgeGroupingKeys = edgeGroupingKeys == null ? Collections.emptyList() :
118       edgeGroupingKeys;
119     this.edgeAggregateFunctions = edgeAggregateFunctions == null ? Collections.emptyList() :
120       edgeAggregateFunctions;
121   }
122 
123   @Override
124   public LG execute(LG graph) {
125     /* First we create tuple representations of each vertex.
126        Those tuples will then be grouped by the respective key fields (the fields containing the values
127        extracted by the key functions) and reduced to assign a super vertex and to calculate aggregates. */
128     DataSet<Tuple> verticesWithSuperVertex = graph.getVertices()
129       .map(new BuildTuplesFromVertices<>(vertexGroupingKeys, vertexAggregateFunctions))
130       .groupBy(getInternalVertexGroupingKeys())
131       .reduceGroup(new ReduceVertexTuples<>(
132         GroupingConstants.VERTEX_TUPLE_RESERVED + vertexGroupingKeys.size(), vertexAggregateFunctions));
133     /* Extract a mapping from vertex-ID to super-vertex-ID from the result of the vertex-reduce step. */
134     DataSet<Tuple2<GradoopId, GradoopId>> idToSuperId = verticesWithSuperVertex
135       .filter(new Not<>(new FilterSuperVertices<>()))
136       .project(GroupingConstants.VERTEX_TUPLE_ID, GroupingConstants.VERTEX_TUPLE_SUPERID);
137 
138     /* Create tuple representations of each edge and update the source- and target-ids of those tuples with
139        with the mapping extracted in the previous step. Edges will then point from and to super-vertices. */
140     DataSet<Tuple> edgesWithUpdatedIds = graph.getEdges()
141       .map(new BuildTuplesFromEdges<>(edgeGroupingKeys, edgeAggregateFunctions))
142       .join(idToSuperId)
143       .where(GroupingConstants.EDGE_TUPLE_SOURCEID)
144       .equalTo(GroupingConstants.VERTEX_TUPLE_ID)
145       .with(new UpdateIdField<>(GroupingConstants.EDGE_TUPLE_SOURCEID))
146       .join(idToSuperId)
147       .where(GroupingConstants.EDGE_TUPLE_TARGETID)
148       .equalTo(GroupingConstants.VERTEX_TUPLE_ID)
149       .with(new UpdateIdField<>(GroupingConstants.EDGE_TUPLE_TARGETID));
150 
151     /* Group the edge-tuples by the key fields and vertex IDs and reduce them to single elements. */
152     DataSet<Tuple> superEdgeTuples = edgesWithUpdatedIds
153       .groupBy(getInternalEdgeGroupingKeys())
154       .reduceGroup(new ReduceEdgeTuples<>(
155         GroupingConstants.EDGE_TUPLE_RESERVED + edgeGroupingKeys.size(), edgeAggregateFunctions))
156       .setCombinable(useGroupCombine);
157 
158     /* Rebuild super-vertices from vertex-tuples. Those new vertices contain the data extracted by the key
159        functions and aggregated by the aggregate functions. */
160     DataSet<V> superVertices = verticesWithSuperVertex
161       .filter(new FilterSuperVertices<>())
162       .map(new BuildSuperVertexFromTuple<>(vertexGroupingKeys, vertexAggregateFunctions,
163         graph.getFactory().getVertexFactory()));
164 
165     /* Rebuild super-edges from edge-tuples. */
166     DataSet<E> superEdges = superEdgeTuples
167       .map(new BuildSuperEdgeFromTuple<>(edgeGroupingKeys, edgeAggregateFunctions,
168         graph.getFactory().getEdgeFactory()));
169 
170     return graph.getFactory().fromDataSets(superVertices, superEdges);
171   }

173   /**
174    * Get the internal grouping keys used for grouping the edge tuples.
175    *
176    * @return The grouping keys, as tuple indices.
177    */
178   private int[] getInternalEdgeGroupingKeys() {
179     return IntStream.range(0, GroupingConstants.EDGE_TUPLE_RESERVED + edgeGroupingKeys.size())
180       .toArray();
181   }
182 
183   /**
184    * Get the internal grouping keys used for grouping the vertex tuples.
185    *
186    * @return The grouping keys, as tuple indices.
187    */
188   private int[] getInternalVertexGroupingKeys() {
189     return IntStream.range(GroupingConstants.VERTEX_TUPLE_RESERVED,
190       GroupingConstants.VERTEX_TUPLE_RESERVED + vertexGroupingKeys.size()).toArray();
191   }
