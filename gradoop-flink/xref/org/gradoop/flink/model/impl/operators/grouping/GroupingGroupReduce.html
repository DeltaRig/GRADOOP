38  /**
39   * Grouping implementation that uses group + groupReduce for building super
40   * vertices and updating the original vertices.
41   *
42   * Algorithmic idea:
43   *
44   * 1) Map vertices to a minimal representation, i.e. {@link VertexGroupItem}.
45   * 2) Group vertices on label and/or property.
46   * 3) Create a super vertex id for each group and collect a non-candidate
47   *    {@link VertexGroupItem} for each group element and one additional
48   *    super vertex tuple that holds the group aggregate.
49   * 4) Filter output of 3)
50   *    a) non-candidate tuples are mapped to {@link VertexWithSuperVertex}
51   *    b) super vertex tuples are used to build final super vertices
52   * 5) Map edges to a minimal representation, i.e. {@link EdgeGroupItem}
53   * 6) Join edges with output of 4a) and replace source/target id with super
54   *    vertex id.
55   * 7) Updated edges are grouped by source and target id and optionally by label
56   *    and/or edge property.
57   * 8) Group combine on the workers and compute aggregate.
58   * 9) Group reduce globally and create final super edges.
59   *
60   * @param <G>  The graph head type.
61   * @param <V>  The vertex type.
62   * @param <E>  The edge type.
63   * @param <LG> The type of the graph.
64   * @param <GC> The type of the graph collection.
65   */
66  public class GroupingGroupReduce<
67    G extends GraphHead,
68    V extends Vertex,
69    E extends Edge,
70    LG extends BaseGraph<G, V, E, LG, GC>,
71    GC extends BaseGraphCollection<G, V, E, LG, GC>> extends Grouping<G, V, E, LG, GC> {
72    /**
73     * Creates grouping operator instance.
74     *
75     * @param useVertexLabels             group on vertex label true/false
76     * @param useEdgeLabels               group on edge label true/false
77     * @param vertexLabelGroups           stores grouping properties for vertex labels
78     * @param edgeLabelGroups             stores grouping properties for edge labels
79     * @param retainVerticesWithoutGroup  a flag to retain vertices that are not affected by the
80     *                                    grouping
81     */
82    GroupingGroupReduce(boolean useVertexLabels, boolean useEdgeLabels,
83      List<LabelGroup> vertexLabelGroups, List<LabelGroup> edgeLabelGroups,
84      boolean retainVerticesWithoutGroup) {
85      super(useVertexLabels, useEdgeLabels, vertexLabelGroups, edgeLabelGroups,
86        retainVerticesWithoutGroup);
87    }
88  
89    @Override
90    protected LG groupInternal(LG graph) {
91  
92      DataSet<V> vertices = isRetainingVerticesWithoutGroup() ?
93        graph.getVertices()
94          .filter(new LabelGroupFilter<>(getVertexLabelGroups(), useVertexLabels())) :
95        graph.getVertices();
96  
97      // map vertex to vertex group item
98      DataSet<VertexGroupItem> verticesForGrouping = vertices.flatMap(
99        new BuildVertexGroupItem<>(useVertexLabels(), getVertexLabelGroups()));
100 
101     // group vertices by label / properties / both
102     DataSet<VertexGroupItem> vertexGroupItems = groupVertices(verticesForGrouping)
103       // apply aggregate function
104       .reduceGroup(new ReduceVertexGroupItems(useVertexLabels()));
105 
106     DataSet<V> superVertices = vertexGroupItems
107       // filter group representative tuples
108       .filter(new FilterSuperVertices())
109       // build super vertices
110       .map(new BuildSuperVertex<>(useVertexLabels(), graph.getFactory().getVertexFactory()));
111 
112     DataSet<VertexWithSuperVertex> vertexToRepresentativeMap = vertexGroupItems
113       // filter group element tuples
114       .filter(new FilterRegularVertices())
115       // build vertex to group representative tuple
116       .map(new BuildVertexWithSuperVertex());
117 
118     DataSet<E> edgesToGroup = graph.getEdges();
119 
120     if (isRetainingVerticesWithoutGroup()) {
121       LG retainedVerticesSubgraph = getSubgraphOfRetainedVertices(graph);
122 
123       // To add support for grouped edges between retained vertices and supervertices,
124       // vertices are their group representatives themselves
125       vertexToRepresentativeMap =
126         updateVertexRepresentatives(vertexToRepresentativeMap,
127           retainedVerticesSubgraph.getVertices());
128 
129       // don't execute grouping on edges between retained vertices
130       // but execute on edges between retained vertices and grouped vertices
131       //   graph.getEdges() - retainedVerticesSubgraph.getEdges()
132       edgesToGroup = subtractEdges(graph.getEdges(), retainedVerticesSubgraph.getEdges());
133     }
134 
135     DataSet<E> superEdges =
136       buildSuperEdges(graph.getFactory().getEdgeFactory(), edgesToGroup, vertexToRepresentativeMap);
137 
138     if (isRetainingVerticesWithoutGroup()) {
139       LG retainedVerticesSubgraph = getSubgraphOfRetainedVertices(graph);
140       superVertices = superVertices.union(retainedVerticesSubgraph.getVertices());
141       superEdges = superEdges.union(retainedVerticesSubgraph.getEdges());
142     }
143 
144     return graph.getFactory().fromDataSets(superVertices, superEdges);
145   }
