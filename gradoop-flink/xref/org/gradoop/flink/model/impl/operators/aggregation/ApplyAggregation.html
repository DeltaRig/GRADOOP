45  /**
46   * Takes a collection of base graphs and user defined aggregate functions as
47   * input. The aggregate functions are applied on each base graph contained in
48   * the collection and the aggregate is stored as additional properties at the graphs.
49   *
50   * @param <G> type of the graph head
51   * @param <V> the vertex type
52   * @param <E> the edge type
53   * @param <LG> type of the logical graph instance
54   * @param <GC> type of the graph collection
55   */
56  public class ApplyAggregation<
57    G extends GraphHead,
58    V extends Vertex,
59    E extends Edge,
60    LG extends BaseGraph<G, V, E, LG, GC>,
61    GC extends BaseGraphCollection<G, V, E, LG, GC>>
62    implements ApplicableUnaryBaseGraphToBaseGraphOperator<GC> {
63  
64    /**
65     * User-defined aggregate functions which get applied on a graph collection.
66     */
67    private final Set<AggregateFunction> aggregateFunctions;
68  
69    /**
70     * Creates a new operator instance.
71     *
72     * @param aggregateFunctions functions to compute aggregate values
73     */
74    public ApplyAggregation(final AggregateFunction... aggregateFunctions) {
75      for (AggregateFunction aggFunc : aggregateFunctions) {
76        checkNotNull(aggFunc);
77      }
78      this.aggregateFunctions = new HashSet<>(Arrays.asList(aggregateFunctions));
79    }
80  
81    @Override
82    public GC executeForGVELayout(GC collection) {
83      DataSet<G> graphHeads = collection.getGraphHeads();
84      DataSet<GradoopId> graphIds = graphHeads.map(new Id<>());
85  
86      DataSet<Tuple2<GradoopId, Map<String, PropertyValue>>> aggregate =
87        aggregateVertices(collection.getVertices(), graphIds)
88          .union(aggregateEdges(collection.getEdges(), graphIds))
89          .groupBy(0)
90          .reduceGroup(new CombinePartitionApplyAggregates(aggregateFunctions));
91  
92      graphHeads = graphHeads
93        .coGroup(aggregate)
94        .where(new Id<>()).equalTo(0)
95        .with(new SetAggregateProperties<>(aggregateFunctions));
96  
97      return collection.getFactory().fromDataSets(graphHeads, collection.getVertices(), collection.getEdges());
98    }
99  
100   @Override
101   public GC executeForTxLayout(GC collection) {
102     DataSet<GraphTransaction> updatedTransactions = collection.getGraphTransactions()
103       .map(new AggregateTransactions(aggregateFunctions));
104 
105     return collection.getFactory().fromTransactions(updatedTransactions);
106   }
107 
108   /**
109    * Applies an aggregate function to the partitions of a vertex data set.
110    *
111    * @param vertices vertex data set
112    * @param graphIds graph ids to aggregate
113    * @return partition aggregate value
114    */
115   private DataSet<Tuple2<GradoopId, Map<String, PropertyValue>>> aggregateVertices(
116     DataSet<V> vertices, DataSet<GradoopId> graphIds) {
117     return vertices
118       .flatMap(new ElementsOfSelectedGraphs<>())
119       .withBroadcastSet(graphIds, ElementsOfSelectedGraphs.GRAPH_IDS)
120       .groupBy(0)
121       .combineGroup(new ApplyAggregateElements<>(aggregateFunctions.stream()
122         .filter(AggregateFunction::isVertexAggregation)
123         .collect(Collectors.toSet())));
124   }
125 
126   /**
127    * Applies an aggregate function to the partitions of an edge data set.
128    *
129    * @param edges edge data set
130    * @param graphIds graph ids to aggregate
131    * @return partition aggregate value
132    */
133   private DataSet<Tuple2<GradoopId, Map<String, PropertyValue>>> aggregateEdges(
134     DataSet<E> edges, DataSet<GradoopId> graphIds) {
135     return edges
136       .flatMap(new ElementsOfSelectedGraphs<>())
137       .withBroadcastSet(graphIds, ElementsOfSelectedGraphs.GRAPH_IDS)
138       .groupBy(0)
139       .combineGroup(new ApplyAggregateElements<>(aggregateFunctions.stream()
140         .filter(AggregateFunction::isEdgeAggregation)
141         .collect(Collectors.toSet())));
142   }
