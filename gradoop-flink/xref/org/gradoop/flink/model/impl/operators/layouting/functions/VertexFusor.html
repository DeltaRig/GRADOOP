33  /**
34   * Simplifies the graph by combining similar vertices to super-vertices using a
35   * Comparison-Function and a threshold.
36   */
37  public class VertexFusor {
38    /**
39     * The VertexCompareFunction to use to find similar vertices
40     */
41    protected VertexCompareFunction compareFunction;
42    /**
43     * Only consider vertices as similar if their similarity is larger then this value
44     */
45    protected double threshold;
46  
47    /**
48     * Construct new VertexFusor
49     *
50     * @param compareFunction The VertexCompareFunction to use to find similar vertices
51     * @param threshold       Only consider vertices as similar if their similarity is larger then
52     *                        this
53     *                        value
54     */
55    public VertexFusor(VertexCompareFunction compareFunction, double threshold) {
56      this.compareFunction = compareFunction;
57      this.threshold = threshold;
58    }
59  
60    /**
61     * Execute the operation. Should be called iteratively. A single call is usually not enough for
62     * practical results.
63     *
64     * @param graph The graph to simplify
65     * @return A tuple containing the vertices and edges of the simplified graph
66     */
67    public LGraph execute(LGraph graph) {
68      DataSet<LVertex> vertices = graph.getVertices();
69      DataSet<LEdge> edges = graph.getEdges();
70  
71      DataSet<Tuple2<LVertex, Boolean>> classifiedVertices = chooseDonorsAndAcceptors(vertices);
72  
73      DataSet<Tuple2<LVertex, LVertex>> fusions = generateFusionCandidates(classifiedVertices, edges);
74  
75      DataSet<LVertex> superVertices = fusions.groupBy("1.0").reduceGroup(new SuperVertexGenerator());
76  
77      DataSet<LVertex> remainingVertices = findRemainingVertices(fusions, vertices, superVertices);
78  
79      vertices = remainingVertices.union(superVertices);
80  
81      edges = fixEdgeReferences(edges, fusions);
82  
83      edges = edges.groupBy(LEdge.SOURCE_ID_POSITION, LEdge.TARGET_ID_POSITION).reduce((a, b) -> {
84        a.addSubEdge(b.getId());
85        a.addSubEdges(b.getSubEdges());
86        return a;
87      });
88  
89      return new LGraph(vertices, edges);
90    }

92    /**
93     * Splits the vertices randomly into donor and acceptor-vertices (~50/50).
94     *
95     * @param vertices vertices to classify
96     * @return {@code Tuple2<LVertex, Boolean>}. If the boolean is true, the vertex is an acceptor
97     */
98    protected DataSet<Tuple2<LVertex, Boolean>> chooseDonorsAndAcceptors(DataSet<LVertex> vertices) {
99      final Random rng = new Random();
100     return vertices.map(v -> new Tuple2<>(v, rng.nextBoolean()))
101       .returns(new TypeHint<Tuple2<LVertex, Boolean>>() {
102       });
103   }
104 
105   /**
106    * Finds connected vertices by joining with the edges. Finds out if the two vertices should be
107    * merged together and if so outputs a tuple for each merge.
108    *
109    * @param classifiedVertices The vertices (split into donors and acceptors)
110    * @param edges              The edges
111    * @return {@code Tuple2<LVertex, LVertex>} for each merge. f0 is the donor and f1 the acceptor
112    * for the
113    * merge.
114    */
115   protected DataSet<Tuple2<LVertex, LVertex>> generateFusionCandidates(
116     DataSet<Tuple2<LVertex, Boolean>> classifiedVertices, DataSet<LEdge> edges) {
117     return edges.join(classifiedVertices).where(LEdge.SOURCE_ID_POSITION).equalTo("0." + LVertex.ID_POSITION)
118       .join(classifiedVertices).where("0." + LEdge.TARGET_ID_POSITION).equalTo("0." + LVertex.ID_POSITION)
119       .with(new CandidateGenerator(compareFunction, threshold)).groupBy("0.0")
120       .reduce((a, b) -> (a.f2 > b.f2) ? a : b).map(c -> new Tuple2<>(c.f0, c.f1))
121       .returns(new TypeHint<Tuple2<LVertex, LVertex>>() { });
122   }
123 
124   /**
125    * There are some vertices that have neither become super-vertices nor have been merged with a
126    * super vertex. Find them so they can be copied to the simplified graph.
127    *
128    * @param fusions       The merges that are to be performed in this iteration
129    * @param vertices      The original vertices of the input graph
130    * @param superVertices The newly created super-vertices
131    * @return All vertices that have to be copied to the output-graph
132    */
133   protected DataSet<LVertex> findRemainingVertices(DataSet<Tuple2<LVertex, LVertex>> fusions,
134     DataSet<LVertex> vertices, DataSet<LVertex> superVertices) {
135     DataSet<LVertex> remainingVertices =
136       vertices.leftOuterJoin(superVertices).where(LVertex.ID_POSITION).equalTo(LVertex.ID_POSITION)
137         .with(new FlatJoinFunction<LVertex, LVertex, LVertex>() {
138           @Override
139           public void join(LVertex lVertex, LVertex lVertex2, Collector<LVertex> collector) {
140             if (lVertex2 == null) {
141               collector.collect(lVertex);
142             }
143           }
144         });
145 
146     remainingVertices =
147       remainingVertices.leftOuterJoin(fusions).where(LVertex.ID_POSITION).equalTo("0." + LVertex.ID_POSITION)
148         .with(new FlatJoinFunction<LVertex, Tuple2<LVertex, LVertex>, LVertex>() {
149           @Override
150           public void join(LVertex lVertex, Tuple2<LVertex, LVertex> lVertexLVertexDoubleTuple3,
151             Collector<LVertex> collector) {
152             if (lVertexLVertexDoubleTuple3 == null) {
153               collector.collect(lVertex);
154             }
155           }
156         });
157     return remainingVertices;
158   }

160   /**
161    * When combining two vertices into one the edges of the old vertices have to be modified to
162    * point to the new vertex.
163    *
164    * @param edges   The edges of the input-graph
165    * @param fusions The merges performed in the current iteration
166    * @return The "fixed" edges for the output-graph
167    */
168   protected DataSet<LEdge> fixEdgeReferences(DataSet<LEdge> edges,
169     DataSet<Tuple2<LVertex, LVertex>> fusions) {
170     edges = edges.leftOuterJoin(fusions).where(LEdge.SOURCE_ID_POSITION).equalTo("0." + LVertex.ID_POSITION)
171       .with(new JoinFunction<LEdge, Tuple2<LVertex, LVertex>, LEdge>() {
172         @Override
173         public LEdge join(LEdge lEdge, Tuple2<LVertex, LVertex> lVertexLVertexDoubleTuple3) {
174           if (lVertexLVertexDoubleTuple3 != null) {
175             lEdge.setSourceId(lVertexLVertexDoubleTuple3.f1.getId());
176           }
177           return lEdge;
178         }
179       });
180 
181     edges = edges.leftOuterJoin(fusions).where(LEdge.TARGET_ID_POSITION).equalTo("0." + LVertex.ID_POSITION)
182       .with(new JoinFunction<LEdge, Tuple2<LVertex, LVertex>, LEdge>() {
183         @Override
184         public LEdge join(LEdge lEdge, Tuple2<LVertex, LVertex> lVertexLVertexDoubleTuple3) {
185           if (lVertexLVertexDoubleTuple3 != null) {
186             lEdge.setTargetId(lVertexLVertexDoubleTuple3.f1.getId());
187           }
188           return lEdge;
189         }
190       });
191     return edges;
192   }

194   /**
195    * Finds out if two given vertices could be merged into one and how "good" this merge would be.
196    * The strange signature of the join function is needed to be able to directly use it in the
197    * join of generateFusionCandidates().
198    */
199   protected static class CandidateGenerator implements
200     FlatJoinFunction<Tuple2<LEdge, Tuple2<LVertex, Boolean>>, Tuple2<LVertex, Boolean>,
201       Tuple3<LVertex, LVertex, Double>> {
202 
203     /**
204      * ComparisonFunction to use to compute similarity of vertices
205      */
206     protected VertexCompareFunction cf;
207     /**
208      * Minimum similarity to allow merge
209      */
210     protected Double threshold;
211 
212     /**
213      * Construct new instance
214      *
215      * @param cf        ComparisonFunction to use to compute similarity of vertices
216      * @param threshold Minimum similarity to allow merge
217      */
218     public CandidateGenerator(VertexCompareFunction cf, Double threshold) {
219       this.cf = cf;
220       this.threshold = threshold;
221     }
222 
223     @Override
224     public void join(Tuple2<LEdge, Tuple2<LVertex, Boolean>> source,
225       Tuple2<LVertex, Boolean> target, Collector<Tuple3<LVertex, LVertex, Double>> collector) throws
226       Exception {
227 
228       LVertex sourceVertex = source.f1.f0;
229       boolean sourceType = source.f1.f1;
230 
231       LVertex targetVertex = target.f0;
232       boolean targetType = target.f1;
233 
234       // Can not merge two vertices of the same type (donor/acceptor)
235       if (sourceType == targetType) {
236         return;
237       }
238 
239       Double similarity = cf.compare(sourceVertex, targetVertex);
240 
241       // Can not merge vertices that are not similar enough
242       if (similarity < threshold) {
243         return;
244       }
245 
246       // The acceptor-vertex (targetType==true) MUST be the second element of the tuple
247       if (targetType) {
248         collector.collect(new Tuple3<>(sourceVertex, targetVertex, similarity));
249       } else {
250         collector.collect(new Tuple3<>(targetVertex, sourceVertex, similarity));
251       }
252 
253     }
254   }

256   /**
257    * Combines multiple vertices into a single super-vertex. The created super-vertex inherits the
258    * id of the acceptor. The position of the super-vertex is a weighted average of all
259    * participating vertices.
260    */
261   protected static class SuperVertexGenerator implements
262     GroupReduceFunction<Tuple2<LVertex, LVertex>, LVertex> {
263 
264     /**
265      * Combine vertices
266      *
267      * @param iterable  The vertices to combine. f0 contains donor-vertices and f1 contains
268      *                  (always the same) acceptor vertex.
269      * @param collector Collector for the resuls
270      */
271     @Override
272     public void reduce(Iterable<Tuple2<LVertex, LVertex>> iterable,
273       Collector<LVertex> collector) {
274       int count = 0;
275       Vector positionSum = new Vector();
276       LVertex self = null;
277 
278       for (Tuple2<LVertex, LVertex> t : iterable) {
279         if (count == 0) {
280           self = t.f1;
281           count = t.f1.getCount();
282           positionSum.mAdd(t.f1.getPosition().mul(t.f1.getCount()));
283         }
284         count += t.f0.getCount();
285         positionSum.mAdd(t.f0.getPosition().mul(t.f0.getCount()));
286         self.addSubVertex(t.f0.getId());
287         self.addSubVertices(t.f0.getSubVertices());
288       }
289 
290       self.setPosition(positionSum.div(count));
291 
292       collector.collect(self);
293     }
294   }
