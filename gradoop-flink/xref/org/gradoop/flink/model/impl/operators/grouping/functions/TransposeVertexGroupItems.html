29  import java.util.HashSet;
30  import java.util.Set;

32  /**
33   * Takes a group of {@link VertexGroupItem} as input where each item represents
34   * the same group but stores a unique super vertex ids. This function creates
35   * a single {@link VertexGroupItem} that represents the whole group and also
36   * outputs a set containing all the super vertex ids of the incoming items.
37   *
38   * Example input:
39   * (1,A)
40   * (2,A)
41   * (3,B)
42   * (4,B)
43   *
44   * Example output:
45   * ((10,A),(10,[1,2]))
46   * ((20,B),(20,[3,4]))
47   */
48  @FunctionAnnotation.ForwardedFields(
49      "f0->f0.f0;" +  // vertexId
50      "f2->f0.f2;" +  // label
51      "f3->f0.f3;" +  // properties
52      "f6->f0.f6"     // label group
53  )
54  @FunctionAnnotation.ReadFields("f4")
55  public class TransposeVertexGroupItems
56    extends ReduceVertexGroupItemBase
57    implements GroupReduceFunction<VertexGroupItem, Tuple2<VertexGroupItem, IdWithIdSet>> {
58    /**
59     * Reduce object instantiation
60     */
61    private final Tuple2<VertexGroupItem, IdWithIdSet> reuseOuterTuple;
62    /**
63     * Reduce object instantiation
64     */
65    private final IdWithIdSet reuseInnerTuple;
66    /**
67     * Creates group reduce function.
68     *
69     * @param useLabel                        true, iff labels are used for grouping
70     */
71    public TransposeVertexGroupItems(boolean useLabel) {
72      super(useLabel);
73      this.reuseOuterTuple = new Tuple2<>();
74      this.reuseInnerTuple = new IdWithIdSet();
75    }
76  
77    @Override
78    public void reduce(Iterable<VertexGroupItem> vertexGroupItems,
79      Collector<Tuple2<VertexGroupItem, IdWithIdSet>> out)
80        throws Exception {
81  
82      GradoopId superVertexId               = null;
83      String groupLabel                     = null;
84      PropertyValueList groupPropertyValues = null;
85      LabelGroup vertexLabelGroup           = null;
86  
87      boolean isFirst = true;
88  
89      Set<GradoopId> superVertexIds = new HashSet<>();
90  
91      for (VertexGroupItem groupItem : vertexGroupItems) {
92        if (isFirst) {
93          superVertexId = GradoopId.get();
94          groupLabel            = groupItem.getGroupLabel();
95          groupPropertyValues   = groupItem.getGroupingValues();
96          vertexLabelGroup      = groupItem.getLabelGroup();
97  
98          isFirst = false;
99        }
100       // store the super vertex id created in the previous combiner
101       superVertexIds.add(groupItem.getSuperVertexId());
102 
103       vertexLabelGroup.aggregate(groupItem.getAggregateValues());
104     }
105 
106     reuseInnerTuple.setId(superVertexId);
107     reuseInnerTuple.setIdSet(GradoopIdSet.fromExisting(superVertexIds));
108 
109     reuseOuterTuple.f0 = createSuperVertexTuple(superVertexId, groupLabel,
110       groupPropertyValues, vertexLabelGroup.getAggregateValueList());
111     reuseOuterTuple.f0.setSuperVertexId(superVertexId);
112     reuseOuterTuple.f0.setLabelGroup(vertexLabelGroup);
113     reuseOuterTuple.f1 = reuseInnerTuple;
114 
115     // collect single item representing the whole group
116     reuseOuterTuple.f0.getLabelGroup().resetAggregateValues();
117     out.collect(reuseOuterTuple);
118   }
119 }
