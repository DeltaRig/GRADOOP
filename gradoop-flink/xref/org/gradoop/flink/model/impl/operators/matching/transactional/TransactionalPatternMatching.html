51  /**
52   * Operator to match a given pattern on a graph collection
53   *
54   * @param <G> type of the graph head
55   * @param <V> the vertex type
56   * @param <E> the edge type
57   * @param <LG> type of the base graph instance
58   * @param <GC> type of the graph collection
59   */
60  public class TransactionalPatternMatching<
61    G extends GraphHead,
62    V extends Vertex,
63    E extends Edge,
64    LG extends BaseGraph<G, V, E, LG, GC>,
65    GC extends BaseGraphCollection<G, V, E, LG, GC>>
66    implements UnaryBaseGraphCollectionToBaseGraphCollectionOperator<GC> {
67  
68    /**
69     * Query Pattern
70     */
71    private String query;
72    /**
73     * Actual used algorithm
74     */
75    private PatternMatchingAlgorithm algorithm;
76  
77    /**
78     * Flag that determines what the result consists of:
79     * false: source graphs with an new property that is true iff the pattern
80     *        was found in this graph
81     * true:  one graph per found embedding
82     */
83    private boolean findEmbeddings;
84  
85    /**
86     * Constructor
87     *
88     * @param algorithm pattern matching algorithm
89     * @param query     given query-pattern
90     * @param findEmbeddings  flag that determines the type of the return
91     *                        false:  source graphs with an new property that is
92     *                                true iff the pattern was found in this graph
93     *                        true:   one graph per found embedding
94     */
95    public TransactionalPatternMatching(
96      String query,
97      PatternMatchingAlgorithm algorithm,
98      boolean findEmbeddings) {
99      this.query = query;
100     this.algorithm = algorithm;
101     this.findEmbeddings = findEmbeddings;
102   }
103 
104   @Override
105   public GC execute(GC collection) {
106 
107     //--------------------------------------------------------------------------
108     // generate graph-id set witch will be used for generating mappings
109     //--------------------------------------------------------------------------
110     DataSet<GradoopId> graphIds = collection.getGraphHeads().map(new Id<>());
111 
112     //--------------------------------------------------------------------------
113     // generate mapping from graph-id to vertex candidates
114     //--------------------------------------------------------------------------
115     DataSet<Tuple2<GradoopId, IdWithCandidates<GradoopId>>> vertexCandidatesWithGraphs =
116       collection.getVertices()
117         .filter(new MatchingVertices<>(query))
118         .map(new BuildIdWithCandidatesAndGraphs<>(query))
119         .flatMap(new ExpandFirstField<>())
120         .join(graphIds)
121         .where(0).equalTo("*")
122         .with(new LeftSide<>());
123 
124     //--------------------------------------------------------------------------
125     // generate mapping from graph-id to edge candidates
126     //--------------------------------------------------------------------------
127     DataSet<Tuple2<GradoopId, TripleWithCandidates<GradoopId>>> edgeCandidatesWithGraphs =
128       collection.getEdges()
129         .filter(new MatchingEdges<>(query))
130         .map(new BuildTripleWithCandidatesAndGraphs<>(query))
131         .flatMap(new ExpandFirstField<>())
132         .join(graphIds)
133         .where(0).equalTo("*")
134         .with(new LeftSide<>());
135 
136     //--------------------------------------------------------------------------
137     // generate graphs with the candidates for their elements
138     //--------------------------------------------------------------------------
139     DataSet<GraphWithCandidates> graphs = vertexCandidatesWithGraphs
140       .coGroup(edgeCandidatesWithGraphs)
141       .where(0).equalTo(0)
142       .with(new BuildGraphWithCandidates());
143 
144     if (findEmbeddings) {
145       return findEmbeddings(collection, graphs);
146     } else {
147       return hasEmbeddings(collection, graphs);
148     }
149   }
150 
151   /**
152    * Returns the input graph collection with a new property for each graph, that
153    * states if it contains the embedding.
154    * @param collection input graph collection
155    * @param graphs graphs with candidates of their elements
156    * @return input graph collection with new boolean property
157    */
158   private GC hasEmbeddings(GC collection, DataSet<GraphWithCandidates> graphs) {
159     //--------------------------------------------------------------------------
160     // run matching algorithm
161     //--------------------------------------------------------------------------
162     DataSet<Tuple2<GradoopId, Boolean>> matches = graphs.map(new HasEmbeddings(algorithm, query));
163 
164     //--------------------------------------------------------------------------
165     // join matches to graph heads
166     //--------------------------------------------------------------------------
167     DataSet<G> newHeads = collection.getGraphHeads()
168       .coGroup(matches)
169       .where(new Id<>()).equalTo(0)
170       .with(new AddMatchesToProperties<>());
171 
172     //--------------------------------------------------------------------------
173     // return updated graph collection
174     //--------------------------------------------------------------------------
175     return collection.getFactory().fromDataSets(
176       newHeads, collection.getVertices(), collection.getEdges());
177   }
178 
179 
180   /**
181    * Finds all embeddings in the given graph and constructs a new graph
182    * collection consisting of these embeddings.
183    * @param collection input graph collection
184    * @param graphs graphs with candidates of their elements
185    * @return collection of found embeddings
186    */
187   private GC findEmbeddings(GC collection, DataSet<GraphWithCandidates> graphs) {
188 
189     //--------------------------------------------------------------------------
190     // run the matching algorithm
191     //--------------------------------------------------------------------------
192     DataSet<Tuple4<GradoopId, GradoopId, GradoopIdSet, GradoopIdSet>> embeddings = graphs
193       .flatMap(new FindEmbeddings(algorithm, query));
194 
195     //--------------------------------------------------------------------------
196     // create new graph heads
197     //--------------------------------------------------------------------------
198     DataSet<G> newHeads = embeddings
199       .map(new Project4To0And1<>())
200       .map(new InitGraphHeadWithLineage<>(collection.getFactory().getGraphHeadFactory()));
201 
202     //--------------------------------------------------------------------------
203     // update vertex graphs
204     //--------------------------------------------------------------------------
205     DataSet<Tuple2<GradoopId, GradoopIdSet>> verticesWithGraphs = embeddings
206       .map(new Project4To0And2AndSwitch<>())
207       .flatMap(new ExpandFirstField<>()).groupBy(0)
208       .reduceGroup(new MergeSecondField<>());
209 
210     DataSet<V> newVertices = verticesWithGraphs
211       .join(collection.getVertices())
212       .where(0).equalTo(new Id<>())
213       .with(new AddGraphsToElements<>());
214 
215     //--------------------------------------------------------------------------
216     // update edge graphs
217     //--------------------------------------------------------------------------
218     DataSet<Tuple2<GradoopId, GradoopIdSet>> edgesWithGraphs = embeddings
219       .map(new Project4To0And3AndSwitch<>())
220       .flatMap(new ExpandFirstField<>()).groupBy(0)
221       .reduceGroup(new MergeSecondField<>());
222 
223     DataSet<E> newEdges = edgesWithGraphs
224       .join(collection.getEdges())
225       .where(0).equalTo(new Id<>())
226       .with(new AddGraphsToElements<>());
227 
228     //--------------------------------------------------------------------------
229     // return the embeddings
230     //--------------------------------------------------------------------------
231     return collection.getFactory().fromDataSets(newHeads, newVertices, newEdges);
232   }
