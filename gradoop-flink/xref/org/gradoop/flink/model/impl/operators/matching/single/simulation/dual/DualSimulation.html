57  /**
58   * Vertex-centric Dual-Simulation.
59   *
60   * @param <G> The graph head type.
61   * @param <V> The vertex type.
62   * @param <E> The edge type.
63   * @param <LG> The graph type.
64   * @param <GC> The graph collection type.
65   */
66  public class DualSimulation<
67    G extends GraphHead,
68    V extends Vertex,
69    E extends Edge,
70    LG extends BaseGraph<G, V, E, LG, GC>,
71    GC extends BaseGraphCollection<G, V, E, LG, GC>> extends PatternMatching<G, V, E, LG, GC> {
72  
73    /**
74     * Logger
75     */
76    private static Logger LOG = LogManager.getLogger(DualSimulation.class);
77  
78    /**
79     * If true, the algorithm uses bulk iteration for the core iteration.
80     * Otherwise it uses delta iteration.
81     */
82    private final boolean useBulkIteration;
83  
84    /**
85     * Creates a new operator instance.
86     *
87     * @param query       GDL based query
88     * @param attachData  attach original data to resulting vertices/edges
89     * @param useBulk     true to use bulk, false to use delta iteration
90     */
91    public DualSimulation(String query, boolean attachData, boolean useBulk) {
92      super(query, attachData, LOG);
93      this.useBulkIteration = useBulk;
94    }
95  
96    @Override
97    protected GC executeForVertex(LG graph)  {
98      DataSet<Tuple1<GradoopId>> matchingVertexIds = PreProcessor
99        .filterVertices(graph, getQuery())
100       .project(0);
101 
102     BaseGraphFactory<G, V, E, LG, GC> graphFactory = graph.getFactory();
103     BaseGraphCollectionFactory<G, V, E, LG, GC> collectionFactory = graph.getCollectionFactory();
104 
105     if (doAttachData()) {
106       return collectionFactory.fromGraph(
107         graphFactory.fromDataSets(matchingVertexIds
108             .join(graph.getVertices())
109             .where(0).equalTo(new Id<>())
110             .with(new RightSide<>())));
111     } else {
112       return collectionFactory.fromGraph(
113         graphFactory.fromDataSets(matchingVertexIds
114             .map(new VertexFromId<>(graphFactory.getVertexFactory()))));
115     }
116   }

118   /**
119    * Performs dual simulation based on the given query.
120    *
121    * @param graph data graph
122    * @return match graph
123    */
124   protected GC executeForPattern(LG graph) {
125     //--------------------------------------------------------------------------
126     // Pre-processing (filter candidates + build initial working set)
127     //--------------------------------------------------------------------------
128 
129     DataSet<TripleWithCandidates<GradoopId>> triples = filterTriples(graph);
130     DataSet<FatVertex> fatVertices = buildInitialWorkingSet(triples);
131 
132     //--------------------------------------------------------------------------
133     // Dual Simulation
134     //--------------------------------------------------------------------------
135 
136     DataSet<FatVertex> result = useBulkIteration ?
137       simulateBulk(fatVertices) : simulateDelta(fatVertices);
138 
139     //--------------------------------------------------------------------------
140     // Post-processing (build maximum match graph)
141     //--------------------------------------------------------------------------
142 
143     return postProcess(graph, result);
144   }

146   /**
147    * Extracts valid triples from the input graph based on the query.
148    *
149    * @param g input graph
150    * @return triples that have a match in the query graph
151    */
152   private DataSet<TripleWithCandidates<GradoopId>> filterTriples(LG g) {
153     // filter vertex-edge-vertex triples by query predicates
154     return PreProcessor.filterTriplets(g, getQuery());
155   }

157   /**
158    * Prepares the initial working set for the bulk iteration.
159    *
160    * @param triples matching triples from the input graph
161    * @return data set containing fat vertices
162    */
163   private DataSet<FatVertex> buildInitialWorkingSet(DataSet<TripleWithCandidates<GradoopId>> triples) {
164     return triples.flatMap(new CloneAndReverse())
165       .groupBy(1) // sourceId
166       .combineGroup(new BuildFatVertex(getQuery()))
167       .groupBy(0) // vertexId
168       .reduceGroup(new GroupedFatVertices());
169   }

171   /**
172    * Performs dual simulation using bulk iteration.
173    *
174    * @param vertices fat vertices
175    * @return remaining fat vertices after dual simulation
176    */
177   private DataSet<FatVertex> simulateBulk(DataSet<FatVertex> vertices) {
178 
179     vertices = log(vertices, new PrintFatVertex(false, "iteration start"),
180       getVertexMapping(), getEdgeMapping());
181 
182     // ITERATION HEAD
183     IterativeDataSet<FatVertex> workSet = vertices.iterate(Integer.MAX_VALUE);
184 
185     // ITERATION BODY
186 
187     // validate neighborhood of each vertex and create deletions
188     DataSet<Deletion> deletions = workSet
189       .filter(new UpdatedFatVertices())
190       .flatMap(new ValidateNeighborhood(getQuery()));
191 
192     deletions = log(deletions, new PrintDeletion(true, "deletion"),
193       getVertexMapping(), getEdgeMapping());
194 
195     // combine deletions to message
196     DataSet<Message> combinedMessages = deletions
197       .groupBy(0)
198       .combineGroup(new CombinedMessages());
199 
200     combinedMessages = log(combinedMessages, new PrintMessage(true, "combined"),
201       getVertexMapping(), getEdgeMapping());
202 
203     // group messages to final message
204     DataSet<Message> messages = combinedMessages
205       .groupBy(0)
206       .reduceGroup(new GroupedMessages());
207 
208     messages = log(messages, new PrintMessage(true, "grouped"),
209       getVertexMapping(), getEdgeMapping());
210 
211     // update candidates and build next working set
212     DataSet<FatVertex> nextWorkingSet = workSet
213       .leftOuterJoin(messages)
214       .where(0).equalTo(0) // vertexId == recipientId
215       .with(new UpdateVertexState(getQuery()))
216       .filter(new ValidFatVertices());
217 
218     nextWorkingSet = log(nextWorkingSet,
219       new PrintFatVertex(true, "next workset"),
220       getVertexMapping(), getEdgeMapping());
221 
222     // ITERATION FOOTER
223     return workSet.closeWith(nextWorkingSet, deletions);
224   }

226   /**
227    * Performs dual simulation using delta iteration.
228    *
229    * @param vertices fat vertices
230    * @return remaining fat vertices after dual simulation
231    */
232   private DataSet<FatVertex> simulateDelta(DataSet<FatVertex> vertices) {
233     // prepare initial working set
234     DataSet<Message> initialWorkingSet = vertices
235       .flatMap(new ValidateNeighborhood(getQuery()))
236       .groupBy(0)
237       .combineGroup(new CombinedMessages())
238       .groupBy(0)
239       .reduceGroup(new GroupedMessages());
240 
241     vertices = log(vertices, new PrintFatVertex(false, "initial solution set"),
242       getVertexMapping(), getEdgeMapping());
243 
244     initialWorkingSet = log(initialWorkingSet,
245       new PrintMessage(false, "initial working set"),
246       getVertexMapping(), getEdgeMapping());
247 
248     // ITERATION HEAD
249     DeltaIteration<FatVertex, Message> iteration = vertices
250       .iterateDelta(initialWorkingSet, Integer.MAX_VALUE, 0);
251 
252     // ITERATION BODY
253 
254     // get updated vertices
255     DataSet<FatVertex> deltas = iteration.getSolutionSet()
256       .join(iteration.getWorkset())
257       .where(0).equalTo(0)
258       .with(new UpdateVertexState(getQuery()));
259 
260     deltas = log(deltas, new PrintFatVertex(true, "solution set delta"),
261       getVertexMapping(), getEdgeMapping());
262 
263     // prepare new messages for the next round from updates
264     DataSet<Message> updates = deltas
265       .filter(new ValidFatVertices())
266       .flatMap(new ValidateNeighborhood(getQuery()))
267       .groupBy(0)
268       .combineGroup(new CombinedMessages())
269       .groupBy(0)
270       .reduceGroup(new GroupedMessages());
271 
272     updates = log(updates, new PrintMessage(true, "next working set"),
273       getVertexMapping(), getEdgeMapping());
274 
275     // ITERATION FOOTER
276     // filter vertices with no candidates after iteration
277     return iteration.closeWith(deltas, updates).filter(new ValidFatVertices());
278   }

280   /**
281    * Extracts vertices and edges from the query result and constructs a
282    * maximum match graph.
283    *
284    * @param graph    input graph
285    * @param vertices valid vertices after simulation
286    * @return maximum match graph
287    */
288   private GC postProcess(LG graph, DataSet<FatVertex> vertices) {
289     BaseGraphFactory<G, V, E, LG, GC> graphFactory = graph.getFactory();
290 
291     DataSet<V> matchVertices = doAttachData() ?
292       PostProcessor.extractVerticesWithData(vertices, graph.getVertices()) :
293       PostProcessor.extractVertices(vertices, graphFactory.getVertexFactory());
294 
295     DataSet<E> matchEdges = doAttachData() ?
296       PostProcessor.extractEdgesWithData(vertices, graph.getEdges()) :
297       PostProcessor.extractEdges(vertices, graphFactory.getEdgeFactory());
298 
299     return graph.getCollectionFactory().fromGraph(graphFactory.fromDataSets(matchVertices, matchEdges));
300   }
