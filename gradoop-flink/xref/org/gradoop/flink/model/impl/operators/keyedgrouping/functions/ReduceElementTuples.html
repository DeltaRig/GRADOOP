16  package org.gradoop.flink.model.impl.operators.keyedgrouping.functions;
17  
18  import org.apache.flink.api.common.functions.GroupReduceFunction;
19  import org.apache.flink.api.java.tuple.Tuple;
20  import org.gradoop.common.model.impl.properties.PropertyValue;
21  import org.gradoop.flink.model.api.functions.AggregateFunction;
22  
23  import java.util.List;
24  
25  /**
26   * Reduce vertex tuples, calculating aggregate values.
27   *
28   * @param <T> The tuple type.
29   */
30  abstract class ReduceElementTuples<T extends Tuple> implements GroupReduceFunction<T, T> {
31  
32    /**
33     * The data offset for tuples. Aggregate values are expected to start at this index.
34     */
35    private final int tupleDataOffset;
36  
37    /**
38     * The aggregate functions.
39     */
40    private final List<AggregateFunction> aggregateFunctions;
41  
42    /**
43     * Instantiate this base class, setting the data offset and aggregate functions.
44     *
45     * @param tupleDataOffset    The data offset for aggregate values in the tuple.
46     * @param aggregateFunctions The aggregate functions.
47     */
48    ReduceElementTuples(int tupleDataOffset, List<AggregateFunction> aggregateFunctions) {
49      this.tupleDataOffset = tupleDataOffset;
50      this.aggregateFunctions = aggregateFunctions;
51    }
52  
53    /**
54     * Calculate aggregate functions and update tuple fields.
55     *
56     * @param superTuple       The tuple storing the current aggregate values.
57     * @param inputTuple       The tuple storing the increment values.
58     */
59    void callAggregateFunctions(T superTuple, T inputTuple) {
60      // Calculate aggregate values.
61      for (int i = 0; i < aggregateFunctions.size(); i++) {
62        final PropertyValue aggregate = superTuple.getField(i + tupleDataOffset);
63        final PropertyValue increment = inputTuple.getField(i + tupleDataOffset);
64        // Delete the increment from the input tuple as it is not needed anymore.
65        inputTuple.setField(PropertyValue.NULL_VALUE, i + tupleDataOffset);
66        // Do not aggregate if the increment is null.
67        if (!increment.equals(PropertyValue.NULL_VALUE)) {
68          if (aggregate.equals(PropertyValue.NULL_VALUE)) {
69            // If the aggregate is null, use the increment as the new initial value.
70            superTuple.setField(increment, i + tupleDataOffset);
71          } else {
72            superTuple.setField(aggregateFunctions.get(i).aggregate(aggregate, increment), i + tupleDataOffset);
73          }
74        }
75      }
76    }
77  }
