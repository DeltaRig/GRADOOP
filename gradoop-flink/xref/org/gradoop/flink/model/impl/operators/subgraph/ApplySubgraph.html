34  /**
35   * Extracts a subgraph from each base graph in a graph collection using
36   * the given filter functions. The graph head stays unchanged for the subgraph.
37   * <p>
38   * The operator is able to:
39   * <ol>
40   * <li>extract vertex-induced subgraph</li>
41   * <li>extract edge-induced subgraph via join + union strategy</li>
42   * <li>extract edge-induced subgraph via project + union + join strategy</li>
43   * <li>extract subgraph based on vertex and edge filter function without verification
44   * (no joins, use {@link Verify} to verify the subgraph)</li>
45   * </ol>
46   *
47   * @param <G> type of the graph head
48   * @param <V> the vertex type
49   * @param <E> the edge type
50   * @param <LG> type of the logical graph instance
51   * @param <GC> type of the graph collection
52   */
53  public class ApplySubgraph<
54    G extends GraphHead,
55    V extends Vertex,
56    E extends Edge,
57    LG extends BaseGraph<G, V, E, LG, GC>,
58    GC extends BaseGraphCollection<G, V, E, LG, GC>> extends Subgraph<G, V, E, LG, GC>
59    implements ApplicableUnaryBaseGraphToBaseGraphOperator<GC> {
60  
61    /**
62     * Creates a new sub graph operator instance.
63     * <p>
64     * If both parameters are not {@code null}, the operator returns the subgraph
65     * defined by filtered vertices and edges.
66     * <p>
67     * If the {@code edgeFilterFunction} is {@code null}, the operator returns the vertex-induced subgraph.
68     * <p>
69     * If the {@code vertexFilterFunction} is {@code null}, the operator returns the edge-induced subgraph.
70     *
71     * @param vertexFilterFunction vertex filter function
72     * @param edgeFilterFunction   edge filter function
73     * @param strategy             sets the execution strategy for the operator
74     */
75    public ApplySubgraph(FilterFunction<V> vertexFilterFunction,
76      FilterFunction<E> edgeFilterFunction, Strategy strategy) {
77      super(vertexFilterFunction, edgeFilterFunction, strategy);
78    }
79  
80    @Override
81    public GC executeForGVELayout(GC collection) {
82  
83      GC result;
84      switch (strategy) {
85      case BOTH:
86        result = subgraph(collection);
87        break;
88      case VERTEX_INDUCED:
89        result = vertexInducedSubgraph(collection);
90        break;
91      case EDGE_INDUCED:
92        result = edgeInducedSubgraph(collection);
93        break;
94      case EDGE_INDUCED_PROJECT_FIRST:
95        result = edgeInducedSubgraphProjectFirst(collection);
96        break;
97      default:
98        throw new IllegalArgumentException("Strategy " + strategy + " is not supported");
99      }
100 
101     return result;
102   }
103 
104   @Override
105   public GC executeForTxLayout(GC collection) {
106     return executeForGVELayout(collection);
107   }
108 
109   /**
110    * Returns one subgraph for each of the given super graphs.
111    * The subgraphs are defined by the vertices that fulfil the vertex filter function.
112    *
113    * @param collection collection of supergraphs
114    * @return collection of vertex-induced subgraphs
115    */
116   private GC vertexInducedSubgraph(GC collection) {
117     DataSet<V> filteredVertices = collection.getVertices().filter(vertexFilterFunction);
118 
119     return collection.getFactory()
120       .fromDataSets(collection.getGraphHeads(), filteredVertices, collection.getEdges())
121       .verify();
122   }
123 
124   /**
125    * Returns one subgraph for each of the given supergraphs.
126    * The subgraphs are defined by the edges that fulfil the vertex filter function.
127    *
128    * @param collection collection of supergraphs
129    * @return collection of edge-induced subgraphs
130    */
131   private GC edgeInducedSubgraph(GC collection) {
132     DataSet<E> filteredEdges = collection.getEdges().filter(edgeFilterFunction);
133     DataSet<V> inducedVertices = filteredEdges
134       .join(collection.getVertices())
135       .where(new SourceId<>()).equalTo(new Id<>())
136       .with(new RightSideWithLeftGraphs<>())
137       .union(filteredEdges
138         .join(collection.getVertices())
139         .where(new TargetId<>()).equalTo(new Id<>())
140         .with(new RightSideWithLeftGraphs<>()))
141       .distinct(new Id<>());
142 
143     return collection.getFactory()
144       .fromDataSets(collection.getGraphHeads(), inducedVertices, filteredEdges);
145   }
146 
147   /**
148    * Returns one subgraph for each of the given supergraphs.
149    * The subgraphs are defined by the edges that fulfil the vertex filter function.
150    *
151    * @param collection collection of supergraph
152    * @return collection of edge-induced subgraph
153    */
154   private GC edgeInducedSubgraphProjectFirst(GC collection) {
155     DataSet<E> filteredEdges = collection.getEdges().filter(edgeFilterFunction);
156     DataSet<V> inducedVertices = filteredEdges
157       .flatMap(new EdgeToSourceAndTargetIdWithGraphIds<>())
158       .distinct(0)
159       .join(collection.getVertices())
160       .where(0).equalTo(new Id<>())
161       .with(new RightSideWithLeftGraphs2To1<>());
162 
163     return collection.getFactory()
164       .fromDataSets(collection.getGraphHeads(), inducedVertices, filteredEdges);
165   }
166 
167   /**
168    * Returns one subgraph for each of the given supergraphs.
169    * The subgraphs are defined by the vertices that fulfil the vertex filter
170    * function and edges that fulfill the edge filter function.
171    * <p>
172    * Note, that the operator does not verify the consistency of the resulting graph.
173    *
174    * @param collection collection of supergraphs
175    * @return collection of subgraphs
176    */
177   private GC subgraph(GC collection) {
178     DataSet<V> newVertices = collection.getVertices().filter(vertexFilterFunction);
179     DataSet<E> newEdges = collection.getEdges().filter(edgeFilterFunction);
180 
181     return collection.getFactory()
182       .fromDataSets(collection.getGraphHeads(), newVertices, newEdges);
183   }
