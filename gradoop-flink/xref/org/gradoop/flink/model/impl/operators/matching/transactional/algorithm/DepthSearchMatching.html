39  /**
40   * This is an implementation of a very straight-forward depth-first pattern matching algorithm.
41   * <p>
42   * The depth-first search is implemented by popping an embedding from a stack, growing it by one vertex
43   * and its corresponding edges, and adding the resulting new edges to the top of the stack.
44   * If it is not possible to apply a step to an existing embedding, the embedding is removed.
45   * This is repeated until all embeddings are either completed (all steps have been applied).
46   */
47  public class DepthSearchMatching implements PatternMatchingAlgorithm {

57     * {@code source-id -> edge-ids}

59    private Map<GradoopId, Set<GradoopId>> sourceDict;
60    /**
61     * {@code target-id -> edge-ids}
62     */
63    private Map<GradoopId, Set<GradoopId>> targetDict;
64    /**
65     * {@code edge id -> source-id, target-id, edge candidates}

83    @Override
84    public List<Embedding<GradoopId>> findEmbeddings(GraphWithCandidates graph,
85      String query) {
86      // construct a handler for the query
87      handler = new QueryHandler(query);
88  
89      // build the plan that determines in which order vertices should added
90      int[] plan = buildQueryPlan();
91  
92      // by using a stack (first in - last out) we implement a depth-first search
93      Stack<Embedding<GradoopId>> embeddings = new Stack<>();
94  
95      // construct an empty embedding
96      Embedding<GradoopId> firstEmbedding = new Embedding<>();
97      firstEmbedding.setVertexMapping(new GradoopId[handler.getVertexCount()]);
98      firstEmbedding.setEdgeMapping(new GradoopId[handler.getEdgeCount()]);
99      embeddings.push(firstEmbedding);
100 
101     // initialize the vertex, edge, source and target maps
102     initializeMaps(graph);
103 
104     List<Embedding<GradoopId>> results = new ArrayList<>();
105 
106     // grow the embeddings until there are none left
107     while (!embeddings.isEmpty()) {
108       Embedding<GradoopId> embedding = embeddings.pop();
109 
110       // find the first step in the plan that has not been applied yet
111       int nextStep = -1;
112       for (int step : plan) {
113         if (embedding.getVertexMapping()[step] == null) {
114           nextStep = step;
115           break;
116         }
117       }
118 
119       // if all steps have been applied, add the embedding to the results
120       if (nextStep == -1) {
121         results.add(embedding);
122         continue;
123       }
124 
125       // compute new embeddings by applying the step on the current embedding
126       List<Embedding<GradoopId>> newEmbeddings = executeStep(embedding, nextStep);
127 
128       // add the new embeddings on top of the stack
129       embeddings.addAll(newEmbeddings);
130     }
131 
132     // reset the maps for the next graph
133     resetMaps();
134 
135     return results;
136   }

138   @Override
139   public Boolean hasEmbedding(GraphWithCandidates graph, String query) {
140 
141     // construct a handler for the query
142     handler = new QueryHandler(query);
143 
144     // build the plan that determines in which order vertices should added
145     int[] plan = buildQueryPlan();
146 
147     // by using a stack (first in - last out) we implement a depth-first search
148     Stack<Embedding<GradoopId>> embeddings = new Stack<>();
149 
150     // construct an empty embedding
151     Embedding<GradoopId> firstEmbedding = new Embedding<>();
152     firstEmbedding.setVertexMapping(new GradoopId[handler.getVertexCount()]);
153     firstEmbedding.setEdgeMapping(new GradoopId[handler.getEdgeCount()]);
154     embeddings.push(firstEmbedding);
155 
156     // initialize the vertex, edge, source and target maps
157     initializeMaps(graph);
158 
159     // grow the embeddings until there are none left
160     while (!embeddings.isEmpty()) {
161       Embedding<GradoopId> embedding = embeddings.pop();
162 
163       // find the first step in the plan that has not been applied yet
164       int nextStep = -1;
165       for (int step : plan) {
166         if (embedding.getVertexMapping()[step] == null) {
167           nextStep = step;
168           break;
169         }
170       }
171 
172       // if there is a complete embedding, return true
173       if (nextStep == -1) {
174         resetMaps();
175         return true;
176       }
177 
178       // compute new embeddings by applying the step on the current embedding
179       List<Embedding<GradoopId>> newEmbeddings = executeStep(embedding, nextStep);
180 
181       // add the new embeddings on top of the stack
182       embeddings.addAll(newEmbeddings);
183     }
184 
185     // reset the maps for the next graph
186     resetMaps();
187 
188     // if no complete embedding could be constructed, return false
189     return false;
190   }

202   /**
203    * Execute a step. A step corresponds to the position of the next vertex in
204    * the vertex mappings of the embedding that shall be matched.
205    *
206    * @param embedding the embedding that has been constructed so far
207    * @param step      number of the next vertex to be matched
208    * @return list of newly constructed embeddings
209    */
210   private List<Embedding<GradoopId>> executeStep(Embedding<GradoopId> embedding, int step) {
211 
212     // map containing the found matches for all pattern edges
213     // this is necessary to be able to construct all valid permutations of edges
214     Map<Long, Set<GradoopId>> edgeMatches = new HashMap<>();
215 
216     List<Embedding<GradoopId>> results = new ArrayList<>();
217 
218     // get the possible vertex matches
219     for (GradoopId id : getCandidates(step)) {
220 
221       // flag to recognize failed vertex matchings
222       boolean failed = false;
223 
224       // if the vertex is already in the embedding, skip it
225       if (Arrays.asList(embedding.getVertexMapping()).contains(id)) {
226         continue;
227       }
228 
229       // get all outgoing edges of the next step vertex
230       Collection<Edge> edges = handler.getEdgesBySourceVertexId((long) step);
231       List<Edge> patternEdges =
232         edges != null ? new ArrayList<>(edges) : new ArrayList<>();
233 
234       // only keep those edges that have at least one vertex already in the
235       // embedding, or are direct loops
236       filterPatternEdges(patternEdges, embedding);
237 
238       // find the matches for each pattern edge
239       for (Edge patternEdge : patternEdges) {
240         edgeMatches.put(patternEdge.getId(), new HashSet<>());
241 
242         // find all edge candidates
243         Set<GradoopId> edgeCandidateIds =
244           sourceDict.get(id) != null ? sourceDict.get(id) : new HashSet<>();
245 
246         // for each candidate, check if it matches the pattern edge
247         for (GradoopId edgeCandidateId : edgeCandidateIds) {
248 
249           Tuple3<GradoopId, GradoopId, boolean[]> edgeCandidate =
250             edgeDict.get(edgeCandidateId);
251 
252           // get the target vertex of the edge
253           GradoopId target = embedding.getVertexMapping()[
254             Math.toIntExact(patternEdge.getTargetVertexId())];
255 
256           // if the pattern edge and the edge candidate are both loops
257           // and the candidate matches, add it to the embedding
258           if (isLoop(patternEdge, edgeCandidate)) {
259             edgeMatches.get(patternEdge.getId()).add(edgeCandidateId);
260 
261             // else, if the candidate matches the pattern edge,
262             // add it to the embedding
263           } else if (matchOutgoingEdge(patternEdge, edgeCandidateId, target)) {
264             edgeMatches.get(patternEdge.getId()).add(edgeCandidateId);
265           }
266         }
267 
268         // if there was no matching edge for a pattern edge, the candidate can be discarded
269         if (edgeMatches.get(patternEdge.getId()).isEmpty()) {
270           failed = true;
271           break;
272         }
273       }
274 
275       if (failed) {
276         continue;
277       }
278 
279       // get all incoming edges of the next step vertex
280       edges = handler.getEdgesByTargetVertexId((long) step);
281       patternEdges = edges != null ? new ArrayList<>(edges) : new ArrayList<>();
282 
283       // only keep those edges that have at least one vertex already in the
284       // embedding, or are direct loops
285       filterPatternEdges(patternEdges, embedding);
286 
287       // find the matches for each pattern edge
288       for (Edge patternEdge : patternEdges) {
289         edgeMatches.put(patternEdge.getId(), new HashSet<>());
290 
291         // find all edge candidates
292         Set<GradoopId> edgeCandidateIds =
293           targetDict.get(id) != null ? targetDict.get(id) : new HashSet<>();
294 
295         // for each candidate, check if it matches the pattern edge
296         for (GradoopId edgeCandidateId : edgeCandidateIds) {
297 
298           Tuple3<GradoopId, GradoopId, boolean[]> edgeCandidate =
299             edgeDict.get(edgeCandidateId);
300 
301           // get the target vertex of the edge
302           GradoopId source = embedding.getVertexMapping()[
303             Math.toIntExact(patternEdge.getSourceVertexId())];
304 
305           // if the pattern edge and the edge candidate are both loops
306           // and the candidate matches, add it to the embedding
307           if (isLoop(patternEdge, edgeCandidate)) {
308             edgeMatches.get(patternEdge.getId()).add(edgeCandidateId);
309 
310             // else, if the candidate matches the pattern edge,
311             // add it to the embedding
312           } else if (matchIncomingEdge(patternEdge, edgeCandidateId, source)) {
313             edgeMatches.get(patternEdge.getId()).add(edgeCandidateId);
314           }
315         }
316 
317         // if there was no matching edge for a pattern edge, the candidate
318         // can be discarded
319         if (edgeMatches.get(patternEdge.getId()).isEmpty()) {
320           failed = true;
321           break;
322         }
323       }
324 
325       if (failed) {
326         continue;
327       }
328       // add all grown embeddings to the results
329       results.addAll(buildNewEmbeddings(embedding, step, id, edgeMatches));
330     }
331     return results;
332   }

334   /**
335    * If both the pattern edge and the edge candidate are loops and the candidate
336    * matches the pattern, return true.
337    *
338    * @param patternEdge pattern edge
339    * @param edgeCandidate edge candidate
340    * @return true iff both are loops and the candidate matches the pattern edge
341    */
342   private boolean isLoop(Edge patternEdge,
343     Tuple3<GradoopId, GradoopId, boolean[]> edgeCandidate) {
344     if (patternEdge.getSourceVertexId()
345       .equals(patternEdge.getTargetVertexId())) {
346       if (edgeCandidate.f0.equals(edgeCandidate.f1)) {
347         return edgeCandidate.f2[(int) patternEdge.getId()];
348       }
349     }
350     return false;
351   }

353   /**
354    * Remove all edges that are neither loops nor have source or target vertex that is in the embedding.
355    *
356    * @param patternEdges edges to be filtered
357    * @param embedding embedding with possible source and target vertices
358    * @return filtered edge list
359    */
360   private List<Edge> filterPatternEdges(
361     List<Edge> patternEdges,
362     Embedding<GradoopId> embedding) {
363     for (int i = 0; i < patternEdges.size(); i++) {
364       long sourceId = patternEdges.get(i).getSourceVertexId();
365       long targetId = patternEdges.get(i).getTargetVertexId();
366       GradoopId[] vertexMapping = embedding.getVertexMapping();
367       if ((vertexMapping[(int) sourceId] == null) &&
368         (vertexMapping[(int) targetId] == null)) {
369         if (sourceId != targetId) {
370           patternEdges.remove(i);
371           i--;
372         }
373       }
374     }
375     return patternEdges;
376   }

378   /**
379    * Extend the given embedding by found matches
380    *
381    * @param startEmbedding embedding that is to be extended
382    * @param step next step
383    * @param vertexMatch vertex chosen for next step
384    * @param edgeMatches found match
385    * @return extended embedding
386    */
387   private List<Embedding<GradoopId>> buildNewEmbeddings(
388     Embedding<GradoopId> startEmbedding,
389     int step,
390     GradoopId vertexMatch,
391     Map<Long, Set<GradoopId>> edgeMatches) {
392 
393     // begin with the start embedding
394     List<Embedding<GradoopId>> result = new ArrayList<>();
395     result.add(startEmbedding);
396 
397     // list of embeddings in next growing step
398     List<Embedding<GradoopId>> temporaryEmbeddings;
399 
400     // get edge matches
401     List<Map.Entry<Long, Set<GradoopId>>> entries =
402       new ArrayList<>(edgeMatches.entrySet());
403 
404     // if no edge matches have been found, the result is an embedding with only
405     // one vertex
406     if (entries.isEmpty()) {
407       Embedding<GradoopId> newEmbedding = copyEmbedding(startEmbedding);
408       newEmbedding.getVertexMapping()[step] = vertexMatch;
409       result.clear();
410       result.add(newEmbedding);
411     }
412 
413     // for each edge with matches
414     for (Map.Entry<Long, Set<GradoopId>> entry : entries) {
415       temporaryEmbeddings = new ArrayList<>();
416       // for each embedding in the result so far
417       for (Embedding<GradoopId> embedding : result) {
418         // for each edge match
419         for (GradoopId id : entry.getValue()) {
420           // check if the edge match already occurs in the embedding
421           boolean contains = false;
422           for (GradoopId edgeId : embedding.getEdgeMapping()) {
423             if (edgeId == null) {
424               continue;
425             }
426             if (edgeId.equals(id)) {
427               contains = true;
428               break;
429             }
430           }
431           // if yes, skip it
432           if (contains) {
433             continue;
434           }
435           // else, copy the embedding, add the edge match and add it to the
436           // results in the next iteration
437           Embedding<GradoopId> newEmbedding = copyEmbedding(embedding);
438           newEmbedding.getVertexMapping()[step] = vertexMatch;
439           newEmbedding.getEdgeMapping()[Math.toIntExact(entry.getKey())] = id;
440           temporaryEmbeddings.add(newEmbedding);
441         }
442       }
443       result = new ArrayList<>(temporaryEmbeddings);
444     }
445     return result;
446   }

448   /**
449    * Creates a copy of an existing embedding.
450    *
451    * @param existing existing embedding
452    * @return copy of the existing embedding
453    */
454   private Embedding<GradoopId> copyEmbedding(Embedding<GradoopId> existing) {
455     Embedding<GradoopId> newEmbedding = new Embedding<>();
456 
457     GradoopId[] vertexCopy = new GradoopId[existing.getVertexMapping().length];
458     System.arraycopy(existing.getVertexMapping(), 0, vertexCopy, 0,
459       existing.getVertexMapping().length);
460 
461     GradoopId[] edgeCopy = new GradoopId[existing.getEdgeMapping().length];
462     System.arraycopy(existing.getEdgeMapping(), 0, edgeCopy, 0,
463       existing.getEdgeMapping().length);
464 
465     newEmbedding.setVertexMapping(vertexCopy);
466     newEmbedding.setEdgeMapping(edgeCopy);
467     return newEmbedding;
468   }

470   /**
471    * Get all vertices that match the pattern of a step.
472    *
473    * @param step next step in the plan
474    * @return list containing the ids of the matching vertices
475    */
476   public List<GradoopId> getCandidates(int step) {
477     List<GradoopId> possibilities = new ArrayList<>(vertexDict.keySet());
478     for (int i = 0; i < possibilities.size(); i++) {
479       GradoopId vertexId = possibilities.get(i);
480       boolean[] candidates = vertexDict.get(vertexId);
481       if (!candidates[step]) {
482         possibilities.remove(i);
483         --i;
484       }
485     }
486     return possibilities;
487   }

489   /**
490    * Returns true, if the edge candidate matches the outgoing pattern edge.
491    *
492    * @param patternEdge outgoing pattern edge
493    * @param edgeCandidate id of the edge candidate
494    * @param target target id of the edge candidate
495    * @return true, iff edge matches pattern edge
496    */
497   private boolean matchOutgoingEdge(Edge patternEdge, GradoopId edgeCandidate,
498     GradoopId target) {
499     Tuple3<GradoopId, GradoopId, boolean[]> edge = edgeDict.get(edgeCandidate);
500     if (edge.f2[(int) patternEdge.getId()]) {
501       GradoopId possibleVertex = edge.f1;
502       return possibleVertex.equals(target);
503     }
504     return false;
505   }

511    * @param edgeCandidate id of the edge candidate
512    * @param source source id of the edge candidate

525   /**
526    * Build HashMaps for faster access.
527    *
528    * @param graph graphs with candidate sets
529    */
530   private void initializeMaps(GraphWithCandidates graph) {
531     for (IdWithCandidates<GradoopId> vertex : graph.getVertexCandidates()) {
532       vertexDict.put(vertex.getId(), vertex.getCandidates());
533     }
534 
535     for (TripleWithCandidates<GradoopId> edge : graph.getEdgeCandidates()) {
536       edgeDict.put(edge.getEdgeId(),
537         new Tuple3<>(edge.getSourceId(), edge.getTargetId(),
538           edge.getCandidates()));
539     }
540 
541     for (TripleWithCandidates<GradoopId> edge : graph.getEdgeCandidates()) {
542       if (!sourceDict.containsKey(edge.getSourceId())) {
543         sourceDict.put(edge.getSourceId(), new HashSet<>());
544       }
545       sourceDict.get(edge.getSourceId()).add(edge.getEdgeId());
546       if (!targetDict.containsKey(edge.getTargetId())) {
547         targetDict.put(edge.getTargetId(), new HashSet<>());
548       }
549       targetDict.get(edge.getTargetId()).add(edge.getEdgeId());
550     }
551   }

553   /**
554    * Method to create query plan, determining how patterns are grown.
555    *
556    * @return plan for walk through
557    */
558   private int[] buildQueryPlan() {
559     int[] queryPlan = new int[handler.getVertices().size()];
560     int step = 0;
561     Set<Long> alreadyVisited = new HashSet<>();
562     Stack<Vertex> stack = new Stack<>();
563     stack.push(handler.getVertexById(0L));
564     alreadyVisited.add(0L);
565     while (!stack.isEmpty()) {
566       Vertex current = stack.pop();
567       Collection<Vertex> neighbors = handler.getNeighbors(current.getId());
568       queryPlan[step] = (int) current.getId();
569       step++;
570       neighbors.stream()
571         .filter(neighbor -> !alreadyVisited.contains(neighbor.getId()))
572         .forEach(neighbor -> {
573             alreadyVisited.add(neighbor.getId());
574             stack.push(neighbor);
575           }
576         );
577     }
578     return queryPlan;
579   }
