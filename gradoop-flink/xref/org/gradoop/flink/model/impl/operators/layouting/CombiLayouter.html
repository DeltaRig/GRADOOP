16  package org.gradoop.flink.model.impl.operators.layouting;
17  
18  import org.gradoop.flink.model.impl.epgm.LogicalGraph;
19  
20  /**
21   * A layouter that combines the {@link CentroidFRLayouter } and the {@link FRLayouter}.
22   */
23  public class CombiLayouter implements LayoutingAlgorithm {
24    /**
25     * Factor by which the k of the CentroidFRLayouter is scaled
26     */
27    private static final double K_FACTOR = 1.3;
28    /**
29     * Number of iterations to perform
30     */
31    private int iterations;
32    /**
33     * Approximate number of vertices in the input-graph
34     */
35    private int numberOfVertices;
36    /**
37     * The ratio of iterations of the CentroidFRLayouter and the FRLayouter
38     */
39    private double quality;
40    /**
41     * The CentroidFRLayouter that is being used
42     */
43    private CentroidFRLayouter centroidFRLayouter;
44    /**
45     * The FRLayouter that is being used
46     */
47    private FRLayouter fRLayouter;
48  
49    /**
50     * Create a new CombiLayouter.
51     *
52     * @param iterations  Number of iterations to perform
53     * @param numVertices Approximate number of vertices in the input-graph
54     * @param quality     Ratio of iterations between the two base algorithms. The higher the value,
55     *                    the more iterations of the FRLayouter are performed. A value of 0.1 is often
56     *                    good enough.
57     */
58    public CombiLayouter(int iterations, int numVertices, double quality) {
59  
60      this.iterations = iterations;
61      this.numberOfVertices = numVertices;
62      this.quality = quality;
63      int centroidIterations = (int) Math.floor(iterations * (1 - quality));
64      int frIterations = (int) Math.ceil(iterations * quality);
65  
66      if (centroidIterations > 0) {
67        centroidFRLayouter = new CentroidFRLayouter(centroidIterations, numVertices);
68        centroidFRLayouter.k(centroidFRLayouter.getK() * K_FACTOR);
69      }
70      if (frIterations > 0) {
71        fRLayouter =
72          new FRLayouter(frIterations, numVertices).useExistingLayout(centroidFRLayouter != null)
73            .startAtIteration(centroidIterations);
74      }
75    }
76  
77    /**
78     * Set the FRLayouter parameter k for both algorithms, overwriting the default value
79     *
80     * @param k The new value
81     * @return this layouter
82     */
83    public CombiLayouter k(double k) {
84      if (centroidFRLayouter != null) {
85        centroidFRLayouter.k(k * K_FACTOR);
86      }
87      if (fRLayouter != null) {
88        fRLayouter.k(k);
89      }
90      return this;
91    }
92  
93    /**
94     * Override default layout-space size
95     * Default:  {@code width = height = Math.sqrt(Math.pow(k, 2) * numberOfVertices) * 0.5}
96     *
97     * @param width  new width
98     * @param height new height
99     * @return this layouter
100    */
101   public CombiLayouter area(int width, int height) {
102     if (centroidFRLayouter != null) {
103       centroidFRLayouter.area(width, height);
104       if (fRLayouter != null) {
105         fRLayouter.area(width, height);
106       }
107     }
108     return this;
109   }
110 
111   /**
112    * Override default maxRepulsionDistance of the FR-Algorithm. Vertices with larger distance
113    * are ignored in repulsion-force calculation
114    * Default-Value is relative to current {@code k}. If {@code k} is overriden, this is changed
115    * accordingly automatically
116    * Default: 2k
117    *
118    * @param maxRepulsionDistance new value
119    * @return this layouter
120    */
121   public CombiLayouter maxRepulsionDistance(int maxRepulsionDistance) {
122     if (fRLayouter != null) {
123       fRLayouter.maxRepulsionDistance(maxRepulsionDistance);
124     }
125     return this;
126   }
127 
128   /**
129    * Use the existing layout as starting point instead of creating a random one.
130    * If used, EVERY vertex in the input-graph MUST have an X and Y property!
131    *
132    * @param useExisting whether to re-use the existing layout or not
133    * @return this layouter
134    */
135   public CombiLayouter useExistingLayout(boolean useExisting) {
136     if (centroidFRLayouter != null) {
137       centroidFRLayouter.useExistingLayout(useExisting);
138     } else {
139       fRLayouter.useExistingLayout(useExisting);
140     }
141     return this;
142   }
143 
144   /**
145    * Gets k. K is the distance in which the attracting and repulsive forces between two connected
146    * vertices are equally strong. The value of k influences the scale of the final graph-layout.
147    * (A small k leads to vertices being closer together)
148    *
149    * @return value of k
150    */
151   public double getK() {
152     if (fRLayouter != null) {
153       return fRLayouter.getK();
154     } else {
155       return centroidFRLayouter.getK() / K_FACTOR;
156     }
157   }
158 
159   /**
160    * Gets maxRepulsionDistance
161    *
162    * @return value of maxRepulsionDistance
163    */
164   public int getMaxRepulsionDistance() {
165     if (fRLayouter == null) {
166       return -1;
167     }
168     return fRLayouter.getMaxRepulsionDistance();
169   }
170 
171   @Override
172   public LogicalGraph execute(LogicalGraph g) {
173     if (centroidFRLayouter != null) {
174       g = centroidFRLayouter.execute(g);
175     }
176     if (fRLayouter != null) {
177       g = fRLayouter.execute(g);
178     }
179     return g;
180   }
181 
182   @Override
183   public int getWidth() {
184     if (fRLayouter != null) {
185       return fRLayouter.getWidth();
186     } else {
187       return centroidFRLayouter.getWidth();
188     }
189   }
190 
191   @Override
192   public int getHeight() {
193     if (fRLayouter != null) {
194       return fRLayouter.getHeight();
195     } else {
196       return centroidFRLayouter.getHeight();
197     }
198   }
199 
200   @Override
201   public String toString() {
202     return "CombiFRLayouter{" + " quality=" + quality + ", iterations=" + iterations + ", k=" +
203       getK() + "," + " " + "with=" + getWidth() + ", height=" + getHeight() +
204       ", maxRepulsionDistance=" + getMaxRepulsionDistance() + ", numberOfVertices=" +
205       numberOfVertices + '}';
206   }
