16  package org.gradoop.flink.model.impl.operators.layouting.functions;
17  
18  import org.apache.flink.api.common.functions.RichJoinFunction;
19  import org.gradoop.flink.model.impl.operators.layouting.util.Force;
20  import org.gradoop.flink.model.impl.operators.layouting.util.LVertex;
21  import org.gradoop.flink.model.impl.operators.layouting.util.Vector;

23  /**
24   * Applies forces to vertices.
25   * Uses simulated annealing with exponentially decreasing temperature.
26   * Confines new coordinates to the layouting-space
27   */
28  public class FRForceApplicator extends RichJoinFunction<LVertex, Force, LVertex> {
29  
30    /**
31     * Width of the layouting-space
32     */
33    protected int layoutWidth;
34    /**
35     * Height of the layouting space
36     */
37    protected int layoutHeight;
38    /**
39     * cache the last computed temperature and re-use if possible to reduce needed computing-power
40     */
41    protected int lastIteration = -1;
42    /**
43     * cache the last computed temperature and re-use if possible to reduce needed computing-power
44     */
45    protected double lastSpeedLimit = 0;
46    /**
47     * Speed at which the cooling-schedule starts
48     */
49    private double startSpeed;
50    /**
51     * Speed at which the cooling-schedule ends
52     */
53    private double endSpeed;
54    /**
55     * Base of the exponentially-decreasing function for the speed
56     */
57    private double base;
58    /**
59     * Maximum number of iterations
60     */
61    private int maxIterations;
62    /**
63     * Calculate temperature as if this number of iterations had already passed
64     */
65    private int previousIterations = 0;
66  
67    /**
68     * Create new FRForceApplicator
69     *
70     * @param width         The width of the layouting-space
71     * @param height        The height of the layouting-space
72     * @param k             A parameter of the FR-Algorithm.
73     * @param maxIterations Number of iterations the FR-Algorithm will have
74     */
75    public FRForceApplicator(int width, int height, double k, int maxIterations) {
76      this.layoutWidth = width;
77      this.layoutHeight = height;
78      this.startSpeed = Math.sqrt((double) width * (double) width + (double) height * (double) height) / 2.0;
79      this.endSpeed = k / 10.0;
80      this.maxIterations = maxIterations;
81      calculateBase();
82    }
83  
84    /**
85     * Gets startSpeed
86     *
87     * @return value of startSpeed
88     */
89    public double getStartSpeed() {
90      return startSpeed;
91    }
92  
93    /**
94     * Sets startSpeed (overrides default)
95     *
96     * @param startSpeed the new value
97     */
98    public void setStartSpeed(double startSpeed) {
99      this.startSpeed = startSpeed;
100     calculateBase();
101   }
102 
103   /**
104    * Gets endSpeed
105    *
106    * @return value of endSpeed
107    */
108   public double getEndSpeed() {
109     return endSpeed;
110   }
111 
112   /**
113    * Sets endSpeed (overrides default)
114    *
115    * @param endSpeed the new value
116    */
117   public void setEndSpeed(double endSpeed) {
118     this.endSpeed = endSpeed;
119     calculateBase();
120   }
121 
122   /**
123    * Gets previousIterations
124    *
125    * @return value of previousIterations
126    */
127   public int getPreviousIterations() {
128     return previousIterations;
129   }
130 
131   /**
132    * Sets previousIterations
133    *
134    * @param previousIterations the new value
135    */
136   public void setPreviousIterations(int previousIterations) {
137     this.previousIterations = previousIterations;
138     calculateBase();
139   }
140 
141   /**
142    * Gets maxIterations
143    *
144    * @return value of maxIterations
145    */
146   public int getMaxIterations() {
147     return maxIterations;
148   }
149 
150   /**
151    * Sets maxIterations
152    *
153    * @param maxIterations the new value
154    */
155   public void setMaxIterations(int maxIterations) {
156     this.maxIterations = maxIterations;
157   }
158 
159   /**
160    * Calculate the base for the exponential function
161    */
162   private void calculateBase() {
163     this.base = Math.pow(endSpeed / startSpeed, 1.0 / (maxIterations + previousIterations - 1));
164   }
165 
166   /**
167    * Calculate the desired speed for a given iteration
168    *
169    * @param iteration Iteration to calculate speed for
170    * @return Desired speed
171    */
172   public double speedForIteration(int iteration) {
173     iteration += previousIterations;
174     // cache last result to avoid costly pow
175     if (iteration != lastIteration) {
176       lastSpeedLimit = startSpeed * Math.pow(base, iteration);
177       lastIteration = iteration;
178     }
179     return lastSpeedLimit;
180   }
181 
182 
183   /**
184    * Apply force to vertex.
185    * Honors speedLimit and vertex-mass. Confines position to layout-area.
186    *
187    * @param vertex     Vertex to move
188    * @param force      Force acting on the vertex
189    * @param speedLimit Current speedLimit
190    */
191   public void apply(LVertex vertex, Force force, double speedLimit) {
192     Vector position = vertex.getPosition();
193     Vector movement = force.getValue();
194     apply(position, movement.div(vertex.getCount()), speedLimit);
195     vertex.setForce(movement);
196     vertex.setPosition(position);
197   }
198 
199 
200   /**
201    * Raw version of apply() using just vectors.
202    *
203    * @param position   Position to modify
204    * @param movement   Movement to apply
205    * @param speedLimit Current speedLimit
206    */
207   public void apply(Vector position, Vector movement, double speedLimit) {
208     position.mAdd(movement.clamped(speedLimit));
209     position.mConfined(0, layoutWidth - 1, 0, layoutHeight - 1);
210   }
211 
212   @Override
213   public LVertex join(LVertex first, Force second) throws Exception {
214     apply(first, second, speedForIteration(getIterationRuntimeContext().getSuperstepNumber()));
215     return first;
216   }
