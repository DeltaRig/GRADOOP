48  /**
49   * The grouping operator determines a structural grouping of vertices and edges
50   * to condense a graph and thus help to uncover insights about patterns and
51   * statistics hidden in the graph.
52   * <p>
53   * The graph grouping operator represents every vertex group by a single super
54   * vertex in the resulting graph; (super) edges between vertices in the
55   * resulting graph represent a group of edges between the vertex group members
56   * of the original graph. Grouping is defined by specifying grouping keys for
57   * vertices and edges, respectively, similarly as for GROUP BY in SQL.
58   * <p>
59   * Consider the following example:
60   * <p>
61   * Input graph:
62   * <p>
63   * Vertices:<br>
64   * (0, "Person", {city: L})<br>
65   * (1, "Person", {city: L})<br>
66   * (2, "Person", {city: D})<br>
67   * (3, "Person", {city: D})<br>
68   * <p>
69   * Edges:{(0,1), (1,0), (1,2), (2,1), (2,3), (3,2)}
70   * <p>
71   * Output graph (grouped on vertex property "city"):
72   * <p>
73   * Vertices:<br>
74   * (0, "Person", {city: L, count: 2})
75   * (2, "Person", {city: D, count: 2})
76   * <p>
77   * Edges:<br>
78   * ((0, 0), {count: 2}) // 2 intra-edges in L<br>
79   * ((2, 2), {count: 2}) // 2 intra-edges in L<br>
80   * ((0, 2), {count: 1}) // 1 inter-edge from L to D<br>
81   * ((2, 0), {count: 1}) // 1 inter-edge from D to L<br>
82   * <p>
83   * In addition to vertex properties, grouping is also possible on edge
84   * properties, vertex- and edge labels as well as combinations of those.
85   *
86   * @param <G>  The graph head type.
87   * @param <V>  The vertex type.
88   * @param <E>  The edge type.
89   * @param <LG> The type of the graph.
90   * @param <GC> The type of the graph collection.
91   */
92  public abstract class Grouping<
93    G extends GraphHead,
94    V extends Vertex,
95    E extends Edge,
96    LG extends BaseGraph<G, V, E, LG, GC>,
97    GC extends BaseGraphCollection<G, V, E, LG, GC>>
98    implements UnaryBaseGraphToBaseGraphOperator<LG> {

157   @Override
158   public LG execute(LG graph) {
159     LG result;
160 
161     if (!useVertexProperties() &&
162       !useEdgeProperties() &&
163       !useVertexLabels() &&
164       !useEdgeLabels()) {
165       result = graph;
166     } else {
167       result = groupInternal(graph);
168     }
169     return result;
170   }

236   /**
237    * Group vertices by either vertex label, vertex property or both.
238    *
239    * @param groupVertices dataset containing vertex representation for grouping
240    * @return unsorted vertex grouping
241    */
242   protected UnsortedGrouping<VertexGroupItem> groupVertices(
243     DataSet<VertexGroupItem> groupVertices) {
244     UnsortedGrouping<VertexGroupItem> vertexGrouping;
245     if (useVertexLabels() && useVertexProperties()) {
246       vertexGrouping = groupVertices.groupBy(2, 3);
247     } else if (useVertexLabels()) {
248       vertexGrouping = groupVertices.groupBy(2);
249     } else {
250       vertexGrouping = groupVertices.groupBy(3);
251     }
252     return vertexGrouping;
253   }
254 
255   /**
256    * Groups edges based on the algorithm parameters.
257    *
258    * @param edges input graph edges
259    * @return grouped edges
260    */
261   protected UnsortedGrouping<EdgeGroupItem> groupEdges(DataSet<EdgeGroupItem> edges) {
262     UnsortedGrouping<EdgeGroupItem> groupedEdges;
263     if (useEdgeProperties() && useEdgeLabels()) {
264       groupedEdges = edges.groupBy(0, 1, 2, 3);
265     } else if (useEdgeLabels()) {
266       groupedEdges = edges.groupBy(0, 1, 2);
267     } else if (useEdgeProperties()) {
268       groupedEdges = edges.groupBy(0, 1, 3);
269     } else {
270       groupedEdges = edges.groupBy(0, 1);
271     }
272     return groupedEdges;
273   }

275   /**
276    * Build super edges by joining them with vertices and their super vertex.
277    *
278    * @param edgeFactory edgeFactory
279    * @param edgesToGroup input edgesToGroup
280    * @param vertexToRepresentativeMap dataset containing tuples of vertex id and super vertex id
281    * @return super edges
282    */
283   protected DataSet<E> buildSuperEdges(
284     EdgeFactory<E> edgeFactory,
285     DataSet<E> edgesToGroup,
286     DataSet<VertexWithSuperVertex> vertexToRepresentativeMap) {
287 
288     DataSet<EdgeGroupItem> edges = edgesToGroup
289       // build edge group items
290       .flatMap(new BuildEdgeGroupItem<>(useEdgeLabels(), getEdgeLabelGroups()))
291       // join edges with vertex-group-map on source-id == vertex-id
292       .join(vertexToRepresentativeMap)
293       .where(0).equalTo(0)
294       .with(new UpdateEdgeGroupItem(0))
295       .withForwardedFieldsFirst("f1;f2;f3;f4")
296       .withForwardedFieldsSecond("f1->f0")
297       // join result with vertex-group-map on target-id == vertex-id
298       .join(vertexToRepresentativeMap)
299       .where(1).equalTo(0)
300       .with(new UpdateEdgeGroupItem(1))
301       .withForwardedFieldsFirst("f0;f2;f3;f4")
302       .withForwardedFieldsSecond("f1->f1");
303 
304     // group + combine
305     DataSet<EdgeGroupItem> combinedEdges = groupEdges(edges)
306       .combineGroup(new CombineEdgeGroupItems(useEdgeLabels()));
307 
308     // group + reduce + build final edges
309     return groupEdges(combinedEdges)
310       .reduceGroup(new ReduceEdgeGroupItems<>(
311         useEdgeLabels(),
312         edgeFactory));
313   }

323   /**

326    * @param graph to filter
327    * @return subgraph
328    */
329   LG getSubgraphOfRetainedVertices(LG graph) {

334   /**
335    * Removes all edges of {@code edgesToSubtract} from {@code edges}.
336    *
337    * @param edges           set of edges
338    * @param edgesToSubtract set of edges to be removed
339    * @return subtracted set of edges
340    */
341   DataSet<E> subtractEdges(DataSet<E> edges, DataSet<E> edgesToSubtract) {
342     return edges
343       .leftOuterJoin(edgesToSubtract)
344       .where("sourceId", "targetId")
345       .equalTo("sourceId", "targetId")
346       .with(new LeftWhenRightIsNull<>());
347   }

349   /**
350    * Is used when {@link Grouping#retainVerticesWithoutGroup} is set.
351    * To add support for grouped edges between retained vertices and supervertices,
352    * retained vertices are singleton groups and are their group representatives themselves.
353    *
354    * @param vertexToRepresentativeMap vertex representative map to add to
355    * @param retainedVertices retained vertices
356    * @return updated vertexToRepresentativeMap
357    */
358   DataSet<VertexWithSuperVertex> updateVertexRepresentatives(
359     DataSet<VertexWithSuperVertex> vertexToRepresentativeMap, DataSet<V> retainedVertices) {
360     return vertexToRepresentativeMap
361       .union(retainedVertices.map(new SetVertexAsSuperVertex<>()));
362   }

730     /**
731      * Creates a new grouping operator instance based on the configured
732      * parameters.
733      *
734      * @param <G> The graph head type.
735      * @param <V> The vertex type.
736      * @param <E> The edge type.
737      * @param <LG> The type of the graph.
738      * @param <GC> The type of the graph collection.
739      * @return grouping operator instance
740      */
741     public <
742       G extends GraphHead,
743       V extends Vertex,
744       E extends Edge,
745       LG extends BaseGraph<G, V, E, LG, GC>,
746       GC extends BaseGraphCollection<G, V, E, LG, GC>> UnaryBaseGraphToBaseGraphOperator<LG> build() {
747 
748       if (strategy == null) {
749         throw new IllegalStateException("A GroupingStrategy has to be set.");
750       }
751 
752       // adding the global aggregators to the associated label groups
753       if (strategy != GroupingStrategy.GROUP_WITH_KEYFUNCTIONS) {
754         // global aggregate functions can be handled separately for KeyedGrouping
755         for (LabelGroup vertexLabelGroup : vertexLabelGroups) {
756           for (AggregateFunction aggregateFunction : globalVertexAggregateFunctions) {
757             vertexLabelGroup.addAggregateFunction(aggregateFunction);
758           }
759         }
760 
761         for (LabelGroup edgeLabelGroup : edgeLabelGroups) {
762           for (AggregateFunction aggregateFunction : globalEdgeAggregateFunctions) {
763             edgeLabelGroup.addAggregateFunction(aggregateFunction);
764           }
765         }
766       }
767 
768       UnaryBaseGraphToBaseGraphOperator<LG> groupingOperator;
769 
770       switch (strategy) {
771       case GROUP_REDUCE:
772         groupingOperator = new GroupingGroupReduce<>(
773           useVertexLabel, useEdgeLabel, vertexLabelGroups, edgeLabelGroups,
774           retainVerticesWithoutGroup);
775         break;
776       case GROUP_COMBINE:
777         groupingOperator = new GroupingGroupCombine<>(
778           useVertexLabel, useEdgeLabel, vertexLabelGroups, edgeLabelGroups,
779           retainVerticesWithoutGroup);
780         break;
781       case GROUP_WITH_KEYFUNCTIONS:
782         if (retainVerticesWithoutGroup) {
783           throw new UnsupportedOperationException("Retaining vertices without group is not yet supported" +
784             " with this strategy.");
785         }
786         groupingOperator = KeyedGroupingUtils.createInstance(
787           useVertexLabel, useEdgeLabel, vertexLabelGroups, edgeLabelGroups,
788           globalVertexAggregateFunctions, globalEdgeAggregateFunctions);
789         break;
790       default:
791         throw new IllegalArgumentException("Unsupported strategy: " + strategy);
792       }
793 
794       return groupingOperator;
795     }
