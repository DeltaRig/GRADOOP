62  /**
63   * A greedy query planner that builds a query plan by iteratively picking the cheapest partial query
64   * plan and extending it.
65   *
66   * @param <G> The graph head type.
67   * @param <V> The vertex type.
68   * @param <E> The edge type.
69   * @param <LG> The graph type.
70   * @param <GC> The graph collection type.
71   */
72  public class GreedyPlanner<
73    G extends GraphHead,
74    V extends org.gradoop.common.model.api.entities.Vertex,
75    E extends org.gradoop.common.model.api.entities.Edge,
76    LG extends BaseGraph<G, V, E, LG, GC>,
77    GC extends BaseGraphCollection<G, V, E, LG, GC>> {

100   /**
101    * Creates a new greedy planner.
102    *
103    * @param graph search graph
104    * @param queryHandler query handler
105    * @param graphStatistics search graph statistics
106    * @param vertexStrategy morphism type for vertex mappings
107    * @param edgeStrategy morphism type for edge mappings
108    */
109   public GreedyPlanner(LG graph, QueryHandler queryHandler,
110     GraphStatistics graphStatistics, MatchStrategy vertexStrategy, MatchStrategy edgeStrategy) {
111     this.graph = graph;
112     this.queryHandler = queryHandler;
113     this.graphStatistics = graphStatistics;
114     this.vertexStrategy = vertexStrategy;
115     this.edgeStrategy = edgeStrategy;
116   }

118   /**
119    * Computes the {@link PlanTableEntry} that wraps the {@link QueryPlan} with the minimum costs
120    * according to the greedy optimization algorithm.
121    *
122    * @return entry with minimum execution costs
123    */
124   public PlanTableEntry plan() {
125     PlanTable planTable = initPlanTable();
126 
127     while (planTable.size() > 1) {
128       PlanTable newPlans = evaluateJoins(planTable);
129 
130       if (newPlans.size() == 0) {
131         // No new plans where generated by joining but there are still multiple disconnected
132         // query graph components. In this case we need to do a cartesian product.
133         newPlans = evaluateCartesianProducts(planTable);
134       }
135       newPlans = evaluateFilter(newPlans);
136       newPlans = evaluateProjection(newPlans);
137 
138       // get plan with minimum costs and remove all plans covered by this plan
139       PlanTableEntry bestEntry = newPlans.min();
140       planTable.removeCoveredBy(bestEntry);
141       planTable.add(bestEntry);
142     }
143 
144     return planTable.get(0);
145   }

167   /**
168    * Creates an initial {@link PlanTableEntry} for each vertex in the query graph and adds it to the
169    * specified {@link PlanTable}. The entry wraps a query plan that filters vertices based on their
170    * predicates and projects properties that are required for further query planning.
171    *
172    * @param planTable plan table
173    */
174   private void createVertexPlans(PlanTable planTable) {
175     for (Vertex vertex : queryHandler.getVertices()) {
176       String vertexVariable = vertex.getVariable();
177       CNF allPredicates = queryHandler.getPredicates();
178       // TODO: this might be moved to the FilterAndProject node in issue #510
179       CNF vertexPredicates = allPredicates.removeSubCNF(vertexVariable);
180       Set<String> projectionKeys = allPredicates.getPropertyKeys(vertexVariable);
181 
182       DataSet<V> vertices =
183         vertex.getLabel().equals(GradoopConstants.DEFAULT_VERTEX_LABEL) ?
184           graph.getVertices() : graph.getVerticesByLabel(vertex.getLabel());
185 
186       FilterAndProjectVerticesNode<V> node = new FilterAndProjectVerticesNode<>(vertices,
187         vertex.getVariable(), vertexPredicates, projectionKeys);
188 
189       planTable.add(new PlanTableEntry(VERTEX, Sets.newHashSet(vertexVariable), allPredicates,
190         new QueryPlanEstimator(new QueryPlan(node), queryHandler, graphStatistics)));
191     }
192   }

194   /**
195    * Creates an initial {@link PlanTableEntry} for each edge in the query graph and adds it to the
196    * specified {@link PlanTable}. The entry wraps a {@link QueryPlan} that filters edges based on
197    * their predicates and projects properties that are required for further query planning.
198    *
199    * @param planTable plan table
200    */
201   private void createEdgePlans(PlanTable planTable) {
202     for (Edge edge : queryHandler.getEdges()) {
203       String edgeVariable = edge.getVariable();
204       String sourceVariable = queryHandler.getVertexById(edge.getSourceVertexId()).getVariable();
205       String targetVariable = queryHandler.getVertexById(edge.getTargetVertexId()).getVariable();
206 
207       CNF allPredicates = queryHandler.getPredicates();
208       // TODO: this might be moved the the FilterAndProject node in issue #510
209       CNF edgePredicates = allPredicates.removeSubCNF(edgeVariable);
210       Set<String> projectionKeys = allPredicates.getPropertyKeys(edgeVariable);
211 
212       boolean isPath = edge.getUpperBound() != 1;
213 
214       DataSet<E> edges =
215         edge.getLabel().equals(GradoopConstants.DEFAULT_EDGE_LABEL) ?
216           graph.getEdges() : graph.getEdgesByLabel(edge.getLabel());
217 
218       FilterAndProjectEdgesNode<E> node = new FilterAndProjectEdgesNode<>(edges,
219         sourceVariable, edgeVariable, targetVariable, edgePredicates, projectionKeys, isPath);
220 
221       PlanTableEntry.Type type = edge.hasVariableLength() ? PATH : EDGE;
222 
223       planTable.add(new PlanTableEntry(type, Sets.newHashSet(edgeVariable), allPredicates,
224         new QueryPlanEstimator(new QueryPlan(node), queryHandler, graphStatistics)));
225     }
226   }

232   /**
233    * Evaluates which entries in the specified plan table can be joined. The joined entries
234    * are added to a new table which is returned.
235    *
236    * @param currentTable query plan table
237    * @return new table containing solely joined plans from the input table
238    */
239   private PlanTable evaluateJoins(PlanTable currentTable) {
240     PlanTable newTable = new PlanTable();
241 
242     for (int i = 0; i < currentTable.size(); i++) {
243       PlanTableEntry leftEntry = currentTable.get(i);
244       if (mayExtend(leftEntry)) {
245         for (int j = 0; j < currentTable.size(); j++) {
246           PlanTableEntry rightEntry = currentTable.get(j);
247           if (i != j) {
248             List<String> joinVariables = getOverlap(leftEntry, rightEntry);
249             if (joinVariables.size() > 0) {
250               if (rightEntry.getType() == PATH && joinVariables.size() == 2) {
251                 // evaluate join with variable length path on source and target vertex
252                 newTable.add(joinEntries(leftEntry, rightEntry, joinVariables.subList(0, 1)));
253                 newTable.add(joinEntries(leftEntry, rightEntry, joinVariables.subList(1, 2)));
254               } else {
255                 // regular join or join with variable length path on source or target vertex
256                 newTable.add(joinEntries(leftEntry, rightEntry, joinVariables));
257               }
258             }
259           }
260         }
261       }
262     }
263     return newTable;
264   }

266   /**
267    * Checks if the given entry may be extended. This is only the case for entries that represents
268    * either a vertex or a partial match graph.
269    *
270    * @param entry plan table entry
271    * @return true, iff the specified entry may be extended
272    */
273   private boolean mayExtend(PlanTableEntry entry) {
274     return entry.getType() == VERTEX || entry.getType() == GRAPH;
275   }

277   /**
278    * Computes the overlapping query variables of the specified entries.
279    *
280    * @param firstEntry first entry
281    * @param secondEntry second entry
282    * @return variables that are available in both input entries
283    */
284   private List<String> getOverlap(PlanTableEntry firstEntry, PlanTableEntry secondEntry) {
285     Set<String> overlap = firstEntry.getAllVariables();
286     overlap.retainAll(secondEntry.getAllVariables());
287     return new ArrayList<>(overlap);
288   }

290   /**
291    * Joins the query plans represented by the specified plan table entries.
292    *
293    * The method considers if the right entry is a variable length path and in that case
294    * creates an {@link ExpandEmbeddingsNode}. In any other case, a regular
295    * {@link JoinEmbeddingsNode} is used to join the query plans.
296    *
297    * @param leftEntry left entry
298    * @param rightEntry right entry
299    * @param joinVariables join variables
300    * @return an entry that represents the join of both input entries
301    */
302   private PlanTableEntry joinEntries(PlanTableEntry leftEntry, PlanTableEntry rightEntry,
303     List<String> joinVariables) {
304 
305     PlanNode node;
306     if (rightEntry.getType() == PATH) {
307       assert joinVariables.size() == 1;
308       node = createExpandNode(leftEntry, rightEntry, joinVariables.get(0));
309     } else {
310       node = new JoinEmbeddingsNode(leftEntry.getQueryPlan().getRoot(),
311         rightEntry.getQueryPlan().getRoot(), joinVariables, vertexStrategy, edgeStrategy);
312     }
313 
314     // update processed variables
315     HashSet<String> processedVariables = Sets.newHashSet(leftEntry.getProcessedVariables());
316     processedVariables.addAll(rightEntry.getProcessedVariables());
317     // create resulting predicates
318     // TODO: this might be moved to the join/expand node in issue #510
319     CNF predicates = mergePredicates(leftEntry, rightEntry);
320 
321     return new PlanTableEntry(GRAPH, processedVariables, predicates,
322       new QueryPlanEstimator(new QueryPlan(node), queryHandler, graphStatistics));
323   }

325   /**
326    * Creates an {@link ExpandEmbeddingsNode} from the specified arguments.
327    *
328    * @param leftEntry left entry
329    * @param rightEntry right entry
330    * @param startVariable vertex variable to expand from
331    *
332    * @return new expand node
333    */
334   private ExpandEmbeddingsNode createExpandNode(PlanTableEntry leftEntry, PlanTableEntry rightEntry,
335     String startVariable) {
336 
337     String pathVariable = rightEntry.getQueryPlan().getRoot()
338       .getEmbeddingMetaData().getEdgeVariables().get(0);
339 
340     Edge queryEdge = queryHandler.getEdgeByVariable(pathVariable);
341     Vertex sourceVertex = queryHandler.getVertexById(queryEdge.getSourceVertexId());
342     Vertex targetVertex = queryHandler.getVertexById(queryEdge.getTargetVertexId());
343 
344     int lowerBound = queryEdge.getLowerBound();
345     int upperBound = queryEdge.getUpperBound();
346     ExpandDirection direction = sourceVertex.getVariable().equals(startVariable) ?
347       ExpandDirection.OUT : ExpandDirection.IN;
348     String endVariable = direction == ExpandDirection.OUT ?
349       targetVertex.getVariable() : sourceVertex.getVariable();
350 
351     return new ExpandEmbeddingsNode(leftEntry.getQueryPlan().getRoot(),
352       rightEntry.getQueryPlan().getRoot(),
353       startVariable, pathVariable, endVariable, lowerBound, upperBound, direction,
354       vertexStrategy, edgeStrategy);
355   }

361   /**
362    * The method checks if a filter can be applied on any of the entries in the specified table. If
363    * this is the case, a {@link FilterEmbeddingsNode} is added to the query plan represented by the
364    * affected entries.
365    *
366    * @param currentTable query plan table
367    * @return input table with possibly updated entries
368    */
369   private PlanTable evaluateFilter(PlanTable currentTable) {
370     PlanTable newTable = new PlanTable();
371 
372     for (PlanTableEntry entry : currentTable) {
373       Set<String> variables = Sets.newHashSet(entry.getProcessedVariables());
374       CNF predicates = entry.getPredicates();
375       CNF subCNF = predicates.removeSubCNF(variables);
376       if (subCNF.size() > 0) {
377         FilterEmbeddingsNode node = new FilterEmbeddingsNode(entry.getQueryPlan().getRoot(),
378           subCNF);
379         newTable.add(new PlanTableEntry(GRAPH, Sets.newHashSet(entry.getProcessedVariables()),
380           predicates, new QueryPlanEstimator(new QueryPlan(node), queryHandler, graphStatistics)));
381       } else {
382         newTable.add(entry);
383       }
384     }
385 
386     return newTable;
387   }

393   /**
394    * The method checks if a filter can be applied on any of the entries in the specified table. If
395    * this is the case, a {@link ProjectEmbeddingsNode} is added to the query plan represented by the
396    * affected entries.
397    *
398    * @param currentTable query plan table
399    * @return input table with possibly updated entries
400    */
401   private PlanTable evaluateProjection(PlanTable currentTable) {
402     PlanTable newTable = new PlanTable();
403 
404     for (PlanTableEntry entry : currentTable) {
405       Set<Pair<String, String>> propertyPairs = entry.getPropertyPairs();
406       Set<Pair<String, String>> projectionPairs = entry.getProjectionPairs();
407 
408       Set<Pair<String, String>> updatedPropertyPairs = propertyPairs.stream()
409         .filter(projectionPairs::contains)
410         .collect(Collectors.toSet());
411 
412       if (updatedPropertyPairs.size() < propertyPairs.size()) {
413         ProjectEmbeddingsNode node = new ProjectEmbeddingsNode(entry.getQueryPlan().getRoot(),
414           new ArrayList<>(updatedPropertyPairs));
415         newTable.add(new PlanTableEntry(GRAPH,
416           Sets.newHashSet(entry.getProcessedVariables()), entry.getPredicates(),
417           new QueryPlanEstimator(new QueryPlan(node), queryHandler, graphStatistics)));
418       } else {
419         newTable.add(entry);
420       }
421     }
422     return newTable;
423   }

429   /**
430    * Evaluates which entries in the specified plan table can be joined. The joined entries
431    * are added to a new table which is returned.
432    *
433    * @param currentTable query plan table
434    * @return new table containing solely joined plans from the input table
435    */
436   private PlanTable evaluateCartesianProducts(PlanTable currentTable) {
437     PlanTable newTable = new PlanTable();
438     for (int i = 0; i < currentTable.size(); i++) {
439       PlanTableEntry leftEntry = currentTable.get(i);
440       for (int j = i + 1; j < currentTable.size(); j++) {
441         PlanTableEntry rightEntry = currentTable.get(j);
442         CNF joinPredicate = getJoinPredicate(leftEntry, rightEntry);
443         if (joinPredicate.size() > 0) {
444           newTable.add(createValueJoinEntry(leftEntry, rightEntry, joinPredicate));
445         } else {
446           // regular join or join with variable length path on source or target vertex
447           newTable.add(createCartesianProductEntry(leftEntry, rightEntry));
448         }
449       }
450     }
451     return newTable;
452   }

455   /**
456    * Computes the overlapping query variables of the specified entries.
457    *
458    * @param leftEntry first entry
459    * @param rightEntry second entry
460    * @return variables that are available in both input entries
461    */
462   private CNF getJoinPredicate(PlanTableEntry leftEntry, PlanTableEntry rightEntry) {
463     Set<String> allVariables = leftEntry.getAllVariables();
464     allVariables.addAll(rightEntry.getAllVariables());
465 
466     CNF leftPredicates = new CNF(leftEntry.getPredicates());
467     CNF rightPredicates = new CNF(rightEntry.getPredicates());
468     leftPredicates.removeSubCNF(rightEntry.getProcessedVariables());
469     rightPredicates.removeSubCNF(leftEntry.getProcessedVariables());
470     CNF predicates = leftPredicates.and(rightPredicates).getSubCNF(allVariables);
471 
472     return new CNF(
473       predicates.getPredicates()
474         .stream()
475         .filter(p ->
476           p.size() == 1 && p.getPredicates().get(0).getComparator().equals(Comparator.EQ)
477         ).collect(Collectors.toList())
478     );
479   }

510   /**
511    * Creates an {@link ValueJoinNode} from the specified arguments.
512    *
513    * @param leftEntry left entry
514    * @param rightEntry right entry
515    * @param joinPredicate join predicate
516    *
517    * @return new value join node
518    */
519   private PlanTableEntry createValueJoinEntry(PlanTableEntry leftEntry,
520     PlanTableEntry rightEntry, CNF joinPredicate) {
521 
522     List<Pair<String, String>> leftProperties = new ArrayList<>();
523     List<Pair<String, String>> rightProperties = new ArrayList<>();
524 
525     for (CNFElement e : joinPredicate.getPredicates()) {
526       ComparisonExpression comparison = e.getPredicates().get(0);
527 
528       Pair<String, String> joinProperty = extractJoinProperty(comparison.getLhs());
529       if (leftEntry.getAllVariables().contains(joinProperty.getKey())) {
530         leftProperties.add(joinProperty);
531       } else {
532         rightProperties.add(joinProperty);
533       }
534 
535       joinProperty = extractJoinProperty(comparison.getRhs());
536       if (leftEntry.getAllVariables().contains(joinProperty.getKey())) {
537         leftProperties.add(joinProperty);
538       } else {
539         rightProperties.add(joinProperty);
540       }
541     }
542 
543     ValueJoinNode node = new ValueJoinNode(
544       leftEntry.getQueryPlan().getRoot(),
545       rightEntry.getQueryPlan().getRoot(),
546       leftProperties, rightProperties,
547       vertexStrategy, edgeStrategy
548     );
549 
550     Set<String> processedVariables = leftEntry.getProcessedVariables();
551     processedVariables.addAll(rightEntry.getProcessedVariables());
552 
553     CNF predicates = mergePredicates(leftEntry, rightEntry);
554 
555     return new PlanTableEntry(
556       GRAPH,
557       processedVariables,
558       predicates,
559       new QueryPlanEstimator(new QueryPlan(node), queryHandler, graphStatistics)
560     );
561   }

563   /**
564    * Turns a QueryComparable into a {@code Pair<Variable, PropertyKey>}
565    * @param comparable query comparable
566    * @return join property
567    */
568   private Pair<String, String> extractJoinProperty(QueryComparable comparable) {
569     if (comparable instanceof PropertySelectorComparable) {
570       PropertySelectorComparable propertySelector = (PropertySelectorComparable) comparable;
571       return Pair.of(propertySelector.getVariable(), propertySelector.getPropertyKey());
572     } else {
573       //TODO #580 Include ElementSelector -> ID needs to be projected as property
574       throw new RuntimeException("Comparable " + comparable + "cant be used for ValueJoin");
575     }
576   }

578   /**
579    * Creates a new predicate that includes only elements that exist in both input predicates
580    *
581    * @param leftEntry left side plant table entry
582    * @param rightEntry right side plan table entry
583    * @return Merged predicates
584    */
585   private CNF mergePredicates(PlanTableEntry leftEntry, PlanTableEntry rightEntry) {
586     CNF leftPredicates = new CNF(leftEntry.getPredicates());
587     CNF rightPredicates = new CNF(rightEntry.getPredicates());
588     leftPredicates.removeSubCNF(rightEntry.getProcessedVariables());
589     rightPredicates.removeSubCNF(leftEntry.getProcessedVariables());
590     return leftPredicates.and(rightPredicates);
591   }
