207   @Override
208   public LogicalGraph execute(LogicalGraph g) {
209 
210     g = createInitialLayout(g);
211 
212     DataSet<EPGMVertex> gradoopVertices = g.getVertices();
213     DataSet<EPGMEdge> gradoopEdges = g.getEdges();
214 
215     DataSet<LVertex> vertices = gradoopVertices.map((v) -> new LVertex(v));
216     DataSet<LEdge> edges = gradoopEdges.map((e) -> new LEdge(e));
217 
218     IterativeDataSet<LVertex> loop = vertices.iterate(iterations);
219     LGraph graph = new LGraph(loop, edges);
220     layout(graph);
221     vertices = loop.closeWith(graph.getVertices());
222 
223     gradoopVertices = vertices
224       .join(gradoopVertices)
225       .where(LVertex.ID_POSITION).equalTo("id")
226       .with(new LVertexEPGMVertexJoinFunction());
227 
228     return g.getFactory().fromDataSets(gradoopVertices, gradoopEdges);
229   }

240       return g;
241     }
242     return new RandomLayouter(getWidth() / 10, getWidth() - (getWidth() / 10), getHeight() / 10,
243       getHeight() - (getHeight() / 10)).execute(g);

246   /**
247    * Perform the actual layouting (calculate and apply forces)
248    *
249    * @param g The Graph to layout. It is modified by this method.
250    */
251   protected void layout(LGraph g) {
252     DataSet<Force> repulsions = repulsionForces(g.getVertices());
253     DataSet<Force> attractions = attractionForces(g.getVertices(), g.getEdges());
254 
255     DataSet<Force> forces = repulsions
256       .union(attractions)
257       .groupBy(Force.ID_POSITION)
258       .reduce((first, second) -> {
259         first.setValue(first.getValue().add(second.getValue()));
260         return first;
261       });
262 
263     g.setVertices(applyForces(g.getVertices(), forces, iterations));
264   }

266   /**
267    * Applies the given forces to the given vertices.
268    *
269    * @param vertices   Vertices to move
270    * @param forces     Forces to apply. At most one per vertex. The id indicates which vertex
271    *                   the force should be applied to
272    * @param iterations Number of iterations that will be performed (NOT the number of the
273    *                   current iteration). It is used to compute the simulated annealing shedule.
274    * @return The input vertices with x and y coordinated changed according to the given force and
275    * current iteration number.
276    */
277   protected DataSet<LVertex> applyForces(DataSet<LVertex> vertices, DataSet<Force> forces,
278     int iterations) {
279     FRForceApplicator applicator =
280       new FRForceApplicator(getWidth(), getHeight(), getK(), iterations);
281     applicator.setPreviousIterations(startAtIteration);
282     return vertices.join(forces).where(LVertex.ID_POSITION).equalTo(Force.ID_POSITION).with(applicator);
283   }

286   /**
287    * Calculates the repulsive forces between the given vertices.
288    *
289    * @param vertices A dataset of vertices
290    * @return Dataset of applied forces. May (and will) contain multiple forces for each vertex.
291    */
292   protected DataSet<Force> repulsionForces(DataSet<LVertex> vertices) {
293     vertices = vertices.map(new FRCellIdMapper(getMaxRepulsionDistance()));
294 
295     KeySelector<LVertex, Integer> selfselector = new FRCellIdSelector(FRCellIdSelector.NeighborType.SELF);
296     FRRepulsionFunction repulsionFunction = new FRRepulsionFunction(getK(), getMaxRepulsionDistance());
297 
298     DataSet<Force> self = vertices
299       .join(vertices)
300       .where(new FRCellIdSelector(FRCellIdSelector.NeighborType.SELF)).equalTo(selfselector)
301       .with((JoinFunction<LVertex, LVertex, Force>) repulsionFunction);
302 
303     DataSet<Force> up = vertices
304       .join(vertices)
305       .where(new FRCellIdSelector(FRCellIdSelector.NeighborType.UP)).equalTo(selfselector)
306       .with((FlatJoinFunction<LVertex, LVertex, Force>) repulsionFunction);
307 
308     DataSet<Force> left = vertices
309       .join(vertices)
310       .where(new FRCellIdSelector(FRCellIdSelector.NeighborType.LEFT)).equalTo(selfselector)
311       .with((FlatJoinFunction<LVertex, LVertex, Force>) repulsionFunction);
312 
313     DataSet<Force> uright = vertices
314       .join(vertices)
315       .where(new FRCellIdSelector(FRCellIdSelector.NeighborType.UPRIGHT)).equalTo(selfselector)
316       .with((FlatJoinFunction<LVertex, LVertex, Force>) repulsionFunction);
317 
318     DataSet<Force> uleft = vertices
319       .join(vertices)
320       .where(new FRCellIdSelector(FRCellIdSelector.NeighborType.UPLEFT)).equalTo(selfselector)
321       .with((FlatJoinFunction<LVertex, LVertex, Force>) repulsionFunction);
322 
323 
324     return self.union(up).union(left).union(uright).union(uleft);
325   }

327   /**
328    * Compute the attractive-forces between all vertices connected by edges.
329    *
330    * @param vertices The vertices
331    * @param edges    The edges between vertices
332    * @return A mapping from VertexId to x and y forces
333    */
334   protected DataSet<Force> attractionForces(DataSet<LVertex> vertices, DataSet<LEdge> edges) {
335     return edges.join(vertices).where(LEdge.SOURCE_ID_POSITION).equalTo(LVertex.ID_POSITION).join(vertices)
336       .where("f0." + LEdge.TARGET_ID_POSITION).equalTo(LVertex.ID_POSITION).with(
337         (first, second) -> new Tuple3<LVertex, LVertex, Integer>(first.f1, second,
338           first.f0.getCount())).returns(new TypeHint<Tuple3<LVertex, LVertex, Integer>>() {
339             }).flatMap(new FRAttractionFunction(getK()));
340   }
