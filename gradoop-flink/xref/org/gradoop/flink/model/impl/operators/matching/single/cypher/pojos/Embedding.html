35  /**
36   * This class represents an Embedding, an ordered List of Embedding Entries. Every entry is
37   * either a reference to a single Edge or Vertex, or a
38   * path (Edge, Vertex, Edge, Vertex, ..., Edge).
39   * The reference is stored via the elements ID. Additionally the embedding ca store an ordered
40   * list of PropertyValues.
41   */
42  public class Embedding implements Value, CopyableValue<Embedding> {

111   /**
112    * Appends all ids to the embeddings
113    *
114    * @param ids list of ids
115    */
116   public void addAll(GradoopId... ids) {
117     byte[] newIds = new byte[idData.length + ids.length * ID_ENTRY_SIZE];
118 
119     System.arraycopy(idData, 0, newIds, 0, idData.length);
120 
121     int offset = idData.length;
122     for (GradoopId id : ids) {
123       newIds[offset] = ID_ENTRY_FLAG;
124       System.arraycopy(id.toByteArray(), 0, newIds, offset + 1, GradoopId.ID_SIZE);
125       offset += ID_ENTRY_SIZE;
126     }
127 
128     idData = newIds;
129   }

141    * @param column position the entry is stored at

144   public byte[] getRawId(int column) {
145     byte[] rawEntry = getRawIdEntry(column);
146 
147     if (rawEntry[0] == ID_LIST_FLAG) {

174   /**
175    * Returns the IDs entries stored at the specified positions
176    * @param columns positions of the entries
177    * @return IDs of the entries stored at the specified positions
178    */
179   public List<GradoopId> getIdsAsList(List<Integer> columns) {
180     int offset;
181     List<GradoopId> ids = new ArrayList<>();
182 
183     for (Integer column : columns) {
184       offset = getIdOffset(column);
185       if (idData[offset] == ID_LIST_FLAG) {
186         ids.addAll(getIdList(column));
187       } else {
188         ids.add(getId(column));
189       }
190     }
191 
192     return ids;
193   }

195   /**
196    * Adds an entry to the embedding.
197    * This can either be an ID representing referencing a Vertex/Edge or a pointer to a path entry
198    * @param id the id that will be added to the embedding
199    * @param isIdList indicates if the id represents a GraphElement or points to a path entry
200    */
201   private void add(GradoopId id, boolean isIdList) {
202     byte[] newIds = new byte[idData.length + 1 + GradoopId.ID_SIZE];
203     System.arraycopy(idData, 0, newIds, 0, idData.length);
204     newIds[idData.length] = isIdList ? ID_LIST_FLAG : ID_ENTRY_FLAG;
205     System.arraycopy(id.toByteArray(), 0, newIds, idData.length + 1, GradoopId.ID_SIZE);
206 
207     idData = newIds;
208   }

235   /**
236    * Adds the list of properties to the embedding
237    *
238    * @param properties new properties
239    */
240   public void addPropertyValues(PropertyValue... properties) {
241     int newPropertiesSize = propertyData.length;
242     for (PropertyValue property : properties) {
243       newPropertiesSize += property.getByteSize() + Integer.BYTES;
244     }
245 
246     byte[] newPropertyData = new byte[newPropertiesSize];
247 
248     System.arraycopy(propertyData, 0, newPropertyData, 0, propertyData.length);
249 
250     int offset = propertyData.length;
251     for (PropertyValue property : properties) {
252       writeProperty(property, newPropertyData, offset);
253       offset += property.getByteSize() + Integer.BYTES;
254     }
255 
256     this.propertyData = newPropertyData;
257   }

265   public PropertyValue getProperty(int column) {
266     int offset = getPropertyOffset(column);

268     int entryLength = Ints.fromByteArray(

294   /**
295    * Returns a list of all property values stored in the embedding
296    * @return List of all property values stored in the embedding
297    */
298   public List<PropertyValue> getProperties() {
299     List<PropertyValue> properties = new ArrayList<>();
300     int offset = 0;
301     int entrySize;
302     while (offset < propertyData.length) {
303       entrySize =
304         Ints.fromByteArray(ArrayUtils.subarray(propertyData, offset, offset + Integer.BYTES));
305 
306       offset += Integer.BYTES;
307 
308       properties.add(PropertyValue.fromRawBytes(
309         ArrayUtils.subarray(propertyData, offset, offset + entrySize)
310       ));
311 
312       offset += entrySize;
313     }
314 
315     return properties;
316   }

318   /**
319    * Returns the offset of the property in the propertyData array
320    * @param column the index of the property
321    * @return Offset of the property in the propertyData array
322    */
323   private int getPropertyOffset(int column) {
324     int i = 0;
325     int offset = 0;
326     int entryLength;
327 
328     while (i < column && offset < propertyData.length) {
329       entryLength =
330         Ints.fromByteArray(ArrayUtils.subarray(propertyData, offset, offset + Integer.BYTES));
331 
332       offset += entryLength + Integer.BYTES;
333       i++;
334     }
335 
336     if (offset >= propertyData.length) {
337       throw new IndexOutOfBoundsException("Cant find Property. " + (i - 1) + " < " + column);
338     }
339 
340     return offset;
341   }

347   /**
348    * Adds an IdList a.k.a. PathEntry to the embedding
349    * @param ids the path that will be added to the embedding
350    */
351   public void add(GradoopId... ids) {
352     GradoopId pointer = GradoopId.get();
353     add(pointer, true);
354 
355     byte[] newIdLists = new byte[idListData.length +
356       GradoopId.ID_SIZE + Integer.BYTES +
357       ids.length * GradoopId.ID_SIZE];
358 
359     System.arraycopy(idListData, 0, newIdLists, 0, idListData.length);
360     writeId(pointer, newIdLists, idListData.length);
361     writeInt(ids.length, newIdLists, idListData.length + GradoopId.ID_SIZE);
362 
363     int offset = idListData.length + GradoopId.ID_SIZE + Integer.BYTES;
364     for (GradoopId id: ids) {
365       writeId(id, newIdLists, offset);
366       offset += GradoopId.ID_SIZE;
367     }
368 
369     idListData = newIdLists;
370   }

372   /**
373    * Returns the ID-List stored at the specified column
374    * @param column the entries index
375    * @return ID-List stored at the specified column
376    */
377   public List<GradoopId> getIdList(int column) {
378     int offset = getIdListOffset(column);
379 
380     int listSize =
381       Ints.fromByteArray(ArrayUtils.subarray(idListData, offset, offset + Integer.BYTES));
382 
383     offset += Integer.BYTES;
384 
385     List<GradoopId> idList = new ArrayList<>(listSize);
386 
387     for (int i = 0; i < listSize; i++) {
388       idList.add(GradoopId.fromByteArray(
389         ArrayUtils.subarray(idListData, offset, offset + GradoopId.ID_SIZE)
390       ));
391       offset += GradoopId.ID_SIZE;
392     }
393 
394     return idList;
395   }

397   /**
398    * Returns the offset of the ID-List in the idListData array
399    * @param column the index of the ID-List entry
400    * @return Offset of the ID-List in the idListData array
401    */
402   private int getIdListOffset(int column) {
403     int pointerOffset = getIdOffset(column);
404 
405     if (idData[pointerOffset++] != ID_LIST_FLAG) {
406       throw new UnsupportedOperationException("Entry is not an IDList");
407     }
408 
409     byte[] pointer = ArrayUtils.subarray(idData, pointerOffset, pointerOffset + GradoopId.ID_SIZE);
410 
411     int offset = 0;
412     byte[] comparePointer;
413     int listSize;
414     boolean found = false;
415 
416     while (!found && offset < idListData.length) {
417       comparePointer = ArrayUtils.subarray(idListData, offset, offset + GradoopId.ID_SIZE);
418       offset += GradoopId.ID_SIZE;
419       found = Arrays.equals(pointer, comparePointer);
420 
421       if (!found) {
422         listSize =
423           Ints.fromByteArray(ArrayUtils.subarray(idListData, offset, offset + Integer.BYTES));
424         offset += GradoopId.ID_SIZE * listSize + Integer.BYTES;
425       }
426     }
427 
428     if (!found) {
429       throw new RuntimeException("Could not find IdList entry");
430     }
431 
432     return offset;
433   }

488   /**
489    * Projects the stored Properties. Only the white-listed properties will be kept
490    * @param propertyWhiteList list of property indices
491    * @return Embedding with the projected property list
492    */
493   public Embedding project(List<Integer> propertyWhiteList) {
494     byte[] newPropertyData = new byte[0];
495     for (int index : propertyWhiteList) {
496       newPropertyData = ArrayUtils.addAll(newPropertyData, getRawProperty(index));
497     }
498 
499     return new Embedding(idData, newPropertyData, idListData);
500   }

502   /**
503    * Reverses the order of the entries stored in the embedding.
504    * The order of the properties will stay untouched.
505    * @return  A new Embedding with reversed entry order
506    */
507   public Embedding reverse() {
508     byte[] newIdData = new byte[idData.length];
509 
510     for (int i = size() - 1; i >= 0; i--) {
511       System.arraycopy(
512         getRawIdEntry(i), 0,
513         newIdData,  (size() - 1 - i) * ID_ENTRY_SIZE,
514         ID_ENTRY_SIZE
515       );
516     }
517 
518     return new Embedding(newIdData, propertyData, idListData);
519   }

685   @Override
686   public String toString() {
687     List<List<GradoopId>> idCollection = new ArrayList<>();
688     for (int i = 0; i < size(); i++) {
689       idCollection.add(getIdAsList(i));
690     }
691 
692     String idString = idCollection
693       .stream()
694       .map(entry -> {
695         if (entry.size() == 1) {
696           return entry.get(0).toString();
697         } else {
698           return entry.stream().map(GradoopId::toString).collect(joining(", ", "[", "]"));
699         }
700       })
701       .collect(joining(", "));
702 
703     String propertyString = getProperties()
704       .stream()
705       .map(PropertyValue::toString)
706       .collect(joining(", "));
707 
708 
709     return "Embedding{ " +
710       "entries: {" + idString + "},  " +
711       "properties: {" + propertyString + "} " +
712       "}";
