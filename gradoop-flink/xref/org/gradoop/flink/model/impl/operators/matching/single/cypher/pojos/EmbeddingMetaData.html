32  /**
33   * This class stores meta data information about a data set of {@link Embedding} objects.
34   *
35   * An {@link Embedding} stores identifiers (single or path) and properties associated with query
36   * elements.
37   *
38   * The meta data contains a mapping between query variables and the column index storing the
39   * associated element/path identifier. Furthermore, the meta data object contains a mapping between
40   * property values associated to property keys at query elements.
41   */
42  public class EmbeddingMetaData implements Serializable {
43    /**
44     * Describes the type of an embedding entry
45     */
46    public enum EntryType {
47      /**
48       * Vertex
49       */
50      VERTEX,
51      /**
52       * Edge
53       */
54      EDGE,
55      /**
56       * Path
57       */
58      PATH
59    }
60  
61    /**
62     * Stores the mapping of query variables to embedding entries
63     */
64    private Map<Pair<String, EntryType>, Integer> entryMapping;
65    /**
66     * Stores where the corresponding PropertyValue of a Variable-PropertyKey-Pair is stored within
67     * the embedding
68     */
69    private Map<Pair<String, String>, Integer> propertyMapping;
70    /**
71     * Stores the direction in which paths are stored in the embedding
72     */
73    private Map<String, ExpandDirection> directionMapping;

76    /**
77     * Initialises an empty EmbeddingMetaData object
78     */
79    public EmbeddingMetaData() {
80      this(new HashMap<>(), new HashMap<>(), new HashMap<>());
81    }

83    /**
84     * Initializes a new EmbeddingMetaData object from the given mappings
85     *
86     * @param entryMapping maps variables to embedding entries
87     * @param propertyMapping maps variable-propertyKey pairs to embedding property data entries
88     * @param directionMapping maps (path) variables to their direction
89     */
90    public EmbeddingMetaData(Map<Pair<String, EntryType>, Integer> entryMapping,
91      Map<Pair<String, String>, Integer> propertyMapping,
92      Map<String, ExpandDirection> directionMapping) {
93      this.entryMapping = entryMapping;
94      this.propertyMapping = propertyMapping;
95      this.directionMapping = directionMapping;
96    }
97  
98    /**
99     * Initializes a new EmbeddingMetaData object using copies of the provided meta data.
100    *
101    * @param metaData meta data to be copied
102    */
103   public EmbeddingMetaData(EmbeddingMetaData metaData) {
104     this.entryMapping = new HashMap<>(metaData.getEntryCount());
105     this.propertyMapping = new HashMap<>(metaData.getPropertyCount());
106     this.directionMapping = new HashMap<>(metaData.getPathCount());
107 
108     metaData.getVariables().forEach(var -> {
109       this.entryMapping.put(
110         Pair.of(var, metaData.getEntryType(var)), metaData.getEntryColumn(var));
111       metaData.getPropertyKeys(var).forEach(key ->
112         this.propertyMapping.put(Pair.of(var, key), metaData.getPropertyColumn(var, key)));
113       if (metaData.getEntryType(var) == EntryType.PATH) {
114         this.directionMapping.put(var, metaData.getDirection(var));
115       }
116     });
117   }

158   /**
159    * Inserts or updates a column mapping entry

161    * @param variable referenced variable
162    * @param entryType entry type
163    * @param column corresponding embedding entry index
164    */
165   public void setEntryColumn(String variable, EntryType entryType, int column) {
166     entryMapping.put(Pair.of(variable, entryType), column);
167   }

169   /**
170    * Returns the position of the embedding entry corresponding to the given variable.
171    * The method checks if the variable is mapped to a vertex or an edge entry.
172    *
173    * @param variable variable name
174    * @return the position of the corresponding embedding entry
175    * @throws NoSuchElementException if there is no column mapped to the specified variable
176    */
177   public int getEntryColumn(String variable) {
178     return entryMapping.get(Pair.of(variable, getEntryType(variable)));
179   }
180 
181   /**
182    * Checks if the specified variable is mapped to a column in the embedding.
183    *
184    * @param variable query variable
185    * @return true, iff the variable is mapped to a column
186    */
187   public boolean containsEntryColumn(String variable) {
188     return Arrays.stream(EntryType.values())
189       .anyMatch(entryType -> entryMapping.containsKey(Pair.of(variable, entryType)));
190   }
191 
192   /**
193    * Returns the entry type of the given variable.
194    *
195    * @param variable query variable
196    * @return Entry type of the referred entry
197    * @throws NoSuchElementException if there is no column mapped to the specified variable
198    */
199   public EntryType getEntryType(String variable) {
200     Optional<EntryType> result = Arrays.stream(EntryType.values())
201       .filter(entryType -> entryMapping.containsKey(Pair.of(variable, entryType)))
202       .findFirst();
203 
204     if (!result.isPresent()) {
205       throw new NoSuchElementException(String.format("no entry for variable %s", variable));
206     }
207     return result.get();
208   }

222   /**
223    * Returns the position of the PropertyValue corresponding to the Variable-PropertyKey-Pair.
224    *
225    * @param variable variable name
226    * @param propertyKey property key
227    * @return the position of the corresponding property value
228    * @throws NoSuchElementException if there is no column mapped to the given variable and key

230   public int getPropertyColumn(String variable, String propertyKey) {
231     Integer column = propertyMapping.get(Pair.of(variable, propertyKey));
232     if (column == null) {
233       throw new NoSuchElementException(

254    * @return direction

257   public ExpandDirection getDirection(String variable) {

265   /**
266    * Returns a list of all variable that are contained in the embedding.
267    * The order of the variables is determined by their position within the embedding.

269    * @return a list of all variables

271   public List<String> getVariables() {
272     return entryMapping.entrySet().stream()
273       .sorted(Comparator.comparingInt(Map.Entry::getValue))
274       .map(entry -> entry.getKey().getLeft())

321   /**
322    * Returns a list of all property keys that are contained in the embedding regarding the
323    * specified variable.
324    * The order of the keys is determined by the position of the property value in the embedding.
325    *
326    * @param variable variable name
327    * @return a list of all property keys contained in the embedding
328    */
329   public List<String> getPropertyKeys(String variable) {
330     return propertyMapping.entrySet().stream()
331       .filter(entry -> entry.getKey().getLeft().equals(variable))
332       .sorted(Comparator.comparingInt(Map.Entry::getValue))
333       .map(entry -> entry.getKey().getRight())
334       .collect(Collectors.toList());
335   }

337   @Override
338   public boolean equals(Object o) {
339     if (this == o) {
340       return true;
341     }
342     if (o == null || getClass() != o.getClass()) {
343       return false;
344     }
345 
346     EmbeddingMetaData metaData = (EmbeddingMetaData) o;
347 
348     return entryMapping.equals(metaData.entryMapping) &&
349       propertyMapping.equals(metaData.propertyMapping);
350   }

357   @Override
358   public String toString() {
359     List<Map.Entry<Pair<String, EntryType>, Integer>> sortedEntries = entryMapping.entrySet()
360       .stream()
361       .sorted(Comparator.comparingInt(Map.Entry::getValue))
362       .collect(Collectors.toList());
363 
364     List<Map.Entry<Pair<String, String>, Integer>> sortiedProperties = propertyMapping.entrySet()
365       .stream()
366       .sorted(Comparator.comparingInt(Map.Entry::getValue))
367       .collect(Collectors.toList());
368 
369     return String.format("EmbeddingMetaData{entryMapping=%s, propertyMapping=%s}",
370       sortedEntries, sortiedProperties);
371   }

373   /**
374    * Returns the variables that fulfil the specified predicate. The variables are ordered by
375    * their appearance in the entry mapping.
376    *
377    * @param predicate predicate for entry types
378    * @return variables that fulfil the predicate
379    */
380   private List<String> getVariables(Predicate<EntryType> predicate) {
381     return entryMapping.entrySet().stream()
382       .filter(entry -> predicate.test(entry.getKey().getRight()))
383       .sorted(Comparator.comparingInt(Map.Entry::getValue))
384       .map(entry -> entry.getKey().getLeft())
385       .collect(Collectors.toList());
386   }
