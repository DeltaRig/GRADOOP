37  /**
38   * Layout a graph using approximate repulsive forces calculated using centroids as described
39   * <a href="https://www.researchgate.net/publication/281348264_Distributed_Graph_Layout_with_Spark">
40   * here</a>
41   * Very fast, even for large inputs.
42   */
43  public class CentroidFRLayouter extends FRLayouter {
44  
45    /**
46     * Fraction of all vertices a centroid should minimally have
47     */
48    public static final double MIN_MASS_FACTOR = 0.0025d;
49    /**
50     * Fraction of all vertices a centroid should maximally have
51     */
52    public static final double MAX_MASS_FACTOR = 0.05d;
53    /**
54     * Name for the Centroid BroadcastSet
55     */
56    public static final String CENTROID_BROADCAST_NAME = "centroids";
57    /**
58     * Name for the Center BroadcastSet
59     */
60    public static final String CENTER_BROADCAST_NAME = "center";
61    /**
62     * DataSet containing the current centroids
63     */
64    private DataSet<Centroid> centroids;
65    /**
66     * DataSet containing the current graph-center
67     */
68    private DataSet<Vector> center;
69  
70    /**
71     * Create new CentroidFRLayouter
72     *
73     * @param iterations  Number of iterations to perform
74     * @param vertexCount Approximate number of vertices in the input-graph
75     */
76    public CentroidFRLayouter(int iterations, int vertexCount) {
77      super(iterations, vertexCount);
78    }
79  
80    @Override
81    public LogicalGraph execute(LogicalGraph g) {
82  
83      g = createInitialLayout(g);
84  
85      DataSet<EPGMVertex> gradoopVertices = g.getVertices();
86      DataSet<EPGMEdge> gradoopEdges = g.getEdges();
87  
88      DataSet<LVertex> vertices = gradoopVertices.map(LVertex::new);
89      DataSet<LEdge> edges = gradoopEdges.map(LEdge::new);
90  
91      centroids = chooseInitialCentroids(vertices);
92  
93      // flink can only iterate over one dataset at once. Create a dataset containing both
94      // centroids and vertices. Split them again at the begin of every iteration
95      DataSet<SimpleGraphElement> graphElements = vertices.map(x -> x);
96      graphElements = graphElements.union(centroids.map(x -> x));
97  
98      IterativeDataSet<SimpleGraphElement> loop = graphElements.iterate(iterations);
99      vertices = loop.filter(x -> x instanceof LVertex).map(x -> (LVertex) x);
100     centroids = loop.filter(x -> x instanceof Centroid).map(x -> (Centroid) x);
101 
102     centroids = calculateNewCentroids(centroids, vertices);
103     center = calculateLayoutCenter(vertices);
104 
105     LGraph graph = new LGraph(vertices, edges);
106     // we have overridden repulsionForces() so layout() will use or new centroid-based solution
107     layout(graph);
108 
109     graphElements = graph.getVertices().map(x -> x);
110     graphElements = graphElements.union(centroids.map(x -> x));
111 
112     graphElements = loop.closeWith(graphElements);
113 
114     vertices = graphElements.filter(x -> x instanceof LVertex).map(x -> (LVertex) x);
115 
116     gradoopVertices = vertices.join(gradoopVertices).where(LVertex.ID_POSITION).equalTo(new Id<>())
117       .with(new LVertexEPGMVertexJoinFunction());
118 
119     return g.getFactory().fromDataSets(gradoopVertices, gradoopEdges);
120   }
121 
122   /* override and calculate repulsionFoces using centroids. Everything else stays like in the
123   original FR */
124   @Override
125   protected DataSet<Force> repulsionForces(DataSet<LVertex> vertices) {
126     return vertices.map(new CentroidRepulsionForceMapper(new FRRepulsionFunction(getK())))
127       .withBroadcastSet(centroids, CENTROID_BROADCAST_NAME)
128       .withBroadcastSet(center, CENTER_BROADCAST_NAME);
129   }
130 
131   /**
132    * Randomly choose some vertex-positions as start centroids
133    *
134    * @param vertices Current (randomly placed) vertices of the graph
135    * @return Random centroids to use (always at least one)
136    */
137   protected DataSet<Centroid> chooseInitialCentroids(DataSet<LVertex> vertices) {
138     // Choose a sample rate that will statistically result in clusters with a mass exactly
139     // between min and max allowed mass
140     final double sampleRate =
141       1.0 / (((MIN_MASS_FACTOR + MAX_MASS_FACTOR) / 2.0) * numberOfVertices);
142     // Because of the randomness of the layouting it is possible that on small graphs no vertex
143     // is chosen as centroid. This would result in problems. Therefore we union with one single
144     // vertex, so there is ALWAYS at least one centroid
145     return vertices.filter((v) -> Math.random() < sampleRate).union(vertices.first(1))
146       .map(v -> new Centroid(v.getPosition(), 0));
147   }
148 
149   /**
150    * Calculate the current centroids for the graph
151    *
152    * @param centroids The old/current centroids
153    * @param vertices  The current vertices of the graph
154    * @return The new centroids (to use for the next iteration)
155    */
156   protected DataSet<Centroid> calculateNewCentroids(DataSet<Centroid> centroids,
157     DataSet<LVertex> vertices) {
158 
159     CentroidUpdater updater =
160       new CentroidUpdater(numberOfVertices, MIN_MASS_FACTOR, MAX_MASS_FACTOR);
161 
162     return updater.updateCentroids(centroids, vertices);
163   }
164 
165   /**
166    * Calculate the current center of the graph-layout
167    *
168    * @param vertices Current vertices of the graph
169    * @return The average of all vertex positions
170    */
171   protected DataSet<Vector> calculateLayoutCenter(DataSet<LVertex> vertices) {
172     return new AverageVertexPositionsFunction().averagePosition(vertices);
173   }
174 
175 
176   @Override
177   public String toString() {
178     return "CentroidFRLayouter{" + "iterations=" + iterations + ", k=" + getK() + ", width=" +
179       getWidth() + ", height=" + getHeight() + ", numberOfVertices=" + numberOfVertices +
180       ", useExistingLayout=" + useExistingLayout + '}';
181   }
