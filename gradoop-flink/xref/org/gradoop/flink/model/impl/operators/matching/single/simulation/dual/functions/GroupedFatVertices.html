16  package org.gradoop.flink.model.impl.operators.matching.single.simulation.dual.functions;
17  
18  import org.apache.flink.api.common.functions.GroupReduceFunction;
19  import org.apache.flink.api.java.functions.FunctionAnnotation;
20  import org.apache.flink.util.Collector;
21  import org.gradoop.flink.model.impl.operators.matching.single.simulation.dual.tuples.FatVertex;
22  
23  /**
24   * Merges multiple fat vertices into a single {@link FatVertex}.
25   * <p>
26   * {@code [fatVertex] -> fatVertex}
27   * <p>
28   * Forwarded fields:
29   * <ul>
30   * <li>f0: vertex id</li>
31   * </ul>
32   * Read fields:
33   * <ul>
34   * <li>f1: vertex query candidates</li>
35   * <li>f2: parent ids</li>
36   * <li>f3: counters for incoming edge candidates</li>
37   * <li>f4: outgoing edges (edgeId, targetId) and their query candidates</li>
38   * </ul>
39   */
40  @FunctionAnnotation.ForwardedFields("f0")
41  @FunctionAnnotation.ReadFields("f1;f2;f3;f4")
42  public class GroupedFatVertices implements GroupReduceFunction<FatVertex, FatVertex> {
43  
44    @Override
45    public void reduce(Iterable<FatVertex> vertices, Collector<FatVertex> collector) throws Exception {
46  
47      boolean first = true;
48      FatVertex result = null;
49  
50      for (FatVertex vertex : vertices) {
51        if (first) {
52          result = vertex;
53          first = false;
54        } else {
55          result = merge(result, vertex);
56        }
57      }
58      collector.collect(result);
59    }
60  
61    /**
62     * Merges two fat vertices.
63     *
64     * @param result base
65     * @param diff   diff to merge
66     * @return base merged with diff
67     */
68    private FatVertex merge(FatVertex result, FatVertex diff) {
69      // update vertex candidates (CA)
70      if (diff.getCandidates().removeAll(result.getCandidates())) {
71        result.getCandidates().addAll(diff.getCandidates());
72      }
73      // update parent ids (P_IDs)
74      result.getParentIds().addAll(diff.getParentIds());
75      // update incoming edge counts (IN_CA)
76      for (int i = 0; i < diff.getIncomingCandidateCounts().length; i++) {
77        result.getIncomingCandidateCounts()[i] += diff
78          .getIncomingCandidateCounts()[i];
79      }
80      // update outgoing edges (OUT_CA)
81      result.getEdgeCandidates().putAll(diff.getEdgeCandidates());
82  
83      return result;
84    }
85  }
