23  import org.gradoop.flink.model.api.epgm.BaseGraph;

31  /**
32   * Computes a vertex sampling of the graph (new graph head will be generated). Retains randomly
33   * chosen vertices of a given relative amount and all edges which source- and target-vertices
34   * were chosen. A degree-dependent value is taken into account to have a bias towards high-degree
35   * vertices. There may retain some unconnected vertices in the sampled graph.
36   *
37   * @param <G>  The graph head type.
38   * @param <V>  The vertex type.
39   * @param <E>  The edge type.
40   * @param <LG> The type of the graph.
41   * @param <GC> The type of the graph collection.
42   */
43  public class RandomNonUniformVertexSampling<
44    G extends GraphHead,
45    V extends Vertex,
46    E extends Edge,
47    LG extends BaseGraph<G, V, E, LG, GC>,
48    GC extends BaseGraphCollection<G, V, E, LG, GC>> extends SamplingAlgorithm<G, V, E, LG, GC> {
49  
50    /**
51     * Relative amount of vertices in the result graph
52     */
53    private final float sampleSize;
54  
55    /**
56     * Seed for the random number generator
57     * If seed is 0, the random generator is created without seed
58     */
59    private final long randomSeed;
60  
61    /**
62     * Creates new RandomNonUniformVertexSampling instance.
63     *
64     * @param sampleSize relative sample size
65     */
66    public RandomNonUniformVertexSampling(float sampleSize) {
67      this(sampleSize, 0L);
68    }
69  
70    /**
71     * Creates new RandomNonUniformVertexSampling instance.
72     *
73     * @param sampleSize relative sample size
74     * @param randomSeed random seed value (can be 0)
75     */
76    public RandomNonUniformVertexSampling(float sampleSize, long randomSeed) {
77      this.sampleSize = sampleSize;
78      this.randomSeed = randomSeed;
79    }
80  
81    @Override
82    public LG sample(LG graph) {
83  
84      graph = graph.callForGraph(new DistinctVertexDegrees<>(
85        SamplingConstants.DEGREE_PROPERTY_KEY,
86        SamplingConstants.IN_DEGREE_PROPERTY_KEY,
87        SamplingConstants.OUT_DEGREE_PROPERTY_KEY,
88        true));
89  
90      DataSet<V> newVertices = graph.getVertices()
91        .map(new VertexToDegreeMap<>(SamplingConstants.DEGREE_PROPERTY_KEY))
92        .max(0)
93        .cross(graph.getVertices())
94        .with(new AddMaxDegreeCrossFunction<>(SamplingConstants.PROPERTY_KEY_MAX_DEGREE));
95  
96      graph = graph.getFactory()
97        .fromDataSets(graph.getGraphHead(), newVertices, graph.getEdges());
98  
99      newVertices = graph.getVertices().filter(new NonUniformVertexRandomFilter<>(
100       sampleSize,
101       randomSeed,
102       SamplingConstants.DEGREE_PROPERTY_KEY,
103       SamplingConstants.PROPERTY_KEY_MAX_DEGREE));
104 
105     newVertices = newVertices
106       .map(new PropertyRemover<>(SamplingConstants.DEGREE_PROPERTY_KEY))
107       .map(new PropertyRemover<>(SamplingConstants.IN_DEGREE_PROPERTY_KEY))
108       .map(new PropertyRemover<>(SamplingConstants.OUT_DEGREE_PROPERTY_KEY))
109       .map(new PropertyRemover<>(SamplingConstants.PROPERTY_KEY_MAX_DEGREE));
110 
111     return graph.getFactory().fromDataSets(newVertices, graph.getEdges()).verify();
112   }
