29  /**
30   * Filters a set of {@link Triple} objects based on a specified predicate. Additionally, the
31   * operator projects all property values to the output {@link Embedding} that are specified in the
32   * given {@code projectionPropertyKeys}.
33   * <p>
34   * {@code Triple -> Embedding( [PropertyEntry(SourceVertexId)], [PropertyEntry(EdgeId)], PropertyEntry
35   * (TargetVertexId)])}
36   * <p>
37   * Example:
38   * <br>
39   * Given a Triple
40   * {@code (
41   *   p1: SourceVertex(0, "Person", {name:"Alice", age:23, location: "Sweden"}),
42   *   e1: Edge(1, "knows", {}),
43   *   p2: TargetVertex(2, "Person", {name:"Bob", age:23})
44   * )},
45   * a predicate {@code "p1.name = "Alice" AND p1.age <= p2.age"} and
46   * projection property keys {@code p1: [name, location], e1: [], p2: [name, location]} the operator creates
47   * an {@link Embedding}:
48   * <br>
49   * {@code ([IdEntry(0),IdEntry(1),IdEntry(2)],[PropertyEntry("Alice"),PropertyEntry("Sweden"),
50   * PropertyEntry("Bob"),PropertyEntry(NULL)])}
51   */
52  public class FilterAndProjectTriples implements PhysicalOperator {
53    /**
54     * Input vertices
55     */
56    private final DataSet<Triple> input;
57    /**
58     * Variable assigned to the source vertex
59     */
60    private final String sourceVariable;
61    /**
62     * Variable assigned to the edge
63     */
64    private final String edgeVariable;
65    /**
66     * Variable assigned to the target vertex
67     */
68    private final String targetVariable;
69    /**
70     * Predicates in conjunctive normal form
71     */
72    private final CNF predicates;
73    /**
74     * Property keys used for projection
75     */
76    private final Map<String, List<String>> projectionPropertyKeys;
77    /**
78     * Match Strategy used for Vertices
79     */
80    private final MatchStrategy vertexMatchStrategy;
81  
82    /**
83     * Operator name used for Flink operator description
84     */
85    private String name;
86  
87    /**
88     * New vertex filter operator
89     *
90     * @param input Candidate vertices
91     * @param sourceVariable Variable assigned to the vertex
92     * @param edgeVariable Variable assigned to the vertex
93     * @param targetVariable Variable assigned to the vertex
94     * @param predicates Predicates used to filter vertices
95     * @param projectionPropertyKeys Property keys used for projection
96     * @param vertexMatchStrategy Vertex match strategy
97     */
98    public FilterAndProjectTriples(DataSet<Triple> input, String sourceVariable, String edgeVariable,
99      String targetVariable, CNF predicates, Map<String, List<String>> projectionPropertyKeys,
100     MatchStrategy vertexMatchStrategy) {
101     this.input = input;
102     this.sourceVariable = sourceVariable;
103     this.edgeVariable = edgeVariable;
104     this.targetVariable = targetVariable;
105     this.predicates = predicates;
106     this.projectionPropertyKeys = projectionPropertyKeys;
107     this.vertexMatchStrategy = vertexMatchStrategy;
108     this.setName("FilterAndProjectTriples");
109   }
110 
111   @Override
112   public DataSet<Embedding> evaluate() {
113     return input.flatMap(
114       new FilterAndProjectTriple(
115         sourceVariable,
116         edgeVariable,
117         targetVariable,
118         predicates,
119         projectionPropertyKeys,
120         vertexMatchStrategy
121       )
122     ).name(getName());
123   }
124 
125   @Override
126   public void setName(String newName) {
127     this.name = newName;
128   }
129 
130   @Override
131   public String getName() {
132     return this.name;
133   }
134 }
