43  /**
44   * Operator deriving a string representation from a graph collection.
45   * The representation follows the concept of a canonical adjacency matrix.
46   *
47   * @param <G> type of the graph head
48   * @param <V> the vertex type
49   * @param <E> the edge type
50   * @param <LG> type of the base graph instance
51   * @param <GC> type of the graph collection
52   */
53  public class CanonicalAdjacencyMatrixBuilder<
54    G extends GraphHead,
55    V extends Vertex,
56    E extends Edge,
57    LG extends BaseGraph<G, V, E, LG, GC>,
58    GC extends BaseGraphCollection<G, V, E, LG, GC>>
59    implements UnaryBaseGraphCollectionToValueOperator<GC, DataSet<String>> {
60  
61    /**
62     * Character used to separate lines. Should be the same regardless of OS.
63     */
64    public static final Character LINE_SEPARATOR = '\n';
65  
66    /**
67     * function describing string representation of graph heads
68     */
69    private final GraphHeadToString<G> graphHeadToString;
70    /**
71     * function describing string representation of vertices
72     */
73    private final VertexToString<V> vertexToString;
74    /**
75     * function describing string representation of edges
76     */
77    private final EdgeToString<E> egeLabelingFunction;
78    /**
79     * sets mode for either directed or undirected graph
80     */
81    private final boolean directed;
82  
83    /**
84     * Creates a new MatrixBuilder instance.
85     *
86     * @param graphHeadToString representation of graph heads
87     * @param vertexToString representation of vertices
88     * @param edgeLabelingFunction representation of edges
89     * @param directed sets mode for either directed or undirected graph
90     */
91    public CanonicalAdjacencyMatrixBuilder(
92      GraphHeadToString<G> graphHeadToString,
93      VertexToString<V> vertexToString,
94      EdgeToString<E> edgeLabelingFunction,
95      boolean directed
96    ) {
97      this.graphHeadToString = graphHeadToString;
98      this.vertexToString = vertexToString;
99      this.egeLabelingFunction = edgeLabelingFunction;
100     this.directed = directed;
101   }
102 
103   @Override
104   public DataSet<String> execute(GC collection) {
105 
106     // 1-10.
107     DataSet<GraphHeadString> graphHeadLabels = getGraphHeadStrings(collection);
108 
109     // 11. add empty head to prevent empty result for empty collection
110 
111     graphHeadLabels = graphHeadLabels
112       .union(collection
113         .getConfig()
114         .getExecutionEnvironment()
115         .fromElements(new GraphHeadString(GradoopId.get(), "")));
116 
117     // 12. label collection
118 
119     return graphHeadLabels
120       .reduceGroup(new ConcatGraphHeadStrings());
121   }
122 
123   /**
124    * Created a dataset of (graph id, canonical label) pairs.
125    *
126    * @param collection input collection
127    * @return (graph id, canonical label) pairs
128    */
129   public DataSet<GraphHeadString> getGraphHeadStrings(GC collection) {
130     // 1. label graph heads
131     DataSet<GraphHeadString> graphHeadLabels = collection.getGraphHeads()
132       .map(graphHeadToString);
133 
134     // 2. label vertices
135     DataSet<VertexString> vertexLabels = collection.getVertices()
136       .flatMap(vertexToString);
137 
138     // 3. label edges
139     DataSet<EdgeString> edgeLabels = collection.getEdges()
140       .flatMap(egeLabelingFunction);
141 
142     if (directed) {
143       // 4. combine labels of parallel edges
144       edgeLabels = edgeLabels
145         .groupBy(0, 1, 2)
146         .reduceGroup(new MultiEdgeStringCombiner());
147 
148       // 5. extend edge labels by vertex labels
149 
150       edgeLabels = edgeLabels
151         .join(vertexLabels)
152         .where(0, 1).equalTo(0, 1) // graphId,sourceId = graphId,vertexId
153         .with(new SourceStringUpdater())
154         .join(vertexLabels)
155         .where(0, 2).equalTo(0, 1) // graphId,targetId = graphId,vertexId
156         .with(new TargetStringUpdater());
157 
158       // 6. extend vertex labels by outgoing vertex+edge labels
159 
160       DataSet<VertexString> outgoingAdjacencyListLabels =
161         edgeLabels.groupBy(0, 1) // graphId, sourceId
162           .reduceGroup(new OutgoingAdjacencyList());
163 
164       // 7. extend vertex labels by outgoing vertex+edge labels
165 
166       DataSet<VertexString> incomingAdjacencyListLabels =
167         edgeLabels.groupBy(0, 2) // graphId, targetId
168           .reduceGroup(new IncomingAdjacencyList());
169 
170       // 8. combine vertex labels
171 
172       vertexLabels = vertexLabels
173         .leftOuterJoin(outgoingAdjacencyListLabels)
174         .where(0, 1).equalTo(0, 1)
175         .with(new LabelCombiner<VertexString>())
176         .leftOuterJoin(incomingAdjacencyListLabels)
177         .where(0, 1).equalTo(0, 1)
178         .with(new LabelCombiner<VertexString>());
179     } else {
180     // undirected graph
181 
182       // 4. union edges with flipped edges and combine labels of parallel edges
183 
184       edgeLabels = edgeLabels
185         .union(edgeLabels
186           .map(new SwitchSourceTargetIds()))
187         .groupBy(0, 1, 2)
188         .reduceGroup(new MultiEdgeStringCombiner());
189 
190       // 5. extend edge labels by vertex labels
191 
192       edgeLabels = edgeLabels
193         .join(vertexLabels)
194         .where(0, 2).equalTo(0, 1) // graphId,targetId = graphId,vertexId
195         .with(new TargetStringUpdater());
196 
197       // 6/7. extend vertex labels by vertex+edge labels
198 
199       DataSet<VertexString> adjacencyListLabels =
200         edgeLabels.groupBy(0, 1) // graphId, sourceId
201           .reduceGroup(new UndirectedAdjacencyList());
202 
203       // 8. combine vertex labels
204 
205       vertexLabels = vertexLabels
206         .leftOuterJoin(adjacencyListLabels)
207         .where(0, 1).equalTo(0, 1)
208         .with(new LabelCombiner<VertexString>());
209     }
210 
211     // 9. create adjacency matrix labels
212 
213     DataSet<GraphHeadString> adjacencyMatrixLabels = vertexLabels
214       .groupBy(0)
215       .reduceGroup(new AdjacencyMatrix());
216 
217     // 10. combine graph labels
218 
219     graphHeadLabels = graphHeadLabels
220       .leftOuterJoin(adjacencyMatrixLabels)
221       .where(0).equalTo(0)
222       .with(new LabelCombiner<GraphHeadString>());
223     return graphHeadLabels;
224   }
