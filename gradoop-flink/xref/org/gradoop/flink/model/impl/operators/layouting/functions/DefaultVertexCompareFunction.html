16  package org.gradoop.flink.model.impl.operators.layouting.functions;
17  
18  import org.gradoop.flink.model.impl.operators.layouting.util.LVertex;
19  import org.gradoop.flink.model.impl.operators.layouting.util.Vector;
20  
21  /**
22   * Compares two vertices based on their position and the forces acting on them
23   */
24  public class DefaultVertexCompareFunction implements VertexCompareFunction {
25    /**
26     * Parameter k that is used in the FR-Algorithm
27     */
28    protected double k;
29  
30    /**
31     * Construct new compare-function
32     *
33     * @param k Parameter k that is used in the FR-Algorithm. Is the distance between two connected
34     *          vertices where attraction and repulsion are equal.
35     */
36    public DefaultVertexCompareFunction(double k) {
37      this.k = k;
38    }
39  
40    @Override
41    public double compare(LVertex v1, LVertex v2) {
42      // compare position-difference in relation to k. Cap values at 0,1.
43      double positionSimilarity =
44        Math.min(1, Math.max(0, 1 - ((v1.getPosition().distance(v2.getPosition()) - k) / k)));
45  
46      // compare difference in forces. Not only consider direction but als magnitude.
47      Vector force1 = v1.getForce().div(v1.getCount());
48      Vector force2 = v2.getForce().div(v2.getCount());
49      double forceSimilarity =
50        1 - (force1.distance(force2) / (force1.magnitude() + force2.magnitude()));
51  
52      return positionSimilarity * forceSimilarity;
53    }
54  }
