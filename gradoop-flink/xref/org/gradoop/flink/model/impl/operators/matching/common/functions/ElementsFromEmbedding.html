44  /**
45   * Extracts {@link EPGMElement} instances from an {@link Embedding}.
46   */
47  public class ElementsFromEmbedding extends RichFlatMapFunction<Tuple1<Embedding<GradoopId>>, Element> {
48  
49    /**
50     * Maps edge candidates to the step in which they are traversed
51     */
52    private final Map<Integer, Step> edgeToStep;
53    /**
54     * Constructs graph heads
55     */
56    private final GraphHeadFactory<? extends GraphHead> graphHeadFactory;
57    /**
58     * Constructs vertices
59     */
60    private final VertexFactory<? extends Vertex> vertexFactory;
61    /**
62     * Constructs edges
63     */
64    private final EdgeFactory<? extends Edge> edgeFactory;
65    /**
66     * Maps query vertex ids to variables
67     */
68    private final Map<Long, String> queryVertexMapping;
69    /**
70     * Maps query edge ids to variables
71     */
72    private final Map<Long, String> queryEdgeMapping;
73    /**
74     * Reuse map for storing the variable mapping
75     */
76    private Map<PropertyValue, PropertyValue> reuseVariableMapping;
77    /**
78     * Constructor
79     *
80     * @param traversalCode     traversal code to retrieve sourceId/targetId
81     * @param graphHeadFactory  graph head factory
82     * @param vertexFactory     vertex factory
83     * @param edgeFactory       edge factory
84     * @param query             query handler
85     */
86    public ElementsFromEmbedding(TraversalCode traversalCode,
87      GraphHeadFactory<? extends GraphHead> graphHeadFactory,
88      VertexFactory<? extends Vertex> vertexFactory,
89      EdgeFactory<? extends Edge> edgeFactory,
90      QueryHandler query) {
91      this.graphHeadFactory = graphHeadFactory;
92      this.vertexFactory = vertexFactory;
93      this.edgeFactory = edgeFactory;
94  
95      this.queryVertexMapping = query.getVertices()
96        .stream()
97        .collect(Collectors.toMap(v -> v.getId(), v -> v.getVariable()));
98  
99      this.queryEdgeMapping = query.getEdges()
100       .stream()
101       .collect(Collectors.toMap(e -> e.getId(), e -> e.getVariable()));
102 
103     List<Step> steps = traversalCode.getSteps();
104     edgeToStep = Maps.newHashMapWithExpectedSize(steps.size());
105 
106     for (Step step : steps) {
107       edgeToStep.put((int) step.getVia(), step);
108     }
109   }
110 
111   @Override
112   public void open(Configuration conf) {
113     this.reuseVariableMapping = new HashMap<>();
114   }
115 
116   @Override
117   public void flatMap(Tuple1<Embedding<GradoopId>> embedding, Collector<Element> out)
118       throws Exception {
119 
120     GradoopId[] vertexMapping = embedding.f0.getVertexMapping();
121     GradoopId[] edgeMapping = embedding.f0.getEdgeMapping();
122 
123 
124     // create graph head for this embedding
125     GraphHead graphHead = graphHeadFactory.createGraphHead();
126 
127     // collect vertices (and assign to graph head)
128     for (int i = 0; i < vertexMapping.length; i++) {
129       if (!isProcessed(vertexMapping, i)) {
130         Vertex v = vertexFactory.initVertex(vertexMapping[i]);
131         v.addGraphId(graphHead.getId());
132         out.collect(v);
133       }
134 
135       reuseVariableMapping.put(
136         PropertyValue.create(queryVertexMapping.get((long) i)),
137         PropertyValue.create(vertexMapping[i])
138       );
139     }
140 
141     // collect edges (and assign to graph head)
142     for (int i = 0; i < edgeMapping.length; i++) {
143       if (!isProcessed(edgeMapping, i)) {
144         Step s = edgeToStep.get(i);
145         // get sourceId/targetId according to traversal step
146         GradoopId sourceId = s.isOutgoing() ?
147           vertexMapping[(int) s.getFrom()] : vertexMapping[(int) s.getTo()];
148         GradoopId targetId = s.isOutgoing() ?
149           vertexMapping[(int) s.getTo()] : vertexMapping[(int) s.getFrom()];
150         Edge e = edgeFactory.initEdge(edgeMapping[i], sourceId, targetId);
151         e.addGraphId(graphHead.getId());
152         out.collect(e);
153       }
154 
155       reuseVariableMapping.put(
156         PropertyValue.create(queryEdgeMapping.get((long) i)),
157         PropertyValue.create(edgeMapping[i])
158       );
159     }
160 
161     graphHead.setProperty(PatternMatching.VARIABLE_MAPPING_KEY, reuseVariableMapping);
162     out.collect(graphHead);
163   }

165   /**
166    * Checks if the the id at the specified index has been processed before.
167    *
168    * @param mapping id mapping
169    * @param i index
170    * @return true, iff the element at position i is present at a position between 0 and i
171    */
172   private boolean isProcessed(GradoopId[] mapping, int i) {
173     for (int j = 0; j < i; j++) {
174       if (mapping[j].equals(mapping[i])) {
175         return true;
176       }
177     }
178     return false;
179   }
