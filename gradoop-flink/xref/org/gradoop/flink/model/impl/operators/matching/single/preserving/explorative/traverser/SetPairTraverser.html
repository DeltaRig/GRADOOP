51  /**
52   * Traverses a graph represented by two DataSets (vertices and edges).
53   *
54   * @param <K> key type
55   */
56  public abstract class SetPairTraverser<K> extends DistributedTraverser<K> {
57    /**
58     * Join strategy used for the join between embeddings and edges
59     */
60    private final JoinOperatorBase.JoinHint edgeStepJoinStrategy;
61    /**
62     * Join strategy used for the join between embeddings and vertices
63     */
64    private final JoinOperatorBase.JoinHint vertexStepJoinStrategy;
65  
66    /**
67     * Creates a new distributed traverser.
68     *
69     * @param traversalCode          describes the graph traversal
70     * @param matchStrategy          matching strategy for vertices and edges
71     * @param vertexCount            number of query vertices
72     * @param edgeCount              number of query edges
73     * @param keyClazz               key type for embedding initialization
74     * @param edgeStepJoinStrategy   Join strategy for edge extension
75     * @param vertexStepJoinStrategy Join strategy for vertex extension
76     * @param vertexMapping          used for debug
77     * @param edgeMapping            used for debug
78     */
79    SetPairTraverser(TraversalCode traversalCode, MatchStrategy matchStrategy, int vertexCount,
80      int edgeCount, Class<K> keyClazz, JoinOperatorBase.JoinHint edgeStepJoinStrategy,
81      JoinOperatorBase.JoinHint vertexStepJoinStrategy,
82      DataSet<Tuple2<K, PropertyValue>> vertexMapping,
83      DataSet<Tuple2<K, PropertyValue>> edgeMapping) {
84      super(traversalCode, matchStrategy, vertexCount, edgeCount, keyClazz,
85        vertexMapping, edgeMapping);
86  
87      Objects.requireNonNull(edgeStepJoinStrategy);
88      Objects.requireNonNull(vertexStepJoinStrategy);
89  
90      this.edgeStepJoinStrategy = edgeStepJoinStrategy;
91      this.vertexStepJoinStrategy = vertexStepJoinStrategy;
92    }
93  
94    /**
95     * Traverses the graph, thereby extracting embeddings of a given pattern.
96     *
97     * @param vertices  vertices including their query candidates
98     * @param edges     edges including their query candidates
99     * @return embeddings contained in the graph
100    */
101   public abstract DataSet<Tuple1<Embedding<K>>> traverse(
102     DataSet<IdWithCandidates<K>> vertices,
103     DataSet<TripleWithCandidates<K>> edges);
104 
105   private JoinOperatorBase.JoinHint getEdgeStepJoinStrategy() {
106     return edgeStepJoinStrategy;
107   }
108 
109   private JoinOperatorBase.JoinHint getVertexStepJoinStrategy() {
110     return vertexStepJoinStrategy;
111   }

113   /**
114    * Builds the initial embeddings from the given vertices.
115    *
116    * @param vertices vertices and their query candidates
117    * @return initial embeddings
118    */
119   DataSet<EmbeddingWithTiePoint<K>> buildInitialEmbeddings(DataSet<IdWithCandidates<K>> vertices) {
120 
121     Step initialStep = getTraversalCode().getStep(0);
122 
123     DataSet<EmbeddingWithTiePoint<K>> initialEmbeddings = vertices
124       .filter(new ElementHasCandidate<>((int) initialStep.getFrom()))
125       .map(new BuildEmbeddingFromVertex<>(getKeyClazz(), initialStep,
126         getVertexCount(), getEdgeCount()));
127 
128     return log(initialEmbeddings,
129       new PrintEmbeddingWithTiePoint<>(), getVertexMapping(), getEdgeMapping());
130   }
131 
132   /**
133    * Extends the given embeddings with valid edges and returns the updated
134    * embeddings.
135    *
136    * @param edges             edges including their candidates
137    * @param embeddings        embeddings
138    * @param superstep         current iteration
139    * @param traverserStrategy iteration strategy
140    * @param forwardedFields   forwarded fields for edge step creation
141    * @return updated embeddings
142    */
143   DataSet<EmbeddingWithTiePoint<K>> traverseEdges(
144     DataSet<TripleWithCandidates<K>> edges,
145     DataSet<EmbeddingWithTiePoint<K>> embeddings,
146     DataSet<Integer> superstep,
147     TraverserStrategy traverserStrategy,
148     String[] forwardedFields) {
149 
150     DataSet<EdgeStep<K>> edgeSteps = edges
151       .filter(new EdgeHasCandidate<>(getTraversalCode()))
152       .withBroadcastSet(superstep, ExplorativePatternMatching.BC_SUPERSTEP)
153       .map(new BuildEdgeStep<>(getTraversalCode()))
154       .withBroadcastSet(superstep, ExplorativePatternMatching.BC_SUPERSTEP)
155       .withForwardedFields(forwardedFields);
156 
157     edgeSteps = log(edgeSteps,
158       new PrintEdgeStep<>(isIterative(), "post-filter-map-edge"),
159       getVertexMapping(), getEdgeMapping());
160 
161     JoinOperator<EmbeddingWithTiePoint<K>, EdgeStep<K>,
162       EmbeddingWithTiePoint<K>> join = embeddings
163       .join(edgeSteps, getEdgeStepJoinStrategy())
164       .where(0).equalTo(1) // tiePointId == sourceId/targetId tie point
165       .with(new UpdateEdgeMapping<>(getTraversalCode(), getMatchStrategy(), traverserStrategy));
166 
167     if (traverserStrategy == TraverserStrategy.SET_PAIR_FOR_LOOP_ITERATION) {
168       embeddings = join.withBroadcastSet(superstep, ExplorativePatternMatching.BC_SUPERSTEP);
169     } else {
170       embeddings = join;
171     }
172 
173     return log(embeddings,
174       new PrintEmbeddingWithTiePoint<>(isIterative(), "post-edge-update"),
175       getVertexMapping(), getEdgeMapping());
176   }
177 
178   /**
179    * Extends the given embeddings with valid vertices and returns the updated
180    * embeddings.
181    *
182    * @param vertices          vertices including their candidates
183    * @param embeddings        embeddings
184    * @param superstep         current super step
185    * @param traverserStrategy iteration strategy
186    * @return updated embeddings
187    */
188   DataSet<EmbeddingWithTiePoint<K>> traverseVertices(
189     DataSet<IdWithCandidates<K>> vertices,
190     DataSet<EmbeddingWithTiePoint<K>> embeddings,
191     DataSet<Integer> superstep,
192     TraverserStrategy traverserStrategy) {
193 
194     DataSet<VertexStep<K>> vertexSteps = vertices
195       .filter(new VertexHasCandidate<>(getTraversalCode()))
196       .withBroadcastSet(superstep, ExplorativePatternMatching.BC_SUPERSTEP)
197       .map(new BuildVertexStep<>());
198 
199     vertexSteps = log(vertexSteps,
200       new PrintVertexStep<>(isIterative(), "post-filter-project-vertex"),
201       getVertexMapping(), getEdgeMapping());
202 
203     JoinOperator<EmbeddingWithTiePoint<K>, VertexStep<K>,
204       EmbeddingWithTiePoint<K>> join = embeddings
205       .join(vertexSteps, getVertexStepJoinStrategy())
206       .where(0).equalTo(0) // tiePointId == vertexId
207       .with(new UpdateVertexMapping<>(getTraversalCode(), getMatchStrategy(), traverserStrategy));
208 
209     if (traverserStrategy == TraverserStrategy.SET_PAIR_FOR_LOOP_ITERATION) {
210       embeddings = join.withBroadcastSet(superstep, ExplorativePatternMatching.BC_SUPERSTEP);
211     } else {
212       embeddings = join;
213     }
214 
215     embeddings = log(embeddings,
216       new PrintEmbeddingWithTiePoint<>(isIterative(), "post-vertex-update"),
217       getVertexMapping(), getEdgeMapping());
218     return embeddings;
219   }
