28  import org.gradoop.gdl.model.Edge;

30  /**
31   * Takes a vertex-edge pair and the corresponding target vertex as input and evaluates, if the triple matches
32   * against the query graph. The output is a {@link TripleWithCandidates} containing all query candidates for
33   * the triple.
34   * <p>
35   * Forwarded fields first:
36   * <br>
37   * {@code f0}:      edge id<br>
38   * {@code f1}:      source vertex id<br>
39   * {@code f3->f2}:  target vertex id
40   */
41  @FunctionAnnotation.ForwardedFieldsFirst("f0;f1;f3->f2")
42  public class MatchingTriples extends RichFlatJoinFunction<TripleWithSourceEdgeCandidates<GradoopId>,
43    IdWithCandidates<GradoopId>, TripleWithCandidates<GradoopId>> {
44    /**
45     * serial version uid
46     */
47    private static final long serialVersionUID = 42L;
48    /**
49     * GDL query
50     */
51    private final String query;
52    /**
53     * Query handler
54     */
55    private transient QueryHandler queryHandler;
56    /**
57     * Reduce instantiations
58     */
59    private final TripleWithCandidates<GradoopId> reuseTriple;
60  
61    /**
62     * Constructor
63     *
64     * @param query GDL query
65     */
66    public MatchingTriples(final String query) {
67      this.query = query;
68      this.reuseTriple = new TripleWithCandidates<>();
69    }
70  
71    @Override
72    public void open(Configuration parameters) throws Exception {
73      super.open(parameters);
74      queryHandler = new QueryHandler(query);
75    }
76  
77    @Override
78    public void join(TripleWithSourceEdgeCandidates<GradoopId> pair,
79      IdWithCandidates<GradoopId> target,
80      Collector<TripleWithCandidates<GradoopId>> collector) throws Exception {
81  
82      boolean[] edgeCandidates = pair.getEdgeCandidates();
83      boolean[] newEdgeCandidates = new boolean[edgeCandidates.length];
84      boolean edgeStillValid = false;
85  
86      for (int i = 0; i < edgeCandidates.length; i++) {
87        if (edgeCandidates[i]) {
88          Edge e = queryHandler.getEdgeById((long) i);
89          if (pair.getSourceCandidates()[e.getSourceVertexId().intValue()] &&
90            target.getCandidates()[e.getTargetVertexId().intValue()]) {
91            newEdgeCandidates[i] = true;
92            edgeStillValid = true;
93          }
94        }
95      }
96  
97      if (edgeStillValid) {
98        reuseTriple.setEdgeId(pair.getEdgeId());
99        reuseTriple.setSourceId(pair.getSourceId());
100       reuseTriple.setTargetId(target.getId());
101       reuseTriple.setCandidates(newEdgeCandidates);
102       collector.collect(reuseTriple);
103     }
104   }
105 }
