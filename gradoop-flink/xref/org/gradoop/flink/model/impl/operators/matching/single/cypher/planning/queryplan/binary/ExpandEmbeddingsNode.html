36  /**
37   * Binary node that wraps an {@link ExpandEmbeddingsBulk} operator.
38   */
39  public class ExpandEmbeddingsNode extends BinaryNode implements JoinNode {
40    /**
41     * Column to expand the embedding from.
42     */
43    private final int expandColumn;
44    /**
45     * Query variable of the first vertex in the path
46     */
47    private final String startVariable;
48    /**
49     * Query variable of the variable length path
50     */
51    private final String pathVariable;
52    /**
53     * Query variable of the last vertex in the path
54     */
55    private final String endVariable;
56    /**
57     * Minimum number of path expansion steps
58     */
59    private final int lowerBound;
60    /**
61     * Maximum number of path expansion steps
62     */
63    private final int upperBound;
64    /**
65     * Column that contains the final vertex of the expansion
66     */
67    private final int closingColumn;
68    /**
69     * Direction in which to expand the embedding
70     */
71    private final ExpandDirection expandDirection;
72    /**
73     * Morphism type for vertices
74     */
75    private final MatchStrategy vertexStrategy;
76    /**
77     * Morphism type for edges
78     */
79    private final MatchStrategy edgeStrategy;
80  
81    /**
82     * Creates a new node.
83     *
84     * @param leftChild left child representing the embeddings to expand
85     * @param rightChild right child representing the edges to expand with
86     * @param startVariable vertex variable on which to start the expansion
87     * @param pathVariable variable representing the path
88     * @param endVariable vertex variable on which to end the expansion
89     * @param lowerBound minimum number of expansions
90     * @param upperBound maximum number of expansions
91     * @param expandDirection edge direction in the expansion
92     * @param vertexStrategy morphism strategy for vertices
93     * @param edgeStrategy morphism strategy for edges
94     */
95    public ExpandEmbeddingsNode(PlanNode leftChild, PlanNode rightChild,
96      String startVariable, String pathVariable, String endVariable,
97      int lowerBound, int upperBound, ExpandDirection expandDirection,
98      MatchStrategy vertexStrategy, MatchStrategy edgeStrategy) {
99      super(leftChild, rightChild);
100     this.pathVariable = pathVariable;
101     this.startVariable = startVariable;
102     this.endVariable = endVariable;
103     this.lowerBound = lowerBound;
104     this.upperBound = upperBound == 0 ? Integer.MAX_VALUE : upperBound;
105     this.expandDirection = expandDirection;
106     this.vertexStrategy = vertexStrategy;
107     this.edgeStrategy = edgeStrategy;
108     this.expandColumn = leftChild.getEmbeddingMetaData().getEntryColumn(startVariable);
109     this.closingColumn = leftChild.getEmbeddingMetaData().containsEntryColumn(endVariable) ?
110       leftChild.getEmbeddingMetaData().getEntryColumn(endVariable) : -1;
111   }
112 
113   @Override
114   public DataSet<Embedding> execute() {
115     ExpandEmbeddings op = new ExpandEmbeddingsBulk(
116       getLeftChild().execute(), getRightChild().execute(),
117       expandColumn, lowerBound, upperBound, expandDirection,
118       getDistinctVertexColumns(getLeftChild().getEmbeddingMetaData()),
119       getDistinctEdgeColumns(getLeftChild().getEmbeddingMetaData()),
120       closingColumn, JoinOperatorBase.JoinHint.OPTIMIZER_CHOOSES);
121     op.setName(toString());
122     return op.evaluate();
123   }
124 
125   @Override
126   protected EmbeddingMetaData computeEmbeddingMetaData() {
127     EmbeddingMetaData inputMetaData = getLeftChild().getEmbeddingMetaData();
128     EmbeddingMetaData metaData = new EmbeddingMetaData(inputMetaData);
129 
130     metaData.setEntryColumn(pathVariable, EmbeddingMetaData.EntryType.PATH,
131       inputMetaData.getEntryCount());
132 
133     metaData.setDirection(pathVariable, expandDirection);
134 
135     if (!inputMetaData.containsEntryColumn(endVariable)) {
136       metaData.setEntryColumn(endVariable, EmbeddingMetaData.EntryType.VERTEX,
137         inputMetaData.getEntryCount() + 1);
138     }
139     return metaData;
140   }
141 
142   /**
143    * According to the specified {@link #vertexStrategy} and the specified
144    * {@link EmbeddingMetaData}, the method returns the columns that need to contain distinct
145    * entries.
146    *
147    * @param metaData meta data for the embedding
148    * @return distinct vertex columns
149    */
150   private List<Integer> getDistinctVertexColumns(EmbeddingMetaData metaData) {
151     return this.vertexStrategy == MatchStrategy.ISOMORPHISM ? metaData.getVertexVariables().stream()
152       .map(metaData::getEntryColumn)
153       .collect(Collectors.toList()) : Collections.emptyList();
154   }
155 
156   /**
157    * According to the specified {@link #edgeStrategy} and the specified
158    * {@link EmbeddingMetaData}, the method returns the columns that need to contain distinct
159    * entries.
160    *
161    * @param metaData meta data for the embedding
162    * @return distinct edge columns
163    */
164   private List<Integer> getDistinctEdgeColumns(EmbeddingMetaData metaData) {
165     return edgeStrategy == MatchStrategy.ISOMORPHISM ?
166       metaData.getEdgeVariables().stream()
167         .map(metaData::getEntryColumn)
168         .collect(Collectors.toList()) : Collections.emptyList();
169   }
170 
171   @Override
172   public String toString() {
173     return String.format("ExpandEmbeddingsNode={" +
174         "startVariable='%s', " +
175         "pathVariable='%s', " +
176         "endVariable='%s', " +
177         "lowerBound=%d, " +
178         "upperBound=%d, " +
179         "expandDirection=%s, " +
180         "vertexMorphismType=%s, " +
181         "edgeMorphismType=%s}",
182       startVariable, pathVariable, endVariable, lowerBound, upperBound, expandDirection,
183       vertexStrategy, edgeStrategy);
184   }
