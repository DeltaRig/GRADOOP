16  package org.gradoop.flink.model.impl.functions.filters;
17  
18  import org.apache.flink.api.common.functions.FilterFunction;
19  
20  /**
21   * A filter function that can be combined using a logical {@code and},
22   * {@code or} and {@code not}.
23   *
24   * @param <T> The type of elements to filter.
25   */
26  public interface CombinableFilter<T> extends FilterFunction<T> {
27  
28    /**
29     * Combine this filter with another filter using a logical {@code and}.
30     * Either filter may be a
31     * {@link org.apache.flink.api.common.functions.RichFunction}.
32     *
33     * @param other The filter that will be combined with this filter.
34     * @return The combined filter.
35     */
36    default CombinableFilter<T> and(FilterFunction<? super T> other) {
37      return new And<>(this, other);
38    }
39  
40    /**
41     * Combine this filter with another filter using a logical {@code or}.
42     * Either filter may be a
43     * {@link org.apache.flink.api.common.functions.RichFunction}.
44     *
45     * @param other The filter that will be combined with this filter.
46     * @return The combined filter.
47     */
48    default CombinableFilter<T> or(FilterFunction<? super T> other) {
49      return new Or<>(this, other);
50    }
51  
52    /**
53     * Invert this filter, i.e. get the logical {@code not} of this filter.
54     * This filter may be a
55     * {@link org.apache.flink.api.common.functions.RichFunction}.
56     *
57     * @return The inverted filter.
58     */
59    default CombinableFilter<T> negate() {
60      return new Not<>(this);
61    }
62  }
