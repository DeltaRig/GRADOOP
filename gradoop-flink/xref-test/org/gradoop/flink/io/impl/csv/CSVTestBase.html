108   /**
109    * Get a logical graph with the schema and all properties of "input_extended_properties" csv
110    * graph from resources directory. A GDL file can not be used while types like
111    * LocalDate etc. are not supported.
112    *
113    * @return the logical graph representing the expected graph
114    */
115   protected LogicalGraph getExtendedLogicalGraph() {
116     GradoopId idUser = GradoopId.get();
117     GradoopId idPost = GradoopId.get();
118     GradoopId idForum = GradoopId.get();
119     GradoopIdSet heads = GradoopIdSet.fromExisting(idForum);
120     Properties properties = Properties.createFromMap(PROPERTY_MAP);
121 
122     DataSet<EPGMGraphHead> graphHead = getExecutionEnvironment().fromElements(
123       new EPGMGraphHead(idForum, "Forum", properties)
124     );
125 
126     DataSet<EPGMVertex> vertices = getExecutionEnvironment().fromElements(
127       new EPGMVertex(idUser, "User", properties, heads),
128       new EPGMVertex(idPost, "Post", properties, heads)
129     );
130 
131     DataSet<EPGMEdge> edges = getExecutionEnvironment().fromElements(
132       new EPGMEdge(GradoopId.get(), "creatorOf", idUser, idPost, properties, heads)
133     );
134 
135     return getConfig().getLogicalGraphFactory().fromDataSets(graphHead, vertices, edges);
136   }
