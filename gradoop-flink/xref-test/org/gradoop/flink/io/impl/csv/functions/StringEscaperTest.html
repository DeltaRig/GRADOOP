27  /**
28   * Tests the string escaper.
29   */
30  public class StringEscaperTest {
31    /**
32     * Characters to be escaped
33     */
34    private static final Set<Character> ESCAPED_CHARACTERS = ImmutableSet
35      .of('\\', ';', ',', '|', ':', '\n', '=');
36    /**
37     * CSV string to be escaped
38     */
39    private static final String UNESCAPED_STRING = "tes\t\\,f\nsfg,:,d|";
40    /**
41     * Escaped CSV string
42     */
43    private static final String ESCAPED_STRING = "tes\t\\\\\\,f\\nsfg\\,\\:\\,d\\|";
44  
45    /**
46     * Test escaping a string.
47     */
48    @Test
49    public void testEscape() {
50      String escapedString = StringEscaper.escape(UNESCAPED_STRING, ESCAPED_CHARACTERS);
51      assertEquals(ESCAPED_STRING, escapedString);
52    }
53  
54    /**
55     * Test unescaping a string.
56     */
57    @Test
58    public void testUnescape() {
59      String string = StringEscaper.unescape(ESCAPED_STRING);
60      assertEquals(UNESCAPED_STRING, string);
61    }
62  
63    /**
64     * Test splitting an escaped string while ignoring escaped delimiters.
65     */
66    @Test
67    public void testSplit() {
68      String[] input = Stream.of("", ",abc;;", "ad,", "|\n\n df", "\\c")
69        .map(s -> StringEscaper.escape(s, ESCAPED_CHARACTERS))
70        .toArray(String[]::new);
71  
72      String delimiter = ",";
73      String[] output = StringEscaper.split(String.join(delimiter, input), delimiter);
74      assertArrayEquals(input, output);
75  
76      delimiter = "c;";
77      output = StringEscaper.split(String.join(delimiter, input), delimiter);
78      assertArrayEquals(input, output);
79  
80      delimiter = "\n\n.;|";
81      output = StringEscaper.split(String.join(delimiter, input), delimiter);
82      assertArrayEquals(input, output);
83    }
84  
85    /**
86     * Test splitting an escaped string with limit while ignoring escaped delimiters.
87     */
88    @Test
89    public void testSplitWithLimit() {
90      String[] input = Stream.of("", ",abc;;", "ad,", "|\n\n df", "\\c")
91        .map(s -> StringEscaper.escape(s, ESCAPED_CHARACTERS))
92        .toArray(String[]::new);
93  
94      // Template to add the correct delimiter in the last string.
95      String template = input[2] + "%s" + input[3] + "%<s" + input[4];
96      String[] expected = {input[0], input[1], ""};
97  
98      String delimiter = ",";
99      expected[2] = String.format(template, delimiter);
100     String[] output = StringEscaper.split(String.join(delimiter, input), delimiter, 3);
101     assertArrayEquals(expected, output);
102 
103     delimiter = "c;";
104     expected[2] = String.format(template, delimiter);
105     output = StringEscaper.split(String.join(delimiter, input), delimiter, 3);
106     assertArrayEquals(expected, output);
107 
108     delimiter = "\n\n.;|";
109     expected[2] = String.format(template, delimiter);
110     output = StringEscaper.split(String.join(delimiter, input), delimiter, 3);
111     assertArrayEquals(expected, output);
112   }
