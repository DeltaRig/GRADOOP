137   /**
138    * Test writing and reading a graph with different labels that result in the same
139    * indexed CSV path because of replaced illegal filename characters in windows.
140    *
141    * @throws Exception if the execution or IO fails.
142    */
143   @Test
144   public void testDifferentLabelsInSameFile() throws Exception {
145     String tmpPath = temporaryFolder.getRoot().getPath();
146     ExecutionEnvironment env = getExecutionEnvironment();
147 
148     // This results in the path "graphs/a_b" because < and > are illegal filename characters.
149     EPGMGraphHead graphHead1 = new EPGMGraphHeadFactory().createGraphHead("a<b");
150     EPGMGraphHead graphHead2 = new EPGMGraphHeadFactory().createGraphHead("a>b");
151     DataSet<EPGMGraphHead> graphHeads = env.fromElements(graphHead1, graphHead2);
152 
153     // This results in the path "vertices/b_c" because < and > are illegal filename characters.
154     EPGMVertex vertex1 = new EPGMVertexFactory().createVertex("B<C");
155     EPGMVertex vertex2 = new EPGMVertexFactory().createVertex("B>C");
156     DataSet<EPGMVertex> vertices = env.fromElements(vertex1, vertex2)
157       .map(new AddToGraph<>(graphHead1))
158       .map(new AddToGraph<>(graphHead2))
159       .withForwardedFields("id;label;properties");
160 
161     // This results in the path "edges/c_d" because < and > are illegal filename characters.
162     EPGMEdge edge1 = new EPGMEdgeFactory().createEdge("c<d", vertex1.getId(), vertex2.getId());
163     EPGMEdge edge2 = new EPGMEdgeFactory().createEdge("c>d", vertex2.getId(), vertex1.getId());
164     DataSet<EPGMEdge> edges = env.fromElements(edge1, edge2)
165       .map(new AddToGraph<>(graphHead1))
166       .withForwardedFields("id;label;properties");
167 
168     LogicalGraph graph = getConfig().getLogicalGraphFactory()
169       .fromDataSets(graphHeads, vertices, edges);
170 
171     checkIndexedCSVWrite(tmpPath, graph);
172   }

174   /**
175    * Test IndexedCSVDataSink to escape strings and labels that contain delimiter characters.
176    * Escape characters get inserted before delimiter characters that are part of strings or labels.
177    * Whitespace gets replaced by their control character sequence.
178    *
179    * @throws Exception if the execution or IO fails.
180    */
181   @Test
182   public void testWriteWithDelimiterCharacters() throws Exception {
183     String tmpPath = temporaryFolder.getRoot().getPath();
184     ExecutionEnvironment env = getExecutionEnvironment();
185 
186     String string1 = "abc;,|:\n=\\ def";
187     String string2 = "def;,|:\n=\\ ghi";
188 
189     List<PropertyValue> list = Arrays.asList(PropertyValue.create(string2), PropertyValue.create(string1));
190     Set<PropertyValue> set = new HashSet<>(list);
191 
192     Map<PropertyValue, PropertyValue> map1 = new HashMap<>();
193     map1.put(PropertyValue.create(string1), PropertyValue.create(string2));
194     map1.put(PropertyValue.create("key"), PropertyValue.create(string1));
195 
196     Map<PropertyValue, PropertyValue> map2 = new HashMap<>();
197     map2.put(PropertyValue.create(string1), PropertyValue.create(1));
198     map2.put(PropertyValue.create("key"), PropertyValue.create(2));
199 
200     Map<PropertyValue, PropertyValue> map3 = new HashMap<>();
201     map3.put(PropertyValue.create(1), PropertyValue.create(string2));
202     map3.put(PropertyValue.create(2), PropertyValue.create(string1));
203 
204     Properties props = Properties.create();
205     props.set(string1, string2);
206     props.set(string2, GradoopTestUtils.BOOL_VAL_1);
207     props.set(GradoopTestUtils.KEY_2, string2);
208     props.set(GradoopTestUtils.KEY_3, list);
209     props.set(GradoopTestUtils.KEY_4, set);
210     props.set(GradoopTestUtils.KEY_5, map1);
211     props.set(GradoopTestUtils.KEY_5, map2);
212     props.set(GradoopTestUtils.KEY_6, map3);
213 
214     EPGMGraphHead graphHead = new EPGMGraphHeadFactory().createGraphHead(string1, props);
215     DataSet<EPGMGraphHead> graphHeads = env.fromElements(graphHead);
216 
217     EPGMVertex vertex = new EPGMVertexFactory().createVertex(string1, props);
218     DataSet<EPGMVertex> vertices = env.fromElements(vertex)
219       .map(new AddToGraph<>(graphHead))
220       .withForwardedFields("id;label;properties");
221 
222     EPGMEdge edge = new EPGMEdgeFactory().createEdge(string1, vertex.getId(), vertex.getId(), props);
223     DataSet<EPGMEdge> edges = env.fromElements(edge)
224       .map(new AddToGraph<>(graphHead))
225       .withForwardedFields("id;label;properties");
226 
227     LogicalGraph graph = getConfig().getLogicalGraphFactory()
228       .fromDataSets(graphHeads, vertices, edges);
229 
230     checkIndexedCSVWrite(tmpPath, graph);
231   }
