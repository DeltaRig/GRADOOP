123   @Test
124   public void testStringEscaping() {
125     EPGMGraphHead graphHead = new EPGMGraphHeadFactory()
126             .initGraphHead(GradoopId.fromString("aaaaaaaaaaaaaaaaaaaaaaaa"), "graph");
127 
128     String key = "Title";
129     String value = "Why was Wojo occasionally known to say \"I'm too lazy!\"?";
130 
131     // init vertex 1
132     Map<String, Object> propertiesMap1 = new HashMap<>();
133     propertiesMap1.put(key, value);
134     GradoopId idVertex1 = GradoopId.fromString("bbbbbbbbbbbbbbbbbbbbbbbb");
135     EPGMVertex vertex1 = new EPGMVertexFactory().initVertex(
136             idVertex1, "weird\nlabe\"l", Properties.createFromMap(propertiesMap1));
137 
138     // create vertex set
139     Set<EPGMVertex> vertices = new HashSet<>();
140     vertices.add(vertex1);
141     // create empty edge set
142     Set<EPGMEdge> edges = new HashSet<>();
143 
144     GraphTransaction transactionMock = mock(GraphTransaction.class);
145     when(transactionMock.getGraphHead()).thenReturn(graphHead);
146     when(transactionMock.getVertices()).thenReturn(vertices);
147     when(transactionMock.getEdges()).thenReturn(edges);
148 
149     DotFileFormatSimple dotFileFormatSimple = new DotFileFormatSimple(true);
150 
151     String expected = "subgraph cluster_gaaaaaaaaaaaaaaaaaaaaaaaa{\n" +
152             "label=\"graph\";\n" +
153             "vbbbbbbbbbbbbbbbbbbbbbbbbaaaaaaaaaaaaaaaaaaaaaaaa [label=\"weird\\nlabe\\\"l\"," +
154             "Title=\"Why was Wojo occasionally known to say \\\"I'm too lazy!\\\"?\"];\n" +
155             "}\n";
156 
157     assertEquals(expected, dotFileFormatSimple.format(transactionMock));
158   }
