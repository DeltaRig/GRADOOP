34    /**
35     * Test getting a logical graph by variable from the loader
36     *
37     * @throws Exception on failure
38     */
39    @Test
40    public void testGetLogicalGraphByVariable() throws Exception {
41      FlinkAsciiGraphLoader loader = getSocialNetworkLoader();
42  
43      String graphVariable = "g1";
44      GradoopId graphId = loader.getGraphHeadByVariable(graphVariable).getId();
45  
46      LogicalGraph graphFromLoader = loader.getLogicalGraphByVariable(graphVariable);
47      LogicalGraph graphFromCollection = loader.getGraphCollection().getGraph(graphId);
48  
49      collectAndAssertTrue(graphFromLoader.equalsByElementData(graphFromCollection));
50    }

52    /**
53     * Test getting the database graph from the loader.
54     * This test will create a loader containing a single graph and check if the database graph
55     * is equal to that graph, i.e. if the graph head label was set correctly.
56     *
57     * @throws Exception if execute fails.
58     */
59    @Test
60    public void testGetLogicalGraph() throws Exception {
61      FlinkAsciiGraphLoader loader = getLoaderFromString("expected:" +
62        GradoopConstants.DB_GRAPH_LABEL +
63        "[" +
64        "(v1:TestVertex)-[e1:TestEdge]->(v2:TestVertex)" +
65        "]");
66      LogicalGraph databaseGraph = loader.getLogicalGraph();
67      LogicalGraph expectedGraph = loader.getLogicalGraphByVariable("expected");
68      collectAndAssertTrue(expectedGraph.equalsByData(databaseGraph));
69    }
70  
71    /**
72     * Test getting a graph collection from the loader
73     *
74     * @throws Exception on failure
75     */
76    @Test
77    public void testGetGraphCollection() throws Exception {
78      FlinkAsciiGraphLoader loader = getSocialNetworkLoader();
79  
80      GraphCollection collectionFromLoader =
81        loader.getGraphCollectionByVariables("g0", "g1", "g2", "g3");
82      GraphCollection collectionFromDb = loader.getGraphCollection();
83  
84      // heads <> heads
85      collectAndAssertTrue(collectionFromLoader.equalsByGraphIds(collectionFromDb));
86  
87      // elements <> elements
88      collectAndAssertTrue(collectionFromLoader.equalsByGraphElementIds(collectionFromDb));
89    }
90  
91    /**
92     * Test getting a graph collection by variables from the loader
93     *
94     * @throws Exception on failure
95     */
96    @Test
97    public void testGetGraphCollectionByVariables() throws Exception {
98      FlinkAsciiGraphLoader loader = getSocialNetworkLoader();
99  
100     String[] graphVariables = new String[]{"g0", "g1", "g2"};
101     List<GradoopId> graphIds = Lists.newArrayList();
102 
103     for (String graphVariable : graphVariables) {
104       graphIds.add(loader.getGraphHeadByVariable(graphVariable).getId());
105     }
106 
107     GradoopIdSet graphIdSet = GradoopIdSet.fromExisting(graphIds);
108 
109     GraphCollection collectionFromLoader = loader.getGraphCollectionByVariables(graphVariables);
110     GraphCollection collectionFromDbViaSet = loader.getGraphCollection().getGraphs(graphIdSet);
111 
112     // heads <> heads
113     collectAndAssertTrue(collectionFromLoader.equalsByGraphIds(collectionFromDbViaSet));
114 
115     // elements <> elements
116     collectAndAssertTrue(collectionFromLoader.equalsByGraphElementIds(collectionFromDbViaSet));
117   }
