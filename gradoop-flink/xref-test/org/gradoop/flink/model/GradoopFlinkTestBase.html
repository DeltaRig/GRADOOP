46  /**
47   * Base class for Flink-based unit tests with the same cluster.
48   */
49  public abstract class GradoopFlinkTestBase {
50  
51    private static final int DEFAULT_PARALLELISM = 4;
52  
53    private static final long TASKMANAGER_MEMORY_SIZE_MB = 512;
54  
55    @ClassRule
56    public static MiniClusterWithClientResource miniClusterResource = getMiniCluster();
57  
58    /**
59     * Flink Execution Environment
60     */
61    private ExecutionEnvironment env;
62  
63    /**
64     * Gradoop Flink configuration
65     */
66    private GradoopFlinkConfig config;
67  
68    /**
69     * The factory to create a logical graph layout.
70     */
71    private LogicalGraphLayoutFactory<EPGMGraphHead, EPGMVertex, EPGMEdge> graphLayoutFactory;
72  
73    /**
74     * The factory to create a graph collection layout.
75     */
76    private GraphCollectionLayoutFactory<EPGMGraphHead, EPGMVertex, EPGMEdge> collectionLayoutFactory;
77  
78    /**
79     * Creates a new instance of {@link GradoopFlinkTestBase}.
80     */
81    public GradoopFlinkTestBase() {
82      this.env = ExecutionEnvironment.getExecutionEnvironment();
83      this.graphLayoutFactory = new GVEGraphLayoutFactory();
84      this.collectionLayoutFactory = new GVECollectionLayoutFactory();
85    }
86  
87    /**
88     * Returns the execution environment for the tests
89     *
90     * @return Flink execution environment
91     */
92    protected ExecutionEnvironment getExecutionEnvironment() {
93      return env;
94    }
95  
96    /**
97     * Returns the configuration for the test
98     *
99     * @return Gradoop Flink configuration
100    */
101   protected GradoopFlinkConfig getConfig() {
102     if (config == null) {
103       setConfig(GradoopFlinkConfig.createConfig(getExecutionEnvironment(),
104         graphLayoutFactory,
105         collectionLayoutFactory));
106     }
107     return config;
108   }
109 
110   /**
111    * Sets the default configuration for the test
112    */
113   protected void setConfig(GradoopFlinkConfig config) {
114     this.config = config;
115   }
116 
117   protected void setCollectionLayoutFactory(
118     GraphCollectionLayoutFactory<EPGMGraphHead, EPGMVertex, EPGMEdge> collectionLayoutFactory) {
119     this.collectionLayoutFactory = collectionLayoutFactory;
120   }

122   //----------------------------------------------------------------------------
123   // Cluster related

125 
126   /**
127    * Custom test cluster start routine,
128    * workaround to set TASK_MANAGER_MEMORY_SIZE.
129    *
130    * TODO: remove, when future issue is fixed
131    * {@see http://mail-archives.apache.org/mod_mbox/flink-dev/201511.mbox/%3CCAC27z=PmPMeaiNkrkoxNFzoR26BOOMaVMghkh1KLJFW4oxmUmw@mail.gmail.com%3E}
132    */
133   private static MiniClusterWithClientResource getMiniCluster() {
134     Configuration config = new Configuration();
135     config.setLong("taskmanager.memory.size", TASKMANAGER_MEMORY_SIZE_MB);
136 
137     return new MiniClusterWithClientResource(
138       new MiniClusterResourceConfiguration.Builder()
139         .setNumberTaskManagers(1)
140         .setNumberSlotsPerTaskManager(DEFAULT_PARALLELISM)
141         .setConfiguration(config).build());
142   }
143

148   protected FlinkAsciiGraphLoader getLoaderFromString(String asciiString) {
149     FlinkAsciiGraphLoader loader = getNewLoader();
150     loader.initDatabaseFromString(asciiString);
151     return loader;
152   }
153 
154   protected FlinkAsciiGraphLoader getLoaderFromFile(String fileName) throws IOException {
155     FlinkAsciiGraphLoader loader = getNewLoader();
156 
157     loader.initDatabaseFromFile(fileName);
158     return loader;
159   }
160 
161   protected FlinkAsciiGraphLoader getLoaderFromStream(InputStream inputStream) throws IOException {
162     FlinkAsciiGraphLoader loader = getNewLoader();
163 
164     loader.initDatabaseFromStream(inputStream);
165     return loader;
166   }

168   /**
169    * Creates a social network as a basis for tests.
170    * <p/>
171    * An image of the network can be found in
172    * gradoop/dev-support/social-network.pdf
173    *
174    * @return graph store containing a simple social network for tests.
175    */
176   protected FlinkAsciiGraphLoader getSocialNetworkLoader() throws IOException {
177     InputStream inputStream = getClass()
178       .getResourceAsStream(GradoopTestUtils.SOCIAL_NETWORK_GDL_FILE);
179     return getLoaderFromStream(inputStream);
180   }

182   /**
183    * Returns an uninitialized loader with the test config.
184    *
185    * @return uninitialized Flink Ascii graph loader
186    */
187   private FlinkAsciiGraphLoader getNewLoader() {
188     return new FlinkAsciiGraphLoader(getConfig());
189   }

195   protected void collectAndAssertTrue(DataSet<Boolean> result) throws Exception {
196     assertTrue("expected true", result.collect().get(0));
197   }
198 
199   protected void collectAndAssertFalse(DataSet<Boolean> result) throws Exception {
200     assertFalse("expected false", result.collect().get(0));
201   }

203   protected <T extends Element> DataSet<T> getEmptyDataSet(T dummy) {
204     return getExecutionEnvironment()
205       .fromElements(dummy)
206       .filter(new False<>());
207   }

209   /**
210    * Returns the encoded file path to a resource.
211    *
212    * @param  relPath the relative path to the resource
213    * @return encoded file path
214    */
215   protected String getFilePath(String relPath) throws UnsupportedEncodingException {
216     return URLDecoder.decode(

218   }
