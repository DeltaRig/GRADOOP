168   @Test
169   public void testExecute() throws Exception {
170     GradoopId a = GradoopId.get();
171     GradoopId b = GradoopId.get();
172     GradoopId c = GradoopId.get();
173     GradoopId d = GradoopId.get();
174     GradoopId e = GradoopId.get();
175     GradoopId f = GradoopId.get();
176 
177     EmbeddingMetaData leftInputMetaData = new EmbeddingMetaData();
178     leftInputMetaData.setEntryColumn("v1", EntryType.VERTEX, 0);
179     leftInputMetaData.setPropertyColumn("v1", "age", 0);
180 
181     Embedding embedding1 = createEmbedding(singletonList(Pair.of(a, singletonList(42))));
182     Embedding embedding2 = createEmbedding(singletonList(Pair.of(b, singletonList(23))));
183 
184     DataSet<Embedding> leftEmbeddings = getExecutionEnvironment()
185       .fromElements(embedding1, embedding2);
186 
187     /*
188      * ----------------------------------
189      * |  v1   | e1    | v2    | v2.age |
190      * ----------------------------------
191      * | id(a) | id(c) | id(e) |  84    | -> Embedding 3
192      * ----------------------------------
193      * | id(b) | id(d) | id(f) |  77    | -> Embedding 4
194      * ----------------------------------
195      */
196     EmbeddingMetaData rightInputMetaData = new EmbeddingMetaData();
197     rightInputMetaData.setEntryColumn("v1", EntryType.VERTEX, 0);
198     rightInputMetaData.setEntryColumn("e1", EntryType.EDGE, 1);
199     rightInputMetaData.setEntryColumn("v2", EntryType.VERTEX, 2);
200     rightInputMetaData.setPropertyColumn("v2", "age", 0);
201 
202     Embedding embedding3 = createEmbedding(asList(
203       Pair.of(a, emptyList()),
204       Pair.of(c, emptyList()),
205       Pair.of(e, singletonList(84))
206     ));
207     Embedding embedding4 = createEmbedding(asList(
208       Pair.of(b, emptyList()),
209       Pair.of(d, emptyList()),
210       Pair.of(f, singletonList(77))
211     ));
212 
213     DataSet<Embedding> rightEmbeddings = getExecutionEnvironment()
214       .fromElements(embedding3, embedding4);
215 
216     MockPlanNode leftChild = new MockPlanNode(leftEmbeddings, leftInputMetaData);
217     MockPlanNode rightChild = new MockPlanNode(rightEmbeddings, rightInputMetaData);
218 
219     JoinEmbeddingsNode node = new JoinEmbeddingsNode(leftChild, rightChild,
220       singletonList("v1"), MatchStrategy.ISOMORPHISM, MatchStrategy.ISOMORPHISM);
221 
222     List<Embedding> result = node.execute().collect();
223     result.sort(Comparator.comparing(o -> o.getProperty(0))); // sort by property value in column 0
224 
225     assertEquals(2, result.size());
226 
227     assertEmbedding(result.get(0), asList(b, d, f), asList(PropertyValue.create(23), PropertyValue.create(77)));
228     assertEmbedding(result.get(1), asList(a, c, e), asList(PropertyValue.create(42), PropertyValue.create(84)));
229   }
