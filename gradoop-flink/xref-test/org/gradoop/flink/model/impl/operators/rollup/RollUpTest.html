38    /**
39     * Executes a rollUp on vertices using a single grouping key and checks if the result
40     * is correct.
41     *
42     * @throws Exception If the execution fails.
43     */
44    @Test
45    public void testVertexRollUpWithSingleGroupingKey() throws Exception {
46      FlinkAsciiGraphLoader loader = getSocialNetworkLoader();
47  
48      LogicalGraph input = loader.getGraphCollectionByVariables("g0", "g1", "g2")
49        .reduce(new ReduceCombination<>());
50  
51      //expected
52      loader.initDatabaseFromString("g0 {vertexRollUpGroupingKeys:\"age\"}[" +
53        "(v0 {count:1L,age:20})" +
54        "(v1 {count:2L,age:30})" +
55        "(v2 {count:2L,age:35})" +
56        "(v3 {count:1L,age:40})" +
57        "(v2)-[e0]->(v0)" +
58        "(v1)-[e1]->(v0)" +
59        "(v3)-[e2]->(v1)" +
60        "(v0)-[e3]->(v1)" +
61        "(v2)-[e4]->(v1)" +
62        "(v2)-[e5]->(v3)" +
63        "(v1)-[e6]->(v1)" +
64        "(v1)-[e7]->(v3)" +
65        "]");
66      GraphCollection expected = loader.getGraphCollection();
67  
68      List<String> vertexGK = Collections.singletonList("age");
69      List<AggregateFunction> vertexAGG =
70        Collections.singletonList(new Count("count"));
71      List<String> edgeGK = Collections.emptyList();
72      List<AggregateFunction> edgeAGG = Collections.emptyList();
73  
74      GraphCollection output = input.groupVerticesByRollUp(vertexGK, vertexAGG, edgeGK, edgeAGG);
75  
76      collectAndAssertTrue(output.equalsByGraphData(expected));
77    }
