90        "(v01:Forum {count : 1L,topic : \"graph\"})" +
91        "(v02:User  {count : 3L,gender : \"male\", max : 30})" +
92        "(v03:User  {count : 1L,gender : \"female\", max : 20})" +
93        "(v04:UserAge  {count : 3L,age : 20, sum : 60})" +
94        "(v05:UserAge  {count : 1L,age : 30, sum : 30})" +
95        "(v02)-->(v00)" +
96        "(v02)-->(v01)" +
97        "(v02)-->(v02)" +
98        "(v02)-->(v03)" +
99        "(v02)-->(v04)" +
100       "(v02)-->(v05)" +
101       "(v03)-->(v01)" +
102       "(v03)-->(v02)" +
103       "(v03)-->(v05)" +
104       "(v04)-->(v00)" +
105       "(v04)-->(v01)" +
106       "(v04)-->(v02)" +
107       "(v04)-->(v03)" +
108       "(v04)-->(v04)" +
109       "(v04)-->(v05)" +
110       "(v05)-->(v01)" +
111       "]");
112 
113     LogicalGraph output = new Grouping.GroupingBuilder()
114       .useVertexLabel(true)
115       .addVertexGroupingKey("topic")
116       .addVertexLabelGroup("User", Lists.newArrayList("gender"),
117         Lists.newArrayList(new Count("count"), new MaxProperty("age", "max")))
118       .addVertexLabelGroup("User", "UserAge", Lists.newArrayList("age"),
119         Lists.newArrayList(new Count("count"), new SumProperty("age", "sum")))
120       .addVertexAggregateFunction(new Count("count"))
121       .setStrategy(getStrategy())
122       .<EPGMGraphHead, EPGMVertex, EPGMEdge, LogicalGraph, GraphCollection>build()
123       .execute(input);
124 
125     collectAndAssertTrue(
126       output.equalsByElementData(loader.getLogicalGraphByVariable("expected")));
127   }
