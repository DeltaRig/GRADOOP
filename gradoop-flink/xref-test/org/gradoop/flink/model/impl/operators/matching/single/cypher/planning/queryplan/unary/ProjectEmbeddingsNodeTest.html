37    @Test
38    public void testMetaDataInitialization() throws Exception {
39      EmbeddingMetaData inputMetaData = new EmbeddingMetaData();
40      inputMetaData.setEntryColumn("a", EntryType.VERTEX, 0);
41      inputMetaData.setEntryColumn("b", EntryType.VERTEX, 1);
42      inputMetaData.setPropertyColumn("a", "age", 0);
43      inputMetaData.setPropertyColumn("b", "name", 1);
44  
45      PlanNode mockNode = new MockPlanNode(null, inputMetaData);
46      List<Pair<String, String>> projectedKeys = new ArrayList<>();
47      projectedKeys.add(Pair.of("a", "age"));
48  
49      ProjectEmbeddingsNode node = new ProjectEmbeddingsNode(mockNode, projectedKeys);
50      EmbeddingMetaData outputMetaData = node.getEmbeddingMetaData();
51      assertEquals(2, outputMetaData.getEntryCount());
52      assertEquals(0, outputMetaData.getEntryColumn("a"));
53      assertEquals(1, outputMetaData.getEntryColumn("b"));
54      assertEquals(1, outputMetaData.getPropertyCount());
55      assertEquals(0, outputMetaData.getPropertyColumn("a", "age"));
56    }

58    @Test
59    public void testExecute() throws Exception {
60      GradoopId vertexAId = GradoopId.get();
61      GradoopId vertexBId = GradoopId.get();
62  
63      EmbeddingMetaData metaData = new EmbeddingMetaData();
64      metaData.setEntryColumn("a", EntryType.VERTEX, 0);
65      metaData.setEntryColumn("b", EntryType.VERTEX, 1);
66      metaData.setPropertyColumn("a", "age", 0);
67      metaData.setPropertyColumn("b", "age", 1);
68  
69      Embedding embedding1 = new Embedding();
70      embedding1.add(vertexAId, PropertyValue.create(42));
71      embedding1.add(vertexBId, PropertyValue.create(23));
72  
73      DataSet<Embedding> input = getExecutionEnvironment().fromElements(embedding1);
74      PlanNode mockChild = new MockPlanNode(input, metaData);
75  
76      List<Pair<String, String>> projectedKeys = new ArrayList<>();
77      projectedKeys.add(Pair.of("a", "age"));
78  
79      List<Embedding> result = new ProjectEmbeddingsNode(mockChild, projectedKeys).execute().collect();
80  
81      assertEquals(1, result.size());
82      assertEquals(result.get(0).getId(0), vertexAId);
83      assertEquals(result.get(0).getId(1), vertexBId);
84      assertEquals(1, result.get(0).getProperties().size());
85      assertEquals(result.get(0).getProperty(0), PropertyValue.create(42));
86    }
