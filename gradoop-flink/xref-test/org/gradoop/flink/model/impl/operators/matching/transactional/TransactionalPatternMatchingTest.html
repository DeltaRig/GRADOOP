126       Collection<EPGMGraphHead> resultHeads = Lists.newArrayList();

128       coll.getGraphHeads().output(new LocalCollectionOutputFormat<>(

130       result.getGraphHeads().output(new LocalCollectionOutputFormat<>(
131         resultHeads
132       ));
133 
134       getExecutionEnvironment().execute();
135 
136       Map<GradoopId, String> lineageIdMap = new HashMap<>();
137 
138       for (EPGMGraphHead original : originalHeads) {
139         lineageIdMap.put(original.getId(), original.getLabel());
140       }
141 
142       int aCount = 0;
143       int bCount = 0;
144       int cCount = 0;
145       int dCount = 0;
146       int eCount = 0;
147       for (EPGMGraphHead head : resultHeads) {
148         GradoopId id = head.getPropertyValue("lineage").getGradoopId();
149         if (lineageIdMap.get(id).equals("A")) {
150           aCount++;
151         }
152         if (lineageIdMap.get(id).equals("B")) {
153           bCount++;
154         }
155         if (lineageIdMap.get(id).equals("C")) {
156           cCount++;
157         }
158         if (lineageIdMap.get(id).equals("D")) {
159           dCount++;
160         }
161         if (lineageIdMap.get(id).equals("E")) {
162           eCount++;
163         }
