35    @Test
36    public void testVertexLabelSpecific() throws Exception {
37      FlinkAsciiGraphLoader loader = getLoaderFromString(getLabelSpecificInput());
38  
39      LogicalGraph input = loader.getLogicalGraphByVariable("input");
40  
41      loader.appendToDatabaseFromString("expected[" +
42        "(v00:Forum {topic : \"rdf\"})" +
43        "(v01:Forum {topic : \"graph\"})" +
44        "(v02:User  {gender : \"male\"})" +
45        "(v03:User  {gender : \"female\"})" +
46        "(v02)-->(v00)" +
47        "(v02)-->(v01)" +
48        "(v02)-->(v02)" +
49        "(v02)-->(v03)" +
50        "(v03)-->(v01)" +
51        "(v03)-->(v02)" +
52        "]");
53  
54      LogicalGraph output = new Grouping.GroupingBuilder()
55        .useVertexLabel(true)
56        .addVertexGroupingKey("topic")
57        .addVertexLabelGroup("User", Lists.newArrayList("gender"))
58        .setStrategy(getStrategy())
59        .<EPGMGraphHead, EPGMVertex, EPGMEdge, LogicalGraph, GraphCollection>build()
60        .execute(input);
61  
62      collectAndAssertTrue(
63        output.equalsByElementData(loader.getLogicalGraphByVariable("expected")));
64    }
