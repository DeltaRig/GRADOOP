35    public void testRepulseJoinFunction() throws Exception {
36      JoinFunction<LVertex, LVertex, Force> jf = new FRRepulsionFunction(1, 20);
37      LVertex v1 = getDummyVertex(1, 1);
38      LVertex v2 = getDummyVertex(2, 3);
39      LVertex v3 = getDummyVertex(7, 5);
40      LVertex v4 = getDummyVertex(1, 1);
41      LVertex v5 = getDummyVertex(30, 30);
42  
43      Vector vec12 = jf.join(v1, v2).getValue().copy();
44      Vector vec13 = jf.join(v1, v3).getValue().copy();
45      Vector vec14 = jf.join(v1, v4).getValue().copy();
46      Vector vec11 = jf.join(v1, v1).getValue().copy();
47      Vector vec15 = jf.join(v1, v5).getValue().copy();
48  
49      Assert.assertTrue(vec12.getX() < 0 && vec12.getY() < 0);
50      Assert.assertTrue(vec12.magnitude() > vec13.magnitude());
51      Assert.assertTrue(vec14.magnitude() > 0);
52      Assert.assertTrue(vec11.magnitude() == 0);
53      Assert.assertTrue(vec15.magnitude() == 0);
54    }

56    @Test
57    public void testRepulseFlatJoin() throws Exception {
58      FRRepulsionFunction jf = new FRRepulsionFunction(1);
59      LVertex v1 = getDummyVertex(1, 1);
60      LVertex v2 = getDummyVertex(2, 3);
61      Vector vec12join = jf.join(v1, v2).getValue().copy();
62  
63      v1 = getDummyVertex(1, 1);
64      v2 = getDummyVertex(2, 3);
65      List<Force> collectorList = new ArrayList<>();
66      ListCollector<Force> collector = new ListCollector<>(collectorList);
67      jf.join(v1, v2, collector);
68  
69      Vector vec12 = collectorList.get(0).getValue().copy();
70      Vector vec21 = collectorList.get(1).getValue().copy();
71  
72      v1 = getDummyVertex(1, 1);
73      v2 = getDummyVertex(2, 3);
74      v1.setSubVertices(generateSubVertices(1));
75      v2.setSubVertices(generateSubVertices(2));
76      Vector vec12joinf6 = jf.join(v1, v2).getValue().copy();
77  
78      Assert.assertEquals(vec12join, vec12);
79      Assert.assertEquals(vec12, vec21.mul(-1));
80      Assert.assertNotEquals(collectorList.get(0).getId(), collectorList.get(1).getId());
81      Assert.assertEquals(vec12join.mul(6), vec12joinf6);
82    }
