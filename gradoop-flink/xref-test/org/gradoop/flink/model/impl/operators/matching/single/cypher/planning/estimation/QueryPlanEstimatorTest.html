91        "n", "e", "m",
92        queryHandler.getPredicates().getSubCNF("e"), Sets.newHashSet(), false);

94        "m", "f", "o",
95        queryHandler.getPredicates().getSubCNF("f"), Sets.newHashSet(), false);
96  
97      // (n)-[e]->
98      JoinEmbeddingsNode neJoin = new JoinEmbeddingsNode(nNode, eNode, Lists.newArrayList("n"),
99        MatchStrategy.ISOMORPHISM, MatchStrategy.ISOMORPHISM);

101     QueryPlan plan = new QueryPlan(neJoin);
102     QueryPlanEstimator estimator = new QueryPlanEstimator(plan, queryHandler, STATS);
103 
104     assertEquals(24L, estimator.getCardinality());
105 
106     // (n)-[e]->(m)
107     JoinEmbeddingsNode nemJoin = new JoinEmbeddingsNode(neJoin, mNode, Lists.newArrayList("m"),
108       MatchStrategy.ISOMORPHISM, MatchStrategy.ISOMORPHISM);
109 
110     plan = new QueryPlan(nemJoin);
111     estimator = new QueryPlanEstimator(plan, queryHandler, STATS);
112 
113     assertEquals(24L, estimator.getCardinality());
114 
115     // (n)-[e]->(m)-[f]->
116     JoinEmbeddingsNode nemfJoin = new JoinEmbeddingsNode(nemJoin, fNode, Lists.newArrayList("m"),
117       MatchStrategy.ISOMORPHISM, MatchStrategy.ISOMORPHISM);
118

120     estimator = new QueryPlanEstimator(plan, queryHandler, STATS);
121 
122     assertEquals(72L, estimator.getCardinality());
123 
124     // (n)-[e]->(m)-[f]->(o)
125     JoinEmbeddingsNode nemfoJoin = new JoinEmbeddingsNode(nemfJoin, oNode, Lists.newArrayList("o"),
126       MatchStrategy.ISOMORPHISM, MatchStrategy.ISOMORPHISM);
127

189   @Test
190   public void testComplexCartesianProduct() throws Exception {
191     String query = "MATCH (a:Person)-[e1:knows]->(b:Person),(c:Forum)-[e2:hasTag]->(d:Tag)";
192     QueryHandler queryHandler = new QueryHandler(query);
193 
194     LeafNode aNode = new FilterAndProjectVerticesNode(null, "a",
195       queryHandler.getPredicates().getSubCNF("a"), Sets.newHashSet());
196     LeafNode bNode = new FilterAndProjectVerticesNode(null, "b",
197       queryHandler.getPredicates().getSubCNF("b"), Sets.newHashSet());
198     LeafNode cNode = new FilterAndProjectVerticesNode(null, "c",
199       queryHandler.getPredicates().getSubCNF("c"), Sets.newHashSet());
200     LeafNode dNode = new FilterAndProjectVerticesNode(null, "d",
201       queryHandler.getPredicates().getSubCNF("d"), Sets.newHashSet());
202 
203     LeafNode e1Node = new FilterAndProjectEdgesNode(null, "a", "e1", "b",
204       queryHandler.getPredicates().getSubCNF("e1"), Sets.newHashSet(), false);
205     LeafNode e2Node = new FilterAndProjectEdgesNode(null, "c", "e2", "d",
206       queryHandler.getPredicates().getSubCNF("e2"), Sets.newHashSet(), false);
207 
208     BinaryNode ae1 = new JoinEmbeddingsNode(aNode, e1Node, Lists.newArrayList("a"),
209       MatchStrategy.HOMOMORPHISM, MatchStrategy.HOMOMORPHISM);
210     BinaryNode ae1b = new JoinEmbeddingsNode(ae1, bNode, Lists.newArrayList("b"),
211       MatchStrategy.HOMOMORPHISM, MatchStrategy.HOMOMORPHISM);
212     BinaryNode ce2 = new JoinEmbeddingsNode(cNode, e2Node, Lists.newArrayList("c"),
213       MatchStrategy.HOMOMORPHISM, MatchStrategy.HOMOMORPHISM);
214     BinaryNode ce2d = new JoinEmbeddingsNode(ce2, dNode, Lists.newArrayList("d"),
215       MatchStrategy.HOMOMORPHISM, MatchStrategy.HOMOMORPHISM);
216 
217     CartesianProductNode crossNode = new CartesianProductNode(ae1b, ce2d, MatchStrategy
218       .HOMOMORPHISM, MatchStrategy.HOMOMORPHISM);
219 
220     QueryPlan ae1bPlan = new QueryPlan(ae1b);
221     QueryPlan ce2dPlan = new QueryPlan(ce2d);
222     QueryPlan crossPlan = new QueryPlan(crossNode);
223 
224     QueryPlanEstimator ae1bEstimator = new QueryPlanEstimator(ae1bPlan, queryHandler, STATS);
225     QueryPlanEstimator ce2dEstimator = new QueryPlanEstimator(ce2dPlan, queryHandler, STATS);
226     QueryPlanEstimator crossEstimator = new QueryPlanEstimator(crossPlan, queryHandler, STATS);
227 
228     assertEquals(ae1bEstimator.getCardinality() * ce2dEstimator.getCardinality(), crossEstimator.getCardinality());
229   }
