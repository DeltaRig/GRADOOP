270 
271     Properties properties = Properties.create();
272     properties.set("name", "Alice");
273     properties.set("age", 25);
274     EPGMVertex a = vertexFactory.createVertex("Person", properties);

277     properties.set("name", "Bob");
278     properties.set("age", 24);
279     EPGMVertex b = vertexFactory.createVertex("Person", properties);

281     properties = Properties.create();
282     properties.set("since", 2013);
283     properties.set("active", true);

285 
286 
287     DataSet<Triple> triples = getExecutionEnvironment().fromElements(new Triple(a, e, b));
288 
289     HashMap<String, List<String>> propertyKeys = new HashMap<>();
290     propertyKeys.put("a", Lists.newArrayList("name", "age"));
291     propertyKeys.put("e", Lists.newArrayList("since"));
292     propertyKeys.put("b", Lists.newArrayList("name"));
293 
294     Embedding result = new FilterAndProjectTriples(triples,
295       "a", "e", "b",
296       predicates, propertyKeys, MatchStrategy.ISOMORPHISM
297     ).evaluate().collect().get(0);
298 
299     assertEquals(result.getProperty(0), PropertyValue.create("Alice"));
300     assertEquals(result.getProperty(1), PropertyValue.create(25));
301     assertEquals(result.getProperty(2), PropertyValue.create(2013));
302     assertEquals(result.getProperty(3), PropertyValue.create("Bob"));
303   }
