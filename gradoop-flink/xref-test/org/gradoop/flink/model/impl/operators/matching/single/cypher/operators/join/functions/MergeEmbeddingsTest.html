492   /**
493    * Creates the input datasets, performs the join and validates the expected result.
494    *
495    * @param distinctVertexColumnsLeft join operator argument
496    * @param distinctEdgeColumnsLeft   join operator argument
497    * @param expectedEmbedding         expected result
498    * @throws Exception on failure
499    */
500   private void testMorphisms(List<Integer> distinctVertexColumnsLeft, List<Integer> distinctVertexColumnsRight,
501     List<Integer> distinctEdgeColumnsLeft, List<Integer> distinctEdgeColumnsRight, List<Embedding> expectedEmbedding) throws Exception {
502 
503     List<Embedding> entries = new ArrayList<>();
504 
505     entries.add(createEmbedding(vertex0, edge0, vertex1));
506     entries.add(createEmbedding(vertex0, edge1, vertex1));
507     entries.add(createEmbedding(vertex0, edge2, vertex0));
508     entries.add(createEmbedding(vertex0, edge3, vertex0));
509 
510     // merge operator
511     MergeEmbeddings op = new MergeEmbeddings(3, Arrays.asList(0, 2), distinctVertexColumnsLeft,
512       distinctVertexColumnsRight, distinctEdgeColumnsLeft, distinctEdgeColumnsRight);
513 
514     ArrayList<Embedding> resultList = new ArrayList<>();
515 
516     // get results
517     for (Embedding left : entries) {
518       for (Embedding right : entries) {
519         if (left.getId(0).equals(right.getId(0)) && left.getId(2).equals(right.getId(2))) {
520           ArrayList<Embedding> tmp = new ArrayList<>();
521           ListCollector<Embedding> collector = new ListCollector<>(tmp);
522           op.join(left, right, collector);
523 
524           resultList.addAll(tmp.stream().map(Embedding::copy).collect(Collectors.toList()));
525         }
526       }
527     }
528 
529     resultList.sort(new EmbeddingComparator());
530     expectedEmbedding.sort(new EmbeddingComparator());
531 
532     // test list equality
533     assertEquals(expectedEmbedding, resultList);
534   }

540   private static class EmbeddingComparator implements Comparator<Embedding> {
541 
542     @Override
543     public int compare(Embedding o1, Embedding o2) {
544       for (int i = 0; i < o1.size(); i++) {
545         if (o1.getId(i).compareTo(o2.getId(i)) < 0) {
546           return -1;
547         } else if (o1.getId(i).compareTo(o2.getId(i)) > 0) {
548           return 1;
549         }
550       }
551       return 0;
552     }
