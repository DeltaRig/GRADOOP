51    /**
52     * Test the verify operator with a subgraph operator.
53     *
54     * @throws Exception when the execution in Flink fails.
55     */
56    @Test
57    public void testVerifyWithSubgraph() throws Exception {
58      FlinkAsciiGraphLoader loader = getSocialNetworkLoader();
59      loader.appendToDatabaseFromString("expected[" +
60        "(eve)-[ekb]->(bob)" +
61        "]");
62      LogicalGraph input = loader.getLogicalGraphByVariable("g0");
63  
64      // Apply a subgraph operator that would result in dangling edges.
65      LogicalGraph subgraph = input.subgraph(
66        new ByProperty<EPGMVertex>("name", PropertyValue.create("Alice")).negate(),
67        new True<>(), Subgraph.Strategy.BOTH);
68  
69      // Make sure that the graph contains dangling edges.
70      List<EPGMEdge> danglingEdges = getDanglingEdges(subgraph);
71      List<EPGMEdge> expectedDanglingEdges = Arrays.asList(loader.getEdgeByVariable("eka"),
72        loader.getEdgeByVariable("akb"), loader.getEdgeByVariable("bka"));
73      Comparator<EPGMEdge> comparator = Comparator.comparing(EPGMEdge::getId);
74      danglingEdges.sort(comparator);
75      expectedDanglingEdges.sort(comparator);
76      assertArrayEquals(expectedDanglingEdges.toArray(), danglingEdges.toArray());
77  
78      // Now run verify and check if those edges were removed.
79      LogicalGraph verifiedSubgraph = subgraph.verify();
80      assertEquals("Verified graph contained dangling edges.",
81        0, getDanglingEdges(verifiedSubgraph).size());
82  
83      // Check if nothing else has been removed (i.e. the result is correct)
84      collectAndAssertTrue(loader.getLogicalGraphByVariable("expected")
85        .equalsByElementData(verifiedSubgraph));
86    }

93    @Test
94    public void testVerifyWithApplySubgraph() throws Exception {
95      FlinkAsciiGraphLoader loader = getSocialNetworkLoader();
96      loader.appendToDatabaseFromString("expected0[" +
97        "(eve)-[ekb]->(bob)" +
98        "] expected1 [" +
99        "(frank)-[fkd]->(dave)" +
100       "]");
101     GraphCollection input = loader.getGraphCollectionByVariables("g0", "g1");
102     GradoopId id0 = loader.getGraphHeadByVariable("g0").getId();
103     GradoopId id1 = loader.getGraphHeadByVariable("g1").getId();
104 
105     // Apply a subgraph operator that would result in dangling edges.
106     GraphCollection subgraph = input.apply(new ApplySubgraph<>(
107       new ByProperty<EPGMVertex>("name", PropertyValue.create("Alice"))
108         .or(new ByProperty<>("name", PropertyValue.create("Carol")))
109         .negate(),
110       new True<>(), Subgraph.Strategy.BOTH));
111 
112     // Make sure that the graphs contain dangling edges.
113     Comparator<EPGMEdge> comparator = Comparator.comparing(EPGMEdge::getId);
114     List<EPGMEdge> danglingEdges0 = getDanglingEdges(subgraph.getGraph(id0));
115     List<EPGMEdge> expectedDanglingEdges0 = Arrays.asList(loader.getEdgeByVariable("eka"),
116       loader.getEdgeByVariable("akb"), loader.getEdgeByVariable("bka"));
117     danglingEdges0.sort(comparator);
118     expectedDanglingEdges0.sort(comparator);
119     assertArrayEquals(expectedDanglingEdges0.toArray(), danglingEdges0.toArray());
120 
121     List<EPGMEdge> danglingEdges1 = getDanglingEdges(subgraph.getGraph(id1));
122     List<EPGMEdge> expectedDanglingEdges1 = Arrays.asList(loader.getEdgeByVariable("fkc"),
123       loader.getEdgeByVariable("ckd"), loader.getEdgeByVariable("dkc"));
124     danglingEdges1.sort(comparator);
125     expectedDanglingEdges1.sort(comparator);
126     assertArrayEquals(expectedDanglingEdges1.toArray(), danglingEdges1.toArray());
127 
128     // Now run verify and check if those edges were removed.
129     GraphCollection verifiedSubgraph = subgraph.verify();
130 
131     assertEquals("Verified graph contained dangling edges.",
132       0, getDanglingEdges(verifiedSubgraph.getGraph(id0)).size());
133     assertEquals("Verified graph contained dangling edges.",
134       0, getDanglingEdges(verifiedSubgraph.getGraph(id1)).size());
135 
136     // Check if nothing else has been removed (i.e. the result is correct)
137     collectAndAssertTrue(loader.getLogicalGraphByVariable("expected0")
138       .equalsByElementData(verifiedSubgraph.getGraph(id0)));
139     collectAndAssertTrue(loader.getLogicalGraphByVariable("expected1")
140       .equalsByElementData(verifiedSubgraph.getGraph(id1)));
141   }

143   /**
144    * Determines all dangling edges of a graph.
145    *
146    * @throws Exception when the execution in Flink fails.
147    */
148   private List<EPGMEdge> getDanglingEdges(LogicalGraph graph) throws Exception {
149     List<EPGMVertex> vertices = new ArrayList<>();
150     List<EPGMEdge> edges = new ArrayList<>();
151     graph.getVertices().output(new LocalCollectionOutputFormat<>(vertices));
152     graph.getEdges().output(new LocalCollectionOutputFormat<>(edges));
153     getExecutionEnvironment().execute();
154     Set<GradoopId> ids = vertices.stream().map(EPGMVertex::getId).collect(Collectors.toSet());
155     return edges.stream()
156       .filter(e -> !ids.contains(e.getSourceId()) || !ids.contains(e.getTargetId()))
157       .collect(Collectors.toList());
158   }

196   /**
197    * Function to test graph element datasets for dangling graph ids.
198    *
199    * @param idSet ids of the corresponding graph or graph collection.
200    * @param elements element dataset to test for dangling grahp ids.
201    * @param <E> element type.
202    * @throws Exception when the execution in Flink fails.
203    */
204   private <E extends GraphElement> void assertGraphContainment(GradoopIdSet idSet, DataSet<E> elements)
205     throws Exception {
206     for (E element : elements.collect()) {
207       GradoopIdSet ids = element.getGraphIds();
208       assertFalse("Element has no graph ids", ids.isEmpty());
209       ids.removeAll(idSet);
210       assertTrue("Element has dangling graph ids", ids.isEmpty());
211     }
