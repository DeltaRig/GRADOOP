50    @Test
51    public void testExecute() throws Exception {
52      String query = "MATCH (a)-->(b) WHERE a.age > b.age";
53      QueryHandler queryHandler = new QueryHandler(query);
54      CNF filterPredicate = queryHandler.getPredicates().getSubCNF(Sets.newHashSet("a", "b"));
55  
56      GradoopId vertexAId = GradoopId.get();
57      GradoopId vertexBId = GradoopId.get();
58      GradoopId vertexCId = GradoopId.get();
59  
60      EmbeddingMetaData metaData = new EmbeddingMetaData();
61      metaData.setEntryColumn("a", EntryType.VERTEX, 0);
62      metaData.setEntryColumn("b", EntryType.VERTEX, 1);
63      metaData.setPropertyColumn("a", "age", 0);
64      metaData.setPropertyColumn("b", "age", 1);
65  
66      Embedding embedding1 = new Embedding();
67      embedding1.add(vertexAId, PropertyValue.create(42));
68      embedding1.add(vertexBId, PropertyValue.create(23));
69  
70      Embedding embedding2 = new Embedding();
71      embedding2.add(vertexAId, PropertyValue.create(42));
72      embedding2.add(vertexCId, PropertyValue.create(84));
73  
74      DataSet<Embedding> input = getExecutionEnvironment().fromElements(embedding1, embedding2);
75      PlanNode mockChild = new MockPlanNode(input, metaData);
76  
77      List<Embedding> result = new FilterEmbeddingsNode(mockChild, filterPredicate).execute().collect();
78  
79      assertEquals(1, result.size());
80      assertEquals(vertexAId, result.get(0).getId(0));
81      assertEquals(vertexBId, result.get(0).getId(1));
82      assertEquals(PropertyValue.create(42), result.get(0).getProperty(0));
83      assertEquals(PropertyValue.create(23), result.get(0).getProperty(1));
84    }
