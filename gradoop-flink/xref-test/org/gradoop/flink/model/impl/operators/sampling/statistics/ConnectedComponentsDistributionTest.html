39  @RunWith(Parameterized.class)

58     * Whether to write the component property to the edges.

60    private final boolean annotateEdges;

62    /**
63     * Test class constructor for parameterized call.
64     *
65     * @param testName Name for test-case
66     * @param propertyKey Property key to store the component id
67     * @param maxIterations Maximum number of iterations
68     * @param annotateEdges Whether to write the component property to the edges
69     */
70    public ConnectedComponentsDistributionTest(String testName, String propertyKey, String maxIterations,
71      String annotateEdges) {
72      this.testName = testName;
73      this.propertyKey = propertyKey;
74      this.maxIterations = maxIterations.equals("MAX") ? Integer.MAX_VALUE : Integer.parseInt(maxIterations);
75      this.annotateEdges = Boolean.parseBoolean(annotateEdges);
76    }

78    @Test
79    public void testComponentDistribution() throws Exception {
80  
81      // Graph containing 3 WCC with: component - #vertices, #edges:
82      // comp1 - 3, 3
83      // comp2 - 4, 12
84      // comp3 - 2, 1
85      String graphString = "g[" +
86        "/* first component */" +
87        "(v0 {id:0, value:\"A\"})" +
88        "(v1 {id:1, value:\"B\"})" +
89        "(v2 {id:2, value:\"C\"})" +
90        "(v0)-[e0]->(v1)" +
91        "(v1)-[e1]->(v2)" +
92        "(v2)-[e2]->(v0)" +
93        "/* second component */" +
94        "(v3 {id:3, value:\"D\"})" +
95        "(v4 {id:4, value:\"E\"})" +
96        "(v5 {id:5, value:\"F\"})" +
97        "(v6 {id:6, value:\"G\"})" +
98        "(v3)-[e4]->(v4)" +
99        "(v3)-[e5]->(v5)" +
100       "(v3)-[e6]->(v6)" +
101       "(v4)-[e7]->(v3)" +
102       "(v4)-[e8]->(v5)" +
103       "(v4)-[e9]->(v6)" +
104       "(v5)-[e10]->(v3)" +
105       "(v5)-[e11]->(v4)" +
106       "(v5)-[e12]->(v6)" +
107       "(v6)-[e13]->(v3)" +
108       "(v6)-[e14]->(v4)" +
109       "(v6)-[e15]->(v5)" +
110       "/* third component */" +
111       "(v7 {id:7, value:\"H\"})" +
112       "(v8 {id:8, value:\"I\"})" +
113       "(v7)-[e3]->(v8)" +
114       "]";
115 
116     LogicalGraph graph = getLoaderFromString(graphString).getLogicalGraphByVariable("g");
117     DataSet<Tuple3<String, Long, Long>> componentDist = graph.callForValue(
118       new ConnectedComponentsDistribution<>(propertyKey, maxIterations, annotateEdges));
119 
120     List<Tuple3<String, Long, Long>> componentDistList = Lists.newArrayList();
121 
122     componentDist.output(new LocalCollectionOutputFormat<>(componentDistList));
123 
124     getExecutionEnvironment().execute();
125 
126     assertEquals("Wrong number of components", 3, componentDistList.size());
127 
128     List<Long> vertexCount = componentDistList.stream()
129       .map(tuple -> tuple.f1).collect(Collectors.toList());
130 
131     assertTrue("Wrong number of vertices per component",
132       vertexCount.contains(4L) && vertexCount.contains(3L) && vertexCount.contains(2L));
133 
134     if (annotateEdges) {
135       for (Tuple3<String, Long, Long> componentTuple : componentDistList) {
136         if (componentTuple.f1 == 4L) {
137           assertEquals("Wrong number of edges for component '" + componentTuple.f0 + "'",
138             12L, (long) componentTuple.f2);
139         }
140         if (componentTuple.f1 == 3L) {
141           assertEquals("Wrong number of edges for component '" + componentTuple.f0 + "'",
142             3L, (long) componentTuple.f2);
143         }
144         if (componentTuple.f1 == 2L) {
145           assertEquals("Wrong number of edges for component '" + componentTuple.f0 + "'",
146             1L, (long) componentTuple.f2);
147         }
148       }
149     }
150   }
