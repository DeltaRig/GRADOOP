223   /**
224    * The common graph validation used by all graph samplings
225    *
226    * @param input The input graph
227    * @param output The sampled graph
228    * @throws Exception on failure
229    */
230   void validateGraph(LogicalGraph input, LogicalGraph output) throws Exception {
231     dbVertices = Lists.newArrayList();
232     dbEdges = Lists.newArrayList();
233     newVertices = Lists.newArrayList();
234     newEdges = Lists.newArrayList();
235 
236     input.getVertices().output(new LocalCollectionOutputFormat<>(dbVertices));
237     input.getEdges().output(new LocalCollectionOutputFormat<>(dbEdges));
238 
239     output.getVertices().output(new LocalCollectionOutputFormat<>(newVertices));
240     output.getEdges().output(new LocalCollectionOutputFormat<>(newEdges));
241 
242     getExecutionEnvironment().execute();
243 
244     assertNotNull("graph was null", output);
245 
246     newVertexIDs = new HashSet<>();
247     for (EPGMVertex vertex : newVertices) {
248       assertTrue("sampled vertex is not part of the original graph", dbVertices.contains(vertex));
249       newVertexIDs.add(vertex.getId());
250     }
251     for (EPGMEdge edge : newEdges) {
252       assertTrue("sampled edge is not part of the original graph", dbEdges.contains(edge));
253       assertTrue("sampled edge has source vertex which is not part of the sampled graph",
254         newVertexIDs.contains(edge.getSourceId()));
255       assertTrue("sampled edge has target vertex which is not part of the sampled graph",
256         newVertexIDs.contains(edge.getTargetId()));
257     }
258   }
