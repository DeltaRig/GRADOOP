106   protected Comparison getComparison() {
107     return new Comparison(
108       new Literal("a.label"),
109       Comparator.EQ,
110       new Literal("Person")
111     );

114   private Comparison invert(
115     Comparison comparison) {
116     ComparableExpression[] arguments = comparison.getComparableExpressions();
117     return new Comparison(
118       arguments[0],
119       comparison.getComparator().getInverse(),
120       arguments[1]
121     );
122   }
