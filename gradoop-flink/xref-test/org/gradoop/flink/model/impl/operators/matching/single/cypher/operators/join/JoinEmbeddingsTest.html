30  public class JoinEmbeddingsTest extends PhysicalOperatorTest {
31    private static GradoopId v0 = GradoopId.get();
32    private static GradoopId v1 = GradoopId.get();
33    private static GradoopId e0 = GradoopId.get();
34    private static GradoopId e1 = GradoopId.get();
35    private static GradoopId e2 = GradoopId.get();
36    private static GradoopId e3 = GradoopId.get();
37  
38    @Test
39    public void testJoin() throws Exception {
40      Embedding l = new Embedding();
41      l.add(v0, PropertyValue.create("Foobar"));
42      l.add(e0, PropertyValue.create(42));
43      l.add(v1);
44      DataSet<Embedding> left = getExecutionEnvironment().fromElements(l);
45  
46      Embedding r = new Embedding();
47      r.add(v1, PropertyValue.create("Baz"));
48      DataSet<Embedding> right = getExecutionEnvironment().fromElements(r);
49  
50      PhysicalOperator join = new JoinEmbeddings(left, right, 1, 2, 0);
51  
52      DataSet<Embedding> result = join.evaluate();
53      assertEquals(1, result.count());
54      assertEveryEmbedding(result, embedding ->
55        embedding.getProperties().equals(Lists.newArrayList(
56          PropertyValue.create("Foobar"),
57          PropertyValue.create(42),
58          PropertyValue.create("Baz")
59        )
60      ));
61      assertEmbeddingExists(result, v0, e0, v1);
62    }
63  
64    @Test
65    public void testSingleJoinPartners() throws Exception {
66      DataSet<Embedding> left = getExecutionEnvironment().fromElements(
67        createEmbedding(v0, e0, v1),
68        createEmbedding(v1, e2, v0)
69      );
70  
71      DataSet<Embedding> right = getExecutionEnvironment().fromElements(
72        createEmbedding(v0),
73        createEmbedding(v1)
74      );
75  
76      PhysicalOperator join = new JoinEmbeddings(left, right, 1, 0, 0);
77  
78      DataSet<Embedding> result = join.evaluate();
79      assertEquals(2, result.count());
80      assertEmbeddingExists(result, v0, e0, v1);
81      assertEmbeddingExists(result, v1, e2, v0);
82    }
83  
84    @Test
85    public void testMultipleJoinPartners() throws Exception {
86      //Single Column
87      DataSet<Embedding> left = getExecutionEnvironment().fromElements(
88        createEmbedding(v0, e0, v1),
89        createEmbedding(v1, e1, v0)
90      );
91  
92      DataSet<Embedding> right = getExecutionEnvironment().fromElements(
93        createEmbedding(v0, e2, v1),
94        createEmbedding(v1, e3, v0)
95      );
96  
97      PhysicalOperator join =
98        new JoinEmbeddings(left, right, 3, Lists.newArrayList(0, 2), Lists.newArrayList(2, 0));
99  
100     DataSet<Embedding> result = join.evaluate();
101     assertEquals(2, result.count());
102     assertEmbeddingExists(result, v0, e0, v1, e3);
103     assertEmbeddingExists(result, v1, e1, v0, e2);
104   }
