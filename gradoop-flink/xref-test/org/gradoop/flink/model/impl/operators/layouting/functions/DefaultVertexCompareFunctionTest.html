26  public class DefaultVertexCompareFunctionTest {
27  
28    @Test
29    public void testCompare() {
30      DefaultVertexCompareFunction cf = new DefaultVertexCompareFunction(10);
31  
32      LVertex v1 = new LVertex(null, new Vector(10, 10), -1, null, new Vector(5, 0));
33      LVertex v2 = new LVertex(null, new Vector(10, 10), -1, null, new Vector(-5, 0));
34      LVertex v3 = new LVertex(null, new Vector(10, 10), -1, null, new Vector(6, 0));
35      Assert.assertEquals(1, cf.compare(v1, v1), 0.0000001);
36      Assert.assertEquals(0, cf.compare(v1, v2), 0.0000001);
37      Assert.assertTrue(cf.compare(v1, v3) > 0.5);
38  
39      LVertex v4 = new LVertex(null, new Vector(10, 20), -1, null, new Vector(5, 0));
40      LVertex v5 = new LVertex(null, new Vector(10, 30), -1, null, new Vector(5, 0));
41      LVertex v6 = new LVertex(null, new Vector(10, 15), -1, null, new Vector(5, 0));
42      Assert.assertEquals(1, cf.compare(v1, v4), 0.0000001);
43      Assert.assertEquals(1, cf.compare(v1, v6), 0.0000001);
44      Assert.assertEquals(0, cf.compare(v1, v5), 0.0000001);
45  
46      LVertex v7 = new LVertex(null, new Vector(10, 10), -1,
47        Arrays.asList(GradoopId.get(), GradoopId.get(), GradoopId.get()), new Vector(50, 0));
48      Assert.assertEquals(1, cf.compare(v7, v7), 0.0000001);
49      Assert.assertEquals(0, cf.compare(v7, v2), 0.0000001);
50      Assert.assertTrue(cf.compare(v7, v3) > 0.5);
51    }
