67    /**
68     * Test MinProperty, VertexMinProperty and EdgeMinProperty with a single graph.
69     *
70     * @throws Exception if the execution fails.
71     */
72    @Test
73    public void testSingleGraphMin() throws Exception {
74      LogicalGraph graph = getLoaderFromString("org:Ga[" +
75          "(:Va{p : 0.5f})-[:ea{p : 2}]->(:Vb{p : 3.1f})" +
76          "(:Vc{p : 0.33f})-[:eb]->(:Vd{p : 0.0f})" +
77          "]"
78        )
79        .getLogicalGraphByVariable("org");
80  
81      MinVertexProperty minVertexProperty =
82        new MinVertexProperty(PROPERTY, VERTEX_AGGREGATE_PROPERTY);
83      MinEdgeProperty minEdgeProperty = new MinEdgeProperty(PROPERTY, EDGE_AGGREGATE_PROPERTY);
84      MinProperty minProperty = new MinProperty(PROPERTY, ELEMENT_AGGREGATE_PROPERTY);
85  
86      graph = graph
87        .aggregate(minVertexProperty)
88        .aggregate(minEdgeProperty)
89        .aggregate(minProperty);
90  
91      GraphHead graphHead = graph.getGraphHead().collect().get(0);
92  
93      assertTrue("edge minimum not set",
94        graphHead.hasProperty(minEdgeProperty.getAggregatePropertyKey()));
95      assertTrue("vertex minimum not set",
96        graphHead.hasProperty(minVertexProperty.getAggregatePropertyKey()));
97      assertTrue("element minimum not set",
98        graphHead.hasProperty(minProperty.getAggregatePropertyKey()));
99      assertEquals(
100       2,
101       graphHead.getPropertyValue(
102         minEdgeProperty.getAggregatePropertyKey()).getInt());
103     assertEquals(
104       0.0f,
105       graphHead.getPropertyValue(
106         minVertexProperty.getAggregatePropertyKey()).getFloat(), 0.00001f);
107     assertEquals(
108       0.0f,
109       graphHead.getPropertyValue(
110         minProperty.getAggregatePropertyKey()).getFloat(), 0.00001f);
111   }

298   /**
299    * Helper function to test count aggregators.
300    * Asserts the count for vertices, edges and all elements.
301    *
302    * @param graphHead graph head containing the aggregate properties
303    * @param expectedVertexCount expected vertex count
304    * @param expectedEdgeCount expected edge count
305    */
306   void assertCounts(GraphHead graphHead, long expectedVertexCount, long expectedEdgeCount) {
307     assertEquals("wrong vertex count", expectedVertexCount,
308       graphHead.getPropertyValue(new VertexCount().getAggregatePropertyKey()).getLong());
309     assertEquals("wrong edge count", expectedEdgeCount,
310       graphHead.getPropertyValue(new EdgeCount().getAggregatePropertyKey()).getLong());
311     assertEquals("wrong element count", expectedEdgeCount + expectedVertexCount,
312       graphHead.getPropertyValue(new Count().getAggregatePropertyKey()).getLong());
313   }

601    * @throws Exception when the execution in Flink fails.

603   @Test
604   public void testAggregationWithPostAggregateForGraph() throws Exception {
605     FlinkAsciiGraphLoader loader = getLoaderFromString("input [" +
606       "(i1 {a: 1L}) (i2 {a: 2L}) (i3 {a: -1L}) (i4 {a: 3L})" +
607       "] expected {sum_a: 5L, sum_a_plusone: 6L} [" +
608       "(i1)(i2)(i3)(i4)" +

610     LogicalGraph input = loader.getLogicalGraphByVariable("input");
