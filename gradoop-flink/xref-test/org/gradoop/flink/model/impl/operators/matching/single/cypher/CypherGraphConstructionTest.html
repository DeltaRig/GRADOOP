26    @Test
27    public void testEdgeConstruction() throws Exception {
28      FlinkAsciiGraphLoader loader = getSocialNetworkLoader();
29  
30      LogicalGraph dbGraph = loader.getLogicalGraph();
31  
32      loader.appendToDatabaseFromString("expected0[" +
33        "(alice)-[akb]->(bob)-[bkc]->(carol)," +
34        "(alice)-[:possible_friend]->(carol)" +
35        "]," +
36        "expected1[" +
37        "(bob)-[bkc]->(carol)-[ckd]->(dave)," +
38        "(bob)-[:possible_friend]->(dave)" +
39        "]");
40  
41      GraphCollection result = dbGraph.query(
42        "MATCH (a:Person)-[e0:knows]->(b:Person)-[e1:knows]->(c:Person) " +
43          "WHERE a.city = 'Leipzig' AND a <> c",
44        "(b)<-[e0]-(a)-[e_new:possible_friend]->(c)<-[e1]-(b)");
45  
46      GraphCollection expectedCollection = loader
47        .getGraphCollectionByVariables("expected0", "expected1");
48  
49      collectAndAssertTrue(result.equalsByGraphElementData(expectedCollection));
50    }
