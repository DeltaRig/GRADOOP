37    @Test
38    public void testGraphHead() throws Exception {
39  
40      GradoopId graphID = GradoopId.get();
41  
42      String label = "A";
43      Properties props = Properties.create();
44      props.set("k1", "v1");
45      props.set("k2", "v2");
46  
47      GraphHead graphHead =
48          new EPGMGraphHeadFactory().initGraphHead(graphID, label, props);
49  
50      HashMap<String, String> newProps = new HashMap<>();
51      newProps.put("k1", "new_k1");
52  
53      TransformationFunction<GraphHead> renameFunction = new RenamePropertyKeys<>(newProps);
54  
55      renameFunction.apply(graphHead, graphHead);
56  
57      assertEquals(2, graphHead.getPropertyCount());
58      assertEquals(label, graphHead.getLabel());
59      assertEquals("v1", graphHead.getPropertyValue("new_k1").toString());
60      assertEquals("v2", graphHead.getPropertyValue("k2").toString());
61      assertNull(graphHead.getPropertyValue("k1"));
62    }
