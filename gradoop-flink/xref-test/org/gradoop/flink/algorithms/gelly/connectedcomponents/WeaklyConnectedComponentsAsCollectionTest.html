24  public class WeaklyConnectedComponentsAsCollectionTest extends GradoopFlinkTestBase {
25  
26    private static final String propertyKey = "componentId";
27  
28    @Test
29    public void testByElementIds() throws Exception {
30      String graph = "input[" +
31      // First component
32      "(v0 {id:0, component:1})" +
33      // Second component
34      "(v1 {id:1, component:2})-[e0]->(v2 {id:2, component:2})" +
35      "(v1)-[e1]->(v3 {id:3, component:2})" +
36      "(v2)-[e2]->(v3)" +
37      "(v3)-[e3]->(v4 {id:4, component:2})" +
38      "(v4)-[e4]->(v5 {id:5, component:2})" +
39      "]";
40  
41      FlinkAsciiGraphLoader loader = getLoaderFromString(graph);
42      LogicalGraph input = loader.getLogicalGraphByVariable("input");
43      GraphCollection result = input
44        .callForCollection(new WeaklyConnectedComponentsAsCollection<>(propertyKey, 10));
45      GraphCollection components = input.splitBy("component");
46  
47      collectAndAssertTrue(result.equalsByGraphElementIds(components));
48    }
