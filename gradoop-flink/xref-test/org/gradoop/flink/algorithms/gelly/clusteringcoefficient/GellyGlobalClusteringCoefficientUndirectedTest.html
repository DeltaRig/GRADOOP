29  /**
30   * Test-class for {@link GellyGlobalClusteringCoefficientUndirected}

32  public class GellyGlobalClusteringCoefficientUndirectedTest
33    extends GellyClusteringCoefficientTestBase {
34  
35    @Override
36    public <
37      G extends GraphHead,
38      V extends Vertex,
39      E extends Edge,
40      LG extends BaseGraph<G, V, E, LG, GC>,
41      GC extends BaseGraphCollection<G, V, E, LG, GC>>
42    ClusteringCoefficientBase<G, V, E, LG, GC> getCCAlgorithm() {
43      return new GellyGlobalClusteringCoefficientUndirected<>();
44    }
45  
46    @Override
47    public void testFullyConnectedGraph() throws Exception {
48      validateGraphProperties(fullGraph);
49      EPGMGraphHead head = fullGraph.getGraphHead().collect().get(0);
50      assertEquals("Wrong global value for fully connected graph, should be 1", 1d,
51        head.getPropertyValue(ClusteringCoefficientBase.PROPERTY_KEY_GLOBAL).getDouble(), 0.0);
52    }
53  
54    @Override
55    public void testNonConnectedGraph() throws Exception {
56      validateGraphProperties(nonConnectedGraph);
57      EPGMGraphHead head = nonConnectedGraph.getGraphHead().collect().get(0);
58      double global = head.getPropertyValue(ClusteringCoefficientBase.PROPERTY_KEY_GLOBAL)
59        .getDouble();
60      assertTrue("Wrong global value for not connected graph, should be 0 or NaN",
61        Double.isNaN(global));
62    }
63  
64    @Override
65    public void testSpecific() throws Exception {
66      String graphString = "halfConnected[" +
67        "/* half connected graph */" +
68        "(v0 {id:0, value:\"A\"})" +
69        "(v1 {id:1, value:\"B\"})" +
70        "(v2 {id:2, value:\"C\"})" +
71        "(v3 {id:3, value:\"D\"})" +
72        "(v0)-[e0]->(v1)" +
73        "(v1)-[e1]->(v0)" +
74        "(v0)-[e2]->(v2)" +
75        "(v2)-[e3]->(v0)" +
76        "(v0)-[e4]->(v3)" +
77        "(v3)-[e5]->(v0)" +
78        "(v1)-[e6]->(v2)" +
79        "(v2)-[e7]->(v1)" +
80        "]";
81  
82      LogicalGraph graph = getLoaderFromString(graphString)
83        .getLogicalGraphByVariable("halfConnected");
84      LogicalGraph result = graph.callForGraph(getCCAlgorithm());
85  
86      validateGraphProperties(result);
87  
88      EPGMGraphHead head = result.getGraphHead().collect().get(0);
89  
90      double global = head.getPropertyValue(ClusteringCoefficientBase.PROPERTY_KEY_GLOBAL)
91        .getDouble();
92      assertEquals("graph has wrong global value, should be 0.6",
93        3d / 5d, global, 0.0);
94    }
95  
96    @Override
97    public void validateGraphProperties(LogicalGraph graph) throws Exception {
98      EPGMGraphHead head = graph.getGraphHead().collect().get(0);
99      assertTrue("No global value stored in graph head",
100       head.hasProperty(ClusteringCoefficientBase.PROPERTY_KEY_GLOBAL));
101   }
