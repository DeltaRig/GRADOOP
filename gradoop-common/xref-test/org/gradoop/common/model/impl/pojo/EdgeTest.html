43    @Test
44    public void createEdgePojoTest() {
45      GradoopId edgeId = GradoopId.get();
46      GradoopId sourceId = GradoopId.get();
47      GradoopId targetId = GradoopId.get();
48      GradoopIdSet graphIds = GradoopIdSet
49        .fromExisting(GradoopId.get(), GradoopId.get());
50  
51      String label = "A";
52      Properties props = Properties.create();
53      props.set("k1", "v1");
54      props.set("k2", "v2");
55  
56      Edge edge = new EPGMEdgeFactory().initEdge(edgeId, label, sourceId, targetId, props, graphIds);
57  
58      assertEquals(edgeId, edge.getId());
59      assertEquals(label, edge.getLabel());
60      assertEquals(sourceId, edge.getSourceId());
61      assertEquals(targetId, edge.getTargetId());
62      assertEquals(2, edge.getPropertyCount());
63      assertEquals("v1", edge.getPropertyValue("k1").getString());
64      assertEquals("v2", edge.getPropertyValue("k2").getString());
65      assertEquals(2, edge.getGraphCount());
66  
67      for (GradoopId graphId : graphIds) {
68        assertTrue(edge.getGraphIds().contains(graphId));
69      }
70    }
