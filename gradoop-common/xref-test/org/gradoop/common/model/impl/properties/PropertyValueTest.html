54    /**
55     * Tests if {@link PropertyValue#create(Object)} works with supported types.
56     */
57    @Test(dataProvider = "supportedTypeProvider", dataProviderClass = PropertyValueTestProvider.class)
58    public void testCreate(Object supportedType) {
59      PropertyValue p = create(supportedType);
60  
61      if (supportedType != null) {
62        assertTrue(p.is(supportedType.getClass()));
63        assertEquals(supportedType, p.get(supportedType.getClass()));
64      } else {
65        assertTrue(p.isNull());
66        assertNull(p.getObject());
67      }
68    }

75      PropertyValue copy = value.copy();

77      assertNotSame(value, copy);
78      if (value instanceof List || value instanceof Map || value instanceof Set) {
79        assertNotSame(supportedType, copy.getObject());
80      }
