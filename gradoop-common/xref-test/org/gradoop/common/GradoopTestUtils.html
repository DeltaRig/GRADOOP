184   /**
185    * Returns a {@link ElementFactoryProvider} able to create EPGM elements.
186    *
187    * @return ElementFactoryProvider for EPGM elements
188    */
189   public static ElementFactoryProvider<EPGMGraphHead, EPGMVertex, EPGMEdge> getEPGMElementFactoryProvider() {
190     if (epgmElementFactoryProvider == null) {
191       epgmElementFactoryProvider = new ElementFactoryProvider<EPGMGraphHead, EPGMVertex, EPGMEdge>() {
192         GraphHeadFactory<EPGMGraphHead> graphHeadFactory = new EPGMGraphHeadFactory();
193         VertexFactory<EPGMVertex> vertexFactory = new EPGMVertexFactory();
194         EdgeFactory<EPGMEdge> edgeFactory = new EPGMEdgeFactory();
195 
196         @Override
197         public GraphHeadFactory<EPGMGraphHead> getGraphHeadFactory() {
198           return graphHeadFactory;
199         }
200 
201         @Override
202         public VertexFactory<EPGMVertex> getVertexFactory() {
203           return vertexFactory;
204         }
205 
206         @Override
207         public EdgeFactory<EPGMEdge> getEdgeFactory() {
208           return edgeFactory;
209         }
210       };
211     }
212     return epgmElementFactoryProvider;
213   }

335   /**
336    * Checks if two given EPGM elements are equal by considering their label and
337    * properties.
338    *
339    * @param element1 first element
340    * @param element2 second element
341    */
342   public static void validateElements(
343     Element element1, Element element2) {
344     assertNotNull("first element was null", element1);
345     assertNotNull("second element was null", element2);
346 
347     assertEquals("id mismatch", element1.getId(), element2.getId());
348     assertEquals("label mismatch", element1.getLabel(), element2.getLabel());
349 
350     if (element1.getPropertyCount() == 0) {
351       assertEquals("property count mismatch",
352         element1.getPropertyCount(), element2.getPropertyCount());
353     } else {
354       List<String> keys1 = Lists.newArrayList(element1.getPropertyKeys());
355       Collections.sort(keys1);
356 
357       List<String> keys2 = Lists.newArrayList(element2.getPropertyKeys());
358       Collections.sort(keys2);
359 
360       Iterator<String> it1 = keys1.iterator();
361       Iterator<String> it2 = keys2.iterator();
362 
363       while (it1.hasNext() && it2.hasNext()) {
364         String key1 = it1.next();
365         String key2 = it2.next();
366         assertEquals("property key mismatch", key1, key2);
367         assertEquals("property value mismatch",
368           element1.getPropertyValue(key1),
369           element2.getPropertyValue(key2));
370       }
371       assertFalse("too many properties in first element", it1.hasNext());
372       assertFalse("too many properties in second element", it2.hasNext());
373     }
374   }

396   public static <T extends Value> T writeAndReadFields(Class<T> clazz, T in) throws IOException {
397     // write to byte[]
398     ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
399     DataOutputView outputView = new DataOutputViewStreamWrapper(outputStream);
400     in.write(outputView);
401     outputStream.flush();
402 
403     T out;
404     try {
405       out = clazz.newInstance();
406     } catch (Exception e) {
407       throw new IOException("Cannot initialize the class: " + clazz, e);
408     }
409 
410     // read from byte[]
411     ByteArrayInputStream inputStream = new ByteArrayInputStream(outputStream.toByteArray());
412     DataInputView inputView = new DataInputViewStreamWrapper(inputStream);
413     out.read(inputView);
414 
415     return out;
416   }

454   /**
455    * Uses reflection to call a private method with arguments.
456    *
457    * @param clazz class which has the method
458    * @param object instance of the class
459    * @param methodName method name
460    * @param args method arguments
461    * @param <T1> return type of method
462    * @param <T2> type of the calling class
463    * @return method result
464    * @throws Exception in case anything goes wrong
465    */
466   @SuppressWarnings("unchecked")
467   public static <T1, T2> T1 call(
468     Class<T2> clazz,
469     T2 object,
470     String methodName,
471     Class<?>[] parameterTypes, Object[] args)
472     throws Exception {
473     Method m = parameterTypes != null ?
474       clazz.getDeclaredMethod(methodName, parameterTypes) : clazz.getDeclaredMethod(methodName);
475     m.setAccessible(true);
476     return (T1) (args != null ?  m.invoke(object, args) : m.invoke(object));
477   }
