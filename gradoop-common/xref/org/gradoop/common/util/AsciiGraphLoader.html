39  /**

137   AsciiGraphLoader<G, V, E> fromString(String asciiGraph,
138                                        ElementFactoryProvider<G, V, E> elementFactoryProvider) {
139     return new AsciiGraphLoader<>(new GDLHandler.Builder()
140       .setDefaultGraphLabel(GradoopConstants.DEFAULT_GRAPH_LABEL)
141       .setDefaultVertexLabel(GradoopConstants.DEFAULT_VERTEX_LABEL)
142       .setDefaultEdgeLabel(GradoopConstants.DEFAULT_EDGE_LABEL)
143       .setErrorStrategy(new BailSyntaxErrorStrategy())
144       .buildFromString(asciiGraph),
145       elementFactoryProvider);
146   }

241    * @return graphHeads that are assigned to the given variables
242    */
243   public Collection<G>  getGraphHeadsByVariables(String... variables) {
244     Collection<G>  result =
245       Sets.newHashSetWithExpectedSize(variables.length);
246     for (String variable : variables) {
247       G graphHead = getGraphHeadByVariable(variable);
248       if (graphHead != null) {
249         result.add(graphHead);
250       }
251     }
252     return result;

440   /**
441    * Initializes GraphHeads and their cache.
442    */
443   private void initGraphHeads() {
444     for (Graph g : gdlHandler.getGraphs()) {
445       if (!graphHeadIds.containsKey(g.getId())) {
446         initGraphHead(g);
447       }
448     }
449 
450     for (Map.Entry<String, Graph> e : gdlHandler.getGraphCache().entrySet()) {
451       updateGraphCache(e.getKey(), e.getValue());
452     }
453   }
454 
455   /**
456    * Initializes vertices and their cache.
457    */
458   private void initVertices() {
459     for (org.gradoop.gdl.model.Vertex v : gdlHandler.getVertices()) {
460       initVertex(v);
461     }
462 
463     for (Map.Entry<String, org.gradoop.gdl.model.Vertex> e : gdlHandler.getVertexCache().entrySet()) {
464       updateVertexCache(e.getKey(), e.getValue());
465     }
466   }

468   /**
469    * Initializes edges and their cache.
470    */
471   private void initEdges() {
472     for (org.gradoop.gdl.model.Edge e : gdlHandler.getEdges()) {
473       initEdge(e);
474     }
475 
476     for (Map.Entry<String, org.gradoop.gdl.model.Edge> e : gdlHandler.getEdgeCache().entrySet()) {
477       updateEdgeCache(e.getKey(), e.getValue());
478     }

481   /**
482    * Creates a new Graph from the GDL Loader.
483    *
484    * @param g graph from GDL Loader
485    * @return graph head
486    */
487   private G initGraphHead(Graph g) {
488     G graphHead = elementFactoryProvider.getGraphHeadFactory().createGraphHead(
489       g.getLabel(), Properties.createFromMap(g.getProperties()));
490     graphHeadIds.put(g.getId(), graphHead.getId());
491     graphHeads.put(graphHead.getId(), graphHead);
492     return graphHead;
493   }
494 
495   /**
496    * Creates a new Vertex from the GDL Loader or updates an existing one.
497    *
498    * @param v vertex from GDL Loader
499    * @return vertex
500    */
501   private V initVertex(org.gradoop.gdl.model.Vertex v) {
502     V vertex;
503     if (!vertexIds.containsKey(v.getId())) {
504       vertex = elementFactoryProvider.getVertexFactory().createVertex(
505         v.getLabel(),
506         Properties.createFromMap(v.getProperties()),
507         createGradoopIdSet(v));
508       vertexIds.put(v.getId(), vertex.getId());
509       vertices.put(vertex.getId(), vertex);
510     } else {
511       vertex = vertices.get(vertexIds.get(v.getId()));
512       vertex.setGraphIds(createGradoopIdSet(v));
513     }
514     return vertex;
515   }
516 
517   /**
518    * Creates a new Edge from the GDL Loader.
519    *
520    * @param e edge from GDL loader
521    * @return edge
522    */
523   private E initEdge(org.gradoop.gdl.model.Edge e) {
524     E edge;
525     if (!edgeIds.containsKey(e.getId())) {
526       edge = elementFactoryProvider.getEdgeFactory().createEdge(
527         e.getLabel(),
528         vertexIds.get(e.getSourceVertexId()),
529         vertexIds.get(e.getTargetVertexId()),
530         Properties.createFromMap(e.getProperties()),
531         createGradoopIdSet(e));
532       edgeIds.put(e.getId(), edge.getId());
533       edges.put(edge.getId(), edge);
534     } else {
535       edge = edges.get(edgeIds.get(e.getId()));
536       edge.setGraphIds(createGradoopIdSet(e));
537     }
538     return edge;
539   }

541   /**

572   /**
573    * Creates a {@code GradoopIDSet} from the long identifiers stored at the
574    * given graph element.
575    *
576    * @param e graph element
577    * @return GradoopIDSet for the given element
578    */
579   private GradoopIdSet createGradoopIdSet(GraphElement e) {
580     GradoopIdSet result = new GradoopIdSet();
581     for (Long graphId : e.getGraphs()) {
582       result.add(graphHeadIds.get(graphId));
583     }
584     return result;
