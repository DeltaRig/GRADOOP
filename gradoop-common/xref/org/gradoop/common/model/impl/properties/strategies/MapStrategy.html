16  package org.gradoop.common.model.impl.properties.strategies;
17  
18  import org.apache.flink.core.memory.DataInputView;
19  import org.apache.flink.core.memory.DataInputViewStreamWrapper;
20  import org.apache.flink.core.memory.DataOutputViewStreamWrapper;
21  import org.gradoop.common.model.impl.properties.PropertyValue;
22  import org.gradoop.common.model.impl.properties.Type;
23  
24  import java.io.ByteArrayOutputStream;
25  import java.io.DataOutputStream;
26  import java.io.IOException;
27  import java.util.HashMap;
28  import java.util.Map;
29  
30  /**
31   * Strategy class for handling {@code PropertyValue} operations with a value of the type
32   * {@code Map}.
33   */
34  public class MapStrategy
35    extends AbstractVariableSizedPropertyValueStrategy<Map<PropertyValue, PropertyValue>> {
36  
37    @Override
38    public Map<PropertyValue, PropertyValue> read(DataInputView inputView, byte typeByte)
39        throws IOException {
40      byte[] rawBytes = readVariableSizedData(inputView, typeByte);
41      DataInputViewStreamWrapper internalInputView = createInputView(rawBytes);
42      return createMap(internalInputView);
43    }
44  
45    @Override
46    public int compare(Map value, Object other) {
47      throw new UnsupportedOperationException("Method compare() is not supported for Map.");
48    }
49  
50    @Override
51    public boolean is(Object value) {
52      if (!(value instanceof Map)) {
53        return false;
54      }
55      for (Map.Entry<?, ?> entry : ((Map<?, ?>) value).entrySet()) {
56        // Map is not a valid property value if it contains any object
57        // that is not a property value itself.
58        if (!(entry.getKey() instanceof PropertyValue) ||
59          !(entry.getValue() instanceof PropertyValue)) {
60          return false;
61        }
62      }
63  
64      return true;
65    }
66  
67    @Override
68    public Class<Map<PropertyValue, PropertyValue>> getType() {
69      return (Class) Map.class;
70    }
71  
72    /**
73     * {@inheritDoc}
74     * @throws IOException if converting the byte array to a Map fails.
75     */
76    @Override
77    public Map<PropertyValue, PropertyValue> get(byte[] bytes) throws IOException {
78      DataInputViewStreamWrapper inputView = createInputView(bytes);
79      Map<PropertyValue, PropertyValue> map;
80  
81      try {
82        if (inputView.skipBytes(PropertyValue.OFFSET) != PropertyValue.OFFSET) {
83          throw new IOException("Malformed entry in PropertyValue Map.");
84        }
85        map = createMap(inputView);
86      } catch (IOException e) {
87        throw new IOException("Error while processing DataInputViewStreamWrapper.");
88      }
89  
90      return map;
91    }
92  
93    @Override
94    public byte getRawType() {
95      return Type.MAP.getTypeByte();
96    }
97  
98    /**
99     * {@inheritDoc}
100    * @throws IOException if converting the value to a byte array fails.
101    */
102   @Override
103   public byte[] getRawBytes(Map<PropertyValue, PropertyValue> value) throws IOException {
104     int size = value.keySet().stream().mapToInt(PropertyValue::byteSize).sum() +
105         value.values().stream().mapToInt(PropertyValue::byteSize).sum() +
106         PropertyValue.OFFSET;
107 
108     try (ByteArrayOutputStream byteStream = new ByteArrayOutputStream(size);
109         DataOutputStream outputStream = new DataOutputStream(byteStream);
110         DataOutputViewStreamWrapper outputView = new DataOutputViewStreamWrapper(outputStream)) {
111 
112       outputStream.write(Type.MAP.getTypeByte());
113       for (Map.Entry<PropertyValue, PropertyValue> entry : value.entrySet()) {
114         entry.getKey().write(outputView);
115         entry.getValue().write(outputView);
116       }
117 
118       return byteStream.toByteArray();
119     } catch (IOException e) {
120       throw new IOException("Error writing PropertyValue with MapStrategy.", e);
121     }
122   }
123 
124   /**
125    * Creates a map with data read from an {@link DataInputViewStreamWrapper}.
126    *
127    * @param inputView {@link DataInputViewStreamWrapper} containing data
128    * @return a map containing the deserialized data
129    * @throws IOException on failure to read input view
130    */
131   private Map<PropertyValue, PropertyValue> createMap(DataInputViewStreamWrapper inputView)
132     throws IOException {
133     PropertyValue key;
134     PropertyValue value;
135 
136     Map<PropertyValue, PropertyValue> map = new HashMap<>();
137 
138     try {
139       while (inputView.available() > 0) {
140         key = new PropertyValue();
141         key.read(inputView);
142 
143         value = new PropertyValue();
144         value.read(inputView);
145 
146         map.put(key, value);
147       }
148     } catch (IOException e) {
149       throw new IOException("Error reading PropertyValue with MapStrategy.", e);
150     }
151     return map;
152   }
