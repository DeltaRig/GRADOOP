16  package org.gradoop.common.model.impl.properties.strategies;
17  
18  import org.apache.flink.core.memory.DataInputView;
19  import org.gradoop.common.model.impl.properties.bytes.Bytes;
20  import org.gradoop.common.model.impl.properties.PropertyValue;
21  import org.gradoop.common.model.impl.properties.Type;
22  
23  import java.io.IOException;
24  
25  /**
26   * Strategy class for handling {@code PropertyValue} operations with a value of the type
27   * {@code Boolean}.
28   */
29  public class BooleanStrategy extends AbstractFixSizedPropertyValueStrategy<Boolean> {
30  
31    @Override
32    public Boolean read(DataInputView inputView, byte typeByte) throws IOException {
33      return inputView.readByte() == -1;
34    }
35  
36    @Override
37    public int compare(Boolean value, Object other) {
38      if (other.getClass() == Boolean.class) {
39        return Boolean.compare(value, (boolean) other);
40      }
41      throw new IllegalArgumentException(String.format(
42        "Incompatible types: %s, %s", value.getClass(), other.getClass()));
43    }
44  
45    @Override
46    public boolean is(Object value) {
47      return value instanceof Boolean;
48    }
49  
50    @Override
51    public Class<Boolean> getType() {
52      return Boolean.class;
53    }
54  
55    @Override
56    public Boolean get(byte[] bytes) {
57      return bytes[1] == -1;
58    }
59  
60    @Override
61    public byte getRawType() {
62      return Type.BOOLEAN.getTypeByte();
63    }
64  
65    /**
66     * {@inheritDoc}
67     *
68     * Warning: Please note that if {@code null} is passed as an argument, it is going to be evaluated
69     * as if it was false.
70     */
71    @Override
72    public byte[] getRawBytes(Boolean value) {
73      byte[] rawBytes = new byte[PropertyValue.OFFSET + Bytes.SIZEOF_BOOLEAN];
74      rawBytes[0] = getRawType();
75      Bytes.putByte(rawBytes, PropertyValue.OFFSET, (byte) (value ? -1 : 0));
76      return rawBytes;
77    }
78  }
