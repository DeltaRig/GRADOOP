312   /**
313    * Returns the value as the specified type. If it is already of the same type as requested, we
314    * just return the value. If not, an {@link UnsupportedOperationException} is thrown.
315    *
316    * @param clazz the requested value type
317    * @param <T> PropertyValue supported type
318    * @return value
319    * @throws UnsupportedOperationException when trying to get the wrong type
320    */
321   @SuppressWarnings("unchecked")
322   public <T> T get(Class<T> clazz) throws UnsupportedOperationException {
323     PropertyValueStrategy strategy = PropertyValueStrategyFactory.get(clazz);
324     if (strategy.is(value)) {
325       return (T) value;
326     }
327     String type = value == null ? "null" : value.getClass().getSimpleName();
328     throw new UnsupportedOperationException("Value '" + value + "' of type " + type +
329       ", cannot be accessed as " + clazz.getSimpleName());
330   }

701   /**
702    * Byte representation:
703    *
704    * byte 1       : type info
705    *
706    * for dynamic length types (e.g. String and BigDecimal)
707    * byte 2       : length (short)
708    * byte 3       : length (short)
709    * byte 4 - end : value bytes
710    *
711    * If the size of the internal byte representation if larger than
712    * {@link #LARGE_PROPERTY_THRESHOLD} (i.e. if a {@code short} is too small to store the length),
713    * then the {@link #FLAG_LARGE} bit will be set in the first byte and the byte representation
714    * will be:
715    * byte 2       ; length (int)
716    * byte 3       : length (int)
717    * byte 4       : length (int)
718    * byte 5       : length (int)
719    * byte 6 - end : value bytes
720    *
721    * for fixed length types (e.g. int, long, float, ...)
722    * byte 2 - end : value bytes
723    *
724    * @param outputView data output to write data to
725    * @throws IOException if write to output view fails.
726    */
727   @Override
728   public void write(DataOutputView outputView) throws IOException {
729     PropertyValueStrategyFactory.get(value).write(value, outputView);
730   }
731 
732   @Override
733   public void read(DataInputView inputView) throws IOException {
734     // type
735     byte typeByte = inputView.readByte();
736     // Apply bitmask to get the actual type.
737     byte type = (byte) (~PropertyValue.FLAG_LARGE & typeByte);
738 
739     PropertyValueStrategy strategy = PropertyValueStrategyFactory.get(type);
740 
741     if (strategy == null) {
742       throw new UnsupportedTypeException("No strategy for type byte from input view found");
743     } else {
744       value = strategy.read(inputView, typeByte);
745     }
746   }
