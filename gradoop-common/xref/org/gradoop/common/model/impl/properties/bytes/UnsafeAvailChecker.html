16  package org.gradoop.common.model.impl.properties.bytes;
17  
18  import org.slf4j.Logger;
19  import org.slf4j.LoggerFactory;
20  
21  import java.lang.reflect.Field;
22  import java.lang.reflect.Method;
23  import java.security.AccessController;
24  import java.security.PrivilegedAction;
25  
26  /**
27   * This implementation reuses much of the code of HBase's UnsafeAvailChecker
28   * (org.apache.hadoop.hbase.util.UnsafeAvailChecker).
29   * This can be found in org.apache.hbase:hbase-common
30   * Much of the code is copied directly or has only small changes.
31   */
32  class UnsafeAvailChecker {
33  
34    /**
35     * Unsafe class name
36     */
37    private static final String CLASS_NAME = "sun.misc.Unsafe";
38  
39    /**
40     * true, when Unsafe available and system has unaligned access capability
41     */
42    private static boolean UNALIGNED = false;
43  
44    /**
45     * Logger
46     */
47    private static final Logger LOG = LoggerFactory.getLogger(UnsafeAvailChecker.class);
48  
49    static {
50      boolean avail = AccessController.doPrivileged(new PrivilegedAction<Boolean>() {
51        @Override
52        public Boolean run() {
53          try {
54            Class<?> clazz = Class.forName(CLASS_NAME);
55            Field f = clazz.getDeclaredField("theUnsafe");
56            f.setAccessible(true);
57            return f.get(null) != null;
58          } catch (ReflectiveOperationException e) {
59            LOG.warn("sun.misc.Unsafe is not available/accessible");
60            return false;
61          }
62        }
63      });
64      // When Unsafe itself is not available/accessible consider unaligned as false.
65      if (avail) {
66        String arch = System.getProperty("os.arch");
67        if ("ppc64".equals(arch) || "ppc64le".equals(arch)) {
68          // java.nio.Bits.unaligned() wrongly returns false on ppc (JDK-8165231),
69          UNALIGNED = true;
70        } else {
71          try {
72            // Using java.nio.Bits#unaligned() to check for unaligned-access capability
73            Class<?> clazz = Class.forName("java.nio.Bits");
74            Method m = clazz.getDeclaredMethod("unaligned");
75            m.setAccessible(true);
76            UNALIGNED = (Boolean) m.invoke(null);
77          } catch (ReflectiveOperationException e) {
78            LOG.warn("java.nio.Bits#unaligned() check failed. Unsafe based read/write of primitive types " +
79              "won't be used", e);
80          }
81        }
82      }
83    }
84  
85    /**
86     * Private constructor to avoid instantiation
87     */
88    private UnsafeAvailChecker() {
89    }
90  
91    /**
92     * True when running JVM is having sun's Unsafe package available in it and underlying
93     * system having unaligned-access capability.
94     *
95     * @return true, if unaligned
96     */
97    static boolean unaligned() {
98      return UNALIGNED;
99    }
100 }
