136   /**
137    * Creates a GradoopId using the given time, machine identifier, process identifier, and counter.
138    * <p>
139    * Note: Implementation taken from org.bson.types.ObjectId
140    *
141    * @param timestamp         the time in seconds
142    * @param machineIdentifier the machine identifier
143    * @param processIdentifier the process identifier
144    * @param counter           the counter
145    * @param checkCounter      if the constructor should test if the counter is between 0 and
146    *                          16777215
147    */
148   private GradoopId(final int timestamp, final int machineIdentifier, final short processIdentifier,
149     final int counter, final boolean checkCounter) {
150     if ((machineIdentifier & HIGH_ORDER_ONE_BYTE) != 0) {
151       throw new IllegalArgumentException("The machine identifier must be between 0" +
152         " and 16777215 (it must fit in three bytes).");
153     }
154     if (checkCounter && ((counter & HIGH_ORDER_ONE_BYTE) != 0)) {
155       throw new IllegalArgumentException("The counter must be between 0" +
156         " and 16777215 (it must fit in three bytes).");
157     }
158 
159     ByteBuffer buffer = ByteBuffer.allocate(12);
160 
161     buffer.put((byte) (timestamp >> 24));
162     buffer.put((byte) (timestamp >> 16));
163     buffer.put((byte) (timestamp >> 8));
164     buffer.put((byte) timestamp);
165 
166     buffer.put((byte) (machineIdentifier >> 16));
167     buffer.put((byte) (machineIdentifier >> 8));
168     buffer.put((byte) machineIdentifier);
169 
170     buffer.put((byte) (processIdentifier >> 8));
171     buffer.put((byte) processIdentifier);
172 
173     buffer.put((byte) (counter >> 16));
174     buffer.put((byte) (counter >> 8));
175     buffer.put((byte) counter);
176 
177     this.bytes = buffer.array();
178   }

180   /**
181    * Creates the machine identifier from the network interface.
182    * <p>
183    * Note: Implementation taken from org.bson.types.ObjectId
184    *
185    * @return a short representing the process
186    */
187   private static int createMachineIdentifier() {
188     int machinePiece;
189     try {
190       StringBuilder sb = new StringBuilder();
191       Enumeration<NetworkInterface> e = NetworkInterface.getNetworkInterfaces();
192       while (e.hasMoreElements()) {
193         NetworkInterface ni = e.nextElement();
194         sb.append(ni.toString());
195         byte[] mac = ni.getHardwareAddress();
196         if (mac != null) {
197           ByteBuffer bb = ByteBuffer.wrap(mac);
198           try {
199             sb.append(bb.getChar());
200             sb.append(bb.getChar());
201             sb.append(bb.getChar());
202           } catch (BufferUnderflowException shortHardwareAddressException) {
203             // mac with less than 6 bytes. continue
204           }
205         }
206       }
207       machinePiece = sb.toString().hashCode();
208     } catch (SocketException t) {
209       machinePiece = new SecureRandom().nextInt();
210     }
211     machinePiece = machinePiece & LOW_ORDER_THREE_BYTES;
212     return machinePiece;
213   }

215   /**
216    * Creates the process identifier.  This does not have to be unique per class loader because
217    * NEXT_COUNTER will provide the uniqueness.
218    * <p>
219    * Note: Implementation taken from org.bson.types.ObjectId
220    *
221    * @return a short representing the process
222    */
223   private static short createProcessIdentifier() {
224     short processId;
225     String processName = java.lang.management.ManagementFactory.getRuntimeMXBean().getName();
226     if (processName.contains("@")) {
227       processId = (short) Integer.parseInt(processName.substring(0, processName.indexOf('@')));
228     } else {
229       processId = (short) java.lang.management.ManagementFactory
230         .getRuntimeMXBean().getName().hashCode();
231     }
232 
233     return processId;
234   }

257    * Returns the Gradoop ID represented by a specified hexadecimal string.

261    * @param string hexadecimal GradoopId representation

264   public static GradoopId fromString(String string) {
265     if (!GradoopId.isValid(string)) {
266       throw new IllegalArgumentException(
267         "invalid hexadecimal representation of a GradoopId: [" + string + "]");
268     }
269 
270     byte[] b = new byte[12];
271     for (int i = 0; i < b.length; i++) {
272       b[i] = (byte) Integer.parseInt(string.substring(i * 2, i * 2 + 2), 16);
273     }
274     return new GradoopId(b);
275   }

277   /**
278    * Checks if a string can be transformed into a GradoopId.
279    * <p>
280    * Note: Implementation taken from org.bson.types.ObjectId
281    *
282    * @param hexString a potential GradoopId as a String.
283    * @return whether the string could be an object id
284    * @throws IllegalArgumentException if hexString is null
285    */
286   public static boolean isValid(final String hexString) {
287     if (hexString == null) {
288       throw new IllegalArgumentException();
289     }
290 
291     int len = hexString.length();
292     if (len != 24) {
293       return false;
294     }
295 
296     for (int i = 0; i < len; i++) {
297       char c = hexString.charAt(i);
298       if (c >= '0' && c <= '9') {
299         continue;
300       }
301       if (c >= 'a' && c <= 'f') {
302         continue;
303       }
304       if (c >= 'A' && c <= 'F') {
305         continue;
306       }
307 
308       return false;
309     }
310 
311     return true;
312   }

374   /**
375    * Performs a byte-wise comparison of this and the specified GradoopId.
376    *
377    * @param other the object to be compared.
378    * @return a negative integer, zero, or a positive integer as this object
379    * is less than, equal to, or greater than the specified object.
380    */
381   @Override
382   public int compareTo(GradoopId other) {
383 
384     for (int i = 0; i < GradoopId.ID_SIZE; i++) {
385       if (this.bytes[i] != other.bytes[i]) {
386         return ((this.bytes[i] & 0xff) < (other.bytes[i] & 0xff)) ? -1 : 1;
387       }
388     }
389     return 0;
390   }

397    * @return GradoopId string representation.
398    */
399   @Override
400   public String toString() {
401     char[] chars = new char[24];
402     int i = 0;
403     for (byte b : bytes) {
404       chars[i++] = HEX_CHARS[b >> 4 & 0xF];
405       chars[i++] = HEX_CHARS[b & 0xF];
406     }
407     return String.valueOf(chars);

503   /**
504    * Compares the given GradoopIds and returns the smaller one. It both are equal, the first
505    * argument is returned.
506    *
507    * @param first  first GradoopId
508    * @param second second GradoopId
509    * @return smaller GradoopId or first if equal
510    */
511   public static GradoopId min(GradoopId first, GradoopId second) {
512     int comparison = first.compareTo(second);
513     return comparison == 0 ? first : (comparison < 0 ? first : second);
514   }

516   /**
517    * Returns a primitive int represented by the given 4 bytes.
518    *
519    * @param b3 byte 3
520    * @param b2 byte 2
521    * @param b1 byte 1
522    * @param b0 byte 0
523    * @return int value
524    */
525   private static int makeInt(final byte b3, final byte b2, final byte b1, final byte b0) {
526     return (b3 << 24) | ((b2 & 0xff) << 16) | ((b1 & 0xff) << 8) | ((b0 & 0xff));
