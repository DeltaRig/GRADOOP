35    /**
36     * Run the operator on a test graph.
37     *
38     * @throws Exception when the execution in Flink fails.
39     */
40    @Test
41    public void testWithGraph() throws Exception {
42      FlinkAsciiGraphLoader loader = getLoaderFromString("input[" +
43        "(va1:a {key: 1L})-[:e {att: 1L}]->(vb1:a {key: 1.0d})-[:e {att: 2L}]->(va2:a {key: 1L})" +
44        "(vc1:a {key: 1L, key2: \"a\"})-[:e {att: 3L}]->(va1)-[:e {att: 4L}]->(b1:b {key: 1L})" +
45        "(vc2:a {key: 1L, key2: \"a\"})" +
46        "(vb2:a {key: 1.0d})-[:e {att: 5L}]->(vb3:a {key: 1.0d})(vd1:a {key: 1L, key2: \"b\"})" +
47        "]" +
48        "expected [" +
49        "(da:a {key: 1L})-[:e {att: 1L}]->(db:a {key: 1.0d})-[:e {att: 2L}]->(da)" +
50        "(dc:a {key: 1L, key2: \"a\"})-[:e {att: 3L}]->(da)-[:e {att: 4L}]->(b:b {key: 1L})" +
51        "(db)-[:e {att: 5L}]->(db)(dd:a {key: 1L, key2: \"b\"})" +
52        "]");
53      LogicalGraph result = loader.getLogicalGraphByVariable("input")
54        .callForGraph(new VertexDeduplication<>("a", Arrays.asList("key", "key2")));
55      collectAndAssertTrue(result.equalsByData(loader.getLogicalGraphByVariable("expected")));
56    }
