31  /**
32   * Test for the {@link CreateVertexFromEdges} function used by
33   * {@link org.gradoop.dataintegration.transformation.EdgeToVertex}.
34   */
35  public class CreateVertexFromEdgesTest extends GradoopFlinkTestBase {
36    /**
37     * Test the function by applying it to some vertices.
38     *
39     * @throws Exception when the execution in Flink fails.
40     */
41    @Test
42    public void testFunction() throws Exception {
43      CreateVertexFromEdges<EPGMVertex, EPGMEdge> function = new CreateVertexFromEdges<>("test",
44        getConfig().getLogicalGraphFactory().getVertexFactory());
45      GradoopId dummy = GradoopId.get();
46      EdgeFactory<EPGMEdge> edgeFactory = getConfig().getLogicalGraphFactory().getEdgeFactory();
47      // Create some test edges, with some having no properties or label.
48      EPGMEdge withoutProperties = edgeFactory.createEdge(dummy, dummy);
49      EPGMEdge withProperties = edgeFactory.createEdge("TestEdge2", dummy, dummy);
50      withProperties.setProperty("TestProperty", 1L);
51      List<EPGMEdge> edges = Arrays.asList(withoutProperties, withProperties);
52      List<Tuple3<EPGMVertex, GradoopId, GradoopId>> result = getExecutionEnvironment()
53        .fromCollection(edges).map(function).collect();
54      // There should be a new vertex for each edge.
55      assertEquals(edges.size(), result.size());
56      // Every ID should be assigned only once.
57      long idCount = result.stream().map(t -> t.f0.getId()).distinct().count();
58      assertEquals("EPGMVertex IDs are not unique.", edges.size(), idCount);
59      for (Tuple3<EPGMVertex, GradoopId, GradoopId> resultTuple : result) {
60        if (resultTuple.f0.getPropertyCount() > 0) {
61          assertEquals(withProperties.getProperties(), resultTuple.f0.getProperties());
62        }
63      }
64    }
65  }
