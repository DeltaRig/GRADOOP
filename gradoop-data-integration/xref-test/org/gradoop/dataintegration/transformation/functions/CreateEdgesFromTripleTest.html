32  /**
33   * Test for the {@link CreateEdgesFromTriple} function used by
34   * {@link org.gradoop.dataintegration.transformation.EdgeToVertex}.
35   */
36  public class CreateEdgesFromTripleTest extends GradoopFlinkTestBase {
37  
38    /**
39     * Test the function by applying it to some tuples.
40     *
41     * @throws Exception when the execution in Flink fails.
42     */
43    @Test
44    public void testFunction() throws Exception {
45      CreateEdgesFromTriple<EPGMVertex, EPGMEdge> function = new CreateEdgesFromTriple<>(
46        getConfig().getLogicalGraphFactory().getEdgeFactory(), "source", "target");
47      VertexFactory<EPGMVertex> vertexFactory = getConfig().getLogicalGraphFactory().getVertexFactory();
48      EPGMVertex testVertex1 = vertexFactory.createVertex();
49      EPGMVertex testVertex2 = vertexFactory.createVertex();
50      GradoopId source1 = GradoopId.get();
51      GradoopId source2 = GradoopId.get();
52      GradoopId target1 = GradoopId.get();
53      GradoopId target2 = GradoopId.get();
54      Tuple3<EPGMVertex, GradoopId, GradoopId> tuple1 = new Tuple3<>(testVertex1, source1, target1);
55      Tuple3<EPGMVertex, GradoopId, GradoopId> tuple2 = new Tuple3<>(testVertex2, source2, target2);
56      List<EPGMEdge> result = getExecutionEnvironment().fromElements(tuple1, tuple2).flatMap(function)
57        .collect();
58      // Check if the correct number of edges were created and if they are distinct.
59      assertEquals(4, result.size());
60      // By id.
61      assertEquals(4, result.stream().map(EPGMElement::getId).count());
62      // By source and target id.
63      assertEquals(4,
64        result.stream().map(e -> Tuple2.of(e.getSourceId(), e.getTargetId())).distinct().count());
65      // Finally check the data of the edges.
66      for (EPGMEdge resultEdge : result) {
67        if (resultEdge.getLabel().equals("source")) {
68          if (resultEdge.getSourceId().equals(source1)) {
69            assertEquals(testVertex1.getId(), resultEdge.getTargetId());
70          } else if (resultEdge.getSourceId().equals(source2)) {
71            assertEquals(testVertex2.getId(), resultEdge.getTargetId());
72          } else {
73            fail("EPGMEdge with invalid source ID created.");
74          }
75        } else if (resultEdge.getLabel().equals("target")) {
76          if (resultEdge.getSourceId().equals(testVertex1.getId())) {
77            assertEquals(target1, resultEdge.getTargetId());
78          } else if (resultEdge.getSourceId().equals(testVertex2.getId())) {
79            assertEquals(target2, resultEdge.getTargetId());
80          } else {
81            fail("EPGMEdge with invalid source ID created.");
82          }
83        } else {
84          fail("EPGMEdge with invalid label created.");
85        }
86      }
87    }
