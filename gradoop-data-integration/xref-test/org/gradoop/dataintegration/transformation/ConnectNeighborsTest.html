67    @Test
68    public void testIncoming() throws Exception {
69      LogicalGraph input = loader.getLogicalGraphByVariable("input");
70      UnaryGraphToGraphOperator operator =
71        new ConnectNeighbors("Center", Neighborhood.EdgeDirection.INCOMING, "V", "neighbor");
72      LogicalGraph expected = loader.getLogicalGraphByVariable("expectedIncoming");
73  
74      collectAndAssertTrue(expected.equalsByData(input.callForGraph(operator)));
75    }
