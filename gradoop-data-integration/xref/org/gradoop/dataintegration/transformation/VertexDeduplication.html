40  /**
41   * Deduplicates vertices based on some attribute.<p>
42   * Given a label and a list of property keys, this operator will find all vertices of that label and condense
43   * duplicates to one vertex. Two (or more) vertices are considered duplicates if all property values,
44   * respective to the list of given keys, are equal.<p>
45   * Other attributes of the vertices are ignored, the new (condensed) vertex will have the attributes of some
46   * (random) vertex of its duplicates.
47   * <p>
48   * This will create no new elements and retain the graph head.
49   *
50   * @param <G>  The graph head type.
51   * @param <V>  The vertex type.
52   * @param <E>  The edge type.
53   * @param <LG> The type of the graph.
54   * @param <GC> The type of the graph collection.
55   */
56  public class VertexDeduplication<
57    G extends GraphHead,
58    V extends Vertex,
59    E extends Edge,
60    LG extends BaseGraph<G, V, E, LG, GC>,
61    GC extends BaseGraphCollection<G, V, E, LG, GC>>
62    implements UnaryBaseGraphToBaseGraphOperator<LG> {
63  
64    /**
65     * The label of vertices to deduplicate.
66     */
67    private final String label;
68  
69    /**
70     * The list of property keys used to determine equal vertices.
71     */
72    private final List<String> propertyKeys;
73  
74    /**
75     * Creates a new instance of this vertex deduplication operator.
76     *
77     * @param label        The label of vertices to deduplicate.
78     * @param propertyKeys The properties used to check if two (or more) vertices of that label are duplicates.
79     */
80    public VertexDeduplication(String label, List<String> propertyKeys) {
81      this.label = Objects.requireNonNull(label);
82      this.propertyKeys = Objects.requireNonNull(propertyKeys);
83    }
84  
85    @Override
86    public LG execute(LG graph) {
87      DataSet<V> annotatedVertices = graph.getVerticesByLabel(label)
88        .groupBy(new GetPropertiesAsList<>(propertyKeys))
89        .reduceGroup(new MarkDuplicatesInGroup<>());
90      DataSet<Tuple2<GradoopId, GradoopId>> vertexToDedupVertex = annotatedVertices
91        .flatMap(new CreateMappingFromMarkedDuplicates<>());
92      DataSet<V> deduplicatedVertices = annotatedVertices
93        .filter(new ByProperty<V>(MarkDuplicatesInGroup.PROPERTY_KEY).negate());
94      DataSet<V> otherVertices = graph.getVertices().filter(new ByLabel<V>(label).negate());
95      DataSet<E> updatesEdges = graph.getEdges()
96        .leftOuterJoin(vertexToDedupVertex)
97        .where(new SourceId<>())
98        .equalTo(0)
99        .with(new EdgeSourceUpdateJoin<>())
100       .leftOuterJoin(vertexToDedupVertex)
101       .where(new TargetId<>())
102       .equalTo(0)
103       .with(new EdgeTargetUpdateJoin<>());
104 
105     return graph.getFactory().fromDataSets(graph.getGraphHead(),
106       otherVertices.union(deduplicatedVertices),
107       updatesEdges);
108   }
