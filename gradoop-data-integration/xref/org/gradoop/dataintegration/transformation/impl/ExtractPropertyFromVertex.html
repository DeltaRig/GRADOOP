39  /**
40   * This operator is executed on the vertex dataset and extracts a property to a newly created vertex
41   * which can be connected to the vertex it was created from. If the property is a list every
42   * element of the list is a new vertex.
43   */
44  public class ExtractPropertyFromVertex implements UnaryGraphToGraphOperator {
45  
46    /**
47     * The vertices the extraction is executed for.
48     */
49    private String forVerticesOfLabel;
50  
51    /**
52     * The property key at the original vertex.
53     */
54    private String originalPropertyName;
55  
56    /**
57     * The label of the extracted vertex.
58     */
59    private String newVertexLabel;
60  
61    /**
62     * The property name of the extracted property at the new vertex.
63     */
64    private String newPropertyName;
65  
66    /**
67     * The direction of the created edge(s).
68     */
69    private EdgeDirection edgeDirection;
70  
71    /**
72     * The label of the newly created edge(s).
73     */
74    private String edgeLabel;
75  
76    /**
77     * If true, all property values got condensed based on their equality.
78     * So that every newly created vertex has a unique property value.
79     * The default value is 'true'.
80     */
81    private boolean condense = true;
82  
83    /**
84     * Constructs a new {@link UnaryGraphToGraphOperator} which extracts properties from vertices into
85     * newly created vertices.
86     *
87     * @param forVerticesOfLabel   The vertices the extraction is executed for.
88     * @param originalPropertyName The property key at the original vertex.
89     * @param newVertexLabel       The label of the extracted vertex.
90     * @param newPropertyName      The property name of the extracted property at the new vertex.
91     */
92    public ExtractPropertyFromVertex(String forVerticesOfLabel, String originalPropertyName,
93                                     String newVertexLabel, String newPropertyName) {
94      this(forVerticesOfLabel, originalPropertyName, newVertexLabel, newPropertyName,
95        EdgeDirection.NONE, null);
96    }
97  
98    /**
99     * Constructs a new {@link UnaryGraphToGraphOperator} which extracts properties from vertices into
100    * newly created vertices and connects original and new vertex with each other.
101    *
102    * @param forVerticesOfLabel   The vertices the extraction is executed for.
103    * @param originalPropertyName The property key at the original vertex.
104    * @param newVertexLabel       The label of the extracted vertex.
105    * @param newPropertyName      The property name of the extracted property at the new vertex.
106    * @param edgeDirection        The direction of the created edge(s).
107    * @param edgeLabel            The label of the newly created edge.
108    */
109   public ExtractPropertyFromVertex(String forVerticesOfLabel, String originalPropertyName,
110                                    String newVertexLabel, String newPropertyName,
111                                    EdgeDirection edgeDirection, String edgeLabel) {
112     this.forVerticesOfLabel = Preconditions.checkNotNull(forVerticesOfLabel);
113     this.originalPropertyName = Preconditions.checkNotNull(originalPropertyName);
114     this.newVertexLabel = Preconditions.checkNotNull(newVertexLabel);
115     this.newPropertyName = Preconditions.checkNotNull(newPropertyName);
116     this.edgeDirection = Preconditions.checkNotNull(edgeDirection);
117     if (!edgeDirection.equals(EdgeDirection.NONE)) {
118       this.edgeLabel = Preconditions.checkNotNull(edgeLabel);
119     }
120   }
121 
122   /**
123    * This method sets whether property value based condensation should be executed or not.
124    * The default value is 'true'.
125    *
126    * @param condense If true, all property values got condensed based on their equality.
127    *                 So that every newly created vertex has a unique property value.
128    */
129   public void setCondensation(boolean condense) {
130     this.condense = condense;
131   }
132 
133   @Override
134   public LogicalGraph execute(LogicalGraph logicalGraph) {
135     // filter the vertices by the given label
136     DataSet<EPGMVertex> filteredVertices = logicalGraph
137       .getVertices()
138       .filter(new ByLabel<>(forVerticesOfLabel));
139 
140     // calculate new vertices and store the origin for linking
141     DataSet<Tuple2<PropertyValue, GradoopId>> candidates = filteredVertices
142       .flatMap(new ExtractPropertyWithOriginId(originalPropertyName));
143 
144     // extract the new vertices
145     DataSet<Tuple2<EPGMVertex, List<GradoopId>>> newVerticesAndOriginIds;
146     if (condense) {
147       newVerticesAndOriginIds = candidates
148         .groupBy(0)
149         .reduceGroup(new CreateNewVertexWithEqualityCondense(
150           logicalGraph.getFactory().getVertexFactory(), newVertexLabel, newPropertyName));
151     } else {
152       newVerticesAndOriginIds = candidates
153         .map(new CreateNewVertex(logicalGraph.getFactory().getVertexFactory(), newVertexLabel,
154           newPropertyName));
155     }
156 
157 
158     DataSet<EPGMVertex> vertices = newVerticesAndOriginIds
159       .map(new Value0Of2<>())
160       .map(new AddToGraphBroadcast<>())
161       .withBroadcastSet(logicalGraph.getGraphHead().map(new Id<>()), AddToGraphBroadcast.GRAPH_ID)
162       .union(logicalGraph.getVertices());
163 
164     // the newly created vertices should be linked to the original vertices
165     DataSet<EPGMEdge> edges = logicalGraph.getEdges();
166     if (!edgeDirection.equals(EdgeDirection.NONE)) {
167       edges = newVerticesAndOriginIds
168         .flatMap(new CreateNewEdges(logicalGraph.getFactory().getEdgeFactory(), edgeDirection,
169           edgeLabel))
170         .map(new AddToGraphBroadcast<>())
171         .withBroadcastSet(logicalGraph.getGraphHead().map(new Id<>()), AddToGraphBroadcast.GRAPH_ID)
172         .union(edges);
173     }
174 
175     return logicalGraph.getFactory()
176       .fromDataSets(logicalGraph.getGraphHead(), vertices, edges);
177   }
