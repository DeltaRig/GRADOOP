16  package org.gradoop.dataintegration.transformation.impl;
17  
18  import org.gradoop.common.model.api.entities.Edge;
19  import org.gradoop.common.model.api.entities.GraphHead;
20  import org.gradoop.common.model.api.entities.Vertex;
21  import org.gradoop.flink.model.api.epgm.BaseGraph;
22  import org.gradoop.dataintegration.transformation.api.PropertyTransformationFunction;
23  import org.gradoop.flink.model.api.epgm.BaseGraphCollection;
24  import org.gradoop.flink.model.api.operators.UnaryBaseGraphToBaseGraphOperator;
25  import org.gradoop.flink.model.impl.operators.transformation.Transformation;
26  
27  /**
28   * Creates a graph with the same structure but a specified property of an element is transformed
29   * by the declared function. The transformed value can be stored under a new key. If the
30   * original key shall be reused the old value can be stored under the key 'key__x' where 'x' is a
31   * version number. This number increases on every continuous transformation.
32   *
33   * @param <G> type of the graph head
34   * @param <V> the vertex type
35   * @param <E> the edge type
36   * @param <LG> type of the logical graph instance
37   * @param <GC> type of the graph collection instance
38   */
39  public class PropertyTransformation<
40    G extends GraphHead,
41    V extends Vertex,
42    E extends Edge,
43    LG extends BaseGraph<G, V, E, LG, GC>,
44    GC extends BaseGraphCollection<G, V, E, LG, GC>>
45    implements UnaryBaseGraphToBaseGraphOperator<LG> {
46  
47    /**
48     * Label of the element whose property shall be transformed.
49     */
50    private String label;
51    /**
52     * Property key.
53     */
54    private String propertyKey;
55    /**
56     * Transformation function which shall be applied to a property of a graph head.
57     */
58    private PropertyTransformationFunction graphHeadTransformationFunction;
59    /**
60     * Transformation function which shall be applied to a property of a vertex.
61     */
62    private PropertyTransformationFunction vertexTransformationFunction;
63    /**
64     * Transformation function which shall be applied to a property of an edge.
65     */
66    private PropertyTransformationFunction edgeTransformationFunction;
67    /**
68     * New property key.
69     */
70    private String newPropertyKey;
71    /**
72     * True, if the history of a property key shall be kept.
73     */
74    private boolean keepHistory;
75  
76    /**
77     * Valued constructor.
78     *
79     * @param propertyKey                      property key
80     * @param graphHeadTransformationFunction  transformation function which shall be applied to a
81     *                                         property of a graph head
82     * @param vertexTransformationFunction     transformation function which shall be applied to a
83     *                                         property of a vertex
84     * @param edgeTransformationFunction       transformation function which shall be applied to a
85     *                                         property of an edge
86     * @param label                            label of the element whose property shall be
87     *                                         transformed (optional)
88     * @param newPropertyKey                   new property key (optional)
89     * @param keepHistory                      flag to enable versioning (false by default)
90     */
91    public PropertyTransformation(String propertyKey,
92        PropertyTransformationFunction graphHeadTransformationFunction,
93        PropertyTransformationFunction vertexTransformationFunction,
94        PropertyTransformationFunction edgeTransformationFunction, String label,
95        String newPropertyKey, boolean keepHistory) {
96      if (graphHeadTransformationFunction == null && vertexTransformationFunction == null &&
97          edgeTransformationFunction == null) {
98        throw new IllegalArgumentException("Provide at least one transformation function.");
99      }
100 
101     this.label = label;
102     this.propertyKey = propertyKey;
103     this.graphHeadTransformationFunction = graphHeadTransformationFunction;
104     this.vertexTransformationFunction = vertexTransformationFunction;
105     this.edgeTransformationFunction = edgeTransformationFunction;
106     this.newPropertyKey = newPropertyKey;
107     this.keepHistory = keepHistory;
108   }
109 
110   /**
111    * Creates a new property transformation operator instance.
112    *
113    * @param propertyKey                      property key
114    * @param graphHeadTransformationFunction  transformation function which shall be applied to a
115    *                                         property of a graph head
116    * @param vertexTransformationFunction     transformation function which shall be applied to a
117    *                                         property of a vertex
118    * @param edgeTransformationFunction       transformation function which shall be applied to a
119    *                                         property of an edge
120    */
121   public PropertyTransformation(String propertyKey,
122       PropertyTransformationFunction graphHeadTransformationFunction,
123       PropertyTransformationFunction vertexTransformationFunction,
124       PropertyTransformationFunction edgeTransformationFunction) {
125     this(propertyKey, graphHeadTransformationFunction, vertexTransformationFunction,
126          edgeTransformationFunction, null, null, false);
127   }
128 
129   /**
130    * Applies the property transformation functions on the given input graph.
131    *
132    * @param graph input graph
133    * @return transformed logical graph
134    */
135   @Override
136   public LG execute(LG graph) {
137     return new Transformation<G, V, E, LG, GC>(
138         graphHeadTransformationFunction == null ? null : new BasePropertyTransformationFunction<>(
139             propertyKey, graphHeadTransformationFunction, label, newPropertyKey, keepHistory),
140         vertexTransformationFunction == null ? null : new BasePropertyTransformationFunction<>(
141             propertyKey, vertexTransformationFunction, label, newPropertyKey, keepHistory),
142         edgeTransformationFunction == null ? null : new BasePropertyTransformationFunction<>(
143             propertyKey, edgeTransformationFunction, label, newPropertyKey, keepHistory))
144       .execute(graph);
145   }
