16  package org.gradoop.dataintegration.transformation.functions;
17  
18  import org.apache.flink.api.common.functions.FlatMapFunction;
19  import org.apache.flink.api.common.typeinfo.TypeInformation;
20  import org.apache.flink.api.java.tuple.Tuple3;
21  import org.apache.flink.api.java.typeutils.ResultTypeQueryable;
22  import org.apache.flink.util.Collector;
23  import org.gradoop.common.model.api.entities.Edge;
24  import org.gradoop.common.model.api.entities.EdgeFactory;
25  import org.gradoop.common.model.api.entities.Vertex;
26  import org.gradoop.common.model.impl.id.GradoopId;
27  
28  import java.util.Objects;
29  
30  /**
31   * A {@link FlatMapFunction} to create two new edges per inserted edge.
32   * Source to new vertex, new vertex to target.
33   *
34   * @param <V> The vertex type.
35   * @param <E> The edge type.
36   */
37  public class CreateEdgesFromTriple<V extends Vertex, E extends Edge>
38    implements FlatMapFunction<Tuple3<V, GradoopId, GradoopId>, E>, ResultTypeQueryable<E> {
39  
40    /**
41     * The edge type created by the factory.
42     */
43    private final Class<E> edgeType;
44  
45    /**
46     * The label of the newly created edge which points to the newly created vertex.
47     */
48    private final String edgeLabelSourceToNew;
49  
50    /**
51     * The label of the newly created edge which starts at the newly created vertex.
52     */
53    private final String edgeLabelNewToTarget;
54  
55    /**
56     * Reduce object instantiations.
57     */
58    private E reuse;
59  
60    /**
61     * The constructor to create the new edges based on the given triple.
62     *
63     * @param factory              The Factory which creates the new edges.
64     * @param edgeLabelSourceToNew The label of the newly created edge which points to the newly
65     *                             created vertex.
66     * @param edgeLabelNewToTarget The label of the newly created edge which starts at the newly
67     *                             created vertex.
68     */
69    public CreateEdgesFromTriple(EdgeFactory<E> factory, String edgeLabelSourceToNew,
70      String edgeLabelNewToTarget) {
71      this.edgeType = Objects.requireNonNull(factory).getType();
72      this.edgeLabelSourceToNew = Objects.requireNonNull(edgeLabelSourceToNew);
73      this.edgeLabelNewToTarget = Objects.requireNonNull(edgeLabelNewToTarget);
74      this.reuse = factory.createEdge(edgeLabelSourceToNew, GradoopId.NULL_VALUE,
75        GradoopId.NULL_VALUE);
76    }
77  
78    @Override
79    public void flatMap(Tuple3<V, GradoopId, GradoopId> triple, Collector<E> out) {
80      reuse.setId(GradoopId.get());
81      reuse.setLabel(edgeLabelSourceToNew);
82      reuse.setSourceId(triple.f1);
83      reuse.setTargetId(triple.f0.getId());
84      out.collect(reuse);
85  
86      reuse.setId(GradoopId.get());
87      reuse.setLabel(edgeLabelNewToTarget);
88      reuse.setSourceId(triple.f0.getId());
89      reuse.setTargetId(triple.f2);
90      out.collect(reuse);
91    }
92  
93    @Override
94    public TypeInformation<E> getProducedType() {
95      return TypeInformation.of(edgeType);
96    }
97  }
