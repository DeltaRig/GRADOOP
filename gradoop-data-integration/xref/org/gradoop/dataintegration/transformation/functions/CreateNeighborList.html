16  package org.gradoop.dataintegration.transformation.functions;
17  
18  import org.apache.flink.api.common.functions.CoGroupFunction;
19  import org.apache.flink.api.java.tuple.Tuple2;
20  import org.apache.flink.util.Collector;
21  import org.gradoop.common.model.impl.id.GradoopId;
22  import org.gradoop.common.model.impl.pojo.EPGMEdge;
23  import org.gradoop.common.model.impl.pojo.EPGMVertex;
24  import org.gradoop.dataintegration.transformation.impl.Neighborhood;
25  import org.gradoop.dataintegration.transformation.impl.NeighborhoodVertex;
26  
27  import java.util.ArrayList;
28  import java.util.Iterator;
29  import java.util.List;
30  import java.util.Objects;
31  
32  /**
33   * This CoGroup operation creates a list of neighbors for each vertex.
34   */
35  public class CreateNeighborList
36    implements CoGroupFunction<EPGMEdge, EPGMVertex, Tuple2<EPGMVertex, List<NeighborhoodVertex>>> {
37  
38    /**
39     * The edge direction to consider.
40     */
41    private final Neighborhood.EdgeDirection edgeDirection;
42  
43    /**
44     * Reduce object instantiations.
45     */
46    private final Tuple2<EPGMVertex, List<NeighborhoodVertex>> reuse;
47  
48    /**
49     * The constructor for the creation of neighbor lists.
50     *
51     * @param edgeDirection The edge direction to consider.
52     */
53    public CreateNeighborList(Neighborhood.EdgeDirection edgeDirection) {
54      this.edgeDirection = Objects.requireNonNull(edgeDirection);
55      reuse = new Tuple2<>();
56    }
57  
58    @Override
59    public void coGroup(Iterable<EPGMEdge> edges, Iterable<EPGMVertex> vertex,
60                        Collector<Tuple2<EPGMVertex, List<NeighborhoodVertex>>> out) {
61      // should only contain one or no vertex
62      Iterator<EPGMVertex> vertexIterator = vertex.iterator();
63      if (vertexIterator.hasNext()) {
64        EPGMVertex v = vertexIterator.next();
65  
66        List<NeighborhoodVertex> neighbors = new ArrayList<>();
67        for (EPGMEdge e : edges) {
68          neighbors.add(new NeighborhoodVertex(getNeighborId(v.getId(), e), e.getLabel()));
69        }
70  
71        reuse.f0 = v;
72        reuse.f1 = neighbors;
73        out.collect(reuse);
74      }
75    }
76  
77    /**
78     * Based on the considered edge direction the neighbor id is returned.
79     *
80     * @param vertexId The vertex id the neighbor id is searched for.
81     * @param edge     The edge the neighbor id is taken from.
82     * @return The GradoopId of the neighbor vertex.
83     */
84    private GradoopId getNeighborId(GradoopId vertexId, EPGMEdge edge) {
85      switch (edgeDirection) {
86      case INCOMING:
87        return edge.getSourceId();
88      case OUTGOING:
89        return edge.getTargetId();
90      default:
91        return vertexId.equals(edge.getSourceId()) ? edge.getTargetId() : edge.getSourceId();
92      }
93    }
94  }
