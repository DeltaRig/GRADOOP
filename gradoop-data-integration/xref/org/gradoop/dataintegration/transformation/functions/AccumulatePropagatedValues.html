16  package org.gradoop.dataintegration.transformation.functions;
17  
18  import org.apache.flink.api.common.functions.CoGroupFunction;
19  import org.apache.flink.api.java.tuple.Tuple2;
20  import org.apache.flink.util.Collector;
21  import org.gradoop.common.model.api.entities.Vertex;
22  import org.gradoop.common.model.impl.id.GradoopId;
23  import org.gradoop.common.model.impl.properties.PropertyValue;
24  
25  import java.util.ArrayList;
26  import java.util.Iterator;
27  import java.util.List;
28  import java.util.Objects;
29  import java.util.Set;
30  
31  /**
32   * This {@link CoGroupFunction} accumulates all properties that might be send to a vertex and
33   * stores them in a {@link PropertyValue} list.
34   *
35   * @param <V> The vertex type.
36   */
37  public class AccumulatePropagatedValues<V extends Vertex>
38    implements CoGroupFunction<Tuple2<GradoopId, PropertyValue>, V, V> {
39  
40    /**
41     * The property key where the PropertyValue list should be stored at the target vertices.
42     */
43    private final String targetVertexPropertyKey;
44  
45    /**
46     * Labels of vertices where the propagated property should be set.
47     */
48    private final Set<String> targetVertexLabels;
49  
50    /**
51     * The constructor of the co group function for accumulation of collected property values.
52     *
53     * @param targetVertexPropertyKey The property key where the PropertyValue list should be
54     *                                stored at the target vertices.
55     * @param targetVertexLabels      The set of labels of elements where the property should be
56     *                                set. (Use {@code null} for all vertices.)
57     */
58    public AccumulatePropagatedValues(String targetVertexPropertyKey,
59      Set<String> targetVertexLabels) {
60      this.targetVertexPropertyKey = Objects.requireNonNull(targetVertexPropertyKey);
61      this.targetVertexLabels = targetVertexLabels;
62    }
63  
64    @Override
65    public void coGroup(Iterable<Tuple2<GradoopId, PropertyValue>> propertyValues,
66                        Iterable<V> elements, Collector<V> out) {
67      // should only contain one vertex, based on the uniqueness of gradoop ids
68      Iterator<V> iterator = elements.iterator();
69      if (!iterator.hasNext()) {
70        return;
71      }
72      V targetVertex = iterator.next();
73      // If the vertex is not whitelisted by the targetVertexLabels list,
74      // forward it without modification.
75      if (targetVertexLabels != null && !targetVertexLabels.contains(targetVertex.getLabel())) {
76        out.collect(targetVertex);
77        return;
78      }
79  
80      // collect values of neighbors
81      List<PropertyValue> values = new ArrayList<>();
82      propertyValues.forEach(t -> values.add(t.f1));
83  
84      // Add to vertex if and only if at least one property was propagated.
85      if (!values.isEmpty()) {
86        targetVertex.setProperty(targetVertexPropertyKey, PropertyValue.create(values));
87      }
88  
89      out.collect(targetVertex);
90    }
91  }
