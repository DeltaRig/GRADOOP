34  /**
35   * A property of a vertex is propagated to its neighbors and aggregated in a Property List.
36   */
37  public class PropagatePropertyToNeighbor implements UnaryGraphToGraphOperator {
38  
39    /**
40     * The label of the vertex the property to propagate is part of.
41     */
42    private final String vertexLabel;
43  
44    /**
45     * The property key of the property to propagate.
46     */
47    private final String propertyKey;
48  
49    /**
50     * The property key where the PropertyValue list should be stored at the target vertices.
51     */
52    private final String targetVertexPropertyKey;
53  
54    /**
55     * Only edges with the inserted labels are used. If all labels are sufficient use {@code null}.
56     */
57    private final Set<String> propagatingEdgeLabels;
58  
59    /**
60     * Only vertices with the inserted labels will store the propagated values.
61     * If all vertices should do it use {@code null}.
62     */
63    private final Set<String> targetVertexLabels;
64  
65    /**
66     * The constructor for the propagate property transformation. Additionally it is possible to
67     * define which edge labels can be used for propagation and / or which vertices could be target
68     * of the Properties.
69     * <p>
70     * Using this constructor, properties will be propagated along all edges and to all
71     * target vertices. {@link #PropagatePropertyToNeighbor(String, String, String, Set, Set)}
72     * can be used when properties should only be propagated along certain edges (selected by their
73     * label) and / or to certain vertices (selected by their label). Using this constructor is
74     * equivalent to {@code PropagatePropertyToNeighbor(vertexLabel, propertyKey,
75     * targetVertexPropertyKey, null, null)}.
76     *
77     * @param vertexLabel             The label of the vertex the property to propagate is part of.
78     * @param propertyKey             The property key of the property to propagate.
79     * @param targetVertexPropertyKey The property key where the PropertyValue list should be stored
80     *                                at the target vertices.
81     */
82    public PropagatePropertyToNeighbor(String vertexLabel, String propertyKey,
83      String targetVertexPropertyKey) {
84      this(vertexLabel, propertyKey, targetVertexPropertyKey, null, null);
85    }
86  
87    /**
88     * The constructor for the propagate property transformation. Additionally it is possible to
89     * define which edge labels can be used for propagation and / or which vertices could be target
90     * of the Properties.
91     *
92     * @param vertexLabel             The label of the vertex the property to propagate is part of.
93     * @param propertyKey             The property key of the property to propagate.
94     * @param targetVertexPropertyKey The property key where the PropertyValue list should be stored
95     *                                at the target vertices.
96     * @param propagatingEdges        Only edges with the inserted labels are used. If all labels
97     *                                are sufficient use {@code null}.
98     * @param targetVertexLabels      Only vertices with the inserted labels will store the
99     *                                propagated values. If all vertices should, use {@code null}.
100    */
101   public PropagatePropertyToNeighbor(String vertexLabel, String propertyKey,
102                                      String targetVertexPropertyKey, Set<String> propagatingEdges,
103                                      Set<String> targetVertexLabels) {
104     this.vertexLabel = Objects.requireNonNull(vertexLabel);
105     this.propertyKey = Objects.requireNonNull(propertyKey);
106     this.targetVertexPropertyKey = Objects.requireNonNull(targetVertexPropertyKey);
107     this.propagatingEdgeLabels = propagatingEdges;
108     this.targetVertexLabels = targetVertexLabels;
109   }
110 
111   @Override
112   public LogicalGraph execute(LogicalGraph graph) {
113     // prepare the edge set, EdgeFilter if propagating edges are given
114     DataSet<EPGMEdge> propagateAlong = graph.getEdges();
115     if (propagatingEdgeLabels != null) {
116       propagateAlong = propagateAlong.filter(new LabelIsIn<>(propagatingEdgeLabels));
117     }
118 
119     DataSet<EPGMVertex> newVertices = graph.getVertices()
120       // Extract properties to propagate
121       .flatMap(new BuildIdPropertyValuePairs<>(vertexLabel, propertyKey))
122       // Propagate along edges.
123       .join(propagateAlong)
124       .where(0).equalTo(new SourceId<>())
125       .with(new BuildTargetVertexIdPropertyValuePairs<>())
126       // Update target vertices.
127       .coGroup(graph.getVertices())
128       .where(0).equalTo(new Id<>())
129       .with(new AccumulatePropagatedValues<>(targetVertexPropertyKey, targetVertexLabels))
130       .map(new AddToGraphBroadcast<>())
131       .withBroadcastSet(graph.getGraphHead().map(new Id<>()), AddToGraphBroadcast.GRAPH_ID);
132 
133     return graph.getFactory().fromDataSets(graph.getGraphHead(), newVertices, graph.getEdges());
134   }
