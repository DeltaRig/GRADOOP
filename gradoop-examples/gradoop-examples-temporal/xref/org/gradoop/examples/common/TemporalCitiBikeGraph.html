191   /**
192    * Function to extract the trip period from properties of a vertex. The names of the properties have to be
193    * {@code starttime} and {@code stoptime} and their type has to be a String.
194    *
195    * @param current The current vertex to transform.
196    * @param transformed A copy of the current vertex used to create the resulting one.
197    * @return The resulting temporal vertex with assigned valid times.
198    */
199   private static TemporalVertex extractTripPeriod(TemporalVertex current, TemporalVertex transformed) {
200     transformed.setLabel(current.getLabel());
201     transformed.setProperties(current.getProperties());
202     SimpleDateFormat format = new SimpleDateFormat("yyyy-MM-dd HH:mm:ss.SSS");
203     try {
204       String startTime = "starttime";
205       if (current.hasProperty(startTime)) {
206         transformed.setValidFrom(format.parse(current.getPropertyValue(startTime).getString()).getTime());
207         transformed.removeProperty(startTime);
208         String stopTime = "stoptime";
209         if (current.hasProperty(stopTime)) {
210           transformed.setValidTo(format.parse(current.getPropertyValue(stopTime).getString()).getTime());
211           transformed.removeProperty(stopTime);
212         }
213       }
214     } catch (ParseException e) {
215       throw new RuntimeException("Can not parse time.");
216     }
217     return transformed;
218   }
