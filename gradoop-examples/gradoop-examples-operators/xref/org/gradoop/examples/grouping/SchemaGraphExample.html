16  package org.gradoop.examples.grouping;
17  
18  import org.apache.flink.api.java.ExecutionEnvironment;
19  import org.gradoop.examples.common.SocialNetworkGraph;
20  import org.gradoop.flink.model.impl.epgm.LogicalGraph;
21  import org.gradoop.flink.util.FlinkAsciiGraphLoader;
22  import org.gradoop.flink.util.GradoopFlinkConfig;
23  
24  import static java.util.Collections.singletonList;
25  import static org.gradoop.flink.model.impl.operators.grouping.Grouping.LABEL_SYMBOL;
26  
27  /**
28   * A self contained example on how to use the grouping operator on Gradoop's {@link LogicalGraph}
29   * class.
30   *
31   * The example used the graph in dev-support/social-network.pdf
32   */
33  public class SchemaGraphExample {
34  
35    /**
36     * Runs the program on the example data graph
37     *
38     * The example provides an overview over the usage of the grouping() method.
39     * Documentation of possible settings for the grouping operator can be found in the projects wiki.
40     *
41     * Using the social network graph {@link SocialNetworkGraph}, the program will:
42     * <ol>
43     *   <li>load the graph from the given gdl string</li>
44     *   <li>group the graph based on vertex and edge labels</li>
45     *   <li>print the resulting schema graph</li>
46     * </ol>
47     *
48     * @param args arguments
49     * @see <a href="https://github.com/dbs-leipzig/gradoop/wiki/Unary-Logical-Graph-Operators">
50     * Gradoop Wiki</a>
51     * @throws Exception if something goes wrong
52     */
53    public static void main(String[] args) throws Exception {
54  
55      // create flink execution environment
56      ExecutionEnvironment env = ExecutionEnvironment.getExecutionEnvironment();
57  
58      // create loader
59      FlinkAsciiGraphLoader loader = new FlinkAsciiGraphLoader(GradoopFlinkConfig.createConfig(env));
60  
61      // load data
62      loader.initDatabaseFromString(SocialNetworkGraph.getGraphGDLString());
63  
64      // load the graph
65      LogicalGraph graph = loader.getLogicalGraph();
66  
67      // use graph grouping to extract the schema
68      LogicalGraph schema = graph.groupBy(singletonList(LABEL_SYMBOL), singletonList(LABEL_SYMBOL));
69  
70      // print results
71      schema.print();
72    }
73  }
