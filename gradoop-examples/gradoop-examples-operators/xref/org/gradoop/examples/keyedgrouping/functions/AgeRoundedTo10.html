16  package org.gradoop.examples.keyedgrouping.functions;
17  
18  import org.apache.flink.api.common.typeinfo.BasicTypeInfo;
19  import org.apache.flink.api.common.typeinfo.TypeInformation;
20  import org.gradoop.common.model.api.entities.Attributed;
21  import org.gradoop.flink.model.api.functions.KeyFunctionWithDefaultValue;
22  
23  /**
24   * A custom key-function extracting the {@code birthday} property and rounding it to the next {@code 10}.<p>
25   * This key-function is extracting the property and dividing it by {@code 10}, rounding it.<p>
26   * We assume that the age divided by {@code 10} fits in a {@link Byte} value (it's range is
27   * {@value Byte#MIN_VALUE} to {@value Byte#MAX_VALUE}).
28   *
29   * @param <E> The type of elements to extract the key from.
30   */
31  public class AgeRoundedTo10<E extends Attributed> implements KeyFunctionWithDefaultValue<E, Byte> {
32  
33    /**
34     * Extracts the {@code birthday} property from the element and divides it by {@code 10}.
35     * The {@link #getDefaultKey() default key} is returned in case the element does not have the property set.
36     *
37     * @param element The element to extract the key from.
38     * @return The property value, divided by {@code 10}, as a {@link Byte}
39     */
40    @Override
41    public Byte getKey(E element) {
42      if (!element.hasProperty("birthday")) {
43        return getDefaultKey();
44      }
45      return (byte) (element.getPropertyValue("birthday").getInt() / 10);
46    }
47  
48    /**
49     * {@inheritDoc}
50     * <p>
51     * We implement this method to store our grouping-key on super-elements. This is useful in most cases,
52     * since the user want's to know which group is represented by a super-element. In our case we multiply
53     * the key by {@code 10}, since it was divided by {@code 10} in the {@link #getKey(Attributed)} method.<p>
54     * The key is stored as a property with key {@code age_rounded}, we also make sure to ignore the default
55     * key, since {@code -10} would not be a correct age and is just used internally as the default value.
56     *
57     * @param element The element where the key should be stored.
58     * @param key     The key to store on the element. A {@link Byte} in this case.
59     */
60    @Override
61    public void addKeyToElement(E element, Object key) {
62      // Manually checking the type is technically not required, but is recommended to avoid hard to find bugs.
63      if (!(key instanceof Byte)) {
64        throw new IllegalArgumentException("Invalid type for key; " + key);
65      }
66      if (getDefaultKey().equals(key)) {
67        return;
68      }
69      final int actualKey = 10 * (byte) key;
70      element.setProperty("age_rounded", actualKey);
71    }
72  
73    /**
74     * {@inheritDoc}
75     * <p>
76     * We have to provide type infos about the keys extracted by this function.<p>
77     * Flink provides commonly used (mostly primitive) types in this class. In most other cases the
78     * {@link TypeInformation#of(Class)} should suffice.
79     *
80     * @return The type information about keys extracted by this class, {@link Byte} in this case.
81     */
82    @Override
83    public TypeInformation<Byte> getType() {
84      return BasicTypeInfo.BYTE_TYPE_INFO;
85    }
86  
87    /**
88     * {@inheritDoc}
89     * <p>
90     * We have to provide a default value if we want to use this key-function with label-specific grouping.
91     * This value is used for labels other than the one for which this function is to be used.
92     *
93     * @return A default value, in this case {@code -1}.
94     */
95    @Override
96    public Byte getDefaultKey() {
97      return -1;
98    }
99  }
