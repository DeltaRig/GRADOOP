16  package org.gradoop.examples.patternmatch;
17  
18  import org.apache.flink.api.java.ExecutionEnvironment;
19  import org.gradoop.examples.common.SocialNetworkGraph;
20  import org.gradoop.flink.model.impl.epgm.GraphCollection;
21  import org.gradoop.flink.model.impl.epgm.LogicalGraph;
22  import org.gradoop.flink.util.FlinkAsciiGraphLoader;
23  import org.gradoop.flink.util.GradoopFlinkConfig;
24  
25  /**
26   * A self-contained example on how to use the query engine in Gradoop.
27   *
28   * The example uses the graph in dev-support/social-network.pdf
29   */
30  public class GDLQueryExample {
31  
32    /**
33     * Runs the program on the example data graph.
34     *
35     * The example provides an overview over the usage of the {@link LogicalGraph#query(String)}
36     * method. It showcases who a user defined (GDL) query can be applied to a graph.
37     * Documentation and usage examples can be found in the projects wiki.
38     *
39     * Using the social network graph {@link SocialNetworkGraph}, the program will:
40     * <ol>
41     *   <li>create the graph based on the given gdl string</li>
42     *   <li>run the query method with a user defined (GDL) query string</li>
43     *   <li>print all found matches</li>
44     * </ol>
45     *
46     * @param args arguments
47     * @see <a href="https://github.com/dbs-leipzig/gradoop/wiki/Unary-Logical-Graph-Operators">
48     * Gradoop Wiki</a>
49     * @throws Exception in case sth goes wrong
50     */
51    public static void main(String[] args) throws Exception {
52      // create flink execution environment
53      ExecutionEnvironment env = ExecutionEnvironment.getExecutionEnvironment();
54  
55      // create loader
56      FlinkAsciiGraphLoader loader = new FlinkAsciiGraphLoader(GradoopFlinkConfig.createConfig(env));
57  
58      // load data
59      loader.initDatabaseFromString(SocialNetworkGraph.getGraphGDLString());
60  
61      // load graph
62      LogicalGraph socialNetwork = loader.getLogicalGraph();
63  
64      // run the query method (result will be a collection of matched subgraphs)
65      GraphCollection matches = socialNetwork.query(
66        "MATCH (u1:Person)<-[:hasModerator]-(f:Forum)" +
67        "(u2:Person)<-[:hasMember]-(f)" +
68        "WHERE u1.name = \"Alice\"");
69  
70      // print the matches
71      matches.print();
72    }
73  }
